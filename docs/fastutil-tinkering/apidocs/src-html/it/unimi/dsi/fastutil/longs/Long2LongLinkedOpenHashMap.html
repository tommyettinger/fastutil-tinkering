047/**  A type-specific linked hash map with with a fast, small-footprint implementation.
048
049        *
050
051        * <P>Instances of this class use a hash table to represent a map. The table is
052
053        * filled up to a specified <em>load factor</em>, and then doubled in size to
054
055        * accommodate new entries. If the table is emptied below <em>one fourth</em>
056
057        * of the load factor, it is halved in size. However, halving is
058
059        * not performed when deleting entries from an iterator, as it would interfere
060
061        * with the iteration process.
062
063        *
064
065        * <p>Note that {@link #clear()} does not modify the hash table size.
066
067        * Rather, a family of {@linkplain #trim() trimming
068
069        * methods} lets you control the size of the table; this is particularly useful
070
071        * if you reuse instances of this class.
072
073        *
074
075        * <P>Iterators generated by this map will enumerate pairs in the same order in which they
076
077        * have been added to the map (addition of pairs whose key is already present
078
079        * in the set does not change the iteration order). Note that this order has nothing in common with the natural
080
081        * order of the keys. The order is kept by means of a doubly linked list, represented
082
083        * <i>via</i> an array of longs parallel to the table.
084
085        *
086
087        * <P>This class implements the interface of a sorted map, so to allow easy
088
089        * access of the iteration order: for instance, you can get the first key
090
091        * in iteration order with {@code firstKey()} without having to create an
092
093        * iterator; however, this class partially violates the {@link java.util.SortedMap}
094
095        * contract because all submap methods throw an exception and {@link
096
097        * #comparator()} returns always {@code null}.
098
099        *
100
101        * <p>Additional methods, such as <code>getAndMoveToFirst()</code>, make it easy
102
103        * to use instances of this class as a cache (e.g., with LRU policy).
104
105        *
106
107        * <P>The iterators provided by the views of this class using are type-specific
108
109        * {@linkplain java.util.ListIterator list iterators}, and can be started at any
110
111        * element <em>which is a key of the map</em>, or
112
113        * a {@link NoSuchElementException} exception will be thrown.
114
115        * If, however, the provided element is not the first or last key in the
116
117        * set, the first access to the list index will require linear time, as in the worst case
118
119        * the entire key set must be scanned in iteration order to retrieve the positional
120
121        * index of the starting key. If you use just the methods of a type-specific {@link it.unimi.dsi.fastutil.BidirectionalIterator},
122
123        * however, all operations will be performed in constant time.
124
125        *
126
127        * @see Hash
128
129        * @see HashCommon
130
131        */

290        private int realSize() {
291         return containsNullKey ? size - 1 : size;
292        }
293        private void ensureCapacity(final int capacity) {
294         final int needed = arraySize(capacity, f);
295         if (needed > n) rehash(needed);
296        }
297        private void tryCapacity(final long capacity) {
298         final int needed = (int)Math.min(1 << 30, Math.max(2, HashCommon.nextPowerOfTwo((long)Math.ceil(capacity / f))));
299         if (needed > n) rehash(needed);
300        }
301        private long removeEntry(final int pos) {
302         final long oldValue = value[pos];
303         size--;
304         fixPointers(pos);
305         shiftKeys(pos);
306         if (size < maxFill / 4 && n > DEFAULT_INITIAL_SIZE) rehash(n / 2);
307         return oldValue;
308        }
309        private long removeNullEntry() {
310         containsNullKey = false;
311         final long oldValue = value[n];
312         size--;
313         fixPointers(n);
314         if (size < maxFill / 4 && n > DEFAULT_INITIAL_SIZE) rehash(n / 2);
315         return oldValue;
316        }
317        @Override
318        public void putAll(Map<? extends Long,? extends Long> m) {
319         if (f <= .5) ensureCapacity(m.size()); // The resulting map will be sized for m.size() elements
320         else tryCapacity(size() + m.size()); // The resulting map will be tentatively sized for size() + m.size() elements
321         super.putAll(m);
322        }
323        private int insert(final long k, final long v) {
324         int pos;
325         if (( (k) == (0) )) {
326          if (containsNullKey) return n;
327          containsNullKey = true;
328          pos = n;
329         }
330         else {
331          long curr;
332          final long[] key = this.key;
333          // The starting point.
334          if (! ( (curr = key[pos = (int)it.unimi.dsi.fastutil.HashCommon.mix( (k) ) & mask]) == (0) )) {
335           if (( (curr) == (k) )) return pos;
336           while(! ( (curr = key[pos = (pos + 1) & mask]) == (0) ))
337            if (( (curr) == (k) )) return pos;
338          }
339         }
340         key[pos] = k;
341         value[pos] = v;
342         if (size == 0) {
343          first = last = pos;
344          // Special case of SET_UPPER_LOWER(link[pos], -1, -1);
345          link[pos] = -1L;
346         }
347         else {
348          link[last] ^= ( ( link[last] ^ ( pos & 0xFFFFFFFFL ) ) & 0xFFFFFFFFL );
349          link[pos] = ( ( last & 0xFFFFFFFFL ) << 32 ) | ( -1 & 0xFFFFFFFFL );
350          last = pos;
351         }
352         if (size++ >= maxFill) rehash(arraySize(size + 1, f));
353         if (ASSERTS) checkTable();
354         return -1;
355        }

358         final int pos = insert(k, v);
359         if (pos < 0) return defRetValue;
360         final long oldValue = value[pos];
361         value[pos] = v;
362         return oldValue;
363        }

369        /** Adds an increment to value currently associated with a key.
370
371         *
372
373         * <P>Note that this method respects the {@linkplain #defaultReturnValue() default return value} semantics: when
374
375         * called with a key that does not currently appears in the map, the key
376
377         * will be associated with the default return value plus

380
381         *
382
383         * @param k the key.

390        public long addTo(final long k, final long incr) {
391         int pos;
392         if (( (k) == (0) )) {
393          if (containsNullKey) return addToValue(n, incr);
394          pos = n;
395          containsNullKey = true;
396         }
397         else {
398          long curr;
399          final long[] key = this.key;
400          // The starting point.
401          if (! ( (curr = key[pos = (int)it.unimi.dsi.fastutil.HashCommon.mix( (k) ) & mask]) == (0) )) {
402           if (( (curr) == (k) )) return addToValue(pos, incr);
403           while(! ( (curr = key[pos = (pos + 1) & mask]) == (0) ))
404            if (( (curr) == (k) )) return addToValue(pos, incr);
405          }
406         }
407         key[pos] = k;
408         value[pos] = defRetValue + incr;
409         if (size == 0) {
410          first = last = pos;
411          // Special case of SET_UPPER_LOWER(link[pos], -1, -1);
412          link[pos] = -1L;
413         }
414         else {
415          link[last] ^= ( ( link[last] ^ ( pos & 0xFFFFFFFFL ) ) & 0xFFFFFFFFL );
416          link[pos] = ( ( last & 0xFFFFFFFFL ) << 32 ) | ( -1 & 0xFFFFFFFFL );
417          last = pos;
418         }
419         if (size++ >= maxFill) rehash(arraySize(size + 1, f));
420         if (ASSERTS) checkTable();
421         return defRetValue;
422        }
423        /** Shifts left entries with the specified hash code, starting at the specified position,
424
425         * and empties the resulting free entry.
426
427         *
428
429         * @param pos a starting position.
430
431         */
432        protected final void shiftKeys(int pos) {
433         // Shift entries with the same hash.
434         int last, slot;
435         long curr;
436         final long[] key = this.key;
437         for(;;) {
438          pos = ((last = pos) + 1) & mask;
439          for(;;) {
440           if (( (curr = key[pos]) == (0) )) {
441            key[last] = (0);
442            return;
443           }
444           slot = (int)it.unimi.dsi.fastutil.HashCommon.mix( (curr) ) & mask;
445           if (last <= pos ? last >= slot || slot > pos : last >= slot && slot > pos) break;
446           pos = (pos + 1) & mask;
447          }
448          key[last] = curr;
449          value[last] = value[pos];
450          fixPointers(pos, last);
451         }
452        }

454        public long remove(final long k) {
455         if (( (k) == (0) )) {
456          if (containsNullKey) return removeNullEntry();
457          return defRetValue;
458         }
459         long curr;
460         final long[] key = this.key;
461         int pos;
462         // The starting point.
463         if (( (curr = key[pos = (int)it.unimi.dsi.fastutil.HashCommon.mix( (k) ) & mask]) == (0) )) return defRetValue;
464         if (( (k) == (curr) )) return removeEntry(pos);
465         while(true) {
466          if (( (curr = key[pos = (pos + 1) & mask]) == (0) )) return defRetValue;
467          if (( (k) == (curr) )) return removeEntry(pos);
468         }
469        }

475        /** Removes the mapping associated with the first key in iteration order.

482        public long removeFirstLong() {
483         if (size == 0) throw new NoSuchElementException();
484         final int pos = first;
485         // Abbreviated version of fixPointers(pos)
486         first = (int) link[pos];
487         if (0 <= first) {
488          // Special case of SET_PREV(link[first], -1)
489          link[first] |= (-1 & 0xFFFFFFFFL) << 32;
490         }
491         size--;
492         final long v = value[pos];
493         if (pos == n) {
494          containsNullKey = false;
495         }
496         else shiftKeys(pos);
497         if (size < maxFill / 4 && n > DEFAULT_INITIAL_SIZE) rehash(n / 2);
498         return v;
499        }
500        /** Removes the mapping associated with the last key in iteration order.
501
502         * @return the value previously associated with the last key in iteration order.
503
504         * @throws NoSuchElementException is this map is empty.
505
506         */
507        public long removeLastLong() {
508         if (size == 0) throw new NoSuchElementException();
509         final int pos = last;
510         // Abbreviated version of fixPointers(pos)
511         last = (int) ( link[pos] >>> 32 );
512         if (0 <= last) {
513          // Special case of SET_NEXT(link[last], -1)
514          link[last] |= -1 & 0xFFFFFFFFL;
515         }
516         size--;
517         final long v = value[pos];
518         if (pos == n) {
519          containsNullKey = false;
520         }
521         else shiftKeys(pos);
522         if (size < maxFill / 4 && n > DEFAULT_INITIAL_SIZE) rehash(n / 2);
523         return v;
524        }
525        private void moveIndexToFirst(final int i) {
526         if (size == 1 || first == i) return;
527         if (last == i) {
528          last = (int) ( link[i] >>> 32 );
529          // Special case of SET_NEXT(link[last], -1);
530          link[last] |= -1 & 0xFFFFFFFFL;
531         }
532         else {
533          final long linki = link[i];
534          final int prev = (int) ( linki >>> 32 );
535          final int next = (int) linki;
536          link[prev] ^= ( ( link[prev] ^ ( linki & 0xFFFFFFFFL ) ) & 0xFFFFFFFFL );
537          link[next] ^= ( ( link[next] ^ ( linki & 0xFFFFFFFF00000000L ) ) & 0xFFFFFFFF00000000L );
538         }
539         link[first] ^= ( ( link[first] ^ ( ( i & 0xFFFFFFFFL ) << 32 ) ) & 0xFFFFFFFF00000000L );
540         link[i] = ( ( -1 & 0xFFFFFFFFL ) << 32 ) | ( first & 0xFFFFFFFFL );
541         first = i;
542        }
543        private void moveIndexToLast(final int i) {
544         if (size == 1 || last == i) return;
545         if (first == i) {
546          first = (int) link[i];
547          // Special case of SET_PREV(link[first], -1);
548          link[first] |= (-1 & 0xFFFFFFFFL) << 32;
549         }
550         else {
551          final long linki = link[i];
552          final int prev = (int) ( linki >>> 32 );
553          final int next = (int) linki;
554          link[prev] ^= ( ( link[prev] ^ ( linki & 0xFFFFFFFFL ) ) & 0xFFFFFFFFL );
555          link[next] ^= ( ( link[next] ^ ( linki & 0xFFFFFFFF00000000L ) ) & 0xFFFFFFFF00000000L );
556         }
557         link[last] ^= ( ( link[last] ^ ( i & 0xFFFFFFFFL ) ) & 0xFFFFFFFFL );
558         link[i] = ( ( last & 0xFFFFFFFFL ) << 32 ) | ( -1 & 0xFFFFFFFFL );
559         last = i;
560        }
561        /** Returns the value to which the given key is mapped; if the key is present, it is moved to the first position of the iteration order.
562
563         *
564
565         * @param k the key.
566
567         * @return the corresponding value, or the {@linkplain #defaultReturnValue() default return value} if no value was present for the given key.
568
569         */
570        public long getAndMoveToFirst(final long k) {
571         if (( (k) == (0) )) {
572          if (containsNullKey) {
573           moveIndexToFirst(n);
574           return value[n];
575          }
576          return defRetValue;
577         }
578         long curr;
579         final long[] key = this.key;
580         int pos;
581         // The starting point.
582         if (( (curr = key[pos = (int)it.unimi.dsi.fastutil.HashCommon.mix( (k) ) & mask]) == (0) )) return defRetValue;
583         if (( (k) == (curr) )) {
584          moveIndexToFirst(pos);
585          return value[pos];
586         }
587         // There's always an unused entry.
588         while(true) {
589          if (( (curr = key[pos = (pos + 1) & mask]) == (0) )) return defRetValue;
590          if (( (k) == (curr) )) {
591           moveIndexToFirst(pos);
592           return value[pos];
593          }
594         }
595        }

631        /** Adds a pair to the map; if the key is already present, it is moved to the first position of the iteration order.
632
633         *
634
635         * @param k the key.
636
637         * @param v the value.
638
639         * @return the old value, or the {@linkplain #defaultReturnValue() default return value} if no value was present for the given key.
640
641         */
642        public long putAndMoveToFirst(final long k, final long v) {
643         int pos;
644         if (( (k) == (0) )) {
645          if (containsNullKey) {
646           moveIndexToFirst(n);
647           return setValue(n, v);
648          }
649          containsNullKey = true;
650          pos = n;
651         }
652         else {
653          long curr;
654          final long[] key = this.key;
655          // The starting point.
656          if (! ( (curr = key[pos = (int)it.unimi.dsi.fastutil.HashCommon.mix( (k) ) & mask]) == (0) )) {
657           if (( (curr) == (k) )) {
658            moveIndexToFirst(pos);
659            return setValue(pos, v);
660           }
661           while(! ( (curr = key[pos = (pos + 1) & mask]) == (0) ))
662            if (( (curr) == (k) )) {
663             moveIndexToFirst(pos);
664             return setValue(pos, v);
665            }
666          }
667         }
668         key[pos] = k;
669         value[pos] = v;
670         if (size == 0) {

1592

1596
1597         * overriden by subclasses implementing different rehashing strategies (e.g.,
1598
1599         * disk-based rehashing). However, you should not override this method
1600

1602
1603         *
1604
1605         * @param newN the new size
1606
1607         */

1609        protected void rehash(final int newN) {
1610         final long key[] = this.key;
1611         final long value[] = this.value;
1612         final int mask = newN - 1; // Note that this is used by the hashing macro
1613         final long newKey[] = new long[newN + 1];
1614         final long newValue[] = new long[newN + 1];
1615         int i = first, prev = -1, newPrev = -1, t, pos;
1616         final long link[] = this.link;
1617         final long newLink[] = new long[newN + 1];
1618         first = -1;
1619         for(int j = size; j-- != 0;) {
1620          if (( (key[i]) == (0) )) pos = newN;
1621          else {
1622           pos = (int)it.unimi.dsi.fastutil.HashCommon.mix( (key[i]) ) & mask;
1623           while (! ( (newKey[pos]) == (0) )) pos = (pos + 1) & mask;
1624          }
1625          newKey[pos] = key[i];
1626          newValue[pos] = value[i];
1627          if (prev != -1) {
1628           newLink[newPrev] ^= ( ( newLink[newPrev] ^ ( pos & 0xFFFFFFFFL ) ) & 0xFFFFFFFFL );
1629           newLink[pos] ^= ( ( newLink[pos] ^ ( ( newPrev & 0xFFFFFFFFL ) << 32 ) ) & 0xFFFFFFFF00000000L );
1630           newPrev = pos;
1631          }
1632          else {
1633           newPrev = first = pos;
1634           // Special case of SET(newLink[pos], -1, -1);
1635           newLink[pos] = -1L;
1636          }
1637          t = i;
1638          i = (int) link[i];
1639          prev = t;
1640         }
1641         this.link = newLink;
1642         this.last = newPrev;
1643         if (newPrev != -1)
1644          // Special case of SET_NEXT(newLink[newPrev], -1);
1645          newLink[newPrev] |= -1 & 0xFFFFFFFFL;
1646         n = newN;
1647         this.mask = mask;
1648         maxFill = maxFill(n, f);
1649         this.key = newKey;
1650         this.value = newValue;
1651        }
