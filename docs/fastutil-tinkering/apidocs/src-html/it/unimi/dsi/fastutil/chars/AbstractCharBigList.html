141        public CharBigListIterator listIterator(final long index) {
142         ensureIndex(index);
143         return new AbstractCharBigListIterator () {
144           long pos = index, last = -1;
145           @Override
146           public boolean hasNext() { return pos < AbstractCharBigList.this.size64(); }
147           @Override
148           public boolean hasPrevious() { return pos > 0; }
149           @Override
150           public char nextChar() { if (! hasNext()) throw new NoSuchElementException(); return AbstractCharBigList.this.getChar(last = pos++); }
151           @Override
152           public char previousChar() { if (! hasPrevious()) throw new NoSuchElementException(); return AbstractCharBigList.this.getChar(last = --pos); }
153           @Override
154           public long nextIndex() { return pos; }
155           @Override
156           public long previousIndex() { return pos - 1; }
157           @Override
158           public void add(char k) {
159            AbstractCharBigList.this.add(pos++, k);
160            last = -1;
161           }
162           @Override
163           public void set(char k) {
164            if (last == -1) throw new IllegalStateException();
165            AbstractCharBigList.this.set(last, k);
166           }
167           @Override
168           public void remove() {
169            if (last == -1) throw new IllegalStateException();
170            AbstractCharBigList.this.removeChar(last);
171            /* If the last operation was a next(), we are removing an element *before* us, and we must decrease pos correspondingly. */
172            if (last < pos) pos--;
173            last = -1;
174           }
175          };
176        }

317        @Override
318        public boolean equals(final Object o) {
319         if (o == this) return true;
320         if (! (o instanceof BigList)) return false;
321         final BigList<?> l = (BigList<?>)o;
322         long s = size64();
323         if (s != l.size64()) return false;
324         if (l instanceof CharBigList) {
325          final CharBigListIterator i1 = listIterator(), i2 = ((CharBigList )l).listIterator();
326          while(s-- != 0) if (i1.nextChar() != i2.nextChar()) return false;
327          return true;
328         }
329         final BigListIterator<?> i1 = listIterator(), i2 = l.listIterator();
330         while(s-- != 0) if (! valEquals(i1.next(), i2.next())) return false;
331         return true;
332        }
333        /** Compares this big list to another object. If the
334         * argument is a {@link BigList}, this method performs a lexicographical comparison; otherwise,
335         * it throws a <code>ClassCastException</code>.
336         *
337         * @param l a big list.
338         * @return if the argument is a {@link BigList}, a negative integer,
339         * zero, or a positive integer as this list is lexicographically less than, equal
340         * to, or greater than the argument.
341         * @throws ClassCastException if the argument is not a big list.
342         */
343
344        @Override
345        public int compareTo(final BigList<? extends Character> l) {
346         if (l == this) return 0;
347         if (l instanceof CharBigList) {
348          final CharBigListIterator i1 = listIterator(), i2 = ((CharBigList )l).listIterator();
349          int r;
350          char e1, e2;
351          while(i1.hasNext() && i2.hasNext()) {
352           e1 = i1.nextChar();
353           e2 = i2.nextChar();
354           if ((r = ( Character.compare((e1),(e2)) )) != 0) return r;
355          }
356          return i2.hasNext() ? -1 : (i1.hasNext() ? 1 : 0);
357         }
358         BigListIterator<? extends Character> i1 = listIterator(), i2 = l.listIterator();
359         int r;
360         while(i1.hasNext() && i2.hasNext()) {
361          if ((r = ((Comparable<? super Character>)i1.next()).compareTo(i2.next())) != 0) return r;
362         }
363         return i2.hasNext() ? -1 : (i1.hasNext() ? 1 : 0);
364        }

563        /** A class implementing a sublist view. */
564        public static class CharSubList extends AbstractCharBigList implements java.io.Serializable {
565         private static final long serialVersionUID = -7046029254386353129L;
566         /** The list this sublist restricts. */
567         protected final CharBigList l;
568         /** Initial (inclusive) index of this sublist. */
569         protected final long from;
570         /** Final (exclusive) index of this sublist. */
571         protected long to;
572         public CharSubList(final CharBigList l, final long from, final long to) {
573          this.l = l;
574          this.from = from;
575          this.to = to;
576         }
577         private boolean assertRange() {
578          assert from <= l.size64();
579          assert to <= l.size64();
580          assert to >= from;
581          return true;
582         }
583         @Override
584         public boolean add(final char k) {
585          l.add(to, k);
586          to++;
587          assert assertRange();
588          return true;
589         }
590         @Override
591         public void add(final long index, final char k) {
592          ensureIndex(index);
593          l.add(from + index, k);
594          to++;
595          assert assertRange();
596         }
597         @Override
598         public boolean addAll(final long index, final Collection<? extends Character> c) {
599          ensureIndex(index);
600          to += c.size();
601          return l.addAll(from + index, c);
602         }
603         @Override
604         public char getChar(long index) {
605          ensureRestrictedIndex(index);
606          return l.getChar(from + index);
607         }
608         @Override
609         public char removeChar(long index) {
610          ensureRestrictedIndex(index);
611          to--;
612          return l.removeChar(from + index);
613         }
614         @Override
615         public char set(long index, char k) {
616          ensureRestrictedIndex(index);
617          return l.set(from + index, k);
618         }
619         @Override
620         public long size64() {
621          return to - from;
622         }
623         @Override
624         public void getElements(final long from, final char[][] a, final long offset, final long length) {
625          ensureIndex(from);
626          if (from + length > size64()) throw new IndexOutOfBoundsException("End index (" + from + length + ") is greater than list size (" + size64() + ")");
627          l.getElements(this.from + from, a, offset, length);
628         }
629         @Override
630         public void removeElements(final long from, final long to) {
631          ensureIndex(from);
632          ensureIndex(to);
633          l.removeElements(this.from + from, this.from + to);
634          this.to -= (to - from);
635          assert assertRange();
636         }
637         @Override
638         public void addElements(final long index, final char a[][], long offset, long length) {
639          ensureIndex(index);
640          l.addElements(this.from + index, a, offset, length);
641          this.to += length;
642          assert assertRange();
643         }
644         @Override
645         public CharBigListIterator listIterator(final long index) {
646          ensureIndex(index);
647          return new AbstractCharBigListIterator () {
648            long pos = index, last = -1;
649            public boolean hasNext() { return pos < size64(); }
650            public boolean hasPrevious() { return pos > 0; }
651            public char nextChar() { if (! hasNext()) throw new NoSuchElementException(); return l.getChar(from + (last = pos++)); }
652            public char previousChar() { if (! hasPrevious()) throw new NoSuchElementException(); return l.getChar(from + (last = --pos)); }
653            public long nextIndex() { return pos; }
654            public long previousIndex() { return pos - 1; }
655            public void add(char k) {
656             if (last == -1) throw new IllegalStateException();
657             CharSubList.this.add(pos++, k);
658             last = -1;
659             assert assertRange();
660            }
661            public void set(char k) {
662             if (last == -1) throw new IllegalStateException();
663             CharSubList.this.set(last, k);
664            }
665            public void remove() {
666             if (last == -1) throw new IllegalStateException();
667             CharSubList.this.removeChar(last);
668             /* If the last operation was a next(), we are removing an element *before* us, and we must decrease pos correspondingly. */
669             if (last < pos) pos--;
670             last = -1;
671             assert assertRange();
672            }
673           };
674         }
675         @Override
676         public CharBigList subList(final long from, final long to) {
677          ensureIndex(from);
678          ensureIndex(to);
679          if (from > to) throw new IllegalArgumentException("Start index (" + from + ") is greater than end index (" + to + ")");
680          return new CharSubList (this, from, to);
681         }
682         @Override
683         public boolean rem(char k) {
