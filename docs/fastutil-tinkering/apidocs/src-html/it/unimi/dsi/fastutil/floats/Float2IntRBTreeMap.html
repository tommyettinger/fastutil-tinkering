030/** A type-specific red-black tree map with a fast, small-footprint implementation.
031        *

073        /** Generates the comparator that will be actually used.
074         *
075         * <P>When a specific {@link Comparator} is specified and stored in {@link
076         * #storedComparator}, we must check whether it is type-specific.  If it is
077         * so, we can used directly, and we store it in {@link #actualComparator}. Otherwise,
078         * we generate on-the-fly an anonymous class that wraps the non-specific {@link Comparator}
079         * and makes it into a type-specific one.
080         */
081        private void setActualComparator() {
082         /* If the provided comparator is already type-specific, we use it. Otherwise,
083                   we use a wrapper anonymous class to fake that it is type-specific. */
084         if (storedComparator == null || storedComparator instanceof FloatComparator) actualComparator = (FloatComparator)storedComparator;
085         else actualComparator = new FloatComparator () {
086           public int compare(float k1, float k2) {
087            return storedComparator.compare((Float.valueOf(k1)), (Float.valueOf(k2)));
088           }
089           public int compare(Float ok1, Float ok2) {
090            return storedComparator.compare(ok1, ok2);
091           }
092          };
093        }

165        /** Compares two keys in the right way.
166         *
167         * <P>This method uses the {@link #actualComparator} if it is non-{@code null}.
168         * Otherwise, it resorts to primitive type comparisons or to {@link Comparable#compareTo(Object) compareTo()}.
169         *
170         * @param k1 the first key.
171         * @param k2 the second key.
172         * @return a number smaller than, equal to or greater than 0, as usual
173         * (i.e., when k1 &lt; k2, k1 = k2 or k1 &gt; k2, respectively).
174         */
175
176        final int compare(final float k1, final float k2) {
177         return actualComparator == null ? ( Float.compare((k1),(k2)) ) : actualComparator.compare(k1, k2);
178        }
179        /** Returns the entry corresponding to the given key, if it is in the tree; {@code null}, otherwise.
180         *
181         * @param k the key to search for.
182         * @return the corresponding entry, or {@code null} if no entry with the given key exists.
183         */
184        final Entry findKey(final float k) {
185         Entry e = tree;
186         int cmp;
187         while (e != null && (cmp = compare(k, e.key)) != 0) e = cmp < 0 ? e.left() : e.right();
188         return e;
189        }
190        /** Locates a key.
191         *
192         * @param k a key.
193         * @return the last entry on a search for the given key; this will be
194         * the given key, if it present; otherwise, it will be either the smallest greater key or the greatest smaller key.
195         */
196        final Entry locateKey(final float k) {
197         Entry e = tree, last = tree;
198         int cmp = 0;
199         while (e != null && (cmp = compare(k, e.key)) != 0) {
200          last = e;
201          e = cmp < 0 ? e.left() : e.right();
202         }
203         return cmp == 0 ? e : last;
204        }

214        /** Adds an increment to value currently associated with a key.
215         *
216         * <P>Note that this method respects the {@linkplain #defaultReturnValue() default return value} semantics: when
217         * called with a key that does not currently appears in the map, the key
218         * will be associated with the default return value plus
219         * the given increment.
220         *
221         * @param k the key.
222         * @param incr the increment.
223         * @return the old value, or the {@linkplain #defaultReturnValue() default return value} if no value was present for the given key.
224         */
225        public int addTo(final float k, final int incr) {
226         Entry e = add(k);
227         final int oldValue = e.value;
228         e.value += incr;
229         return oldValue;
230        }

238        /** Returns a node with key k in the balanced tree, creating one with defRetValue if necessary.
239         *
240         * @param k the key
241         * @return a node with key k. If a node with key k already exists, then that node is returned,
242         * otherwise a new node with defRetValue is created ensuring that the tree is balanced
243         * after creation of the node.
244         */
245        private Entry add(final float k) {
246         /* After execution of this method, modified is true iff a new entry has been inserted. */
247         modified = false;
248         int maxDepth = 0;
249         Entry e;
250         if (tree == null) { // The case of the empty tree is treated separately.
251          count++;
252          e = tree = lastEntry = firstEntry = new Entry (k, defRetValue);
253         }
254         else {
255          Entry p = tree;
256          int cmp, i = 0;
257          while(true) {
258           if ((cmp = compare(k, p.key)) == 0) {
259            // We clean up the node path, or we could have stale references later.
260            while(i-- != 0) nodePath[i] = null;
261            return p;
262           }
263           nodePath[i] = p;
264           if (dirPath[i++] = cmp > 0) {
265            if (p.succ()) {
266             count++;
267             e = new Entry (k, defRetValue);
268             if (p.right == null) lastEntry = e;
269             e.left = p;
270             e.right = p.right;
271             p.right(e);
272             break;
273            }
274            p = p.right;
275           }
276           else {
277            if (p.pred()) {
278             count++;
279             e = new Entry (k, defRetValue);
280             if (p.left == null) firstEntry = e;
281             e.right = p;
282             e.left = p.left;
283             p.left(e);
284             break;
285            }
286            p = p.left;
287           }
288          }
289          modified = true;
290          maxDepth = i--;
291          while(i > 0 && ! nodePath[i].black()) {
292           if (! dirPath[i - 1]) {
293            Entry y = nodePath[i - 1].right;
294            if (! nodePath[i - 1].succ() && ! y.black()) {
295             nodePath[i].black(true);
296             y.black(true);
297             nodePath[i - 1].black(false);
298             i -= 2;
299            }
300            else {
301             Entry x;
302             if (! dirPath[i]) y = nodePath[i];
303             else {
304              x = nodePath[i];
305              y = x.right;
306              x.right = y.left;
307              y.left = x;
308              nodePath[i - 1].left = y;
309              if (y.pred()) {
310               y.pred(false);
311               x.succ(y);
312              }
313             }
314             x = nodePath[i - 1];
315             x.black(false);
316             y.black(true);
317             x.left = y.right;
318             y.right = x;
319             if (i < 2) tree = y;
320             else {
321              if (dirPath[i - 2]) nodePath[i - 2].right = y;
322              else nodePath[i - 2].left = y;
323             }
324             if (y.succ()) {
325              y.succ(false);
326              x.pred(y);
327             }
328             break;
329            }
330           }
331           else {
332            Entry y = nodePath[i - 1].left;
333            if (! nodePath[i - 1].pred() && ! y.black()) {
334             nodePath[i].black(true);
335             y.black(true);
336             nodePath[i - 1].black(false);
337             i -= 2;
338            }
339            else {
340             Entry x;
341             if (dirPath[i]) y = nodePath[i];
342             else {
343              x = nodePath[i];
344              y = x.left;
345              x.left = y.right;
346              y.right = x;
347              nodePath[i - 1].right = y;
348              if (y.succ()) {
349               y.succ(false);
350               x.pred(y);
351              }
352             }
353             x = nodePath[i - 1];
354             x.black(false);
355             y.black(true);
356             x.right = y.left;
357             y.left = x;
358             if (i < 2) tree = y;
359             else {
360              if (dirPath[i - 2]) nodePath[i - 2].right = y;
361              else nodePath[i - 2].left = y;
362             }
363             if (y.pred()){
364              y.pred(false);
365              x.succ(y);
366             }
367             break;
368            }
369           }
370          }
371         }
372         tree.black(true);
373         // We clean up the node path, or we could have stale references later.
374         while(maxDepth-- != 0) nodePath[maxDepth] = null;
375         if (ASSERTS) {
376          checkNodePath();
377          checkTree(tree, 0, -1);
378         }
379         return e;
380        }

384        @Override
385        public int remove(final float k) {
386         modified = false;
387         if (tree == null) return defRetValue;
388         Entry p = tree;
389         int cmp;
390         int i = 0;
391         final float kk = k;
392         while(true) {
393          if ((cmp = compare(kk, p.key)) == 0) break;
394          dirPath[i] = cmp > 0;
395          nodePath[i] = p;
396          if (dirPath[i++]) {
397           if ((p = p.right()) == null) {
398            // We clean up the node path, or we could have stale references later.
399            while(i-- != 0) nodePath[i] = null;
400            return defRetValue;
401           }
402          }
403          else {
404           if ((p = p.left()) == null) {
405            // We clean up the node path, or we could have stale references later.
406            while(i-- != 0) nodePath[i] = null;
407            return defRetValue;
408           }
409          }
410         }
411         if (p.left == null) firstEntry = p.next();
412         if (p.right == null) lastEntry = p.prev();
413         if (p.succ()) {
414          if (p.pred()) {
415           if (i == 0) tree = p.left;
416           else {
417            if (dirPath[i - 1]) nodePath[i - 1].succ(p.right);
418            else nodePath[i - 1].pred(p.left);
419           }
420          }
421          else {
422           p.prev().right = p.right;
423           if (i == 0) tree = p.left;
424           else {
425            if (dirPath[i - 1]) nodePath[i - 1].right = p.left;
426            else nodePath[i - 1].left = p.left;
427           }
428          }
429         }
430         else {
431          boolean color;
432          Entry r = p.right;
433          if (r.pred()) {
434           r.left = p.left;
435           r.pred(p.pred());
436           if (! r.pred()) r.prev().right = r;
437           if (i == 0) tree = r;
438           else {
439            if (dirPath[i - 1]) nodePath[i - 1].right = r;
440            else nodePath[i - 1].left = r;
441           }
442           color = r.black();
443           r.black(p.black());
444           p.black(color);
445           dirPath[i] = true;
446           nodePath[i++] = r;
447          }
448          else {
449           Entry s;
450           int j = i++;
451           while(true) {
452            dirPath[i] = false;
453            nodePath[i++] = r;
454            s = r.left;
455            if (s.pred()) break;
456            r = s;
457           }
458           dirPath[j] = true;
459           nodePath[j] = s;
460           if (s.succ()) r.pred(s);
461           else r.left = s.right;
462           s.left = p.left;
463           if (! p.pred()) {
464            p.prev().right = s;
465            s.pred(false);
466           }
467           s.right(p.right);
468           color = s.black();
469           s.black(p.black());
470           p.black(color);
471           if (j == 0) tree = s;
472           else {
473            if (dirPath[j - 1]) nodePath[j - 1].right = s;
474            else nodePath[j - 1].left = s;
475           }
476          }
477         }
478         int maxDepth = i;
479         if (p.black()) {
480          for(; i > 0; i--) {
481           if (dirPath[i - 1] && ! nodePath[i - 1].succ() ||
482             ! dirPath[i - 1] && ! nodePath[i - 1].pred()) {
483            Entry x = dirPath[i - 1] ? nodePath[i - 1].right : nodePath[i - 1].left;
484            if (! x.black()) {
485             x.black(true);
486             break;
487            }
488           }
489           if (! dirPath[i - 1]) {
490            Entry w = nodePath[i - 1].right;
491            if (! w.black()) {
492             w.black(true);
493             nodePath[i - 1].black(false);
494             nodePath[i - 1].right = w.left;
495             w.left = nodePath[i - 1];
496             if (i < 2) tree = w;
497             else {
498              if (dirPath[i - 2]) nodePath[i - 2].right = w;
499              else nodePath[i - 2].left = w;
500             }
501             nodePath[i] = nodePath[i - 1];
502             dirPath[i] = false;
503             nodePath[i - 1] = w;
504             if (maxDepth == i++) maxDepth++;
505             w = nodePath[i - 1].right;
506            }
507            if ((w.pred() || w.left.black()) &&
508              (w.succ() || w.right.black())) {
509             w.black(false);
510            }
511            else {
512             if (w.succ() || w.right.black()) {
513              Entry y = w.left;
514              y.black (true);
515              w.black(false);
516              w.left = y.right;
517              y.right = w;
518              w = nodePath[i - 1].right = y;
519              if (w.succ()) {
520               w.succ(false);
521               w.right.pred(w);
522              }
523             }
524             w.black(nodePath[i - 1].black());
525             nodePath[i - 1].black(true);
526             w.right.black(true);
527             nodePath[i - 1].right = w.left;
528             w.left = nodePath[i - 1];
529             if (i < 2) tree = w;
530             else {
531              if (dirPath[i - 2]) nodePath[i - 2].right = w;
532              else nodePath[i - 2].left = w;
533             }
534             if (w.pred()) {
535              w.pred(false);
536              nodePath[i - 1].succ(w);
537             }
538             break;
539            }
540           }
541           else {
542            Entry w = nodePath[i - 1].left;
543            if (! w.black()) {
544             w.black (true);
545             nodePath[i - 1].black(false);
546             nodePath[i - 1].left = w.right;
547             w.right = nodePath[i - 1];
548             if (i < 2) tree = w;
549             else {
550              if (dirPath[i - 2]) nodePath[i - 2].right = w;
551              else nodePath[i - 2].left = w;
552             }
553             nodePath[i] = nodePath[i - 1];
554             dirPath[i] = true;
555             nodePath[i - 1] = w;
556             if (maxDepth == i++) maxDepth++;
557             w = nodePath[i - 1].left;
558            }
559            if ((w.pred() || w.left.black()) &&
560              (w.succ() || w.right.black())) {
561             w.black(false);
562            }
563            else {
564             if (w.pred() || w.left.black()) {
565              Entry y = w.right;
566              y.black(true);
567              w.black (false);
568              w.right = y.left;
569              y.left = w;
570              w = nodePath[i - 1].left = y;
571              if (w.pred()) {
572               w.pred(false);
573               w.left.succ(w);
574              }
575             }
576             w.black(nodePath[i - 1].black());
577             nodePath[i - 1].black(true);
578             w.left.black(true);
579             nodePath[i - 1].left = w.right;
580             w.right = nodePath[i - 1];
581             if (i < 2) tree = w;
582             else {
583              if (dirPath[i - 2]) nodePath[i - 2].right = w;
584              else nodePath[i - 2].left = w;
585             }
586             if (w.succ()) {
587              w.succ(false);
588              nodePath[i - 1].pred(w);
589             }
590             break;
591            }
592           }
593          }
594          if (tree != null) tree.black(true);
595         }
596         modified = true;
597         count--;
598         // We clean up the node path, or we could have stale references later.
599         while(maxDepth-- != 0) nodePath[maxDepth] = null;
600         if (ASSERTS) {
601          checkNodePath();
602          checkTree(tree, 0, -1);
603         }
604         return p.value;
605        }
606        @Override

626        /** This class represent an entry in a tree map.
627         *
628         * <P>We use the only "metadata", i.e., {@link Entry#info}, to store
629         * information about color, predecessor status and successor status.
630         *
631         * <P>Note that since the class is recursive, it can be
632         * considered equivalently a tree.
633         */
634        private static final class Entry extends AbstractFloat2IntMap.BasicEntry implements Cloneable {
635         /** The the bit in this mask is true, the node is black. */
636         private final static int BLACK_MASK = 1;
637         /** If the bit in this mask is true, {@link #right} points to a successor. */
638         private final static int SUCC_MASK = 1 << 31;
639         /** If the bit in this mask is true, {@link #left} points to a predecessor. */
640         private final static int PRED_MASK = 1 << 30;
641         /** The pointers to the left and right subtrees. */
642         Entry left, right;
643         /** This integers holds different information in different bits (see {@link #SUCC_MASK} and {@link #PRED_MASK}. */
644         int info;
645         Entry() {
646          super((0), (0));
647         }
648         /** Creates a new entry with the given key and value.
649                 *
650                 * @param k a key.
651                 * @param v a value.
652                 */
653         Entry(final float k, final int v) {
654          super(k, v);
655          info = SUCC_MASK | PRED_MASK;
656         }
657         /** Returns the left subtree.
658                 *
659                 * @return the left subtree ({@code null} if the left

854         *
855         * <P>This class can iterate in both directions on a threaded tree.

857        private class TreeIterator {
858         /** The entry that will be returned by the next call to {@link java.util.ListIterator#previous()} (or {@code null} if no previous entry exists). */
859         Entry prev;
860         /** The entry that will be returned by the next call to {@link java.util.ListIterator#next()} (or {@code null} if no next entry exists). */
861         Entry next;
862         /** The last entry that was returned (or {@code null} if we did not iterate or used {@link #remove()}). */
863         Entry curr;
864         /** The current index (in the sense of a {@link java.util.ListIterator}). Note that this value is not meaningful when this {@link TreeIterator} has been created using the nonempty constructor.*/
865         int index = 0;
866         TreeIterator() {
867          next = firstEntry;
868         }
869         TreeIterator(final float k) {
870          if ((next = locateKey(k)) != null) {
871           if (compare(next.key, k) <= 0) {
872            prev = next;
873            next = next.next();
874           }
875           else prev = next.prev();
876          }
877         }
878         public boolean hasNext() { return next != null; }
879         public boolean hasPrevious() { return prev != null; }
880         void updateNext() {
881          next = next.next();
882         }
883         Entry nextEntry() {
884          if (! hasNext()) throw new NoSuchElementException();
885          curr = prev = next;
886          index++;
887          updateNext();
888          return curr;
889         }
890         void updatePrevious() {
891          prev = prev.prev();
892         }
893         Entry previousEntry() {
894          if (! hasPrevious()) throw new NoSuchElementException();
895          curr = next = prev;
896          index--;
897          updatePrevious();
898          return curr;
899         }
900         public int nextIndex() {
901          return index;
902         }
903         public int previousIndex() {
904          return index - 1;
905         }
906         public void remove() {
907          if (curr == null) throw new IllegalStateException();
908          /* If the last operation was a next(), we are removing an entry that preceeds
909                           the current index, and thus we must decrement it. */
910          if (curr == prev) index--;
911          next = prev = curr;
912          updatePrevious();
913          updateNext();
914          Float2IntRBTreeMap.this.remove(curr.key);
915          curr = null;
916         }
917         public int skip(final int n) {
918          int i = n;
919          while(i-- != 0 && hasNext()) nextEntry();
920          return n - i - 1;
921         }
922         public int back(final int n) {
923          int i = n;
924          while(i-- != 0 && hasPrevious()) previousEntry();
925          return n - i - 1;
926         }

1339         /** Locates the first entry.
1340                 *
1341                 * @return the first entry of this submap, or {@code null} if the submap is empty.
1342                 */
1343         public Float2IntRBTreeMap.Entry firstEntry() {
1344          if (tree == null) return null;
1345          // If this submap goes to -infinity, we return the main map first entry; otherwise, we locate the start of the map.
1346          Float2IntRBTreeMap.Entry e;
1347          if (bottom) e = firstEntry;
1348          else {
1349           e = locateKey(from);
1350           // If we find either the start or something greater we're OK.
1351           if (compare(e.key, from) < 0) e = e.next();
1352          }
1353          // Finally, if this submap doesn't go to infinity, we check that the resulting key isn't greater than the end.
1354          if (e == null || ! top && compare(e.key, to) >= 0) return null;
1355          return e;
1356         }

1358                 *

1362          if (tree == null) return null;
1363          // If this submap goes to infinity, we return the main map last entry; otherwise, we locate the end of the map.
1364          Float2IntRBTreeMap.Entry e;
1365          if (top) e = lastEntry;
1366          else {
1367           e = locateKey(to);

1369           if (compare(e.key, to) >= 0) e = e.prev();
1370          }
1371          // Finally, if this submap doesn't go to -infinity, we check that the resulting key isn't smaller than the start.
1372          if (e == null || ! bottom && compare(e.key, from) < 0) return null;
1373          return e;
1374         }

1387         /** An iterator for subranges.
1388                 *
1389                 * <P>This class inherits from {@link TreeIterator}, but overrides the methods that
1390                 * update the pointer after a {@link java.util.ListIterator#next()} or {@link java.util.ListIterator#previous()}. If we would
1391                 * move out of the range of the submap we just overwrite the next or previous
1392                 * entry with {@code null}.
1393                 */
1394         private class SubmapIterator extends TreeIterator {
1395          SubmapIterator() {
1396           next = firstEntry();
1397          }
1398          SubmapIterator(final float k) {
1399           this();
1400           if (next != null) {
1401            if (! bottom && compare(k, next.key) < 0) prev = null;
1402            else if (! top && compare(k, (prev = lastEntry()).key) >= 0) next = null;
1403            else {
1404             next = locateKey(k);
1405             if (compare(next.key, k) <= 0) {
1406              prev = next;
1407              next = next.next();
1408             }
1409             else prev = next.prev();
1410            }
1411           }
1412          }
1413          void updatePrevious() {
1414           prev = prev.prev();
1415           if (! bottom && prev != null && Float2IntRBTreeMap.this.compare(prev.key, from) < 0) prev = null;
1416          }
1417          void updateNext() {
1418           next = next.next();
1419           if (! top && next != null && Float2IntRBTreeMap.this.compare(next.key, to) >= 0) next = null;
1420          }
1421         }

1574        /** Reads the given number of entries from the input stream, returning the corresponding tree.
1575         *
1576         * @param s the input stream.
1577         * @param n the (positive) number of entries to read.
1578         * @param pred the entry containing the key that preceeds the first key in the tree.
1579         * @param succ the entry containing the key that follows the last key in the tree.
1580         */
1581
1582        private Entry readTree(final java.io.ObjectInputStream s, final int n, final Entry pred, final Entry succ) throws java.io.IOException, ClassNotFoundException {
1583         if (n == 1) {
1584          final Entry top = new Entry ( s.readFloat(), s.readInt());
1585          top.pred(pred);
1586          top.succ(succ);
1587          top.black(true);
1588          return top;
1589         }
1590         if (n == 2) {
1591          /* We handle separately this case so that recursion will
1592                         *always* be on nonempty subtrees. */
1593          final Entry top = new Entry ( s.readFloat(), s.readInt());
1594          top.black(true);
1595          top.right(new Entry ( s.readFloat(), s.readInt()));
1596          top.right.pred(top);
1597          top.pred(pred);
1598          top.right.succ(succ);
1599          return top;
1600         }
1601         // The right subtree is the largest one.
1602         final int rightN = n / 2, leftN = n - rightN - 1;
1603         final Entry top = new Entry ();
1604         top.left(readTree(s, leftN, pred, top));
1605         top.key = s.readFloat();
1606         top.value = s.readInt();
1607         top.black(true);
1608         top.right(readTree(s, rightN, top, succ));
1609         if (n + 2 == ((n + 2) & -(n + 2))) top.right.black(false); // Quick test for determining whether n + 2 is a power of 2.
1610         return top;
1611        }
