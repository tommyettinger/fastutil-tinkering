101        /** A class to wrap arrays in iterators. */
102        private static class ArrayIterator extends AbstractByteListIterator {
103         private final byte[] array;
104         private final int offset, length;
105         private int curr;
106         public ArrayIterator(final byte[] array, final int offset, final int length) {
107          this.array = array;
108          this.offset = offset;
109          this.length = length;
110         }
111         @Override
112         public boolean hasNext() { return curr < length; }
113         @Override
114         public boolean hasPrevious() { return curr > 0; }
115         @Override
116         public byte nextByte() {
117          if (! hasNext()) throw new NoSuchElementException();
118          return array[offset + curr++];
119         }
120         @Override
121         public byte previousByte() {
122          if (! hasPrevious()) throw new NoSuchElementException();
123          return array[offset + --curr];
124         }
125         @Override
126         public int skip(int n) {
127          if (n <= length - curr) {
128           curr += n;
129           return n;
130          }
131          n = length - curr;
132          curr = length;
133          return n;
134         }
135         @Override
136         public int back(int n) {
137          if (n <= curr) {
138           curr -= n;
139           return n;
140          }
141          n = curr;
142          curr = 0;
143          return n;
144         }
145         @Override
146         public int nextIndex() {
147          return curr;
148         }
149         @Override
150         public int previousIndex() {
151          return curr - 1;
152         }
153        }

180        /** Unwraps an iterator into an array starting at a given offset for a given number of elements.
181         *
182         * <P>This method iterates over the given type-specific iterator and stores the elements
183         * returned, up to a maximum of <code>length</code>, in the given array starting at <code>offset</code>.
184         * The number of actually unwrapped elements is returned (it may be less than <code>max</code> if
185         * the iterator emits less than <code>max</code> elements).
186         *
187         * @param i a type-specific iterator.
188         * @param array an array to contain the output of the iterator.
189         * @param offset the first element of the array to be returned.
190         * @param max the maximum number of elements to unwrap.
191         * @return the number of elements unwrapped.
192         */
193        public static int unwrap(final ByteIterator i, final byte array[], int offset, final int max) {
194         if (max < 0) throw new IllegalArgumentException("The maximum number of elements (" + max + ") is negative");
195         if (offset < 0 || offset + max > array.length) throw new IllegalArgumentException();
196         int j = max;
197         while(j-- != 0 && i.hasNext()) array[offset++] = i.nextByte();
198         return max - j - 1;
199        }
200        /** Unwraps an iterator into an array.
201         *
202         * <P>This method iterates over the given type-specific iterator and stores the
203         * elements returned in the given array. The iteration will stop when the
204         * iterator has no more elements or when the end of the array has been reached.
205         *
206         * @param i a type-specific iterator.
207         * @param array an array to contain the output of the iterator.
208         * @return the number of elements unwrapped.
209         */
210        public static int unwrap(final ByteIterator i, final byte array[]) {
211         return unwrap(i, array, 0, array.length);
212        }
213        /** Unwraps an iterator, returning an array, with a limit on the number of elements.
214         *
215         * <P>This method iterates over the given type-specific iterator and returns an array
216         * containing the elements returned by the iterator. At most <code>max</code> elements
217         * will be returned.
218         *
219         * @param i a type-specific iterator.
220         * @param max the maximum number of elements to be unwrapped.
221         * @return an array containing the elements returned by the iterator (at most <code>max</code>).
222         */
223
224        public static byte[] unwrap(final ByteIterator i, int max) {
225         if (max < 0) throw new IllegalArgumentException("The maximum number of elements (" + max + ") is negative");
226         byte array[] = new byte[16];
227         int j = 0;
228         while(max-- != 0 && i.hasNext()) {
229          if (j == array.length) array = ByteArrays.grow(array, j + 1);
230          array[j++] = i.nextByte();
231         }
232         return ByteArrays.trim(array, j);
233        }

240         * @return an array containing the elements returned by the iterator.

242        public static byte[] unwrap(final ByteIterator i) {

245        /** Unwraps an iterator into a type-specific collection,  with a limit on the number of elements.
246         *
247         * <P>This method iterates over the given type-specific iterator and stores the elements
248         * returned, up to a maximum of <code>max</code>, in the given type-specific collection.
249         * The number of actually unwrapped elements is returned (it may be less than <code>max</code> if
250         * the iterator emits less than <code>max</code> elements).
251         *
252         * @param i a type-specific iterator.
253         * @param c a type-specific collection array to contain the output of the iterator.
254         * @param max the maximum number of elements to unwrap.
255         * @return the number of elements unwrapped. Note that
256         * this is the number of elements returned by the iterator, which is not necessarily the number
257         * of elements that have been added to the collection (because of duplicates).
258         */
259        public static int unwrap(final ByteIterator i, final ByteCollection c, final int max) {
260         if (max < 0) throw new IllegalArgumentException("The maximum number of elements (" + max + ") is negative");
261         int j = max;
262         while(j-- != 0 && i.hasNext()) c.add(i.nextByte());
263         return max - j - 1;
264        }
265        /** Unwraps an iterator into a type-specific collection.
266         *
267         * <P>This method iterates over the given type-specific iterator and stores the
268         * elements returned in the given type-specific collection. The returned count on the number
269         * unwrapped elements is a long, so that it will work also with very large collections.
270         *
271         * @param i a type-specific iterator.
272         * @param c a type-specific collection to contain the output of the iterator.
273         * @return the number of elements unwrapped. Note that
274         * this is the number of elements returned by the iterator, which is not necessarily the number
275         * of elements that have been added to the collection (because of duplicates).
276         */
277        public static long unwrap(final ByteIterator i, final ByteCollection c) {
278         long n = 0;
279         while(i.hasNext()) {
280          c.add(i.nextByte());
281          n++;
282         }
283         return n;
284        }

420        private static class IntervalIterator extends AbstractByteListIterator {
421         private final byte from, to;
422         byte curr;
423         public IntervalIterator(final byte from, final byte to) {
424          this.from = this.curr = from;
425          this.to = to;
426         }
427         @Override
428         public boolean hasNext() { return curr < to; }
429         @Override
430         public boolean hasPrevious() { return curr > from; }
431         @Override
432         public byte nextByte() {
433          if (! hasNext()) throw new NoSuchElementException();
434          return curr++;
435         }
436         @Override
437         public byte previousByte() {
438          if (! hasPrevious()) throw new NoSuchElementException();
439          return --curr;
440         }
441         @Override
442         public int nextIndex() { return curr - from; }
443         @Override
444         public int previousIndex() { return curr - from - 1; }
445         @Override
446         public int skip(int n) {
447          if (curr + n <= to) {
448           curr += n;
449           return n;
450          }
451          n = to - curr;
452          curr = to;
453          return n;
454         }
455         @Override
456         public int back(int n) {
457          if (curr - n >= from) {
458           curr -= n;
459           return n;
460          }
461          n = curr - from ;
462          curr = from;
463          return n;
464         }
465        }

478        private static class IteratorConcatenator extends AbstractByteIterator {
479         final ByteIterator a[];
480         int offset, length, lastOffset = -1;
481         public IteratorConcatenator(final ByteIterator a[], int offset, int length) {
482          this.a = a;
483          this.offset = offset;
484          this.length = length;
485          advance();
486         }
487         private void advance() {
488          while(length != 0) {
489           if (a[offset].hasNext()) break;
490           length--;
491           offset++;
492          }
493          return;
494         }
495         @Override
496         public boolean hasNext() {
497          return length > 0;
498         }
499         @Override
500         public byte nextByte() {
501          if (! hasNext()) throw new NoSuchElementException();
502          byte next = a[lastOffset = offset].nextByte();
503          advance();
504          return next;
505         }
506         @Override
507         public void remove() {
508          if (lastOffset == -1) throw new IllegalStateException();
509          a[lastOffset].remove();
510         }
511         @Override
512         public int skip(int n) {
513          lastOffset = -1;
514          int skipped = 0;
515          while(skipped < n && length != 0) {
516           skipped += a[offset].skip(n - skipped);
517           if (a[offset].hasNext()) break;
518           length--;
519           offset++;
520          }
521          return skipped;
522         }
523        }
