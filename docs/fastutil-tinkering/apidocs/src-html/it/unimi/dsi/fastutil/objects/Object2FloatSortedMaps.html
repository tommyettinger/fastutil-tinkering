023/** A class providing static methods and objects that do useful things with type-specific sorted maps.

029        /** Returns a comparator for entries based on a given comparator on keys.
030         *
031         * @param comparator a comparator on keys.
032         * @return the associated comparator on entries.
033         */
034        public static <K> Comparator<? super Map.Entry<K, ?>> entryComparator(final Comparator <K> comparator) {
035         return new Comparator<Map.Entry<K, ?>>() {
036          public int compare(Map.Entry<K, ?> x, Map.Entry<K, ?> y) {
037           return comparator.compare(x.getKey(), y.getKey());
038          }
039         };
040        }

090        /** An immutable class representing a type-specific singleton sorted map.
091         *
092         * <P>This class may be useful to implement your own in case you subclass
093         * a type-specific sorted map.
094         */
095        public static class Singleton <K> extends Object2FloatMaps.Singleton <K> implements Object2FloatSortedMap <K>, java.io.Serializable, Cloneable {
096         private static final long serialVersionUID = -7046029254386353129L;
097         protected final Comparator <? super K> comparator;
098         protected Singleton(final K key, final float value, Comparator <? super K> comparator) {
099          super(key, value);
100          this.comparator = comparator;
101         }
102         protected Singleton(final K key, final float value) {
103          this(key, value, null);
104         }
105         @SuppressWarnings("unchecked")
106         final int compare(final K k1, final K k2) {
107          return comparator == null ? ( ((Comparable<K>)(k1)).compareTo(k2) ) : comparator.compare(k1, k2);
108         }
109         @Override
110         public Comparator <? super K> comparator() { return comparator; }
111         @SuppressWarnings("unchecked")
112         @Override
113         public ObjectSortedSet<Object2FloatMap.Entry <K> > object2FloatEntrySet() { if (entries == null) entries = ObjectSortedSets.singleton((Object2FloatMap.Entry <K>)new AbstractObject2FloatMap.BasicEntry <K>(key, value), (Comparator<? super Object2FloatMap.Entry <K> >)entryComparator(comparator)); return (ObjectSortedSet<Object2FloatMap.Entry <K> >)entries; }
114         /** {@inheritDoc}
115                 * @deprecated Please use the corresponding type-specific method instead. */
116         @Deprecated
117         @Override
118         @SuppressWarnings({ "rawtypes", "unchecked" })
119         public ObjectSortedSet<Map.Entry<K, Float>> entrySet() { return (ObjectSortedSet)object2FloatEntrySet(); }
120         @Override
121         public ObjectSortedSet <K> keySet() { if (keys == null) keys = ObjectSortedSets.singleton(key, comparator); return (ObjectSortedSet <K>)keys; }
122         @SuppressWarnings("unchecked")
123         @Override
124         public Object2FloatSortedMap <K> subMap(final K from, final K to) { if (compare(from, key) <= 0 && compare(key, to) < 0) return this; return EMPTY_MAP; }
125         @SuppressWarnings("unchecked")
126         @Override
127         public Object2FloatSortedMap <K> headMap(final K to) { if (compare(key, to) < 0) return this; return EMPTY_MAP; }
128         @SuppressWarnings("unchecked")
129         @Override
130         public Object2FloatSortedMap <K> tailMap(final K from) { if (compare(from, key) <= 0) return this; return EMPTY_MAP; }
131         @Override
132         public K firstKey() { return key; }
133         @Override
134         public K lastKey() { return key; }
135        }

182         /** A synchronized wrapper class for sorted maps. */
183        public static class SynchronizedSortedMap <K> extends Object2FloatMaps.SynchronizedMap <K> implements Object2FloatSortedMap <K>, java.io.Serializable {
184         private static final long serialVersionUID = -7046029254386353129L;
185         protected final Object2FloatSortedMap <K> sortedMap;
186         protected SynchronizedSortedMap(final Object2FloatSortedMap <K> m, final Object sync) {
187          super(m, sync);
188          sortedMap = m;
189         }
190         protected SynchronizedSortedMap(final Object2FloatSortedMap <K> m) {
191          super(m);
192          sortedMap = m;
193         }
194         @Override
195         public Comparator <? super K> comparator() { synchronized(sync) { return sortedMap.comparator(); } }
196         @Override
197         public ObjectSortedSet<Object2FloatMap.Entry <K> > object2FloatEntrySet() { if (entries == null) entries = ObjectSortedSets.synchronize(sortedMap.object2FloatEntrySet(), sync); return (ObjectSortedSet<Object2FloatMap.Entry <K> >)entries; }
198         /** {@inheritDoc}
199                 * @deprecated Please use the corresponding type-specific method instead. */
200         @Deprecated
201         @Override
202         @SuppressWarnings({ "rawtypes", "unchecked" })
203         public ObjectSortedSet<Map.Entry<K, Float>> entrySet() { return (ObjectSortedSet)object2FloatEntrySet(); }
204         @Override
205         public ObjectSortedSet <K> keySet() { if (keys == null) keys = ObjectSortedSets.synchronize(sortedMap.keySet(), sync); return (ObjectSortedSet <K>)keys; }
206         @Override
207         public Object2FloatSortedMap <K> subMap(final K from, final K to) { return new SynchronizedSortedMap <K>(sortedMap.subMap(from, to), sync); }
208         @Override
209         public Object2FloatSortedMap <K> headMap(final K to) { return new SynchronizedSortedMap <K>(sortedMap.headMap(to), sync); }
210         @Override
211         public Object2FloatSortedMap <K> tailMap(final K from) { return new SynchronizedSortedMap <K>(sortedMap.tailMap(from), sync); }
212         @Override
213         public K firstKey() { synchronized(sync) { return sortedMap.firstKey(); } }
214         @Override
215         public K lastKey() { synchronized(sync) { return sortedMap.lastKey(); } }
216        }
217        /** Returns a synchronized type-specific sorted map backed by the given type-specific sorted map.
218         *
219         * @param m the sorted map to be wrapped in a synchronized sorted map.
220         * @return a synchronized view of the specified sorted map.
221         * @see java.util.Collections#synchronizedSortedMap(SortedMap)
222         */
223        public static <K> Object2FloatSortedMap <K> synchronize(final Object2FloatSortedMap <K> m) { return new SynchronizedSortedMap <K>(m); }
224        /** Returns a synchronized type-specific sorted map backed by the given type-specific sorted map, using an assigned object to synchronize.

230         */
231        public static <K> Object2FloatSortedMap <K> synchronize(final Object2FloatSortedMap <K> m, final Object sync) { return new SynchronizedSortedMap <K>(m, sync); }
