029        /** Returns a comparator for entries based on a given comparator on keys.
030         *
031         * @param comparator a comparator on keys.
032         * @return the associated comparator on entries.
033         */
034        public static Comparator<? super Map.Entry<Long, ?>> entryComparator(final LongComparator comparator) {
035         return new Comparator<Map.Entry<Long, ?>>() {
036          public int compare(Map.Entry<Long, ?> x, Map.Entry<Long, ?> y) {
037           return comparator.compare(x.getKey(), y.getKey());
038          }
039         };
040        }

106        /** An immutable class representing a type-specific singleton sorted map.
107         *
108         * <P>This class may be useful to implement your own in case you subclass
109         * a type-specific sorted map.
110         */
111        public static class Singleton extends Long2IntMaps.Singleton implements Long2IntSortedMap , java.io.Serializable, Cloneable {
112         private static final long serialVersionUID = -7046029254386353129L;
113         protected final LongComparator comparator;
114         protected Singleton(final long key, final int value, LongComparator comparator) {
115          super(key, value);
116          this.comparator = comparator;
117         }
118         protected Singleton(final long key, final int value) {
119          this(key, value, null);
120         }
121        
122         final int compare(final long k1, final long k2) {
123          return comparator == null ? ( Long.compare((k1),(k2)) ) : comparator.compare(k1, k2);
124         }
125         @Override
126         public LongComparator comparator() { return comparator; }
127        
128         @Override
129         public ObjectSortedSet<Long2IntMap.Entry > long2IntEntrySet() { if (entries == null) entries = ObjectSortedSets.singleton((Long2IntMap.Entry )new AbstractLong2IntMap.BasicEntry (key, value), (Comparator<? super Long2IntMap.Entry >)entryComparator(comparator)); return (ObjectSortedSet<Long2IntMap.Entry >)entries; }
130         /** {@inheritDoc}
131                 * @deprecated Please use the corresponding type-specific method instead. */
132         @Deprecated
133         @Override
134         @SuppressWarnings({ "rawtypes", "unchecked" })
135         public ObjectSortedSet<Map.Entry<Long, Integer>> entrySet() { return (ObjectSortedSet)long2IntEntrySet(); }
136         @Override
137         public LongSortedSet keySet() { if (keys == null) keys = LongSortedSets.singleton(key, comparator); return (LongSortedSet )keys; }
138        
139         @Override
140         public Long2IntSortedMap subMap(final long from, final long to) { if (compare(from, key) <= 0 && compare(key, to) < 0) return this; return EMPTY_MAP; }
141        
142         @Override
143         public Long2IntSortedMap headMap(final long to) { if (compare(key, to) < 0) return this; return EMPTY_MAP; }
144        
145         @Override
146         public Long2IntSortedMap tailMap(final long from) { if (compare(from, key) <= 0) return this; return EMPTY_MAP; }
147         @Override
148         public long firstLongKey() { return key; }
149         @Override
150         public long lastLongKey() { return key; }
151         /** {@inheritDoc}
152                 * @deprecated Please use the corresponding type-specific method instead. */
153         @Deprecated
154         @Override
155         public Long2IntSortedMap headMap(Long oto) { return headMap(((oto).longValue())); }
156         /** {@inheritDoc}
157                 * @deprecated Please use the corresponding type-specific method instead. */
158         @Deprecated
159         @Override
160         public Long2IntSortedMap tailMap(Long ofrom) { return tailMap(((ofrom).longValue())); }
161         /** {@inheritDoc}
162                 * @deprecated Please use the corresponding type-specific method instead. */
163         @Deprecated
164         @Override
165         public Long2IntSortedMap subMap(Long ofrom, Long oto) { return subMap(((ofrom).longValue()), ((oto).longValue())); }
166         /** {@inheritDoc}
167                 * @deprecated Please use the corresponding type-specific method instead. */
168         @Deprecated
169         @Override
170         public Long firstKey() { return (Long.valueOf(firstLongKey())); }
171         /** {@inheritDoc}
172                 * @deprecated Please use the corresponding type-specific method instead. */
173         @Deprecated
174         @Override
175         public Long lastKey() { return (Long.valueOf(lastLongKey())); }
176        }

223         /** A synchronized wrapper class for sorted maps. */
224        public static class SynchronizedSortedMap extends Long2IntMaps.SynchronizedMap implements Long2IntSortedMap , java.io.Serializable {
225         private static final long serialVersionUID = -7046029254386353129L;
226         protected final Long2IntSortedMap sortedMap;
227         protected SynchronizedSortedMap(final Long2IntSortedMap m, final Object sync) {
228          super(m, sync);
229          sortedMap = m;
230         }
231         protected SynchronizedSortedMap(final Long2IntSortedMap m) {
232          super(m);
233          sortedMap = m;
234         }
235         @Override
236         public LongComparator comparator() { synchronized(sync) { return sortedMap.comparator(); } }
237         @Override
238         public ObjectSortedSet<Long2IntMap.Entry > long2IntEntrySet() { if (entries == null) entries = ObjectSortedSets.synchronize(sortedMap.long2IntEntrySet(), sync); return (ObjectSortedSet<Long2IntMap.Entry >)entries; }
239         /** {@inheritDoc}

241         @Deprecated

245         @Override
246         public LongSortedSet keySet() { if (keys == null) keys = LongSortedSets.synchronize(sortedMap.keySet(), sync); return (LongSortedSet )keys; }
247         @Override
248         public Long2IntSortedMap subMap(final long from, final long to) { return new SynchronizedSortedMap (sortedMap.subMap(from, to), sync); }
249         @Override
