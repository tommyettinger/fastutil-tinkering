016package it.unimi.dsi.fastutil.objects;
017import java.util.Collection;
018import java.util.Set;
019/** A class providing static methods and objects that do useful things with type-specific sets.
020        *
021        * @see java.util.Collections
022        */
023public class ReferenceSets {
024        private ReferenceSets() {}
025        /** An immutable class representing the empty set and implementing a type-specific set interface.
026         *
027         * <P>This class may be useful to implement your own in case you subclass
028         * a type-specific set.
029         */
030        public static class EmptySet <K> extends ReferenceCollections.EmptyCollection <K> implements ReferenceSet <K>, java.io.Serializable, Cloneable {
031         private static final long serialVersionUID = -7046029254386353129L;
032         protected EmptySet() {}
033         @Override
034         public boolean remove(Object ok) { throw new UnsupportedOperationException(); }
035         @Override
036         public Object clone() { return EMPTY_SET; }
037         @Override
038         @SuppressWarnings("rawtypes")
039         public boolean equals(final Object o) { return o instanceof Set && ((Set)o).isEmpty(); }
040         private Object readResolve() { return EMPTY_SET; }
041        }
042        /** An empty set (immutable). It is serializable and cloneable.
043         */
044        @SuppressWarnings("rawtypes")
045        public static final EmptySet EMPTY_SET = new EmptySet();
046        /** Returns an empty set (immutable). It is serializable and cloneable.
047         *
048         * <P>This method provides a typesafe access to {@link #EMPTY_SET}.
049         * @return an empty set (immutable).
050         */
051        @SuppressWarnings("unchecked")
052        public static <K> ReferenceSet <K> emptySet() {
053         return EMPTY_SET;
054        }
055        /** An immutable class representing a type-specific singleton set.
056         *
057         * <P>This class may be useful to implement your own in case you subclass
058         * a type-specific set.  */
059        public static class Singleton <K> extends AbstractReferenceSet <K> implements java.io.Serializable, Cloneable {
060         private static final long serialVersionUID = -7046029254386353129L;
061         protected final K element;
062         protected Singleton(final K element) {
063          this.element = element;
064         }
065         @Override
066         public boolean contains(final Object k) { return ( (k) == (element) ); }
067         @Override
068         public boolean remove(final Object k) { throw new UnsupportedOperationException(); }
069         @Override
070         public ObjectListIterator <K> iterator() { return ObjectIterators.singleton(element); }
071         @Override
072         public int size() { return 1; }
073         @Override
074         public boolean addAll(final Collection<? extends K> c) { throw new UnsupportedOperationException(); }
075         @Override
076         public boolean removeAll(final Collection<?> c) { throw new UnsupportedOperationException(); }
077         @Override
078         public boolean retainAll(final Collection<?> c) { throw new UnsupportedOperationException(); }
079         @Override
080         public Object clone() { return this; }
081        }
082        /** Returns a type-specific immutable set containing only the specified element. The returned set is serializable and cloneable.
083         *
084         * @param element the only element of the returned set.
085         * @return a type-specific immutable set containing just <code>element</code>.
086         */
087        public static <K> ReferenceSet <K> singleton(final K element) {
088         return new Singleton <K>(element);
089        }
090        /** A synchronized wrapper class for sets. */
091        public static class SynchronizedSet <K> extends ReferenceCollections.SynchronizedCollection <K> implements ReferenceSet <K>, java.io.Serializable {
092         private static final long serialVersionUID = -7046029254386353129L;
093         protected SynchronizedSet(final ReferenceSet <K> s, final Object sync) {
094          super(s, sync);
095         }
096         protected SynchronizedSet(final ReferenceSet <K> s) {
097          super(s);
098         }
099         @Override
100         public boolean remove(final Object k) { synchronized(sync) { return collection.remove(k); } }
101        }
102        /** Returns a synchronized type-specific set backed by the given type-specific set.
103         *
104         * @param s the set to be wrapped in a synchronized set.
105         * @return a synchronized view of the specified set.
106         * @see java.util.Collections#synchronizedSet(Set)
107         */
108        public static <K> ReferenceSet <K> synchronize(final ReferenceSet <K> s) { return new SynchronizedSet <K>(s); }
109        /** Returns a synchronized type-specific set backed by the given type-specific set, using an assigned object to synchronize.
110         *
111         * @param s the set to be wrapped in a synchronized set.
112         * @param sync an object that will be used to synchronize the access to the set.
113         * @return a synchronized view of the specified set.
114         * @see java.util.Collections#synchronizedSet(Set)
115         */
116        public static <K> ReferenceSet <K> synchronize(final ReferenceSet <K> s, final Object sync) { return new SynchronizedSet <K>(s, sync); }
117        /** An unmodifiable wrapper class for sets. */
118        public static class UnmodifiableSet <K> extends ReferenceCollections.UnmodifiableCollection <K> implements ReferenceSet <K>, java.io.Serializable {
119         private static final long serialVersionUID = -7046029254386353129L;
120         protected UnmodifiableSet(final ReferenceSet <K> s) {
121          super(s);
122         }
123         @Override
124         public boolean remove(final Object k) { throw new UnsupportedOperationException(); }
125         @Override
126         public boolean equals(final Object o) { if (o == this) return true; return collection.equals(o); }
127         @Override
128         public int hashCode() { return collection.hashCode(); }
129        }
130        /** Returns an unmodifiable type-specific set backed by the given type-specific set.
131         *
132         * @param s the set to be wrapped in an unmodifiable set.
133         * @return an unmodifiable view of the specified set.
134         * @see java.util.Collections#unmodifiableSet(Set)
135         */
136        public static <K> ReferenceSet <K> unmodifiable(final ReferenceSet <K> s) { return new UnmodifiableSet <K>(s); }
137}
