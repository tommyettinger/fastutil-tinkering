016package it.unimi.dsi.fastutil.floats;
017import java.util.Iterator;
018import java.util.NoSuchElementException;
019/** A class providing static methods and objects that do useful things with type-specific iterators.
020        *
021        * @see Iterator
022        */
023public class FloatBigListIterators {
024        private FloatBigListIterators() {}
025        /** A class returning no elements and a type-specific big list iterator interface.
026         *
027         * <P>This class may be useful to implement your own in case you subclass
028         * a type-specific iterator.
029         */
030        public static class EmptyBigListIterator extends AbstractFloatBigListIterator implements java.io.Serializable, Cloneable {
031         private static final long serialVersionUID = -7046029254386353129L;
032         protected EmptyBigListIterator() {}
033         @Override
034         public boolean hasNext() { return false; }
035         @Override
036         public boolean hasPrevious() { return false; }
037         @Override
038         public float nextFloat() { throw new NoSuchElementException(); }
039         @Override
040         public float previousFloat() { throw new NoSuchElementException(); }
041         @Override
042         public long nextIndex() { return 0; }
043         @Override
044         public long previousIndex() { return -1; }
045         @Override
046         public long skip(long n) { return 0; };
047         @Override
048         public long back(long n) { return 0; };
049         @Override
050         public Object clone() { return EMPTY_BIG_LIST_ITERATOR; }
051         private Object readResolve() { return EMPTY_BIG_LIST_ITERATOR; }
052        }
053        /** An empty iterator (immutable). It is serializable and cloneable.
054         *
055         * <P>The class of this objects represent an abstract empty iterator
056         * that can iterate as a type-specific (list) iterator.
057         */
058
059        public final static EmptyBigListIterator EMPTY_BIG_LIST_ITERATOR = new EmptyBigListIterator();
060        /** An iterator returning a single element. */
061        private static class SingletonBigListIterator extends AbstractFloatBigListIterator {
062         private final float element;
063         private int curr;
064         public SingletonBigListIterator(final float element) {
065          this.element = element;
066         }
067         @Override
068         public boolean hasNext() { return curr == 0; }
069         @Override
070         public boolean hasPrevious() { return curr == 1; }
071         @Override
072         public float nextFloat() {
073          if (! hasNext()) throw new NoSuchElementException();
074          curr = 1;
075          return element;
076         }
077         @Override
078         public float previousFloat() {
079          if (! hasPrevious()) throw new NoSuchElementException();
080          curr = 0;
081          return element;
082         }
083         @Override
084         public long nextIndex() {
085          return curr;
086         }
087         @Override
088         public long previousIndex() {
089          return curr - 1;
090         }
091        }
092        /** Returns an iterator that iterates just over the given element.
093         *
094         * @param element the only element to be returned by a type-specific list iterator.
095         * @return  an iterator that iterates just over <code>element</code>.
096         */
097        public static FloatBigListIterator singleton(final float element) {
098         return new SingletonBigListIterator (element);
099        }
100          /** An unmodifiable wrapper class for big list iterators. */
101        public static class UnmodifiableBigListIterator extends AbstractFloatBigListIterator {
102         protected final FloatBigListIterator i;
103         public UnmodifiableBigListIterator(final FloatBigListIterator i) {
104          this.i = i;
105         }
106         @Override
107         public boolean hasNext() { return i.hasNext(); }
108         @Override
109         public boolean hasPrevious() { return i.hasPrevious(); }
110         @Override
111         public float nextFloat() { return i.nextFloat(); }
112         @Override
113         public float previousFloat() { return i.previousFloat(); }
114         @Override
115         public long nextIndex() { return i.nextIndex(); }
116         @Override
117         public long previousIndex() { return i.previousIndex(); }
118        }
119        /** Returns an unmodifiable list iterator backed by the specified list iterator.
120         *
121         * @param i the list iterator to be wrapped in an unmodifiable list iterator.
122         * @return an unmodifiable view of the specified list iterator.
123         */
124        public static FloatBigListIterator unmodifiable(final FloatBigListIterator i) { return new UnmodifiableBigListIterator (i); }
125        /** A class exposing a list iterator as a big-list iterator.. */
126        public static class BigListIteratorListIterator extends AbstractFloatBigListIterator {
127         protected final FloatListIterator i;
128         protected BigListIteratorListIterator(final FloatListIterator i) {
129          this.i = i;
130         }
131         private int intDisplacement(long n) {
132          if (n < Integer.MIN_VALUE || n > Integer.MAX_VALUE) throw new IndexOutOfBoundsException("This big iterator is restricted to 32-bit displacements");
133          return (int)n;
134         }
135         @Override
136         public void set(float ok) { i.set(ok); }
137         @Override
138         public void add(float ok) { i.add(ok); }
139         @Override
140         public int back(int n) { return i.back(n); }
141         @Override
142         public long back(long n) { return i.back(intDisplacement(n)); }
143         @Override
144         public void remove() { i.remove(); }
145         @Override
146         public int skip(int n) { return i.skip(n); }
147         @Override
148         public long skip(long n) { return i.skip(intDisplacement(n)); }
149         @Override
150         public boolean hasNext() { return i.hasNext(); }
151         @Override
152         public boolean hasPrevious() { return i.hasPrevious(); }
153         @Override
154         public float nextFloat() { return i.nextFloat(); }
155         @Override
156         public float previousFloat() { return i.previousFloat(); }
157         @Override
158         public long nextIndex() { return i.nextIndex(); }
159         @Override
160         public long previousIndex() { return i.previousIndex(); }
161        }
162         /** Returns a big-list iterator backed by the specified list iterator.
163          *
164          * @param i the list iterator to adapted to the big-list-iterator interface.
165          * @return a big-list iterator backed by the specified list iterator.
166          */
167        public static FloatBigListIterator asBigListIterator(final FloatListIterator i) { return new BigListIteratorListIterator (i); }
168}
