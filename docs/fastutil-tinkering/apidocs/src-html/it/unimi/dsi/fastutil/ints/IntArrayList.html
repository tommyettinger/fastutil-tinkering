016package it.unimi.dsi.fastutil.ints;
017import java.util.Arrays;
018import java.util.Collection;
019import java.util.Iterator;
020import java.util.RandomAccess;
021import java.util.NoSuchElementException;
022/** A type-specific array-based list; provides some additional methods that use polymorphism to avoid (un)boxing.
023        *
024        * <P>This class implements a lightweight, fast, open, optimized,
025        * reuse-oriented version of array-based lists. Instances of this class
026        * represent a list with an array that is enlarged as needed when new entries
027        * are created (by doubling its current length), but is
028        * <em>never</em> made smaller (even on a {@link #clear()}). A family of
029        * {@linkplain #trim() trimming methods} lets you control the size of the
030        * backing array; this is particularly useful if you reuse instances of this class.
031        * Range checks are equivalent to those of {@link java.util}'s classes, but
032        * they are delayed as much as possible. The backing array is exposed by the
033        * {@link #elements()} method.
034        *
035        * <p>This class implements the bulk methods <code>removeElements()</code>,
036        * <code>addElements()</code> and <code>getElements()</code> using
037        * high-performance system calls (e.g., {@link
038        * System#arraycopy(Object,int,Object,int,int) System.arraycopy()} instead of
039        * expensive loops.
040        *
041        * @see java.util.ArrayList
042        */
043public class IntArrayList extends AbstractIntList implements RandomAccess, Cloneable, java.io.Serializable {
044        private static final long serialVersionUID = -7046029254386353130L;
045        /** The initial default capacity of an array list. */
046        public final static int DEFAULT_INITIAL_CAPACITY = 16;
047        /** The backing array. */
048        protected transient int a[];
049        /** The current actual size of the list (never greater than the backing-array length). */
050        protected int size;
051        private static final boolean ASSERTS = false;
052        /** Creates a new array list using a given array.
053         *
054         * <P>This constructor is only meant to be used by the wrapping methods.
055         *
056         * @param a the array that will be used to back this array list.
057         */
058        protected IntArrayList(final int a[], boolean dummy) {
059         this.a = a;
060        }
061        /** Creates a new array list with given capacity.
062         *
063         * @param capacity the initial capacity of the array list (may be 0).
064         */
065
066        public IntArrayList(final int capacity) {
067         if (capacity < 0) throw new IllegalArgumentException("Initial capacity (" + capacity + ") is negative");
068         a = new int[capacity];
069        }
070        /** Creates a new array list with {@link #DEFAULT_INITIAL_CAPACITY} capacity.
071         */
072        public IntArrayList() {
073         this(DEFAULT_INITIAL_CAPACITY);
074        }
075        /** Creates a new array list and fills it with a given collection.
076         *
077         * @param c a collection that will be used to fill the array list.
078         */
079        public IntArrayList(final Collection<? extends Integer> c) {
080         this(c.size());
081         size = IntIterators.unwrap(IntIterators.asIntIterator(c.iterator()), a);
082        }
083        /** Creates a new array list and fills it with a given type-specific collection.
084         *
085         * @param c a type-specific collection that will be used to fill the array list.
086         */
087        public IntArrayList(final IntCollection c) {
088         this(c.size());
089         size = IntIterators.unwrap(c.iterator(), a);
090        }
091        /** Creates a new array list and fills it with a given type-specific list.
092         *
093         * @param l a type-specific list that will be used to fill the array list.
094         */
095        public IntArrayList(final IntList l) {
096         this(l.size());
097         l.getElements(0, a, 0, size = l.size());
098        }
099        /** Creates a new array list and fills it with the elements of a given array.
100         *
101         * @param a an array whose elements will be used to fill the array list.
102         */
103        public IntArrayList(final int a[]) {
104         this(a, 0, a.length);
105        }
106        /** Creates a new array list and fills it with the elements of a given array.
107         *
108         * @param a an array whose elements will be used to fill the array list.
109         * @param offset the first element to use.
110         * @param length the number of elements to use.
111         */
112        public IntArrayList(final int a[], final int offset, final int length) {
113         this(length);
114         System.arraycopy(a, offset, this.a, 0, length);
115         size = length;
116        }
117        /** Creates a new array list and fills it with the elements returned by an iterator..
118         *
119         * @param i an iterator whose returned elements will fill the array list.
120         */
121        public IntArrayList(final Iterator<? extends Integer> i) {
122         this();
123         while(i.hasNext()) this.add(((i.next()).intValue()));
124        }
125        /** Creates a new array list and fills it with the elements returned by a type-specific iterator..
126         *
127         * @param i a type-specific iterator whose returned elements will fill the array list.
128         */
129        public IntArrayList(final IntIterator i) {
130         this();
131         while(i.hasNext()) this.add(i.nextInt());
132        }
133        /** Returns the backing array of this list.
134         *
135         * @return the backing array.
136         */
137        public int[] elements() {
138         return a;
139        }
140        /** Wraps a given array into an array list of given size.
141         *
142         * <P>Note it is guaranteed
143         * that the type of the array returned by {@link #elements()} will be the same
144         * (see the comments in the class documentation).
145         *
146         * @param a an array to wrap.
147         * @param length the length of the resulting array list.
148         * @return a new array list of the given size, wrapping the given array.
149         */
150        public static IntArrayList wrap(final int a[], final int length) {
151         if (length > a.length) throw new IllegalArgumentException("The specified length (" + length + ") is greater than the array size (" + a.length + ")");
152         final IntArrayList l = new IntArrayList (a, false);
153         l.size = length;
154         return l;
155        }
156        /** Wraps a given array into an array list.
157         *
158         * <P>Note it is guaranteed
159         * that the type of the array returned by {@link #elements()} will be the same
160         * (see the comments in the class documentation).
161         *
162         * @param a an array to wrap.
163         * @return a new array list wrapping the given array.
164         */
165        public static IntArrayList wrap(final int a[]) {
166         return wrap(a, a.length);
167        }
168        /** Ensures that this array list can contain the given number of entries without resizing.
169         *
170         * @param capacity the new minimum capacity for this array list.
171         */
172
173        public void ensureCapacity(final int capacity) {
174         a = IntArrays.ensureCapacity(a, capacity, size);
175         if (ASSERTS) assert size <= a.length;
176        }
177        /** Grows this array list, ensuring that it can contain the given number of entries without resizing,
178         * and in case enlarging it at least by a factor of two.
179         *
180         * @param capacity the new minimum capacity for this array list.
181         */
182
183        private void grow(final int capacity) {
184         a = IntArrays.grow(a, capacity, size);
185         if (ASSERTS) assert size <= a.length;
186        }
187        @Override
188        public void add(final int index, final int k) {
189         ensureIndex(index);
190         grow(size + 1);
191         if (index != size) System.arraycopy(a, index, a, index + 1, size - index);
192         a[index] = k;
193         size++;
194         if (ASSERTS) assert size <= a.length;
195        }
196        @Override
197        public boolean add(final int k) {
198         grow(size + 1);
199         a[size++] = k;
200         if (ASSERTS) assert size <= a.length;
201         return true;
202        }
203        @Override
204        public int getInt(final int index) {
205         if (index >= size) throw new IndexOutOfBoundsException("Index (" + index + ") is greater than or equal to list size (" + size + ")");
206         return a[index];
207        }
208        @Override
209        public int indexOf(final int k) {
210         for(int i = 0; i < size; i++) if (( (k) == (a[i]) )) return i;
211         return -1;
212        }
213        @Override
214        public int lastIndexOf(final int k) {
215         for(int i = size; i-- != 0;) if (( (k) == (a[i]) )) return i;
216         return -1;
217        }
218        @Override
219        public int removeInt(final int index) {
220         if (index >= size) throw new IndexOutOfBoundsException("Index (" + index + ") is greater than or equal to list size (" + size + ")");
221         final int old = a[index];
222         size--;
223         if (index != size) System.arraycopy(a, index + 1, a, index, size - index);
224         if (ASSERTS) assert size <= a.length;
225         return old;
226        }
227        @Override
228        public boolean rem(final int k) {
229         int index = indexOf(k);
230         if (index == -1) return false;
231         removeInt(index);
232         if (ASSERTS) assert size <= a.length;
233         return true;
234        }
235        @Override
236        public int set(final int index, final int k) {
237         if (index >= size) throw new IndexOutOfBoundsException("Index (" + index + ") is greater than or equal to list size (" + size + ")");
238         int old = a[index];
239         a[index] = k;
240         return old;
241        }
242        @Override
243        public void clear() {
244         size = 0;
245         if (ASSERTS) assert size <= a.length;
246        }
247        @Override
248        public int size() {
249         return size;
250        }
251        @Override
252        public void size(final int size) {
253         if (size > a.length) ensureCapacity(size);
254         if (size > this.size) Arrays.fill(a, this.size, size, (0));
255         this.size = size;
256        }
257        @Override
258        public boolean isEmpty() {
259         return size == 0;
260        }
261        /** Trims this array list so that the capacity is equal to the size.
262         *
263         * @see java.util.ArrayList#trimToSize()
264         */
265        public void trim() {
266         trim(0);
267        }
268        /** Trims the backing array if it is too large.
269         *
270         * If the current array length is smaller than or equal to
271         * <code>n</code>, this method does nothing. Otherwise, it trims the
272         * array length to the maximum between <code>n</code> and {@link #size()}.
273         *
274         * <P>This method is useful when reusing lists.  {@linkplain #clear() Clearing a
275         * list} leaves the array length untouched. If you are reusing a list
276         * many times, you can call this method with a typical
277         * size to avoid keeping around a very large array just
278         * because of a few large transient lists.
279         *
280         * @param n the threshold for the trimming.
281         */
282
283        public void trim(final int n) {
284         // TODO: use Arrays.trim() and preserve type only if necessary
285         if (n >= a.length || size == a.length) return;
286         final int t[] = new int[Math.max(n, size)];
287         System.arraycopy(a, 0, t, 0, size);
288         a = t;
289         if (ASSERTS) assert size <= a.length;
290        }
291           /** Copies element of this type-specific list into the given array using optimized system calls.
292         *
293         * @param from the start index (inclusive).
294         * @param a the destination array.
295         * @param offset the offset into the destination array where to store the first element copied.
296         * @param length the number of elements to be copied.
297         */
298        @Override
299        public void getElements(final int from, final int[] a, final int offset, final int length) {
300         IntArrays.ensureOffsetLength(a, offset, length);
301         System.arraycopy(this.a, from, a, offset, length);
302        }
303        /** Removes elements of this type-specific list using optimized system calls.
304         *
305         * @param from the start index (inclusive).
306         * @param to the end index (exclusive).
307         */
308        @Override
309        public void removeElements(final int from, final int to) {
310         it.unimi.dsi.fastutil.Arrays.ensureFromTo(size, from, to);
311         System.arraycopy(a, to, a, from, size - to);
312         size -= (to - from);
313        }
314        /** Adds elements to this type-specific list using optimized system calls.
315         *
316         * @param index the index at which to add elements.
317         * @param a the array containing the elements.
318         * @param offset the offset of the first element to add.
319         * @param length the number of elements to add.
320         */
321        @Override
322        public void addElements(final int index, final int a[], final int offset, final int length) {
323         ensureIndex(index);
324         IntArrays.ensureOffsetLength(a, offset, length);
325         grow(size + length);
326         System.arraycopy(this.a, index, this.a, index + length, size - index);
327         System.arraycopy(a, offset, this.a, index, length);
328         size += length;
329        }
330        @Override
331        public int[] toArray(int a[]) {
332         if (a == null || a.length < size) a = new int[size];
333         System.arraycopy(this.a, 0, a, 0, size);
334         return a;
335        }
336        @Override
337        public boolean addAll(int index, final IntCollection c) {
338         ensureIndex(index);
339         int n = c.size();
340         if (n == 0) return false;
341         grow(size + n);
342         if (index != size) System.arraycopy(a, index, a, index + n, size - index);
343         final IntIterator i = c.iterator();
344         size += n;
345         while(n-- != 0) a[index++] = i.nextInt();
346         if (ASSERTS) assert size <= a.length;
347         return true;
348        }
349        @Override
350        public boolean addAll(final int index, final IntList l) {
351         ensureIndex(index);
352         final int n = l.size();
353         if (n == 0) return false;
354         grow(size + n);
355         if (index != size) System.arraycopy(a, index, a, index + n, size - index);
356         l.getElements(0, a, index, n);
357         size += n;
358         if (ASSERTS) assert size <= a.length;
359         return true;
360        }
361        @Override
362        public boolean removeAll(final IntCollection c) {
363         final int[] a = this.a;
364         int j = 0;
365         for(int i = 0; i < size; i++)
366          if (! c.contains(a[i])) a[j++] = a[i];
367         final boolean modified = size != j;
368         size = j;
369         return modified;
370        }
371        @Override
372        public boolean removeAll(final Collection<?> c) {
373         final int[] a = this.a;
374         int j = 0;
375         for(int i = 0; i < size; i++)
376          if (! c.contains((Integer.valueOf(a[i])))) a[j++] = a[i];
377         final boolean modified = size != j;
378         size = j;
379         return modified;
380        }
381        @Override
382        public IntListIterator listIterator(final int index) {
383         ensureIndex(index);
384         return new AbstractIntListIterator () {
385           int pos = index, last = -1;
386           public boolean hasNext() { return pos < size; }
387           public boolean hasPrevious() { return pos > 0; }
388           public int nextInt() { if (! hasNext()) throw new NoSuchElementException(); return a[last = pos++]; }
389           public int previousInt() { if (! hasPrevious()) throw new NoSuchElementException(); return a[last = --pos]; }
390           public int nextIndex() { return pos; }
391           public int previousIndex() { return pos - 1; }
392           public void add(int k) {
393            IntArrayList.this.add(pos++, k);
394            last = -1;
395           }
396           public void set(int k) {
397            if (last == -1) throw new IllegalStateException();
398            IntArrayList.this.set(last, k);
399           }
400           public void remove() {
401            if (last == -1) throw new IllegalStateException();
402            IntArrayList.this.removeInt(last);
403            /* If the last operation was a next(), we are removing an element *before* us, and we must decrease pos correspondingly. */
404            if (last < pos) pos--;
405            last = -1;
406           }
407          };
408        }
409        @Override
410        public IntArrayList clone() {
411         IntArrayList c = new IntArrayList (size);
412         System.arraycopy(a, 0, c.a, 0, size);
413         c.size = size;
414         return c;
415        }
416        /** Compares this type-specific array list to another one.
417         *
418         * <P>This method exists only for sake of efficiency. The implementation
419         * inherited from the abstract implementation would already work.
420         *
421         * @param l a type-specific array list.
422         * @return true if the argument contains the same elements of this type-specific array list.
423         */
424        public boolean equals(final IntArrayList l) {
425         if (l == this) return true;
426         int s = size();
427         if (s != l.size()) return false;
428         final int[] a1 = a;
429         final int[] a2 = l.a;
430         while(s-- != 0) if (a1[s] != a2[s]) return false;
431         return true;
432        }
433        /** Compares this array list to another array list.
434         *
435         * <P>This method exists only for sake of efficiency. The implementation
436         * inherited from the abstract implementation would already work.
437         *
438         * @param l an array list.
439         * @return a negative integer,
440         * zero, or a positive integer as this list is lexicographically less than, equal
441         * to, or greater than the argument.
442         */
443
444        public int compareTo(final IntArrayList l) {
445         final int s1 = size(), s2 = l.size();
446         final int a1[] = a, a2[] = l.a;
447         int e1, e2;
448         int r, i;
449         for(i = 0; i < s1 && i < s2; i++) {
450          e1 = a1[i];
451          e2 = a2[i];
452          if ((r = ( Integer.compare((e1),(e2)) )) != 0) return r;
453         }
454         return i < s2 ? -1 : (i < s1 ? 1 : 0);
455        }
456        private void writeObject(java.io.ObjectOutputStream s) throws java.io.IOException {
457         s.defaultWriteObject();
458         for(int i = 0; i < size; i++) s.writeInt(a[i]);
459        }
460
461        private void readObject(java.io.ObjectInputStream s) throws java.io.IOException, ClassNotFoundException {
462         s.defaultReadObject();
463         a = new int[size];
464         for(int i = 0; i < size; i++) a[i] = s.readInt();
465        }
