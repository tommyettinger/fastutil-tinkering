016package it.unimi.dsi.fastutil.objects;
017import java.util.Comparator;
018import it.unimi.dsi.fastutil.ints.IntArrays;
019/** A class providing static methods and objects that do useful things with semi-indirect heaps.
020        *
021        * <P>A semi-indirect heap is based on a <em>reference array</em>. Elements of
022        * a semi-indirect heap are integers that index the reference array (note that
023        * in an <em>indirect</em> heap you can also map elements of the reference
024        * array to heap positions).
025        */
026public class ObjectSemiIndirectHeaps {
027        private ObjectSemiIndirectHeaps() {}
028        /** Moves the given element down into the semi-indirect heap until it reaches the lowest possible position.
029         *
030         * @param refArray the reference array.
031         * @param heap the semi-indirect heap (starting at 0).
032         * @param size the number of elements in the heap.
033         * @param i the index in the heap of the element to be moved down.
034         * @param c a type-specific comparator, or {@code null} for the natural order.
035         * @return the new position in the heap of the element of heap index <code>i</code>.
036         */
037        @SuppressWarnings("unchecked")
038        public static <K> int downHeap(final K[] refArray, final int[] heap, final int size, int i, final Comparator <K> c) {
039         assert i < size;
040         final int e = heap[i];
041         final K E = refArray[e];
042         int child;
043         if (c == null)
044          while ((child = (i << 1) + 1) < size) {
045           int t = heap[child];
046           final int right = child + 1;
047           if (right < size && ( ((Comparable<K>)(refArray[heap[right]])).compareTo(refArray[t]) < 0 )) t = heap[child = right];
048           if (( ((Comparable<K>)(E)).compareTo(refArray[t]) <= 0 )) break;
049           heap[i] = t;
050           i = child;
051          }
052         else
053          while ((child = (i << 1) + 1) < size) {
054           int t = heap[child];
055           final int right = child + 1;
056           if (right < size && c.compare(refArray[heap[right]], refArray[t]) < 0) t = heap[child = right];
057           if (c.compare(E, refArray[t]) <= 0) break;
058           heap[i] = t;
059           i = child;
060          }
061         heap[i] = e;
062         return i;
063        }
064        /** Moves the given element up in the semi-indirect heap until it reaches the highest possible position.
065         *
066         * @param refArray the reference array.
067         * @param heap the semi-indirect heap (starting at 0).
068         * @param size the number of elements in the heap.
069         * @param i the index in the heap of the element to be moved up.
070         * @param c a type-specific comparator, or {@code null} for the natural order.
071         * @return the new position in the heap of the element of heap index <code>i</code>.
072         */
073        @SuppressWarnings("unchecked")
074        public static <K> int upHeap(final K[] refArray, final int[] heap, final int size, int i, final Comparator <K> c) {
075         assert i < size;
076         final int e = heap[i];
077         final K E = refArray[e];
078         if (c == null)
079          while (i != 0) {
080           final int parent = (i - 1) >>> 1;
081           final int t = heap[parent];
082           if (( ((Comparable<K>)(refArray[t])).compareTo(E) <= 0 )) break;
083           heap[i] = t;
084           i = parent;
085          }
086         else
087          while (i != 0) {
088           final int parent = (i - 1) >>> 1;
089           final int t = heap[parent];
090           if (c.compare(refArray[t], E) <= 0) break;
091           heap[i] = t;
092           i = parent;
093          }
094         heap[i] = e;
095         return i;
096        }
097        /** Creates a semi-indirect heap in the given array.
098         *
099         * @param refArray the reference array.
100         * @param offset the first element of the reference array to be put in the heap.
101         * @param length the number of elements to be put in the heap.
102         * @param heap the array where the heap is to be created.
103         * @param c a type-specific comparator, or {@code null} for the natural order.
104         */
105        public static <K> void makeHeap(final K[] refArray, final int offset, final int length, final int[] heap, final Comparator <K> c) {
106         ObjectArrays.ensureOffsetLength(refArray, offset, length);
107         if (heap.length < length) throw new IllegalArgumentException("The heap length (" + heap.length + ") is smaller than the number of elements (" + length + ")");
108         int i = length;
109         while(i-- != 0) heap[i] = offset + i;
110         i = length >>> 1;
111         while(i-- != 0) downHeap(refArray, heap, length, i, c);
112        }
113        /** Creates a semi-indirect heap, allocating its heap array.
114         *
115         * @param refArray the reference array.
116         * @param offset the first element of the reference array to be put in the heap.
117         * @param length the number of elements to be put in the heap.
118         * @param c a type-specific comparator, or {@code null} for the natural order.
119         * @return the heap array.
120         */
121        public static <K> int[] makeHeap(final K[] refArray, final int offset, final int length, final Comparator <K> c) {
122         final int[] heap = length <= 0 ? IntArrays.EMPTY_ARRAY : new int[length];
123         makeHeap(refArray, offset, length, heap, c);
124         return heap;
125        }
126        /** Creates a semi-indirect heap from a given index array.
127         *
128         * @param refArray the reference array.
129         * @param heap an array containing indices into <code>refArray</code>.
130         * @param size the number of elements in the heap.
131         * @param c a type-specific comparator, or {@code null} for the natural order.
132         */
133        public static <K> void makeHeap(final K[] refArray, final int[] heap, final int size, final Comparator <K> c) {
134         int i = size >>> 1;
135         while(i-- != 0) downHeap(refArray, heap, size, i, c);
136        }
137        /** Retrieves the front of a heap in a given array.
138         *
139         * <p>The <em>front</em> of a semi-indirect heap is the set of indices whose associated elements in the reference array
140         * are equal to the element associated to the first index.
141         *
142         * <p>In several circumstances you need to know the front, and scanning linearly the entire heap is not
143         * the best strategy. This method simulates (using a partial linear scan) a breadth-first visit that
144         * terminates when all visited nodes are larger than the element associated
145         * to the top index, which implies that no elements of the front can be found later.
146         * In most cases this trick yields a significant improvement.
147         *
148         * @param refArray the reference array.
149         * @param heap an array containing indices into <code>refArray</code>.
150         * @param size the number of elements in the heap.
151         * @param a an array large enough to hold the front (e.g., at least long as <code>refArray</code>).
152         * @return the number of elements actually written (starting from the first position of <code>a</code>).
153         */
154        @SuppressWarnings("unchecked")
155        public static <K> int front(final K[] refArray, final int[] heap, final int size, final int[] a) {
156         final K top = refArray[heap[0]];
157         int j = 0, // The current position in a
158          l = 0, // The first position to visit in the next level (inclusive)
159          r = 1, // The last position to visit in the next level (exclusive)
160          f = 0; // The first position (in the heap array) of the next level
161         for(int i = 0; i < r; i++) {
162          if (i == f) { // New level
163           if (l >= r) break; // If we are crossing the two bounds, we're over
164           f = (f << 1) + 1; // Update the first position of the next level...
165           i = l; // ...and jump directly to position l
166           l = -1; // Invalidate l
167          }
168          if (( ((Comparable<K>)(top)).compareTo(refArray[heap[i]]) == 0 )) {
169           a[j++] = heap[i];
170           if (l == -1) l = i * 2 + 1; // If this is the first time in this level, set l
171           r = Math.min(size, i * 2 + 3); // Update r, but do not go beyond size
172          }
173         }
174         return j;
175        }
176        /** Retrieves the front of a heap in a given array using a given comparator.
177         *
178         * <p>The <em>front</em> of a semi-indirect heap is the set of indices whose associated elements in the reference array
179         * are equal to the element associated to the first index.
180         *
181         * <p>In several circumstances you need to know the front, and scanning linearly the entire heap is not
182         * the best strategy. This method simulates (using a partial linear scan) a breadth-first visit that
183         * terminates when all visited nodes are larger than the element associated
184         * to the top index, which implies that no elements of the front can be found later.
185         * In most cases this trick yields a significant improvement.
186         *
187         * @param refArray the reference array.
188         * @param heap an array containing indices into <code>refArray</code>.
189         * @param size the number of elements in the heap.
190         * @param a an array large enough to hold the front (e.g., at least long as <code>refArray</code>).
191         * @param c a type-specific comparator.
192         * @return the number of elements actually written (starting from the first position of <code>a</code>).
193         */
194        public static <K> int front(final K[] refArray, final int[] heap, final int size, final int[] a, final Comparator <K> c) {
195         final K top = refArray[heap[0]];
196         int j = 0, // The current position in a
197          l = 0, // The first position to visit in the next level (inclusive)
198          r = 1, // The last position to visit in the next level (exclusive)
199          f = 0; // The first position (in the heap array) of the next level
200         for(int i = 0; i < r; i++) {
201          if (i == f) { // New level
202           if (l >= r) break; // If we are crossing the two bounds, we're over
203           f = (f << 1) + 1; // Update the first position of the next level...
204           i = l; // ...and jump directly to position l
205           l = -1; // Invalidate l
206          }
207          if (c.compare(top, refArray[heap[i]]) == 0) {
208           a[j++] = heap[i];
209           if (l == -1) l = i * 2 + 1; // If this is the first time in this level, set l
210           r = Math.min(size, i * 2 + 3); // Update r, but do not go beyond size
211          }
212         }
213         return j;
214        }
