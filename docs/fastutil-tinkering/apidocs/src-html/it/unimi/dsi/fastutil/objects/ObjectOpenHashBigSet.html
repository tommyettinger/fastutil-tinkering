026/**  A type-specific hash big set with with a fast, small-footprint implementation.
027        *
028        * <P>Instances of this class use a hash table to represent a big set: the number
029        * of elements in the set is limited only by the amount of core memory. The table
030        * (backed by a {@linkplain it.unimi.dsi.fastutil.BigArrays big array}) is
031        * filled up to a specified <em>load factor</em>, and then doubled in size to
032        * accommodate new entries. If the table is emptied below <em>one fourth</em>
033        * of the load factor, it is halved in size. However, halving is
034        * not performed when deleting entries from an iterator, as it would interfere
035        * with the iteration process.
036        *
037        * <p>Note that {@link #clear()} does not modify the hash table size.
038        * Rather, a family of {@linkplain #trim() trimming
039        * methods} lets you control the size of the table; this is particularly useful
040        * if you reuse instances of this class.
041        *
042        * <p>The methods of this class are about 30% slower than those of the corresponding non-big set.
043        *
044        * @see Hash
045        * @see HashCommon
046        */
047public class ObjectOpenHashBigSet <K> extends AbstractObjectSet <K> implements java.io.Serializable, Cloneable, Hash, Size64 {
048        private static final long serialVersionUID = 0L;
049        private static final boolean ASSERTS = false;
050        /** The big array of keys. */
051        protected transient K[][] key;
052        /** The mask for wrapping a position counter. */
053        protected transient long mask;
054        /** The mask for wrapping a segment counter. */
055        protected transient int segmentMask;
056        /** The mask for wrapping a base counter. */
057        protected transient int baseMask;
058        /** Whether this set contains the null key. */
059        protected transient boolean containsNull;
060        /** The current table size (always a power of 2). */
061        protected transient long n;
062        /** Threshold after which we rehash. It must be the table size times {@link #f}. */
063        protected transient long maxFill;
064        /** The acceptable load factor. */
065        protected final float f;
066        /** Number of entries in the set. */
067        protected long size;
068        /** Initialises the mask values. */
069        private void initMasks() {
070         mask = n - 1;
071         /* Note that either we have more than one segment, and in this case all segments
072                 * are BigArrays.SEGMENT_SIZE long, or we have exactly one segment whose length
073                 * is a power of two. */
074         segmentMask = key[0].length - 1;
075         baseMask = key.length - 1;
076        }
077        /** Creates a new hash big set.
078         *
079         * <p>The actual table size will be the least power of two greater than <code>expected</code>/<code>f</code>.
080         *
081         * @param expected the expected number of elements in the set.
082         * @param f the load factor.
083         */
084        @SuppressWarnings("unchecked")
085        public ObjectOpenHashBigSet(final long expected, final float f) {
086         if (f <= 0 || f > 1) throw new IllegalArgumentException("Load factor must be greater than 0 and smaller than or equal to 1");
087         if (n < 0) throw new IllegalArgumentException("The expected number of elements must be nonnegative");
088         this.f = f;
089         n = bigArraySize(expected, f);
090         maxFill = maxFill(n, f);
091         key = (K[][]) ObjectBigArrays.newBigArray(n);
092         initMasks();
093        }

194        private long realSize() {
195         return containsNull ? size - 1 : size;
196        }
197        private void ensureCapacity(final long capacity) {
198         final long needed = bigArraySize(capacity, f);
199         if (needed > n) rehash(needed);
200        }
201        @Override
202        public boolean addAll(Collection<? extends K> c) {
203         final long size = c instanceof Size64 ? ((Size64)c).size64() : c.size();
204         // The resulting collection will be at least c.size() big
205         if (f <= .5) ensureCapacity(size); // The resulting collection will be sized for c.size() elements
206         else ensureCapacity(size64() + size); // The resulting collection will be sized for size() + c.size() elements
207         return super.addAll(c);
208        }
209        @Override
210        public boolean add(final K k) {
211         int displ, base;
212         if (( (k) == null )) {
213          if (containsNull) return false;
214          containsNull = true;
215         }
216         else {
217          K curr;
218          final K[][] key = this.key;
219          final long h = ( it.unimi.dsi.fastutil.HashCommon.mix( (long)( (k).hashCode() ) ) );
220          // The starting point.
221          if (! ( (curr = key[base = (int)((h & mask) >>> BigArrays.SEGMENT_SHIFT)][displ = (int)(h & segmentMask)]) == null )) {
222           if (( (curr).equals(k) )) return false;
223           while(! ( (curr = key[base = (base + ((displ = (displ + 1) & segmentMask) == 0 ? 1 : 0)) & baseMask][displ]) == null ))
224            if (( (curr).equals(k) )) return false;
225          }
226          key[base][displ] = k;
227         }
228         if (size++ >= maxFill) rehash(2 * n);
229         if (ASSERTS) checkTable();
230         return true;
231        }
232        /** Add a random element if not present, get the existing value if already present.
233         *
234         * This is equivalent to (but faster than) doing a:
235         * <pre>
236         * K exist = set.get(k);
237         * if (exist == null) {
238         *   set.add(k);
239         *   exist = k;
240         * }
241         * </pre>
242         */
243        public K addOrGet(final K k) {
244         int displ, base;
245         if (( (k) == null )) {
246          if (containsNull) return null;
247          containsNull = true;
248         }
249         else {
250          K curr;
251          final K[][] key = this.key;
252          final long h = ( it.unimi.dsi.fastutil.HashCommon.mix( (long)( (k).hashCode() ) ) );
253          // The starting point.
254          if (! ( (curr = key[base = (int)((h & mask) >>> BigArrays.SEGMENT_SHIFT)][displ = (int)(h & segmentMask)]) == null )) {
255           if (( (curr).equals(k) )) return curr;
256           while(! ( (curr = key[base = (base + ((displ = (displ + 1) & segmentMask) == 0 ? 1 : 0)) & baseMask][displ]) == null ))
257            if (( (curr).equals(k) )) return curr;
258          }
259          key[base][displ] = k;
260         }
261         if (size++ >= maxFill) rehash(2 * n);
262         if (ASSERTS) checkTable();
263         return k;
264        }
265        /** Shifts left entries with the specified hash code, starting at the specified position,
266         * and empties the resulting free entry.
267         *
268         * @param pos a starting position.
269         */
270        protected final void shiftKeys(long pos) {
271         // Shift entries with the same hash.
272         long last, slot;
273         final K[][] key = this.key;
274         for(;;) {
275          pos = ((last = pos) + 1) & mask;
276          for(;;) {
277           if (( (ObjectBigArrays.get(key, pos)) == null )) {
278            ObjectBigArrays.set(key, last, (null));
279            return;
280           }
281           slot = ( it.unimi.dsi.fastutil.HashCommon.mix( (long)( (ObjectBigArrays.get(key, pos)).hashCode() ) ) ) & mask;
282           if (last <= pos ? last >= slot || slot > pos : last >= slot && slot > pos) break;
283           pos = (pos + 1) & mask;
284          }
285          ObjectBigArrays.set(key, last, ObjectBigArrays.get(key, pos));
286         }
287        }
288        private boolean removeEntry(final int base, final int displ) {
289         shiftKeys(base * (long)BigArrays.SEGMENT_SIZE + displ);
290         if (--size < maxFill / 4 && n > DEFAULT_INITIAL_SIZE) rehash(n / 2);
291         return true;
292        }
293        private boolean removeNullEntry() {
294         containsNull = false;
295         if (--size < maxFill / 4 && n > DEFAULT_INITIAL_SIZE) rehash(n / 2);
296         return true;
297        }
298        @Override
299        public boolean remove(final Object k) {
300         if (( (k) == null )) {
301          if (containsNull) return removeNullEntry();
302          return false;
303         }
304         K curr;
305         final K[][] key = this.key;
306         final long h = ( it.unimi.dsi.fastutil.HashCommon.mix( (long)( (k).hashCode() ) ) );
307         int displ, base;
308         // The starting point.
309         if (( (curr = key[base = (int)((h & mask) >>> BigArrays.SEGMENT_SHIFT)][displ = (int)(h & segmentMask)]) == null )) return false;
310         if (( (curr).equals(k) )) return removeEntry(base, displ);
311         while(true) {
312          if (( (curr = key[base = (base + ((displ = (displ + 1) & segmentMask) == 0 ? 1 : 0)) & baseMask][displ]) == null )) return false;
313          if (( (curr).equals(k) )) return removeEntry(base, displ);
314         }
315        }
316        @Override
317        public boolean contains(final Object k) {
318         if (( (k) == null )) return containsNull;
319         K curr;
320         final K[][] key = this.key;
321         final long h = ( it.unimi.dsi.fastutil.HashCommon.mix( (long)( (k).hashCode() ) ) );
322         int displ, base;
323         // The starting point.
324         if (( (curr = key[base = (int)((h & mask) >>> BigArrays.SEGMENT_SHIFT)][displ = (int)(h & segmentMask)]) == null )) return false;
325         if (( (curr).equals(k) )) return true;
326         while(true) {
327          if (( (curr = key[base = (base + ((displ = (displ + 1) & segmentMask) == 0 ? 1 : 0)) & baseMask][displ]) == null )) return false;
328          if (( (curr).equals(k) )) return true;
329         }
330        }
331        /** Returns the element of this set that is equal to the given key, or {@code null}.
332         * @return the element of this set that is equal to the given key, or {@code null}.
333         */
334        public K get(final Object k) {
335         if (k == null) return null; // This is correct independently of the value of containsNull
336         K curr;
337         final K[][] key = this.key;
338         final long h = ( it.unimi.dsi.fastutil.HashCommon.mix( (long)( (k).hashCode() ) ) );
339         int displ, base;
340         // The starting point.
341         if (( (curr = key[base = (int)((h & mask) >>> BigArrays.SEGMENT_SHIFT)][displ = (int)(h & segmentMask)]) == null )) return null;
342         if (( (curr).equals(k) )) return curr;
343         while(true) {
344          if (( (curr = key[base = (base + ((displ = (displ + 1) & segmentMask) == 0 ? 1 : 0)) & baseMask][displ]) == null )) return null;
345          if (( (curr).equals(k) )) return curr;
346         }
347        }

350         */

363        /** An iterator over a hash big set. */
364        private class SetIterator extends AbstractObjectIterator <K> {
365         /** The base of the last entry returned, if positive or zero; initially, the number of components
366                        of the key array. If negative, the last element returned was
367                        that of index {@code - base - 1} from the {@link #wrapped} list. */
368         int base = key.length;
369         /** The displacement of the last entry returned; initially, zero. */
370         int displ;
371         /** The index of the last entry that has been returned (or {@link Long#MIN_VALUE} if {@link #base} is negative).
372                        It is -1 if either we did not return an entry yet, or the last returned entry has been removed. */
373         long last = -1;
374         /** A downward counter measuring how many entries must still be returned. */
375         long c = size;
376         /** A boolean telling us whether we should return the null key. */
377         boolean mustReturnNull = ObjectOpenHashBigSet.this.containsNull;
378         /** A lazily allocated list containing elements that have wrapped around the table because of removals. */
379         ObjectArrayList <K> wrapped;
380         public boolean hasNext() {
381          return c != 0;
382         }
383         public K next() {
384          if (! hasNext()) throw new NoSuchElementException();
385          c--;
386          if (mustReturnNull) {
387           mustReturnNull = false;
388           last = n;
389           return (null);
390          }
391          final K[][] key = ObjectOpenHashBigSet.this.key;
392          for(;;) {
393           if (displ == 0 && base <= 0) {
394            // We are just enumerating elements from the wrapped list.
395            last = Long.MIN_VALUE;
396            return wrapped.get(- (--base) - 1);
397           }
398           if (displ-- == 0) displ = key[--base].length - 1;
399           final K k = key[base][displ];
400           if (! ( (k) == null )) {
401            last = base * (long)BigArrays.SEGMENT_SIZE + displ;
402            return k;
403           }
404          }
405         }
406         /** Shifts left entries with the specified hash code, starting at the specified position,
407                 * and empties the resulting free entry.
408                 *
409                 * @param pos a starting position.
410                 */
411         private final void shiftKeys(long pos) {
412          // Shift entries with the same hash.
413          long last, slot;
414          K curr;
415          final K[][] key = ObjectOpenHashBigSet.this.key;
416          for(;;) {
417           pos = ((last = pos) + 1) & mask;
418           for(;;) {
419            if(( (curr = ObjectBigArrays.get(key, pos)) == null )) {
420             ObjectBigArrays.set(key, last, (null));
421             return;
422            }
423            slot = ( it.unimi.dsi.fastutil.HashCommon.mix( (long)( (curr).hashCode() ) ) ) & mask;
424            if (last <= pos ? last >= slot || slot > pos : last >= slot && slot > pos) break;
425            pos = (pos + 1) & mask;
426           }
427           if (pos < last) { // Wrapped entry.
428            if (wrapped == null) wrapped = new ObjectArrayList <K>();
429            wrapped.add(ObjectBigArrays.get(key, pos));
430           }
431           ObjectBigArrays.set(key, last, curr);
432          }
433         }
434         public void remove() {
435          if (last == -1) throw new IllegalStateException();
436          if (last == n) ObjectOpenHashBigSet.this.containsNull = false;
437          else if (base >= 0) shiftKeys(last);
438          else {
439           // We're removing wrapped entries.
440           ObjectOpenHashBigSet.this.remove(wrapped.set(- base - 1, null));
441           last = -1; // Note that we must not decrement size
442           return;
443          }
444          size--;
445          last = -1; // You can no longer remove this entry.
446          if (ASSERTS) checkTable();
447         }
448        }
449        @Override
450        public ObjectIterator <K> iterator() {
451         return new SetIterator();
452        }
453        /** A no-op for backward compatibility. The kind of tables implemented by
454         * this class never need rehashing.
455         *
456         * <P>If you need to reduce the table size to fit exactly
457         * this set, use {@link #trim()}.
458         *
459         * @return true.
460         * @see #trim()
461         * @deprecated A no-op.

465         return true;

467        /** Rehashes this set, making the table as small as possible.
468         *
469         * <P>This method rehashes the table to the smallest size satisfying the
470         * load factor. It can be used when the set will not be changed anymore, so
471         * to optimize access speed and size.
472         *
473         * <P>If the table size is already the minimum possible, this method
474         * does nothing.
475         *
476         * @return true if there was enough memory to trim the set.
477         * @see #trim(long)
478         */
479        public boolean trim() {
480         final long l = bigArraySize(size, f);
481         if (l >= n || size > maxFill(l, f)) return true;
482         try {
483          rehash(l);
484         }
485         catch(OutOfMemoryError cantDoIt) { return false; }
486         return true;
487        }
488        /** Rehashes this set if the table is too large.
489         *
490         * <P>Let <var>N</var> be the smallest table size that can hold
491         * <code>max(n,{@link #size64()})</code> entries, still satisfying the load factor. If the current
492         * table size is smaller than or equal to <var>N</var>, this method does
493         * nothing. Otherwise, it rehashes this set in a table of size
494         * <var>N</var>.
495         *
496         * <P>This method is useful when reusing sets.  {@linkplain #clear() Clearing a
497         * set} leaves the table size untouched. If you are reusing a set
498         * many times, you can call this method with a typical
499         * size to avoid keeping around a very large table just
500         * because of a few large transient sets.
501         *
502         * @param n the threshold for the trimming.
503         * @return true if there was enough memory to trim the set.
504         * @see #trim()
505         */
506        public boolean trim(final long n) {
507         final long l = bigArraySize(n, f);
508         if (this.n <= l) return true;
509         try {
510          rehash(l);
511         }
512         catch(OutOfMemoryError cantDoIt) { return false; }
513         return true;
514        }
515        /** Resizes the set.
516         *
517         * <P>This method implements the basic rehashing strategy, and may be
518         * overriden by subclasses implementing different rehashing strategies (e.g.,
519         * disk-based rehashing). However, you should not override this method
520         * unless you understand the internal workings of this class.
521         *
522         * @param newN the new size
523         */
524        @SuppressWarnings("unchecked")
525        protected void rehash(final long newN) {
526         final K key[][] = this.key;
527         final K newKey[][] = (K[][]) ObjectBigArrays.newBigArray(newN);
528         final long mask = newN - 1; // Note that this is used by the hashing macro
529         final int newSegmentMask = newKey[0].length - 1;
530         final int newBaseMask = newKey.length - 1;
531         int base = 0, displ = 0, b, d;
532         long h;
533         K k;
534         for(long i = realSize(); i-- != 0;) {
535          while(( (key[base][displ]) == null )) base = (base + ((displ = (displ + 1) & segmentMask) == 0 ? 1 : 0));
536          k = key[base][displ];
537          h = ( it.unimi.dsi.fastutil.HashCommon.mix( (long)( (k).hashCode() ) ) );
538          // The starting point.
539          if (! ( (newKey[b = (int)((h & mask) >>> BigArrays.SEGMENT_SHIFT)][d = (int)(h & newSegmentMask)]) == null ))
540           while(! ( (newKey[b = (b + ((d = (d + 1) & newSegmentMask) == 0 ? 1 : 0)) & newBaseMask][d]) == null ));
541          newKey[b][d] = k;
542          base = (base + ((displ = (displ + 1) & segmentMask) == 0 ? 1 : 0));
543         }
544         this.n = newN;
545         this.key = newKey;
546         initMasks();
547         maxFill = maxFill(n, f);
548        }

562        /** Returns a deep copy of this big set.
563         *
564         * <P>This method performs a deep copy of this big hash set; the data stored in the
565         * set, however, is not cloned. Note that this makes a difference only for object keys.
566         *
567         *  @return a deep copy of this big set.
568         */
569        @Override
570        @SuppressWarnings("unchecked")
571        public ObjectOpenHashBigSet <K> clone() {
572         ObjectOpenHashBigSet <K> c;
573         try {
574          c = (ObjectOpenHashBigSet <K>)super.clone();
575         }
576         catch(CloneNotSupportedException cantHappen) {
577          throw new InternalError();
