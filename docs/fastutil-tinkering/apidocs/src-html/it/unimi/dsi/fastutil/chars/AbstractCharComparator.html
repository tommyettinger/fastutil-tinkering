016package it.unimi.dsi.fastutil.chars;
017/**  An abstract class facilitating the creation of type-specific {@linkplain java.util.Comparator comparators}.
018        *
019        * <P>To create a type-specific comparator you need both a method comparing
020        * primitive types and a method comparing objects. However, if you have the
021        * first one you can just inherit from this class and get for free the second
022        * one.
023        *
024        * @see java.util.Comparator
025        */
026public abstract class AbstractCharComparator implements CharComparator , java.io.Serializable {
027        private static final long serialVersionUID = 0L;
028        protected AbstractCharComparator() {}
029        /** {@inheritDoc}
030         * <p>This implementation delegates to the corresponding type-specific method.
031         * @deprecated Please use the corresponding type-specific method instead. */
032        @Deprecated
033        @Override
034        public int compare(Character ok1, Character ok2) {
035         return compare(ok1.charValue(), ok2.charValue());
036        }
037}
