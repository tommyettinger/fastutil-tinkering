028/** A type-specific hash map with a fast, small-footprint implementation.
029        *
030        * <P>Instances of this class use a hash table to represent a map. The table is
031        * filled up to a specified <em>load factor</em>, and then doubled in size to
032        * accommodate new entries. If the table is emptied below <em>one fourth</em>
033        * of the load factor, it is halved in size. However, halving is
034        * not performed when deleting entries from an iterator, as it would interfere
035        * with the iteration process.
036        *
037        * <p>Note that {@link #clear()} does not modify the hash table size.
038        * Rather, a family of {@linkplain #trim() trimming
039        * methods} lets you control the size of the table; this is particularly useful
040        * if you reuse instances of this class.
041        *
042        * @see Hash
043        * @see HashCommon
044        */
045public class Reference2LongOpenHashMap <K> extends AbstractReference2LongMap <K> implements java.io.Serializable, Cloneable, Hash {
046        private static final long serialVersionUID = 0L;
047        private static final boolean ASSERTS = false;
048        /** The array of keys. */
049        protected transient K[] key;
050        /** The array of values. */
051        protected transient long[] value;
052        /** The mask for wrapping a position counter. */
053        protected transient int mask;
054        /** Whether this set contains the key zero. */
055        protected transient boolean containsNullKey;
056        /** The current table size. */
057        protected transient int n;
058        /** Threshold after which we rehash. It must be the table size times {@link #f}. */
059        protected transient int maxFill;
060        /** Number of entries in the set (including the key zero, if present). */
061        protected int size;
062        /** The acceptable load factor. */
063        protected final float f;
064        /** Cached set of entries. */
065        protected transient FastEntrySet <K> entries;
066        /** Cached set of keys. */
067        protected transient ReferenceSet <K> keys;
068        /** Cached collection of values. */
069        protected transient LongCollection values;
070        /** Creates a new hash map.
071         *
072         * <p>The actual table size will be the least power of two greater than <code>expected</code>/<code>f</code>.
073         *
074         * @param expected the expected number of elements in the hash set.
075         * @param f the load factor.
076         */
077        @SuppressWarnings("unchecked")
078        public Reference2LongOpenHashMap(final int expected, final float f) {
079         if (f <= 0 || f > 1) throw new IllegalArgumentException("Load factor must be greater than 0 and smaller than or equal to 1");
080         if (expected < 0) throw new IllegalArgumentException("The expected number of elements must be nonnegative");
081         this.f = f;
082         n = arraySize(expected, f);
083         mask = n - 1;
084         maxFill = maxFill(n, f);
085         key = (K[]) new Object[n + 1];
086         value = new long[n + 1];
087        }

133        /** Creates a new hash map using the elements of two parallel arrays.
134         *
135         * @param k the array of keys of the new hash map.
136         * @param v the array of corresponding values in the new hash map.
137         * @param f the load factor.
138         * @throws IllegalArgumentException if <code>k</code> and <code>v</code> have different lengths.
139         */
140        public Reference2LongOpenHashMap(final K[] k, final long[] v, final float f) {
141         this(k.length, f);
142         if (k.length != v.length) throw new IllegalArgumentException("The key array and the value array have different lengths (" + k.length + " and " + v.length + ")");
143         for(int i = 0; i < k.length; i++) this.put(k[i], v[i]);
144        }

154        private int realSize() {
155         return containsNullKey ? size - 1 : size;
156        }
157        private void ensureCapacity(final int capacity) {
158         final int needed = arraySize(capacity, f);
159         if (needed > n) rehash(needed);
160        }
161        private void tryCapacity(final long capacity) {
162         final int needed = (int)Math.min(1 << 30, Math.max(2, HashCommon.nextPowerOfTwo((long)Math.ceil(capacity / f))));
163         if (needed > n) rehash(needed);
164        }
165        private long removeEntry(final int pos) {
166         final long oldValue = value[pos];
167         size--;
168         shiftKeys(pos);
169         if (size < maxFill / 4 && n > DEFAULT_INITIAL_SIZE) rehash(n / 2);
170         return oldValue;
171        }
172        private long removeNullEntry() {
173         containsNullKey = false;
174         key[n] = null;
175         final long oldValue = value[n];
176         size--;
177         if (size < maxFill / 4 && n > DEFAULT_INITIAL_SIZE) rehash(n / 2);
178         return oldValue;
179        }
180        @Override
181        public void putAll(Map<? extends K,? extends Long> m) {
182         if (f <= .5) ensureCapacity(m.size()); // The resulting map will be sized for m.size() elements
183         else tryCapacity(size() + m.size()); // The resulting map will be tentatively sized for size() + m.size() elements
184         super.putAll(m);
185        }
186        private int insert(final K k, final long v) {
187         int pos;
188         if (( (k) == (null) )) {
189          if (containsNullKey) return n;
190          containsNullKey = true;
191          pos = n;
192         }
193         else {
194          K curr;
195          final K[] key = this.key;
196          // The starting point.
197          if (! ( (curr = key[pos = ( it.unimi.dsi.fastutil.HashCommon.mix( System.identityHashCode(k) ) ) & mask]) == (null) )) {
198           if (( (curr) == (k) )) return pos;
199           while(! ( (curr = key[pos = (pos + 1) & mask]) == (null) ))
200            if (( (curr) == (k) )) return pos;
201          }
202         }
203         key[pos] = k;
204         value[pos] = v;
205         if (size++ >= maxFill) rehash(arraySize(size + 1, f));
206         if (ASSERTS) checkTable();
207         return -1;
208        }
209        @Override
210        public long put(final K k, final long v) {
211         final int pos = insert(k, v);
212         if (pos < 0) return defRetValue;
213         final long oldValue = value[pos];
214         value[pos] = v;
215         return oldValue;
216        }
217        private long addToValue(final int pos, final long incr) {
218         final long oldValue = value[pos];
219         value[pos] = oldValue + incr;
220         return oldValue;
221        }
222        /** Adds an increment to value currently associated with a key.
223         *
224         * <P>Note that this method respects the {@linkplain #defaultReturnValue() default return value} semantics: when
225         * called with a key that does not currently appears in the map, the key
226         * will be associated with the default return value plus
227         * the given increment.
228         *
229         * @param k the key.
230         * @param incr the increment.
231         * @return the old value, or the {@linkplain #defaultReturnValue() default return value} if no value was present for the given key.
232         */
233        public long addTo(final K k, final long incr) {
234         int pos;
235         if (( (k) == (null) )) {
236          if (containsNullKey) return addToValue(n, incr);
237          pos = n;
238          containsNullKey = true;
239         }
240         else {
241          K curr;
242          final K[] key = this.key;
243          // The starting point.
244          if (! ( (curr = key[pos = ( it.unimi.dsi.fastutil.HashCommon.mix( System.identityHashCode(k) ) ) & mask]) == (null) )) {
245           if (( (curr) == (k) )) return addToValue(pos, incr);
246           while(! ( (curr = key[pos = (pos + 1) & mask]) == (null) ))
247            if (( (curr) == (k) )) return addToValue(pos, incr);
248          }
249         }
250         key[pos] = k;
251         value[pos] = defRetValue + incr;
252         if (size++ >= maxFill) rehash(arraySize(size + 1, f));
253         if (ASSERTS) checkTable();
254         return defRetValue;
255        }
256        /** Shifts left entries with the specified hash code, starting at the specified position,
257         * and empties the resulting free entry.
258         *
259         * @param pos a starting position.
260         */
261        protected final void shiftKeys(int pos) {
262         // Shift entries with the same hash.
263         int last, slot;
264         K curr;
265         final K[] key = this.key;
266         for(;;) {
267          pos = ((last = pos) + 1) & mask;
268          for(;;) {
269           if (( (curr = key[pos]) == (null) )) {
270            key[last] = (null);
271            return;
272           }
273           slot = ( it.unimi.dsi.fastutil.HashCommon.mix( System.identityHashCode(curr) ) ) & mask;
274           if (last <= pos ? last >= slot || slot > pos : last >= slot && slot > pos) break;
275           pos = (pos + 1) & mask;
276          }
277          key[last] = curr;
278          value[last] = value[pos];
279         }
280        }
281        @SuppressWarnings("unchecked")
282        public long removeLong(final Object k) {
283         if (( ((K) k) == (null) )) {
284          if (containsNullKey) return removeNullEntry();
285          return defRetValue;
286         }
287         K curr;
288         final K[] key = this.key;
289         int pos;
290         // The starting point.
291         if (( (curr = key[pos = ( it.unimi.dsi.fastutil.HashCommon.mix( System.identityHashCode(k) ) ) & mask]) == (null) )) return defRetValue;
292         if (( (k) == (curr) )) return removeEntry(pos);
293         while(true) {
294          if (( (curr = key[pos = (pos + 1) & mask]) == (null) )) return defRetValue;
295          if (( (k) == (curr) )) return removeEntry(pos);
296         }
297        }
298        @Override
299        @SuppressWarnings("unchecked")
300        public long getLong(final Object k) {
301         if (( ((K) k) == (null) )) return containsNullKey ? value[n] : defRetValue;
302         K curr;
303         final K[] key = this.key;
304         int pos;
305         // The starting point.
306         if (( (curr = key[pos = ( it.unimi.dsi.fastutil.HashCommon.mix( System.identityHashCode(k) ) ) & mask]) == (null) )) return defRetValue;
307         if (( (k) == (curr) )) return value[pos];
308         // There's always an unused entry.
309         while(true) {
310          if (( (curr = key[pos = (pos + 1) & mask]) == (null) )) return defRetValue;
311          if (( (k) == (curr) )) return value[pos];
312         }
313        }

316        public boolean containsKey(final Object k) {
317         if (( ((K) k) == (null) )) return containsNullKey;
318         K curr;
319         final K[] key = this.key;
320         int pos;
321         // The starting point.
322         if (( (curr = key[pos = ( it.unimi.dsi.fastutil.HashCommon.mix( System.identityHashCode(k) ) ) & mask]) == (null) )) return false;

324         // There's always an unused entry.
325         while(true) {
326          if (( (curr = key[pos = (pos + 1) & mask]) == (null) )) return false;

328         }
329        }

332         final long value[] = this.value;

336         return false;

394         }
395         @Override

397          final long oldValue = value[index];
398          value[index] = v;
399          return oldValue;
400         }
401         /** {@inheritDoc}
402                 * @deprecated Please use the corresponding type-specific method instead. */
403         @Deprecated
404         @Override
405         public Long getValue() {

407         }
408         /** {@inheritDoc}
409                 * @deprecated Please use the corresponding type-specific method instead. */

414         }
415         @SuppressWarnings("unchecked")
416         @Override
417         public boolean equals(final Object o) {
418          if (!(o instanceof Map.Entry)) return false;
419          Map.Entry<K, Long> e = (Map.Entry<K, Long>)o;
420          return ( (key[index]) == ((e.getKey())) ) && ( (value[index]) == (((e.getValue()).longValue())) );
421         }
422         @Override
423         public int hashCode() {
424          return ( System.identityHashCode(key[index]) ) ^ it.unimi.dsi.fastutil.HashCommon.long2int(value[index]);
425         }

431        /** An iterator over a hash map. */
432        private class MapIterator {
433         /** The index of the last entry returned, if positive or zero; initially, {@link #n}. If negative, the last
434                        entry returned was that of the key of index {@code - pos - 1} from the {@link #wrapped} list. */
435         int pos = n;
436         /** The index of the last entry that has been returned (more precisely, the value of {@link #pos} if {@link #pos} is positive,
437                        or {@link Integer#MIN_VALUE} if {@link #pos} is negative). It is -1 if either
438                        we did not return an entry yet, or the last returned entry has been removed. */
439         int last = -1;
440         /** A downward counter measuring how many entries must still be returned. */
441         int c = size;
442         /** A boolean telling us whether we should return the entry with the null key. */
443         boolean mustReturnNullKey = Reference2LongOpenHashMap.this.containsNullKey;
444         /** A lazily allocated list containing keys of entries that have wrapped around the table because of removals. */
445         ReferenceArrayList <K> wrapped;
446         public boolean hasNext() {
447          return c != 0;
448         }
449         public int nextEntry() {
450          if (! hasNext()) throw new NoSuchElementException();
451          c--;
452          if (mustReturnNullKey) {
453           mustReturnNullKey = false;
454           return last = n;
455          }
456          final K key[] = Reference2LongOpenHashMap.this.key;
457          for(;;) {
458           if (--pos < 0) {
459            // We are just enumerating elements from the wrapped list.
460            last = Integer.MIN_VALUE;
461            final K k = wrapped.get(- pos - 1);
462            int p = ( it.unimi.dsi.fastutil.HashCommon.mix( System.identityHashCode(k) ) ) & mask;
463            while (! ( (k) == (key[p]) )) p = (p + 1) & mask;
464            return p;
465           }
466           if (! ( (key[pos]) == (null) )) return last = pos;
467          }
468         }
469         /** Shifts left entries with the specified hash code, starting at the specified position,
470                 * and empties the resulting free entry.
471                 *
472                 * @param pos a starting position.
473                 */
474         private final void shiftKeys(int pos) {
475          // Shift entries with the same hash.
476          int last, slot;
477          K curr;
478          final K[] key = Reference2LongOpenHashMap.this.key;
479          for(;;) {
480           pos = ((last = pos) + 1) & mask;
481           for(;;) {
482            if (( (curr = key[pos]) == (null) )) {
483             key[last] = (null);
484             return;
485            }
486            slot = ( it.unimi.dsi.fastutil.HashCommon.mix( System.identityHashCode(curr) ) ) & mask;
487            if (last <= pos ? last >= slot || slot > pos : last >= slot && slot > pos) break;
488            pos = (pos + 1) & mask;
489           }
490           if (pos < last) { // Wrapped entry.
491            if (wrapped == null) wrapped = new ReferenceArrayList <K>(2);
492            wrapped.add(key[pos]);
493           }
494           key[last] = curr;
495           value[last] = value[pos];
496          }
497         }
498         public void remove() {
499          if (last == -1) throw new IllegalStateException();
500          if (last == n) {
501           containsNullKey = false;
502           key[n] = null;
503          }
504          else if (pos >= 0) shiftKeys(last);
505          else {
506           // We're removing wrapped entries.
507           Reference2LongOpenHashMap.this.removeLong(wrapped.set(- pos - 1, null));
508           last = -1; // Note that we must not decrement size
509           return;
510          }
511          size--;
512          last = -1; // You can no longer remove this entry.
513          if (ASSERTS) checkTable();
514         }
515         public int skip(final int n) {
516          int i = n;
517          while(i-- != 0 && hasNext()) nextEntry();
518          return n - i - 1;
519         }
520        }

541        private final class MapEntrySet extends AbstractObjectSet<Reference2LongMap.Entry <K> > implements FastEntrySet <K> {
542         public ObjectIterator<Reference2LongMap.Entry <K> > iterator() {
543          return new EntryIterator();
544         }
545         public ObjectIterator<Reference2LongMap.Entry <K> > fastIterator() {
546          return new FastEntryIterator();
547         }
548         @Override
549         @SuppressWarnings("unchecked")
550         public boolean contains(final Object o) {
551          if (!(o instanceof Map.Entry)) return false;
552          final Map.Entry<?,?> e = (Map.Entry<?,?>)o;
553          if (e.getValue() == null || ! (e.getValue() instanceof Long)) return false;
554          final K k = ((K) e.getKey());
555          final long v = ((((Long)( e.getValue())).longValue()));
556          if (( (k) == (null) )) return Reference2LongOpenHashMap.this.containsNullKey && ( (value[n]) == (v) );
557          K curr;
558          final K[] key = Reference2LongOpenHashMap.this.key;
559          int pos;
560          // The starting point.
561          if (( (curr = key[pos = ( it.unimi.dsi.fastutil.HashCommon.mix( System.identityHashCode(k) ) ) & mask]) == (null) )) return false;
562          if (( (k) == (curr) )) return ( (value[pos]) == (v) );
563          // There's always an unused entry.
564          while(true) {
565           if (( (curr = key[pos = (pos + 1) & mask]) == (null) )) return false;
566           if (( (k) == (curr) )) return ( (value[pos]) == (v) );
567          }
568         }
569         @Override
570         @SuppressWarnings("unchecked")
571         public boolean remove(final Object o) {
572          if (!(o instanceof Map.Entry)) return false;
573          final Map.Entry<?,?> e = (Map.Entry<?,?>)o;
574          if (e.getValue() == null || ! (e.getValue() instanceof Long)) return false;
575          final K k = ((K) e.getKey());
576          final long v = ((((Long)( e.getValue())).longValue()));
577          if (( (k) == (null) )) {
578           if (containsNullKey && ( (value[n]) == (v) )) {
579            removeNullEntry();
580            return true;
581           }
582           return false;
583          }
584          K curr;
585          final K[] key = Reference2LongOpenHashMap.this.key;
586          int pos;
587          // The starting point.
588          if (( (curr = key[pos = ( it.unimi.dsi.fastutil.HashCommon.mix( System.identityHashCode(k) ) ) & mask]) == (null) )) return false;
589          if (( (curr) == (k) )) {
590           if (( (value[pos]) == (v) )) {
591            removeEntry(pos);
592            return true;
593           }
594           return false;
595          }
596          while(true) {
597           if (( (curr = key[pos = (pos + 1) & mask]) == (null) )) return false;

603           }
604          }

710        /** Rehashes the map, making the table as small as possible.
711         *
712         * <P>This method rehashes the table to the smallest size satisfying the
713         * load factor. It can be used when the set will not be changed anymore, so
714         * to optimize access speed and size.

717         * does nothing.

720         * @see #trim(int)
721         */
722        public boolean trim() {
723         final int l = arraySize(size, f);
724         if (l >= n || size > maxFill(l, f)) return true;

731        /** Rehashes this map if the table is too large.
732         *
733         * <P>Let <var>N</var> be the smallest table size that can hold
734         * <code>max(n,{@link #size()})</code> entries, still satisfying the load factor. If the current
735         * table size is smaller than or equal to <var>N</var>, this method does
736         * nothing. Otherwise, it rehashes this map in a table of size
737         * <var>N</var>.
738         *
739         * <P>This method is useful when reusing maps.  {@linkplain #clear() Clearing a
740         * map} leaves the table size untouched. If you are reusing a map
741         * many times, you can call this method with a typical
742         * size to avoid keeping around a very large table just
743         * because of a few large transient maps.
744         *
745         * @param n the threshold for the trimming.
746         * @return true if there was enough memory to trim the map.
747         * @see #trim()
748         */
749        public boolean trim(final int n) {
750         final int l = HashCommon.nextPowerOfTwo((int)Math.ceil(n / f));
751         if (l >= n || size > maxFill(l, f)) return true;
752         try {
753          rehash(l);
754         }
755         catch(OutOfMemoryError cantDoIt) { return false; }
756         return true;
757        }
758        /** Rehashes the map.
759         *
760         * <P>This method implements the basic rehashing strategy, and may be
761         * overriden by subclasses implementing different rehashing strategies (e.g.,
762         * disk-based rehashing). However, you should not override this method
763         * unless you understand the internal workings of this class.
764         *
765         * @param newN the new size
766         */
767        @SuppressWarnings("unchecked")
768        protected void rehash(final int newN) {
769         final K key[] = this.key;
770         final long value[] = this.value;
771         final int mask = newN - 1; // Note that this is used by the hashing macro
772         final K newKey[] = (K[]) new Object[newN + 1];
773         final long newValue[] = new long[newN + 1];
774         int i = n, pos;
775         for(int j = realSize(); j-- != 0;) {
776          while(( (key[--i]) == (null) ));
777          if (! ( (newKey[pos = ( it.unimi.dsi.fastutil.HashCommon.mix( System.identityHashCode(key[i]) ) ) & mask]) == (null) ))
778           while (! ( (newKey[pos = (pos + 1) & mask]) == (null) ));
779          newKey[pos] = key[i];
780          newValue[pos] = value[i];
781         }
782         newValue[newN] = value[n];
783         n = newN;
784         this.mask = mask;
785         maxFill = maxFill(n, f);
786         this.key = newKey;
787         this.value = newValue;
788        }

814        /** Returns a hash code for this map.
815         *
816         * This method overrides the generic method provided by the superclass.
817         * Since <code>equals()</code> is not overriden, it is important
818         * that the value returned by this method is the same value as
819         * the one returned by the overriden method.
820         *
821         * @return a hash code for this map.
822         */
823        @Override
824        public int hashCode() {
825         int h = 0;
826         for(int j = realSize(), i = 0, t = 0; j-- != 0;) {
827          while(( (key[i]) == (null) )) i++;
828          if (this != key[i])
829           t = ( System.identityHashCode(key[i]) );
830           t ^= it.unimi.dsi.fastutil.HashCommon.long2int(value[i]);
831          h += t;
832          i++;
833         }
834         // Zero / null keys have hash zero.
835         if (containsNullKey) h += it.unimi.dsi.fastutil.HashCommon.long2int(value[n]);
836         return h;
837        }

849        @SuppressWarnings("unchecked")
850        private void readObject(java.io.ObjectInputStream s) throws java.io.IOException, ClassNotFoundException {
851         s.defaultReadObject();
852         n = arraySize(size, f);
853         maxFill = maxFill(n, f);
854         mask = n - 1;
855         final K key[] = this.key = (K[]) new Object[n + 1];
856         final long value[] = this.value = new long[n + 1];
857         K k;
858         long v;
859         for(int i = size, pos; i-- != 0;) {
860          k = (K) s.readObject();
861          v = s.readLong();
862          if (( (k) == (null) )) {
863           pos = n;
864           containsNullKey = true;
865          }
866          else {
867           pos = ( it.unimi.dsi.fastutil.HashCommon.mix( System.identityHashCode(k) ) ) & mask;
868           while (! ( (key[pos]) == (null) )) pos = (pos + 1) & mask;
869          }
870          key[pos] = k;
871          value[pos] = v;
872         }
873         if (ASSERTS) checkTable();
874        }
