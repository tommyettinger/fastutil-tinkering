038/** A class providing static methods and objects that do useful things with type-specific arrays.
039        *
040        * In particular, the <code>ensureCapacity()</code>, <code>grow()</code>,
041        * <code>trim()</code> and <code>setLength()</code> methods allow to handle
042        * arrays much like array lists. This can be very useful when efficiency (or
043        * syntactic simplicity) reasons make array lists unsuitable.
044        *
045        * <P><strong>Warning:</strong> if your array is not of type {@code Object[]},
046        * {@link #ensureCapacity(Object[],int,int)} and {@link #grow(Object[],int,int)}
047        * will use {@linkplain java.lang.reflect.Array#newInstance(Class,int) reflection}
048        * to preserve your array type. Reflection is <em>significantly slower</em> than using <code>new</code>.
049        * This phenomenon is particularly
050        * evident in the first growth phases of an array reallocated with doubling (or similar) logic.
051        *
052        * <h2>Sorting</h2>
053        *
054        * <p>There are several sorting methods available. The main theme is that of letting you choose
055        * the sorting algorithm you prefer (i.e., trading stability of mergesort for no memory allocation in quicksort).
056        * Several algorithms provide a parallel version, that will use the {@linkplain Runtime#availableProcessors() number of cores available}.
057        *
058        * <p>All comparison-based algorithm have an implementation based on a type-specific comparator.
059        *
060        * <p>If you are fine with not knowing exactly which algorithm will be run (in particular, not knowing exactly whether a support array will be allocated),
061        * the dual-pivot parallel sorts in {@link java.util.Arrays}
062        * are about 50% faster than the classical single-pivot implementation used here.
063        *
064        * <p>In any case, if sorting time is important I suggest that you benchmark your sorting load
065        * with your data distribution and on your architecture.
066        *
067        * @see java.util.Arrays
068        */

332        private static <K> int med3(final K x[], final int a, final int b, final int c, Comparator <K> comp) {
333         final int ab = comp.compare(x[a], x[b]);
334         final int ac = comp.compare(x[a], x[c]);
335         final int bc = comp.compare(x[b], x[c]);
336         return (ab < 0 ?
337          (bc < 0 ? b : ac < 0 ? c : a) :
338          (bc > 0 ? b : ac > 0 ? c : a));
339        }
340        private static <K> void selectionSort(final K[] a, final int from, final int to, final Comparator <K> comp) {
341         for(int i = from; i < to - 1; i++) {
342          int m = i;
343          for(int j = i + 1; j < to; j++) if (comp.compare(a[j], a[m]) < 0) m = j;
344          if (m != i) {
345           final K u = a[i];
346           a[i] = a[m];
347           a[m] = u;
348          }
349         }
350        }
351        private static <K> void insertionSort(final K[] a, final int from, final int to, final Comparator <K> comp) {
352         for (int i = from; ++i < to;) {
353          K t = a[i];
354          int j = i;
355          for (K u = a[j - 1]; comp.compare(t, u) < 0; u = a[--j - 1]) {
356           a[j] = u;
357           if (from == j - 1) {
358            --j;
359            break;
360           }
361          }
362          a[j] = t;
363         }
364        }
365        /** Sorts the specified range of elements according to the order induced by the specified
366         * comparator using quicksort.
367         *
368         * <p>The sorting algorithm is a tuned quicksort adapted from Jon L. Bentley and M. Douglas
369         * McIlroy, &ldquo;Engineering a Sort Function&rdquo;, <i>Software: Practice and Experience</i>, 23(11), pages
370         * 1249&minus;1265, 1993.
371         *
372         * <p>Note that this implementation does not allocate any object, contrarily to the implementation
373         * used to sort primitive types in {@link java.util.Arrays}, which switches to mergesort on large inputs.
374         *
375         * @param x the array to be sorted.
376         * @param from the index of the first element (inclusive) to be sorted.
377         * @param to the index of the last element (exclusive) to be sorted.
378         * @param comp the comparator to determine the sorting order.
379         *
380         */
381        public static <K> void quickSort(final K[] x, final int from, final int to, final Comparator <K> comp) {
382         final int len = to - from;
383         // Selection sort on smallest arrays
384         if (len < QUICKSORT_NO_REC) {
385          selectionSort(x, from, to, comp);
386          return;
387         }
388         // Choose a partition element, v
389         int m = from + len / 2;
390         int l = from;
391         int n = to - 1;
392         if (len > QUICKSORT_MEDIAN_OF_9) { // Big arrays, pseudomedian of 9
393          int s = len / 8;
394          l = med3(x, l, l + s, l + 2 * s, comp);
395          m = med3(x, m - s, m, m + s, comp);
396          n = med3(x, n - 2 * s, n - s, n, comp);
397         }
398         m = med3(x, l, m, n, comp); // Mid-size, med of 3
399         final K v = x[m];
400         // Establish Invariant: v* (<v)* (>v)* v*
401         int a = from, b = a, c = to - 1, d = c;
402         while(true) {
403          int comparison;
404          while (b <= c && (comparison = comp.compare(x[b], v)) <= 0) {
405           if (comparison == 0) swap(x, a++, b);
406           b++;
407          }
408          while (c >= b && (comparison = comp.compare(x[c], v)) >=0) {
409           if (comparison == 0) swap(x, c, d--);
410           c--;
411          }
412          if (b > c) break;
413          swap(x, b++, c--);
414         }
415         // Swap partition elements back to middle
416         int s;
417         s = Math.min(a - from, b - a);
418         swap(x, from, b - s, s);
419         s = Math.min(d - c, to - d - 1);
420         swap(x, b, to - s, s);
421         // Recursively sort non-partition-elements
422         if ((s = b - a) > 1) quickSort(x, from, from + s, comp);
423         if ((s = d - c) > 1) quickSort(x, to - s, to, comp);
424        }
425        /** Sorts an array according to the order induced by the specified
426         * comparator using quicksort.
427         *
428         * <p>The sorting algorithm is a tuned quicksort adapted from Jon L. Bentley and M. Douglas
429         * McIlroy, &ldquo;Engineering a Sort Function&rdquo;, <i>Software: Practice and Experience</i>, 23(11), pages
430         * 1249&minus;1265, 1993.
431         *
432         * <p>Note that this implementation does not allocate any object, contrarily to the implementation
433         * used to sort primitive types in {@link java.util.Arrays}, which switches to mergesort on large inputs.
434         *
435         * @param x the array to be sorted.
436         * @param comp the comparator to determine the sorting order.
437         *
438         */
439        public static <K> void quickSort(final K[] x, final Comparator <K> comp) {
440         quickSort(x, 0, x.length, comp);
441        }
442        protected static class ForkJoinQuickSortComp <K> extends RecursiveAction {
443         private static final long serialVersionUID = 1L;
444         private final int from;
445         private final int to;
446         private final K[] x;
447         private final Comparator <K> comp;
448         public ForkJoinQuickSortComp(final K[] x , final int from , final int to, final Comparator <K> comp) {
449          this.from = from;
450          this.to = to;
451          this.x = x;
452          this.comp = comp;
453         }
454         @Override
455         protected void compute() {
456          final K[] x = this.x;
457          final int len = to - from;
458          if (len < PARALLEL_QUICKSORT_NO_FORK) {
459           quickSort(x, from, to, comp);
460           return;
461          }
462          // Choose a partition element, v
463          int m = from + len / 2;
464          int l = from;
465          int n = to - 1;
466          int s = len / 8;
467          l = med3(x, l, l + s, l + 2 * s, comp);
468          m = med3(x, m - s, m, m + s, comp);
469          n = med3(x, n - 2 * s, n - s, n, comp);
470          m = med3(x, l, m, n, comp);
471          final K v = x[m];
472          // Establish Invariant: v* (<v)* (>v)* v*
473          int a = from, b = a, c = to - 1, d = c;
474          while (true) {
475           int comparison;
476           while (b <= c && (comparison = comp.compare(x[b], v)) <= 0) {
477            if (comparison == 0) swap(x, a++, b);
478            b++;
479           }
480           while (c >= b && (comparison = comp.compare(x[c], v)) >= 0) {
481            if (comparison == 0) swap(x, c, d--);
482            c--;
483           }
484           if (b > c) break;
485           swap(x, b++, c--);
486          }
487          // Swap partition elements back to middle
488          int t;
489          s = Math.min(a - from, b - a);
490          swap(x, from, b - s, s);
491          s = Math.min(d - c, to - d - 1);
492          swap(x, b, to - s, s);
493          // Recursively sort non-partition-elements
494          s = b - a;
495          t = d - c;
496          if (s > 1 && t > 1) invokeAll(new ForkJoinQuickSortComp <K>(x, from, from + s, comp), new ForkJoinQuickSortComp <K>(x, to - t, to, comp));
497          else if (s > 1) invokeAll(new ForkJoinQuickSortComp <K>(x, from, from + s, comp));
498          else invokeAll(new ForkJoinQuickSortComp <K>(x, to - t, to, comp));
499         }
500        }

503         *
504         * <p>The sorting algorithm is a tuned quicksort adapted from Jon L. Bentley and M. Douglas
505         * McIlroy, &ldquo;Engineering a Sort Function&rdquo;, <i>Software: Practice and Experience</i>, 23(11), pages
506         * 1249&minus;1265, 1993.
507         *
