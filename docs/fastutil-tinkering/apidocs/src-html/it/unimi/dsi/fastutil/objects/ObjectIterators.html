113        /** A class to wrap arrays in iterators. */
114        private static class ArrayIterator <K> extends AbstractObjectListIterator <K> {
115         private final K[] array;
116         private final int offset, length;
117         private int curr;
118         public ArrayIterator(final K[] array, final int offset, final int length) {
119          this.array = array;
120          this.offset = offset;
121          this.length = length;
122         }
123         @Override
124         public boolean hasNext() { return curr < length; }
125         @Override
126         public boolean hasPrevious() { return curr > 0; }
127         @Override
128         public K next() {
129          if (! hasNext()) throw new NoSuchElementException();
130          return array[offset + curr++];
131         }
132         @Override
133         public K previous() {
134          if (! hasPrevious()) throw new NoSuchElementException();
135          return array[offset + --curr];
136         }
137         @Override
138         public int skip(int n) {
139          if (n <= length - curr) {
140           curr += n;
141           return n;
142          }
143          n = length - curr;
144          curr = length;
145          return n;
146         }
147         @Override
148         public int back(int n) {
149          if (n <= curr) {
150           curr -= n;
151           return n;
152          }
153          n = curr;
154          curr = 0;
155          return n;
156         }
157         @Override
158         public int nextIndex() {
159          return curr;
160         }
161         @Override
162         public int previousIndex() {
163          return curr - 1;
164         }
165        }

192        /** Unwraps an iterator into an array starting at a given offset for a given number of elements.
193         *
194         * <P>This method iterates over the given type-specific iterator and stores the elements
195         * returned, up to a maximum of <code>length</code>, in the given array starting at <code>offset</code>.
196         * The number of actually unwrapped elements is returned (it may be less than <code>max</code> if
197         * the iterator emits less than <code>max</code> elements).
198         *
199         * @param i a type-specific iterator.
200         * @param array an array to contain the output of the iterator.
201         * @param offset the first element of the array to be returned.
202         * @param max the maximum number of elements to unwrap.
203         * @return the number of elements unwrapped.
204         */
205        public static <K> int unwrap(final Iterator <? extends K> i, final K array[], int offset, final int max) {
206         if (max < 0) throw new IllegalArgumentException("The maximum number of elements (" + max + ") is negative");
207         if (offset < 0 || offset + max > array.length) throw new IllegalArgumentException();
208         int j = max;
209         while(j-- != 0 && i.hasNext()) array[offset++] = i.next();
210         return max - j - 1;
211        }
212        /** Unwraps an iterator into an array.
213         *
214         * <P>This method iterates over the given type-specific iterator and stores the
215         * elements returned in the given array. The iteration will stop when the
216         * iterator has no more elements or when the end of the array has been reached.
217         *
218         * @param i a type-specific iterator.
219         * @param array an array to contain the output of the iterator.
220         * @return the number of elements unwrapped.
221         */
222        public static <K> int unwrap(final Iterator <? extends K> i, final K array[]) {
223         return unwrap(i, array, 0, array.length);
224        }
225        /** Unwraps an iterator, returning an array, with a limit on the number of elements.
226         *
227         * <P>This method iterates over the given type-specific iterator and returns an array
228         * containing the elements returned by the iterator. At most <code>max</code> elements
229         * will be returned.
230         *
231         * @param i a type-specific iterator.
232         * @param max the maximum number of elements to be unwrapped.
233         * @return an array containing the elements returned by the iterator (at most <code>max</code>).
234         */
235        @SuppressWarnings("unchecked")
236        public static <K> K[] unwrap(final Iterator <? extends K> i, int max) {
237         if (max < 0) throw new IllegalArgumentException("The maximum number of elements (" + max + ") is negative");
238         K array[] = (K[]) new Object[16];
239         int j = 0;
240         while(max-- != 0 && i.hasNext()) {
241          if (j == array.length) array = ObjectArrays.grow(array, j + 1);
242          array[j++] = i.next();
243         }
244         return ObjectArrays.trim(array, j);
245        }

257        /** Unwraps an iterator into a type-specific collection,  with a limit on the number of elements.
258         *
259         * <P>This method iterates over the given type-specific iterator and stores the elements
260         * returned, up to a maximum of <code>max</code>, in the given type-specific collection.
261         * The number of actually unwrapped elements is returned (it may be less than <code>max</code> if
262         * the iterator emits less than <code>max</code> elements).
263         *
264         * @param i a type-specific iterator.
265         * @param c a type-specific collection array to contain the output of the iterator.
266         * @param max the maximum number of elements to unwrap.
267         * @return the number of elements unwrapped. Note that
268         * this is the number of elements returned by the iterator, which is not necessarily the number
269         * of elements that have been added to the collection (because of duplicates).
270         */
271        public static <K> int unwrap(final Iterator <K> i, final ObjectCollection <? super K> c, final int max) {
272         if (max < 0) throw new IllegalArgumentException("The maximum number of elements (" + max + ") is negative");
273         int j = max;
274         while(j-- != 0 && i.hasNext()) c.add(i.next());
275         return max - j - 1;
276        }
277        /** Unwraps an iterator into a type-specific collection.
278         *
279         * <P>This method iterates over the given type-specific iterator and stores the
280         * elements returned in the given type-specific collection. The returned count on the number
281         * unwrapped elements is a long, so that it will work also with very large collections.
282         *
283         * @param i a type-specific iterator.
284         * @param c a type-specific collection to contain the output of the iterator.
285         * @return the number of elements unwrapped. Note that
286         * this is the number of elements returned by the iterator, which is not necessarily the number
287         * of elements that have been added to the collection (because of duplicates).
288         */
289        public static <K> long unwrap(final Iterator <K> i, final ObjectCollection <? super K> c) {
290         long n = 0;
291         while(i.hasNext()) {
292          c.add(i.next());
293          n++;
294         }
295         return n;
296        }

430        private static class IteratorConcatenator <K> extends AbstractObjectIterator <K> {
431         final ObjectIterator <? extends K> a[];
432         int offset, length, lastOffset = -1;
433         public IteratorConcatenator(final ObjectIterator <? extends K> a[], int offset, int length) {
434          this.a = a;
435          this.offset = offset;
436          this.length = length;
437          advance();
438         }
439         private void advance() {
440          while(length != 0) {
441           if (a[offset].hasNext()) break;
442           length--;
443           offset++;
444          }
445          return;
446         }
447         @Override
448         public boolean hasNext() {
449          return length > 0;
450         }
451         @Override
452         public K next() {
453          if (! hasNext()) throw new NoSuchElementException();
454          K next = a[lastOffset = offset].next();
455          advance();
456          return next;
457         }
458         @Override
459         public void remove() {
460          if (lastOffset == -1) throw new IllegalStateException();
461          a[lastOffset].remove();
462         }
463         @Override
464         public int skip(int n) {
465          lastOffset = -1;
466          int skipped = 0;
467          while(skipped < n && length != 0) {
468           skipped += a[offset].skip(n - skipped);
469           if (a[offset].hasNext()) break;
470           length--;
471           offset++;
472          }
473          return skipped;
474         }
475        }
