016package it.unimi.dsi.fastutil.booleans;
017/**  An abstract class facilitating the creation of type-specific {@linkplain it.unimi.dsi.fastutil.BidirectionalIterator bidirectional iterators}.
018        *
019        * <P>To create a type-specific bidirectional iterator, besides what is needed
020        * for an iterator you need both a method returning the previous element as
021        * primitive type and a method returning the previous element as an
022        * object. However, if you inherit from this class you need just one (anyone).
023        *
024        * <P>This class implements also a trivial version of {@link #back(int)} that
025        * uses type-specific methods.
026        */
027public abstract class AbstractBooleanBidirectionalIterator extends AbstractBooleanIterator implements BooleanBidirectionalIterator {
028        protected AbstractBooleanBidirectionalIterator() {}
029        /** <p>This implementation delegates to {@link #previous()}. */
030        @Override
031        public boolean previousBoolean() { return previous().booleanValue(); }
032        /** {@inheritDoc}
033         * <p>This implementation delegates to the corresponding type-specific method.
034         * @deprecated Please use the corresponding type-specific method instead. */
035        @Deprecated
036        @Override
037        public Boolean previous() { return Boolean.valueOf(previousBoolean()); }
038        /** {@inheritDoc}
039         *  <P>This implemntation iterates the type-specific version of {@link #previous()} for
040         * at most {@code n} times, stopping if {@link #hasPrevious()} becomes false.
041         *
042         * @param n the number of elements to skip backwards.
043         * @return the number of elements actually skipped.
044         * @see #previous()
045         */
046        @Override
047        public int back(final int n) {
048         int i = n;
049         while(i-- != 0 && hasPrevious()) previousBoolean();
050         return n - i - 1;
051        }
052}
