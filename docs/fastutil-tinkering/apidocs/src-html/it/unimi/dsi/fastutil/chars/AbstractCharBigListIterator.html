016package it.unimi.dsi.fastutil.chars;
017import java.util.ListIterator;
018import it.unimi.dsi.fastutil.BigListIterator;
019/**  An abstract class facilitating the creation of type-specific {@linkplain it.unimi.dsi.fastutil.BigListIterator big-list iterators}.
020        *
021        * <p>This implementation provides (deprecated) implementations of {@link ListIterator#previousIndex()} and {@link ListIterator#nextIndex()} that
022        * just invoke the corresponding {@link BigListIterator} methods.
023        *
024        * @see java.util.ListIterator
025        * @see it.unimi.dsi.fastutil.BigListIterator
026        */
027public abstract class AbstractCharBigListIterator extends AbstractCharBidirectionalIterator implements CharBigListIterator {
028        protected AbstractCharBigListIterator() {}
029        /** {@inheritDoc}
030         * <P>This implementation always throws an {@link UnsupportedOperationException}. */
031        @Override
032        public void set(char k) {
033         throw new UnsupportedOperationException();
034        }
035        /** {@inheritDoc}
036         * <P>This implementation always throws an {@link UnsupportedOperationException}. */
037        @Override
038        public void add(char k) {
039         throw new UnsupportedOperationException();
040        }
041        /** {@inheritDoc}
042         * <p>This implementation delegates to the corresponding type-specific method.
043         * @deprecated Please use the corresponding type-specific method instead. */
044        @Deprecated
045        @Override
046        public void set(Character ok) {
047         set(ok.charValue());
048        }
049        /** {@inheritDoc}
050         * <p>This implementation delegates to the corresponding type-specific method.
051         * @deprecated Please use the corresponding type-specific method instead. */
052        @Deprecated
053        @Override
054        public void add(Character ok) {
055         add(ok.charValue());
056        }
057        /** {@inheritDoc}
058         * <P>This implementaion iterates the type-specific version of {@link #next()} for at most
059         * {@code n} times, stopping if {@link #hasNext()} becomes false.
060         *
061         * @param n the number of elements to skip.
062         * @return the number of elements actually skipped.
063         */
064        @Override
065        public long skip(final long n) {
066         long i = n;
067         while(i-- != 0 && hasNext()) nextChar();
068         return n - i - 1;
069        }
070        /** {@inheritDoc}
071         *  <P>This implemntation iterates the type-specific version of {@link #previous()} for
072         * at most {@code n} times, stopping if {@link #hasPrevious()} becomes false.
073         *
074         * @param n the number of elements to skip backwards.
075         * @return the number of elements actually skipped.
076         */
077        public long back(final long n) {
078         long i = n;
079         while(i-- != 0 && hasPrevious()) previousChar();
080         return n - i - 1;
081        }
082}
