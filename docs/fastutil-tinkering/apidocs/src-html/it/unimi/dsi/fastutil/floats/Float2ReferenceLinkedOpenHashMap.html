046/**  A type-specific linked hash map with with a fast, small-footprint implementation.
047
048        *
049
050        * <P>Instances of this class use a hash table to represent a map. The table is
051
052        * filled up to a specified <em>load factor</em>, and then doubled in size to
053
054        * accommodate new entries. If the table is emptied below <em>one fourth</em>
055
056        * of the load factor, it is halved in size. However, halving is
057
058        * not performed when deleting entries from an iterator, as it would interfere
059
060        * with the iteration process.
061
062        *
063
064        * <p>Note that {@link #clear()} does not modify the hash table size.
065
066        * Rather, a family of {@linkplain #trim() trimming
067
068        * methods} lets you control the size of the table; this is particularly useful
069
070        * if you reuse instances of this class.
071
072        *
073
074        * <P>Iterators generated by this map will enumerate pairs in the same order in which they
075
076        * have been added to the map (addition of pairs whose key is already present
077
078        * in the set does not change the iteration order). Note that this order has nothing in common with the natural
079
080        * order of the keys. The order is kept by means of a doubly linked list, represented
081
082        * <i>via</i> an array of longs parallel to the table.
083
084        *
085
086        * <P>This class implements the interface of a sorted map, so to allow easy
087
088        * access of the iteration order: for instance, you can get the first key
089
090        * in iteration order with {@code firstKey()} without having to create an
091
092        * iterator; however, this class partially violates the {@link java.util.SortedMap}
093
094        * contract because all submap methods throw an exception and {@link
095
096        * #comparator()} returns always {@code null}.
097
098        *
099
100        * <p>Additional methods, such as <code>getAndMoveToFirst()</code>, make it easy
101
102        * to use instances of this class as a cache (e.g., with LRU policy).
103
104        *
105
106        * <P>The iterators provided by the views of this class using are type-specific
107
108        * {@linkplain java.util.ListIterator list iterators}, and can be started at any
109
110        * element <em>which is a key of the map</em>, or
111
112        * a {@link NoSuchElementException} exception will be thrown.
113
114        * If, however, the provided element is not the first or last key in the
115
116        * set, the first access to the list index will require linear time, as in the worst case
117
118        * the entire key set must be scanned in iteration order to retrieve the positional
119
120        * index of the starting key. If you use just the methods of a type-specific {@link it.unimi.dsi.fastutil.BidirectionalIterator},
121
122        * however, all operations will be performed in constant time.
123
124        *
125
126        * @see Hash
127
128        * @see HashCommon
129
130        */

292        private void ensureCapacity(final int capacity) {
293         final int needed = arraySize(capacity, f);

296        private void tryCapacity(final long capacity) {
297         final int needed = (int)Math.min(1 << 30, Math.max(2, HashCommon.nextPowerOfTwo((long)Math.ceil(capacity / f))));
298         if (needed > n) rehash(needed);

300        private V removeEntry(final int pos) {
301         final V oldValue = value[pos];
302         value[pos] = null;
303         size--;
304         fixPointers(pos);
305         shiftKeys(pos);
306         if (size < maxFill / 4 && n > DEFAULT_INITIAL_SIZE) rehash(n / 2);
307         return oldValue;
308        }
309        private V removeNullEntry() {
310         containsNullKey = false;
311         final V oldValue = value[n];
312         value[n] = null;
313         size--;
314         fixPointers(n);
315         if (size < maxFill / 4 && n > DEFAULT_INITIAL_SIZE) rehash(n / 2);
316         return oldValue;
317        }
318        @Override
319        public void putAll(Map<? extends Float,? extends V> m) {
320         if (f <= .5) ensureCapacity(m.size()); // The resulting map will be sized for m.size() elements
321         else tryCapacity(size() + m.size()); // The resulting map will be tentatively sized for size() + m.size() elements
322         super.putAll(m);
323        }
324        private int insert(final float k, final V v) {
325         int pos;
326         if (( Float.floatToIntBits(k) == 0 )) {
327          if (containsNullKey) return n;
328          containsNullKey = true;
329          pos = n;
330         }
331         else {
332          float curr;
333          final float[] key = this.key;
334          // The starting point.
335          if (! ( Float.floatToIntBits(curr = key[pos = it.unimi.dsi.fastutil.HashCommon.mix( it.unimi.dsi.fastutil.HashCommon.float2int(k) ) & mask]) == 0 )) {
336           if (( Float.floatToIntBits(curr) == Float.floatToIntBits(k) )) return pos;
337           while(! ( Float.floatToIntBits(curr = key[pos = (pos + 1) & mask]) == 0 ))
338            if (( Float.floatToIntBits(curr) == Float.floatToIntBits(k) )) return pos;
339          }
340         }
341         key[pos] = k;
342         value[pos] = v;
343         if (size == 0) {
344          first = last = pos;
345          // Special case of SET_UPPER_LOWER(link[pos], -1, -1);
346          link[pos] = -1L;
347         }
348         else {
349          link[last] ^= ( ( link[last] ^ ( pos & 0xFFFFFFFFL ) ) & 0xFFFFFFFFL );
350          link[pos] = ( ( last & 0xFFFFFFFFL ) << 32 ) | ( -1 & 0xFFFFFFFFL );
351          last = pos;
352         }
353         if (size++ >= maxFill) rehash(arraySize(size + 1, f));
354         if (ASSERTS) checkTable();
355         return -1;
356        }
357        @Override
358        public V put(final float k, final V v) {
359         final int pos = insert(k, v);
360         if (pos < 0) return defRetValue;
361         final V oldValue = value[pos];
362         value[pos] = v;
363         return oldValue;
364        }
365        /** Shifts left entries with the specified hash code, starting at the specified position,
366
367         * and empties the resulting free entry.
368
369         *
370
371         * @param pos a starting position.
372
373         */
374        protected final void shiftKeys(int pos) {
375         // Shift entries with the same hash.
376         int last, slot;
377         float curr;
378         final float[] key = this.key;
379         for(;;) {
380          pos = ((last = pos) + 1) & mask;
381          for(;;) {
382           if (( Float.floatToIntBits(curr = key[pos]) == 0 )) {
383            key[last] = (0);
384            value[last] = null;
385            return;
386           }
387           slot = it.unimi.dsi.fastutil.HashCommon.mix( it.unimi.dsi.fastutil.HashCommon.float2int(curr) ) & mask;
388           if (last <= pos ? last >= slot || slot > pos : last >= slot && slot > pos) break;
389           pos = (pos + 1) & mask;
390          }
391          key[last] = curr;
392          value[last] = value[pos];
393          fixPointers(pos, last);
394         }
395        }

397        public V remove(final float k) {
398         if (( Float.floatToIntBits(k) == 0 )) {
399          if (containsNullKey) return removeNullEntry();
400          return defRetValue;
401         }
402         float curr;
403         final float[] key = this.key;
404         int pos;
405         // The starting point.
406         if (( Float.floatToIntBits(curr = key[pos = it.unimi.dsi.fastutil.HashCommon.mix( it.unimi.dsi.fastutil.HashCommon.float2int(k) ) & mask]) == 0 )) return defRetValue;
407         if (( Float.floatToIntBits(k) == Float.floatToIntBits(curr) )) return removeEntry(pos);
408         while(true) {
409          if (( Float.floatToIntBits(curr = key[pos = (pos + 1) & mask]) == 0 )) return defRetValue;
410          if (( Float.floatToIntBits(k) == Float.floatToIntBits(curr) )) return removeEntry(pos);
411         }
412        }

418        /** Removes the mapping associated with the first key in iteration order.
419
420         * @return the value previously associated with the first key in iteration order.
421

423
424         */
425        public V removeFirst() {
426         if (size == 0) throw new NoSuchElementException();
427         final int pos = first;
428         // Abbreviated version of fixPointers(pos)
429         first = (int) link[pos];
430         if (0 <= first) {
431          // Special case of SET_PREV(link[first], -1)
432          link[first] |= (-1 & 0xFFFFFFFFL) << 32;
433         }
434         size--;
435         final V v = value[pos];
436         if (pos == n) {
437          containsNullKey = false;
438          value[n] = null;
439         }
440         else shiftKeys(pos);
441         if (size < maxFill / 4 && n > DEFAULT_INITIAL_SIZE) rehash(n / 2);
442         return v;
443        }

445
446         * @return the value previously associated with the last key in iteration order.

448         * @throws NoSuchElementException is this map is empty.

450         */
451        public V removeLast() {
452         if (size == 0) throw new NoSuchElementException();
453         final int pos = last;
454         // Abbreviated version of fixPointers(pos)
455         last = (int) ( link[pos] >>> 32 );
456         if (0 <= last) {
457          // Special case of SET_NEXT(link[last], -1)
458          link[last] |= -1 & 0xFFFFFFFFL;
459         }
460         size--;
461         final V v = value[pos];
462         if (pos == n) {
463          containsNullKey = false;
464          value[n] = null;
465         }
466         else shiftKeys(pos);
467         if (size < maxFill / 4 && n > DEFAULT_INITIAL_SIZE) rehash(n / 2);
468         return v;
469        }
470        private void moveIndexToFirst(final int i) {
471         if (size == 1 || first == i) return;
472         if (last == i) {
473          last = (int) ( link[i] >>> 32 );
474          // Special case of SET_NEXT(link[last], -1);
475          link[last] |= -1 & 0xFFFFFFFFL;
476         }
477         else {
478          final long linki = link[i];
479          final int prev = (int) ( linki >>> 32 );
480          final int next = (int) linki;
481          link[prev] ^= ( ( link[prev] ^ ( linki & 0xFFFFFFFFL ) ) & 0xFFFFFFFFL );
482          link[next] ^= ( ( link[next] ^ ( linki & 0xFFFFFFFF00000000L ) ) & 0xFFFFFFFF00000000L );
483         }
484         link[first] ^= ( ( link[first] ^ ( ( i & 0xFFFFFFFFL ) << 32 ) ) & 0xFFFFFFFF00000000L );
485         link[i] = ( ( -1 & 0xFFFFFFFFL ) << 32 ) | ( first & 0xFFFFFFFFL );
486         first = i;
487        }
488        private void moveIndexToLast(final int i) {
489         if (size == 1 || last == i) return;
490         if (first == i) {
491          first = (int) link[i];
492          // Special case of SET_PREV(link[first], -1);
493          link[first] |= (-1 & 0xFFFFFFFFL) << 32;
494         }
495         else {
496          final long linki = link[i];
497          final int prev = (int) ( linki >>> 32 );
498          final int next = (int) linki;
499          link[prev] ^= ( ( link[prev] ^ ( linki & 0xFFFFFFFFL ) ) & 0xFFFFFFFFL );
500          link[next] ^= ( ( link[next] ^ ( linki & 0xFFFFFFFF00000000L ) ) & 0xFFFFFFFF00000000L );
501         }
502         link[last] ^= ( ( link[last] ^ ( i & 0xFFFFFFFFL ) ) & 0xFFFFFFFFL );
503         link[i] = ( ( last & 0xFFFFFFFFL ) << 32 ) | ( -1 & 0xFFFFFFFFL );
504         last = i;
505        }

515        public V getAndMoveToFirst(final float k) {
516         if (( Float.floatToIntBits(k) == 0 )) {
517          if (containsNullKey) {
518           moveIndexToFirst(n);
519           return value[n];
520          }
521          return defRetValue;
522         }
523         float curr;
524         final float[] key = this.key;
525         int pos;
526         // The starting point.
527         if (( Float.floatToIntBits(curr = key[pos = it.unimi.dsi.fastutil.HashCommon.mix( it.unimi.dsi.fastutil.HashCommon.float2int(k) ) & mask]) == 0 )) return defRetValue;
528         if (( Float.floatToIntBits(k) == Float.floatToIntBits(curr) )) {
529          moveIndexToFirst(pos);
530          return value[pos];
531         }
532         // There's always an unused entry.
533         while(true) {
534          if (( Float.floatToIntBits(curr = key[pos = (pos + 1) & mask]) == 0 )) return defRetValue;
535          if (( Float.floatToIntBits(k) == Float.floatToIntBits(curr) )) {
536           moveIndexToFirst(pos);
537           return value[pos];
538          }
539         }
540        }

576        /** Adds a pair to the map; if the key is already present, it is moved to the first position of the iteration order.
577
578         *
579
580         * @param k the key.
581

587        public V putAndMoveToFirst(final float k, final V v) {
588         int pos;
589         if (( Float.floatToIntBits(k) == 0 )) {
590          if (containsNullKey) {
591           moveIndexToFirst(n);
592           return setValue(n, v);
593          }
594          containsNullKey = true;
595          pos = n;
596         }
597         else {
598          float curr;
599          final float[] key = this.key;
600          // The starting point.
601          if (! ( Float.floatToIntBits(curr = key[pos = it.unimi.dsi.fastutil.HashCommon.mix( it.unimi.dsi.fastutil.HashCommon.float2int(k) ) & mask]) == 0 )) {
602           if (( Float.floatToIntBits(curr) == Float.floatToIntBits(k) )) {
603            moveIndexToFirst(pos);
604            return setValue(pos, v);
605           }
606           while(! ( Float.floatToIntBits(curr = key[pos = (pos + 1) & mask]) == 0 ))
607            if (( Float.floatToIntBits(curr) == Float.floatToIntBits(k) )) {
608             moveIndexToFirst(pos);
609             return setValue(pos, v);
610            }
611          }
612         }
613         key[pos] = k;
614         value[pos] = v;
615         if (size == 0) {
616          first = last = pos;
617          // Special case of SET_UPPER_LOWER(link[pos], -1, -1);
618          link[pos] = -1L;
619         }
620         else {
621          link[first] ^= ( ( link[first] ^ ( ( pos & 0xFFFFFFFFL ) << 32 ) ) & 0xFFFFFFFF00000000L );
622          link[pos] = ( ( -1 & 0xFFFFFFFFL ) << 32 ) | ( first & 0xFFFFFFFFL );
623          first = pos;
624         }
625         if (size++ >= maxFill) rehash(arraySize(size, f));
626         if (ASSERTS) checkTable();
627         return defRetValue;
628        }

1509        /** Rehashes the map.
1510
1511         *
1512
1513         * <P>This method implements the basic rehashing strategy, and may be
1514
1515         * overriden by subclasses implementing different rehashing strategies (e.g.,
1516
1517         * disk-based rehashing). However, you should not override this method
1518
1519         * unless you understand the internal workings of this class.
1520
1521         *
1522
1523         * @param newN the new size
1524
1525         */
1526        @SuppressWarnings("unchecked")
1527        protected void rehash(final int newN) {
1528         final float key[] = this.key;
1529         final V value[] = this.value;
1530         final int mask = newN - 1; // Note that this is used by the hashing macro
1531         final float newKey[] = new float[newN + 1];
1532         final V newValue[] = (V[]) new Object[newN + 1];
1533         int i = first, prev = -1, newPrev = -1, t, pos;
1534         final long link[] = this.link;
1535         final long newLink[] = new long[newN + 1];
1536         first = -1;
1537         for(int j = size; j-- != 0;) {
1538          if (( Float.floatToIntBits(key[i]) == 0 )) pos = newN;
1539          else {
1540           pos = it.unimi.dsi.fastutil.HashCommon.mix( it.unimi.dsi.fastutil.HashCommon.float2int(key[i]) ) & mask;
1541           while (! ( Float.floatToIntBits(newKey[pos]) == 0 )) pos = (pos + 1) & mask;
1542          }
1543          newKey[pos] = key[i];
1544          newValue[pos] = value[i];
1545          if (prev != -1) {
1546           newLink[newPrev] ^= ( ( newLink[newPrev] ^ ( pos & 0xFFFFFFFFL ) ) & 0xFFFFFFFFL );
1547           newLink[pos] ^= ( ( newLink[pos] ^ ( ( newPrev & 0xFFFFFFFFL ) << 32 ) ) & 0xFFFFFFFF00000000L );
1548           newPrev = pos;
1549          }
1550          else {
1551           newPrev = first = pos;
1552           // Special case of SET(newLink[pos], -1, -1);
1553           newLink[pos] = -1L;
1554          }
1555          t = i;
1556          i = (int) link[i];
1557          prev = t;
1558         }
1559         this.link = newLink;
1560         this.last = newPrev;
1561         if (newPrev != -1)
1562          // Special case of SET_NEXT(newLink[newPrev], -1);
1563          newLink[newPrev] |= -1 & 0xFFFFFFFFL;
1564         n = newN;
1565         this.mask = mask;
1566         maxFill = maxFill(n, f);
1567         this.key = newKey;
1568         this.value = newValue;
1569        }
