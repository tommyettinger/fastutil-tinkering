039/** A type-specific hash set with a fast, small-footprint implementation whose {@linkplain it.unimi.dsi.fastutil.Hash.Strategy hashing strategy}
040
041        * is specified at creation time.
042
043        *
044
045        * <P>Instances of this class use a hash table to represent a set. The table is
046
047        * filled up to a specified <em>load factor</em>, and then doubled in size to
048
049        * accommodate new entries. If the table is emptied below <em>one fourth</em>
050
051        * of the load factor, it is halved in size. However, halving is
052
053        * not performed when deleting entries from an iterator, as it would interfere
054
055        * with the iteration process.
056
057        *
058
059        * <p>Note that {@link #clear()} does not modify the hash table size.
060
061        * Rather, a family of {@linkplain #trim() trimming
062
063        * methods} lets you control the size of the table; this is particularly useful
064
065        * if you reuse instances of this class.
066
067        *
068
069        * @see Hash
070
071        * @see HashCommon
072
073        */

333        private void tryCapacity(final long capacity) {
334         final int needed = (int)Math.min(1 << 30, Math.max(2, HashCommon.nextPowerOfTwo((long)Math.ceil(capacity / f))));

350        @Override
351        public boolean add(final double k) {
352         int pos;
353         if (( strategy.equals( (k), (0) ) )) {
354          if (containsNull) return false;
355          containsNull = true;
356          key[n] = k;
357         }
358         else {
359          double curr;
360          final double[] key = this.key;
361          // The starting point.
362          if (! ( Double.doubleToLongBits(curr = key[pos = ( it.unimi.dsi.fastutil.HashCommon.mix( strategy.hashCode(k) ) ) & mask]) == 0 )) {
363           if (( strategy.equals( (curr), (k) ) )) return false;
364           while(! ( Double.doubleToLongBits(curr = key[pos = (pos + 1) & mask]) == 0 ))
365            if (( strategy.equals( (curr), (k) ) )) return false;
366          }
367          key[pos] = k;
368         }
369         if (size++ >= maxFill) rehash(arraySize(size + 1, f));
370         if (ASSERTS) checkTable();
371         return true;
372        }
373        /** Shifts left entries with the specified hash code, starting at the specified position,
374
375         * and empties the resulting free entry.
376
377         *
378
379         * @param pos a starting position.
380
381         */
382        protected final void shiftKeys(int pos) {
383         // Shift entries with the same hash.
384         int last, slot;
385         double curr;
386         final double[] key = this.key;
387         for(;;) {
388          pos = ((last = pos) + 1) & mask;
389          for(;;) {
390           if (( Double.doubleToLongBits(curr = key[pos]) == 0 )) {
391            key[last] = (0);
392            return;
393           }
394           slot = ( it.unimi.dsi.fastutil.HashCommon.mix( strategy.hashCode(curr) ) ) & mask;
395           if (last <= pos ? last >= slot || slot > pos : last >= slot && slot > pos) break;
396           pos = (pos + 1) & mask;
397          }
398          key[last] = curr;
399         }
400        }
401        private boolean removeEntry(final int pos) {
402         size--;
403         shiftKeys(pos);
404         if (size < maxFill / 4 && n > DEFAULT_INITIAL_SIZE) rehash(n / 2);
405         return true;
406        }
407        private boolean removeNullEntry() {
408         containsNull = false;
409         key[n] = (0);
410         size--;
411         if (size < maxFill / 4 && n > DEFAULT_INITIAL_SIZE) rehash(n / 2);
412         return true;
413        }

415        @Override
416        public boolean remove(final double k) {
417         if (( strategy.equals( ( k), (0) ) )) {
418          if (containsNull) return removeNullEntry();
419          return false;
420         }
421         double curr;
422         final double[] key = this.key;
423         int pos;
424         // The starting point.
425         if (( Double.doubleToLongBits(curr = key[pos = ( it.unimi.dsi.fastutil.HashCommon.mix( strategy.hashCode(k) ) ) & mask]) == 0 )) return false;
426         if (( strategy.equals( (k), (curr) ) )) return removeEntry(pos);
427         while(true) {
428          if (( Double.doubleToLongBits(curr = key[pos = (pos + 1) & mask]) == 0 )) return false;
429          if (( strategy.equals( (k), (curr) ) )) return removeEntry(pos);
430         }
431        }

433        @Override
434        public boolean contains(final double k) {
435         if (( strategy.equals( ( k), (0) ) )) return containsNull;
436         double curr;
437         final double[] key = this.key;
438         int pos;
439         // The starting point.
440         if (( Double.doubleToLongBits(curr = key[pos = ( it.unimi.dsi.fastutil.HashCommon.mix( strategy.hashCode(k) ) ) & mask]) == 0 )) return false;
441         if (( strategy.equals( (k), (curr) ) )) return true;
442         while(true) {
443          if (( Double.doubleToLongBits(curr = key[pos = (pos + 1) & mask]) == 0 )) return false;
444          if (( strategy.equals( (k), (curr) ) )) return true;
445         }
446        }

499        /** An iterator over a hash set. */
500        private class SetIterator extends AbstractDoubleIterator {
501         /** The index of the last entry returned, if positive or zero; initially, {@link #n}. If negative, the last
502
503                        element returned was that of index {@code - pos - 1} from the {@link #wrapped} list. */
504         int pos = n;
505         /** The index of the last entry that has been returned (more precisely, the value of {@link #pos} if {@link #pos} is positive,
506
507                        or {@link Integer#MIN_VALUE} if {@link #pos} is negative). It is -1 if either
508
509                        we did not return an entry yet, or the last returned entry has been removed. */
510         int last = -1;
511         /** A downward counter measuring how many entries must still be returned. */
512         int c = size;
513         /** A boolean telling us whether we should return the null key. */
514         boolean mustReturnNull = DoubleOpenCustomHashSet.this.containsNull;
515         /** A lazily allocated list containing elements that have wrapped around the table because of removals. */
516         DoubleArrayList wrapped;
517         public boolean hasNext() {
518          return c != 0;
519         }
520         public double nextDouble() {
521          if (! hasNext()) throw new NoSuchElementException();
522          c--;
523          if (mustReturnNull) {
524           mustReturnNull = false;
525           last = n;
526           return key[n];
527          }
528          final double key[] = DoubleOpenCustomHashSet.this.key;
529          for(;;) {
530           if (--pos < 0) {
531            // We are just enumerating elements from the wrapped list.
532            last = Integer.MIN_VALUE;
533            return wrapped.getDouble(- pos - 1);
534           }
535           if (! ( Double.doubleToLongBits(key[pos]) == 0 )) return key[last = pos];
536          }
537         }
538         /** Shifts left entries with the specified hash code, starting at the specified position,
539
540                 * and empties the resulting free entry.
541
542                 *
543
544                 * @param pos a starting position.
545
546                 */
547         private final void shiftKeys(int pos) {
548          // Shift entries with the same hash.
549          int last, slot;
550          double curr;
551          final double[] key = DoubleOpenCustomHashSet.this.key;
552          for(;;) {
553           pos = ((last = pos) + 1) & mask;
554           for(;;) {
555            if (( Double.doubleToLongBits(curr = key[pos]) == 0 )) {
556             key[last] = (0);
557             return;
558            }
559            slot = ( it.unimi.dsi.fastutil.HashCommon.mix( strategy.hashCode(curr) ) ) & mask;
560            if (last <= pos ? last >= slot || slot > pos : last >= slot && slot > pos) break;
561            pos = (pos + 1) & mask;
562           }
563           if (pos < last) { // Wrapped entry.
564            if (wrapped == null) wrapped = new DoubleArrayList (2);
565            wrapped.add(key[pos]);
566           }
567           key[last] = curr;
568          }
569         }
570         public void remove() {
571          if (last == -1) throw new IllegalStateException();
572          if (last == n) {
573           DoubleOpenCustomHashSet.this.containsNull = false;
574           DoubleOpenCustomHashSet.this.key[n] = (0);
575          }
576          else if (pos >= 0) shiftKeys(last);
577          else {
578           // We're removing wrapped entries.
579           DoubleOpenCustomHashSet.this.remove(wrapped.getDouble(- pos - 1));
580           last = -1; // Note that we must not decrement size
581           return;
582          }
583          size--;
584          last = -1; // You can no longer remove this entry.
585          if (ASSERTS) checkTable();
586         }
587        }

615        /** Rehashes this set, making the table as small as possible.
616
617         *
618
619         * <P>This method rehashes the table to the smallest size satisfying the
620
621         * load factor. It can be used when the set will not be changed anymore, so
622
623         * to optimize access speed and size.
624
625         *
626
627         * <P>If the table size is already the minimum possible, this method
628
629         * does nothing.
630
631         *

633         * @return true if there was enough memory to trim the set.
634
635         * @see #trim(int)
636
637         */
638        public boolean trim() {
639         final int l = arraySize(size, f);
640         if (l >= n || size > maxFill(l, f)) return true;
641         try {
642          rehash(l);
643         }
644         catch(OutOfMemoryError cantDoIt) { return false; }
645         return true;
646        }
647        /** Rehashes this set if the table is too large.
648
649         *
650
651         * <P>Let <var>N</var> be the smallest table size that can hold
652
653         * <code>max(n,{@link #size()})</code> entries, still satisfying the load factor. If the current
654
655         * table size is smaller than or equal to <var>N</var>, this method does
656
657         * nothing. Otherwise, it rehashes this set in a table of size
658
659         * <var>N</var>.
660
661         *
662
663         * <P>This method is useful when reusing sets.  {@linkplain #clear() Clearing a
664
665         * set} leaves the table size untouched. If you are reusing a set
666
667         * many times, you can call this method with a typical
668
669         * size to avoid keeping around a very large table just
670
671         * because of a few large transient sets.
672
673         *
674
675         * @param n the threshold for the trimming.
676
677         * @return true if there was enough memory to trim the set.
678
679         * @see #trim()
680
681         */
682        public boolean trim(final int n) {
683         final int l = HashCommon.nextPowerOfTwo((int)Math.ceil(n / f));
684         if (l >= n || size > maxFill(l, f)) return true;
685         try {
686          rehash(l);
687         }
688         catch(OutOfMemoryError cantDoIt) { return false; }
689         return true;
690        }
691        /** Rehashes the set.
692
693         *
694
695         * <P>This method implements the basic rehashing strategy, and may be
696
697         * overriden by subclasses implementing different rehashing strategies (e.g.,
698
699         * disk-based rehashing). However, you should not override this method
700
701         * unless you understand the internal workings of this class.
702
703         *
704
705         * @param newN the new size
706
707         */
708
709        protected void rehash(final int newN) {
710         final double key[] = this.key;
711         final int mask = newN - 1; // Note that this is used by the hashing macro
712         final double newKey[] = new double[newN + 1];
713         int i = n, pos;
714         for(int j = realSize(); j-- != 0;) {
715          while(( Double.doubleToLongBits(key[--i]) == 0 ));
716          if (! ( Double.doubleToLongBits(newKey[pos = ( it.unimi.dsi.fastutil.HashCommon.mix( strategy.hashCode(key[i]) ) ) & mask]) == 0 ))
717           while (! ( Double.doubleToLongBits(newKey[pos = (pos + 1) & mask]) == 0 ));
718          newKey[pos] = key[i];
719         }
720         n = newN;
721         this.mask = mask;
722         maxFill = maxFill(n, f);
723         this.key = newKey;
724        }
725        /** Returns a deep copy of this set.
726
727         *
728

753        /** Returns a hash code for this set.
754
755         *
756
757         * This method overrides the generic method provided by the superclass.
758
759         * Since <code>equals()</code> is not overriden, it is important
760
761         * that the value returned by this method is the same value as
762
763         * the one returned by the overriden method.
764
765         *
766

770        @Override

772         int h = 0;
773         for(int j = realSize(), i = 0; j-- != 0;) {
774          while(( Double.doubleToLongBits(key[i]) == 0 )) i++;
775           h += ( strategy.hashCode(key[i]) );
776          i++;
777         }
778         // Zero / null have hash zero.
779         return h;
780        }

788         s.defaultReadObject();
789         n = arraySize(size, f);

791         mask = n - 1;
792         final double key[] = this.key = new double[n + 1];
793         double k;
