025/**  A type-specific hash set with with a fast, small-footprint implementation.
026        *
027        * <P>Instances of this class use a hash table to represent a set. The table is
028        * enlarged as needed by doubling its size when new entries are created, but it is <em>never</em> made
029        * smaller (even on a {@link #clear()}). A family of {@linkplain #trim() trimming
030        * methods} lets you control the size of the table; this is particularly useful
031        * if you reuse instances of this class.
032        *
033        * @see Hash
034        * @see HashCommon
035        */
036public class ObjectOpenHashSet <K> extends AbstractObjectSet <K> implements java.io.Serializable, Cloneable, Hash {
037        private static final long serialVersionUID = 0L;
038        private static final boolean ASSERTS = false;
039        /** The array of keys. */
040        protected transient K[] key;
041        /** The mask for wrapping a position counter. */
042        protected transient int mask;
043        /** Whether this set contains the null key. */
044        protected transient boolean containsNull;
045        /** The current table size. Note that an additional element is allocated for storing the null key. */
046        protected transient int n;
047        /** Threshold after which we rehash. It must be the table size times {@link #f}. */
048        protected transient int maxFill;
049        /** Number of entries in the set (including the null key, if present). */
050        protected int size;
051        /** The acceptable load factor. */
052        protected final float f;
053        /** Creates a new hash set.
054         *
055         * <p>The actual table size will be the least power of two greater than <code>expected</code>/<code>f</code>.
056         *
057         * @param expected the expected number of elements in the hash set.
058         * @param f the load factor.
059         */
060        @SuppressWarnings("unchecked")
061        public ObjectOpenHashSet(final int expected, final float f) {
062         if (f <= 0 || f > 1) throw new IllegalArgumentException("Load factor must be greater than 0 and smaller than or equal to 1");
063         if (expected < 0) throw new IllegalArgumentException("The expected number of elements must be nonnegative");
064         this.f = f;
065         n = arraySize(expected, f);
066         mask = n - 1;
067         maxFill = maxFill(n, f);
068         key = (K[]) new Object[n + 1];
069        }

170        private int realSize() {
171         return containsNull ? size - 1 : size;
172        }
173        private void ensureCapacity(final int capacity) {
174         final int needed = arraySize(capacity, f);
175         if (needed > n) rehash(needed);
176        }
177        private void tryCapacity(final long capacity) {
178         final int needed = (int)Math.min(1 << 30, Math.max(2, HashCommon.nextPowerOfTwo((long)Math.ceil(capacity / f))));
179         if (needed > n) rehash(needed);
180        }
181        @Override
182        public boolean addAll(Collection<? extends K> c) {
183         // The resulting collection will be at least c.size() big
184         if (f <= .5) ensureCapacity(c.size()); // The resulting collection will be sized for c.size() elements
185         else tryCapacity(size() + c.size()); // The resulting collection will be tentatively sized for size() + c.size() elements
186         return super.addAll(c);
187        }
188        @Override
189        public boolean add(final K k) {
190         int pos;
191         if (( (k) == null )) {
192          if (containsNull) return false;
193          containsNull = true;
194         }
195         else {
196          K curr;
197          final K[] key = this.key;
198          // The starting point.
199          if (! ( (curr = key[pos = ( it.unimi.dsi.fastutil.HashCommon.mix( (k).hashCode() ) ) & mask]) == null )) {
200           if (( (curr).equals(k) )) return false;
201           while(! ( (curr = key[pos = (pos + 1) & mask]) == null ))
202            if (( (curr).equals(k) )) return false;
203          }
204          key[pos] = k;
205         }
206         if (size++ >= maxFill) rehash(arraySize(size + 1, f));
207         if (ASSERTS) checkTable();
208         return true;
209        }
210        /** Add a random element if not present, get the existing value if already present.
211         *
212         * This is equivalent to (but faster than) doing a:
213         * <pre>
214         * K exist = set.get(k);
215         * if (exist == null) {
216         *   set.add(k);
217         *   exist = k;
218         * }
219         * </pre>
220         */
221        public K addOrGet(final K k) {
222         int pos;
223         if (( (k) == null )) {
224          if (containsNull) return key [n];
225          containsNull = true;
226         }
227         else {
228          K curr;
229          final K[] key = this.key;
230          // The starting point.
231          if (! ( (curr = key[pos = ( it.unimi.dsi.fastutil.HashCommon.mix( (k).hashCode() ) ) & mask]) == null )) {
232           if (( (curr).equals(k) )) return curr;
233           while(! ( (curr = key[pos = (pos + 1) & mask]) == null ))
234            if (( (curr).equals(k) )) return curr;
235          }
236          key[pos] = k;
237         }
238         if (size++ >= maxFill) rehash(arraySize(size + 1, f));
239         if (ASSERTS) checkTable();
240         return k;
241        }
242        /** Shifts left entries with the specified hash code, starting at the specified position,
243         * and empties the resulting free entry.
244         *
245         * @param pos a starting position.
246         */
247        protected final void shiftKeys(int pos) {
248         // Shift entries with the same hash.
249         int last, slot;
250         K curr;
251         final K[] key = this.key;
252         for(;;) {
253          pos = ((last = pos) + 1) & mask;
254          for(;;) {
255           if (( (curr = key[pos]) == null )) {
256            key[last] = (null);
257            return;
258           }
259           slot = ( it.unimi.dsi.fastutil.HashCommon.mix( (curr).hashCode() ) ) & mask;
260           if (last <= pos ? last >= slot || slot > pos : last >= slot && slot > pos) break;
261           pos = (pos + 1) & mask;
262          }
263          key[last] = curr;
264         }
265        }
266        private boolean removeEntry(final int pos) {
267         size--;
268         shiftKeys(pos);
269         if (size < maxFill / 4 && n > DEFAULT_INITIAL_SIZE) rehash(n / 2);
270         return true;
271        }
272        private boolean removeNullEntry() {
273         containsNull = false;
274         key[n] = (null);
275         size--;
276         if (size < maxFill / 4 && n > DEFAULT_INITIAL_SIZE) rehash(n / 2);
277         return true;
278        }
279        @SuppressWarnings("unchecked")
280        @Override
281        public boolean remove(final Object k) {
282         if (( ((K) k) == null )) {
283          if (containsNull) return removeNullEntry();
284          return false;
285         }
286         K curr;
287         final K[] key = this.key;
288         int pos;
289         // The starting point.
290         if (( (curr = key[pos = ( it.unimi.dsi.fastutil.HashCommon.mix( (k).hashCode() ) ) & mask]) == null )) return false;
291         if (( (k).equals(curr) )) return removeEntry(pos);
292         while(true) {
293          if (( (curr = key[pos = (pos + 1) & mask]) == null )) return false;
294          if (( (k).equals(curr) )) return removeEntry(pos);
295         }
296        }
297        @SuppressWarnings("unchecked")
298        @Override
299        public boolean contains(final Object k) {
300         if (( ((K) k) == null )) return containsNull;
301         K curr;
302         final K[] key = this.key;
303         int pos;
304         // The starting point.
305         if (( (curr = key[pos = ( it.unimi.dsi.fastutil.HashCommon.mix( (k).hashCode() ) ) & mask]) == null )) return false;
306         if (( (k).equals(curr) )) return true;
307         while(true) {
308          if (( (curr = key[pos = (pos + 1) & mask]) == null )) return false;
309          if (( (k).equals(curr) )) return true;
310         }
311        }
312        /** Returns the element of this set that is equal to the given key, or {@code null}.
313         * @return the element of this set that is equal to the given key, or {@code null}.
314         */
315        @SuppressWarnings("unchecked")
316        public K get(final Object k) {
317         if (( ((K) k) == null )) return key[n]; // This is correct independently of the value of containsNull and of the map being custom
318         K curr;
319         final K[] key = this.key;
320         int pos;
321         // The starting point.
322         if (( (curr = key[pos = ( it.unimi.dsi.fastutil.HashCommon.mix( (k).hashCode() ) ) & mask]) == null )) return null;
323         if (( (k).equals(curr) )) return curr;
324         // There's always an unused entry.
325         while(true) {
326          if (( (curr = key[pos = (pos + 1) & mask]) == null )) return null;
327          if (( (k).equals(curr) )) return curr;
328         }
329        }
330        /* Removes all elements from this set.
331         *
332         * <P>To increase object reuse, this method does not change the table size.

368        /** An iterator over a hash set. */
369        private class SetIterator extends AbstractObjectIterator <K> {
370         /** The index of the last entry returned, if positive or zero; initially, {@link #n}. If negative, the last
371                        element returned was that of index {@code - pos - 1} from the {@link #wrapped} list. */
372         int pos = n;
373         /** The index of the last entry that has been returned (more precisely, the value of {@link #pos} if {@link #pos} is positive,
374                        or {@link Integer#MIN_VALUE} if {@link #pos} is negative). It is -1 if either
375                        we did not return an entry yet, or the last returned entry has been removed. */
376         int last = -1;
377         /** A downward counter measuring how many entries must still be returned. */
378         int c = size;
379         /** A boolean telling us whether we should return the null key. */
380         boolean mustReturnNull = ObjectOpenHashSet.this.containsNull;
381         /** A lazily allocated list containing elements that have wrapped around the table because of removals. */
382         ObjectArrayList <K> wrapped;
383         public boolean hasNext() {
384          return c != 0;
385         }
386         public K next() {
387          if (! hasNext()) throw new NoSuchElementException();
388          c--;
389          if (mustReturnNull) {
390           mustReturnNull = false;
391           last = n;
392           return key[n];
393          }
394          final K key[] = ObjectOpenHashSet.this.key;
395          for(;;) {
396           if (--pos < 0) {
397            // We are just enumerating elements from the wrapped list.
398            last = Integer.MIN_VALUE;
399            return wrapped.get(- pos - 1);
400           }
401           if (! ( (key[pos]) == null )) return key[last = pos];
402          }
403         }
404         /** Shifts left entries with the specified hash code, starting at the specified position,
405                 * and empties the resulting free entry.
406                 *
407                 * @param pos a starting position.
408                 */
409         private final void shiftKeys(int pos) {
410          // Shift entries with the same hash.
411          int last, slot;
412          K curr;
413          final K[] key = ObjectOpenHashSet.this.key;
414          for(;;) {
415           pos = ((last = pos) + 1) & mask;
416           for(;;) {
417            if (( (curr = key[pos]) == null )) {
418             key[last] = (null);
419             return;
420            }
421            slot = ( it.unimi.dsi.fastutil.HashCommon.mix( (curr).hashCode() ) ) & mask;
422            if (last <= pos ? last >= slot || slot > pos : last >= slot && slot > pos) break;
423            pos = (pos + 1) & mask;
424           }
425           if (pos < last) { // Wrapped entry.
426            if (wrapped == null) wrapped = new ObjectArrayList <K>(2);
427            wrapped.add(key[pos]);
428           }
429           key[last] = curr;
430          }
431         }
432         public void remove() {
433          if (last == -1) throw new IllegalStateException();
434          if (last == n) {
435           ObjectOpenHashSet.this.containsNull = false;
436           ObjectOpenHashSet.this.key[n] = (null);
437          }
438          else if (pos >= 0) shiftKeys(last);
439          else {
440           // We're removing wrapped entries.
441           ObjectOpenHashSet.this.remove(wrapped.set(- pos - 1, null));
442           last = -1; // Note that we must not decrement size
443           return;
444          }
445          size--;
446          last = -1; // You can no longer remove this entry.
447          if (ASSERTS) checkTable();
448         }
449        }
450        @Override
451        public ObjectIterator <K> iterator() {

454        /** A no-op for backward compatibility. The kind of tables implemented by
455         * this class never need rehashing.
456         *
457         * <P>If you need to reduce the table size to fit exactly
458         * this set, use {@link #trim()}.
459         *
460         * @return true.
461         * @see #trim()
462         * @deprecated A no-op.
463         */

468        /** Rehashes this set, making the table as small as possible.
469         *
470         * <P>This method rehashes the table to the smallest size satisfying the
471         * load factor. It can be used when the set will not be changed anymore, so
472         * to optimize access speed and size.
473         *
474         * <P>If the table size is already the minimum possible, this method
475         * does nothing.
476         *
477         * @return true if there was enough memory to trim the set.
478         * @see #trim(int)
479         */
480        public boolean trim() {
481         final int l = arraySize(size, f);
482         if (l >= n || size > maxFill(l, f)) return true;
483         try {
484          rehash(l);
485         }
486         catch(OutOfMemoryError cantDoIt) { return false; }
487         return true;
488        }
489        /** Rehashes this set if the table is too large.
490         *
491         * <P>Let <var>N</var> be the smallest table size that can hold
492         * <code>max(n,{@link #size()})</code> entries, still satisfying the load factor. If the current
493         * table size is smaller than or equal to <var>N</var>, this method does
494         * nothing. Otherwise, it rehashes this set in a table of size
495         * <var>N</var>.
496         *
497         * <P>This method is useful when reusing sets.  {@linkplain #clear() Clearing a
498         * set} leaves the table size untouched. If you are reusing a set
499         * many times, you can call this method with a typical
500         * size to avoid keeping around a very large table just
501         * because of a few large transient sets.
502         *
503         * @param n the threshold for the trimming.
504         * @return true if there was enough memory to trim the set.
505         * @see #trim()
506         */
507        public boolean trim(final int n) {
508         final int l = HashCommon.nextPowerOfTwo((int)Math.ceil(n / f));
509         if (l >= n || size > maxFill(l, f)) return true;
510         try {
511          rehash(l);
512         }
513         catch(OutOfMemoryError cantDoIt) { return false; }
514         return true;
515        }
516        /** Rehashes the set.
517         *
518         * <P>This method implements the basic rehashing strategy, and may be
519         * overriden by subclasses implementing different rehashing strategies (e.g.,
520         * disk-based rehashing). However, you should not override this method
521         * unless you understand the internal workings of this class.
522         *
523         * @param newN the new size
524         */
525        @SuppressWarnings("unchecked")
526        protected void rehash(final int newN) {
527         final K key[] = this.key;
528         final int mask = newN - 1; // Note that this is used by the hashing macro
529         final K newKey[] = (K[]) new Object[newN + 1];
530         int i = n, pos;
531         for(int j = realSize(); j-- != 0;) {
532          while(( (key[--i]) == null ));
533          if (! ( (newKey[pos = ( it.unimi.dsi.fastutil.HashCommon.mix( (key[i]).hashCode() ) ) & mask]) == null ))
534           while (! ( (newKey[pos = (pos + 1) & mask]) == null ));
535          newKey[pos] = key[i];
536         }
537         n = newN;
538         this.mask = mask;
539         maxFill = maxFill(n, f);
540         this.key = newKey;
541        }
542        /** Returns a deep copy of this set.
543         *
544         * <P>This method performs a deep copy of this hash set; the data stored in the
545         * set, however, is not cloned. Note that this makes a difference only for object keys.
546         *
547         *  @return a deep copy of this set.
548         */
549        @Override
550        @SuppressWarnings("unchecked")
551        public ObjectOpenHashSet <K> clone() {
552         ObjectOpenHashSet <K> c;
553         try {
554          c = (ObjectOpenHashSet <K>)super.clone();
555         }
556         catch(CloneNotSupportedException cantHappen) {
557          throw new InternalError();
558         }
559         c.key = key.clone();
560         c.containsNull = containsNull;
561         return c;
562        }
563        /** Returns a hash code for this set.
564         *
565         * This method overrides the generic method provided by the superclass.
566         * Since <code>equals()</code> is not overriden, it is important
567         * that the value returned by this method is the same value as
568         * the one returned by the overriden method.
569         *
570         * @return a hash code for this set.
571         */
572        @Override
573        public int hashCode() {
574         int h = 0;
575         for(int j = realSize(), i = 0; j-- != 0;) {
576          while(( (key[i]) == null )) i++;
577          if (this != key[i])
578           h += ( (key[i]).hashCode() );
579          i++;
580         }
581         // Zero / null have hash zero.
582         return h;
583        }
584        private void writeObject(java.io.ObjectOutputStream s) throws java.io.IOException {
585         final ObjectIterator <K> i = iterator();
586         s.defaultWriteObject();
587         for(int j = size; j-- != 0;) s.writeObject(i.next());
588        }
589        @SuppressWarnings("unchecked")
590        private void readObject(java.io.ObjectInputStream s) throws java.io.IOException, ClassNotFoundException {
591         s.defaultReadObject();
592         n = arraySize(size, f);
593         maxFill = maxFill(n, f);
594         mask = n - 1;
595         final K key[] = this.key = (K[]) new Object[n + 1];
596         K k;
597         for(int i = size, pos; i-- != 0;) {
598          k = (K) s.readObject();
599          if (( (k) == null )) {
600           pos = n;
601           containsNull = true;
602          }
603          else {
604           if (! ( (key[pos = ( it.unimi.dsi.fastutil.HashCommon.mix( (k).hashCode() ) ) & mask]) == null ))
605            while (! ( (key[pos = (pos + 1) & mask]) == null ));
606          }
607          key[pos] = k;
608         }
609         if (ASSERTS) checkTable();
610        }
611        private void checkTable() {}
