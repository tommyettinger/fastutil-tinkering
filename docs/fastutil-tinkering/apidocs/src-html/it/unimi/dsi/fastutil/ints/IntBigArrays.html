041/** A class providing static methods and objects that do useful things with {@linkplain BigArrays big arrays}.
042        *
043        * <p>In particular, the <code>ensureCapacity()</code>, <code>grow()</code>,
044        * <code>trim()</code> and <code>setLength()</code> methods allow to handle
045        * big arrays much like array lists.
046        *
047        * <P>Note that {@link it.unimi.dsi.fastutil.io.BinIO} and {@link it.unimi.dsi.fastutil.io.TextIO}
048        * contain several methods that make it possible to load and save big arrays of primitive types as sequences
049        * of elements in {@link java.io.DataInput} format (i.e., not as objects) or as sequences of lines of text.
050        *
051        * @see BigArrays
052        */

054        private IntBigArrays() {}
055        /** A static, final, empty big array. */

057        /** Returns the element of the given big array of specified index.

065        }

129        /** Copies a big array from the specified source big array, beginning at the specified position, to the specified position of the destination big array.
130         * Handles correctly overlapping regions of the same big array.
131         *
132         * @param srcArray the source big array.
133         * @param srcPos the starting position in the source big array.
134         * @param destArray the destination big array.
135         * @param destPos the starting position in the destination data.
136         * @param length the number of elements to be copied.
137         */
138        public static void copy(final int[][] srcArray, final long srcPos, final int[][] destArray, final long destPos, long length) {
139         if (destPos <= srcPos) {
140          int srcSegment = segment(srcPos);
141          int destSegment = segment(destPos);
142          int srcDispl = displacement(srcPos);
143          int destDispl = displacement(destPos);
144          int l;
145          while(length > 0) {
146           l = (int)Math.min(length, Math.min(srcArray[srcSegment].length - srcDispl, destArray[destSegment].length - destDispl));
147           System.arraycopy(srcArray[srcSegment], srcDispl, destArray[destSegment], destDispl, l);
148           if ((srcDispl += l) == SEGMENT_SIZE) {
149            srcDispl = 0;
150            srcSegment++;
151           }
152           if ((destDispl += l) == SEGMENT_SIZE) {
153            destDispl = 0;
154            destSegment++;
155           }
156           length -= l;
157          }
158         }
159         else {
160          int srcSegment = segment(srcPos + length);
161          int destSegment = segment(destPos + length);
162          int srcDispl = displacement(srcPos + length);
163          int destDispl = displacement(destPos + length);
164          int l;
165          while(length > 0) {
166           if (srcDispl == 0) {
167            srcDispl = SEGMENT_SIZE;
168            srcSegment--;
169           }
170           if (destDispl == 0) {
171            destDispl = SEGMENT_SIZE;
172            destSegment--;
173           }
174           l = (int)Math.min(length, Math.min(srcDispl, destDispl));
175           System.arraycopy(srcArray[srcSegment], srcDispl - l, destArray[destSegment], destDispl - l, l);
176           srcDispl -= l;
177           destDispl -= l;
178           length -= l;
179          }
180         }
181        }
182        /** Copies a big array from the specified source big array, beginning at the specified position, to the specified position of the destination array.
183         *
184         * @param srcArray the source big array.
185         * @param srcPos the starting position in the source big array.
186         * @param destArray the destination array.
187         * @param destPos the starting position in the destination data.
188         * @param length the number of elements to be copied.
189         */
190        public static void copyFromBig(final int[][] srcArray, final long srcPos, final int[] destArray, int destPos, int length) {
191         int srcSegment = segment(srcPos);
192         int srcDispl = displacement(srcPos);
193         int l;
194         while(length > 0) {
195          l = Math.min(srcArray[srcSegment].length - srcDispl, length);
196          System.arraycopy(srcArray[srcSegment], srcDispl, destArray, destPos, l);
197          if ((srcDispl += l) == SEGMENT_SIZE) {
198           srcDispl = 0;
199           srcSegment++;
200          }
201          destPos += l;
202          length -= l;
203         }
204        }
205        /** Copies an array from the specified source array, beginning at the specified position, to the specified position of the destination big array.
206         *
207         * @param srcArray the source array.
208         * @param srcPos the starting position in the source array.
209         * @param destArray the destination big array.
210         * @param destPos the starting position in the destination data.
211         * @param length the number of elements to be copied.
212         */
213        public static void copyToBig(final int[] srcArray, int srcPos, final int[][] destArray, final long destPos, long length) {
214         int destSegment = segment(destPos);
215         int destDispl = displacement(destPos);
216         int l;
217         while(length > 0) {
218          l = (int)Math.min(destArray[destSegment].length - destDispl, length);
219          System.arraycopy(srcArray, srcPos, destArray[destSegment], destDispl, l);
220          if ((destDispl += l) == SEGMENT_SIZE) {
221           destDispl = 0;
222           destSegment++;
223          }
224          srcPos += l;
225          length -= l;
226         }
227        }
228        /** Creates a new big array.
229         *
230         * @param length the length of the new big array.
231         * @return a new big array of given length.
232         */
233        public static int[][] newBigArray(final long length) {
234         if (length == 0) return EMPTY_BIG_ARRAY;
235         ensureLength(length);
236         final int baseLength = (int)((length + SEGMENT_MASK) >>> SEGMENT_SHIFT);
237         int[][] base = new int[baseLength][];
238         final int residual = (int)(length & SEGMENT_MASK);
239         if (residual != 0) {
240          for(int i = 0; i < baseLength - 1; i++) base[i] = new int[SEGMENT_SIZE];
241          base[baseLength - 1] = new int[residual];
242         }
243         else for(int i = 0; i < baseLength; i++) base[i] = new int[SEGMENT_SIZE];
244         return base;
245        }
246        /** Turns a standard array into a big array.
247         *
248         * <P>Note that the returned big array might contain as a segment the original array.
249         *
250         * @param array an array.
251         * @return a new big array with the same length and content of <code>array</code>.
252         */
253        public static int[][] wrap(final int[] array) {
254         if (array.length == 0) return EMPTY_BIG_ARRAY;
255         if (array.length <= SEGMENT_SIZE) return new int[][] { array };
256         final int[][] bigArray = newBigArray(array.length);
257         for(int i = 0; i < bigArray.length; i++) System.arraycopy(array, (int)start(i), bigArray[i], 0, bigArray[i].length);
258         return bigArray;
259        }
260        /** Ensures that a big array can contain the given number of entries.
261         *
262         * <P>If you cannot foresee whether this big array will need again to be
263         * enlarged, you should probably use <code>grow()</code> instead.
264         *
265         * <p><strong>Warning:</strong> the returned array might use part of the segments of the original
266         * array, which must be considered read-only after calling this method.
267         *
268         * @param array a big array.
269         * @param length the new minimum length for this big array.
270         * @return <code>array</code>, if it contains <code>length</code> entries or more; otherwise,
271         * a big array with <code>length</code> entries whose first <code>length(array)</code>
272         * entries are the same as those of <code>array</code>.
273         */
274        public static int[][] ensureCapacity(final int[][] array, final long length) {
275         return ensureCapacity(array, length, length(array));
276        }
277        /** Ensures that a big array can contain the given number of entries, preserving just a part of the big array.
278         *
279         * <p><strong>Warning:</strong> the returned array might use part of the segments of the original
280         * array, which must be considered read-only after calling this method.
281         *
282         * @param array a big array.
283         * @param length the new minimum length for this big array.
284         * @param preserve the number of elements of the big array that must be preserved in case a new allocation is necessary.
285         * @return <code>array</code>, if it can contain <code>length</code> entries or more; otherwise,
286         * a big array with <code>length</code> entries whose first <code>preserve</code>
287         * entries are the same as those of <code>array</code>.
288         */
289        public static int[][] ensureCapacity(final int[][] array, final long length, final long preserve) {
290         final long oldLength = length(array);
291         if (length > oldLength) {
292          ensureLength(length);
293          final int valid = array.length - (array.length == 0 || array.length > 0 && array[array.length - 1].length == SEGMENT_SIZE ? 0 : 1);
294          final int baseLength = (int)((length + SEGMENT_MASK) >>> SEGMENT_SHIFT);
295          final int[][] base = Arrays.copyOf(array, baseLength);
296          final int residual = (int)(length & SEGMENT_MASK);
297          if (residual != 0) {
298           for(int i = valid; i < baseLength - 1; i++) base[i] = new int[SEGMENT_SIZE];
299           base[baseLength - 1] = new int[residual];
300          }
301          else for(int i = valid; i < baseLength; i++) base[i] = new int[SEGMENT_SIZE];
302          if (preserve - (valid * (long)SEGMENT_SIZE) > 0) copy(array, valid * (long)SEGMENT_SIZE, base, valid * (long)SEGMENT_SIZE, preserve - (valid * (long)SEGMENT_SIZE));
303          return base;
304         }
305         return array;
306        }
307        /** Grows the given big array to the maximum between the given length and
308         * the current length multiplied by two, provided that the given
309         * length is larger than the current length.
310         *
311         * <P>If you want complete control on the big array growth, you
312         * should probably use <code>ensureCapacity()</code> instead.
313         *
314         * <p><strong>Warning:</strong> the returned array might use part of the segments of the original
315         * array, which must be considered read-only after calling this method.
316         *
317         * @param array a big array.
318         * @param length the new minimum length for this big array.
319         * @return <code>array</code>, if it can contain <code>length</code>
320         * entries; otherwise, a big array with
321         * max(<code>length</code>,<code>length(array)</code>/&phi;) entries whose first
322         * <code>length(array)</code> entries are the same as those of <code>array</code>.
323         * */
324        public static int[][] grow(final int[][] array, final long length) {
325         final long oldLength = length(array);
326         return length > oldLength ? grow(array, length, oldLength) : array;
327        }
328        /** Grows the given big array to the maximum between the given length and
329         * the current length multiplied by two, provided that the given
330         * length is larger than the current length, preserving just a part of the big array.
331         *
332         * <P>If you want complete control on the big array growth, you
333         * should probably use <code>ensureCapacity()</code> instead.
334         *
335         * <p><strong>Warning:</strong> the returned array might use part of the segments of the original
336         * array, which must be considered read-only after calling this method.
337         *
338         * @param array a big array.
339         * @param length the new minimum length for this big array.
340         * @param preserve the number of elements of the big array that must be preserved in case a new allocation is necessary.
341         * @return <code>array</code>, if it can contain <code>length</code>
342         * entries; otherwise, a big array with
343         * max(<code>length</code>,<code>length(array)</code>/&phi;) entries whose first
344         * <code>preserve</code> entries are the same as those of <code>array</code>.
345         * */
346        public static int[][] grow(final int[][] array, final long length, final long preserve) {
347         final long oldLength = length(array);
348         return length > oldLength ? ensureCapacity(array, Math.max(2 * oldLength, length), preserve) : array;
349        }
350        /** Trims the given big array to the given length.
351         *
352         * <p><strong>Warning:</strong> the returned array might use part of the segments of the original
353         * array, which must be considered read-only after calling this method.
354         *
355         * @param array a big array.
356         * @param length the new maximum length for the big array.
357         * @return <code>array</code>, if it contains <code>length</code>
358         * entries or less; otherwise, a big array with
359         * <code>length</code> entries whose entries are the same as
360         * the first <code>length</code> entries of <code>array</code>.
361         *
362         */
363        public static int[][] trim(final int[][] array, final long length) {
364         ensureLength(length);
365         final long oldLength = length(array);
366         if (length >= oldLength) return array;
367         final int baseLength = (int)((length + SEGMENT_MASK) >>> SEGMENT_SHIFT);
368         final int[][] base = Arrays.copyOf(array, baseLength);
369         final int residual = (int)(length & SEGMENT_MASK);
370         if (residual != 0) base[baseLength - 1] = IntArrays.trim(base[baseLength - 1], residual);
371         return base;
372        }
373        /** Sets the length of the given big array.
374         *
375         * <p><strong>Warning:</strong> the returned array might use part of the segments of the original
376         * array, which must be considered read-only after calling this method.
377         *
378         * @param array a big array.
379         * @param length the new length for the big array.
380         * @return <code>array</code>, if it contains exactly <code>length</code>
381         * entries; otherwise, if it contains <em>more</em> than
382         * <code>length</code> entries, a big array with <code>length</code> entries
383         * whose entries are the same as the first <code>length</code> entries of
384         * <code>array</code>; otherwise, a big array with <code>length</code> entries
385         * whose first <code>length(array)</code> entries are the same as those of
386         * <code>array</code>.
387         *
388         */
389        public static int[][] setLength(final int[][] array, final long length) {
390         final long oldLength = length(array);
391         if (length == oldLength) return array;
392         if (length < oldLength) return trim(array, length);
393         return ensureCapacity(array, length);
394        }

419        /** Fills the given big array with the given value.
420         *
421         * <P>This method uses a backward loop. It is significantly faster than the corresponding
422         * method in {@link java.util.Arrays}.
423         *
424         * @param array a big array.
425         * @param value the new value for all elements of the big array.
426         */
427        public static void fill(final int[][] array, final int value) {
428         for(int i = array.length; i-- != 0;) Arrays.fill(array[i], value);
429        }
430        /** Fills a portion of the given big array with the given value.
431         *
432         * <P>If possible (i.e., <code>from</code> is 0) this method uses a
433         * backward loop. In this case, it is significantly faster than the
434         * corresponding method in {@link java.util.Arrays}.
435         *
436         * @param array a big array.
437         * @param from the starting index of the portion to fill.
438         * @param to the end index of the portion to fill.
439         * @param value the new value for all elements of the specified portion of the big array.
440         */
441        public static void fill(final int[][] array, final long from, long to, final int value) {
442         final long length = length(array);
443         BigArrays.ensureFromTo(length, from, to);
444         int fromSegment = segment(from);
445         int toSegment = segment(to);
446         int fromDispl = displacement(from);
447         int toDispl = displacement(to);
448         if (fromSegment == toSegment) {
449          Arrays.fill(array[fromSegment], fromDispl, toDispl, value);
450          return;
451         }
452         if (toDispl != 0) Arrays.fill(array[toSegment], 0, toDispl, value);
453         while(--toSegment > fromSegment) Arrays.fill(array[toSegment], value);
454         Arrays.fill(array[fromSegment], fromDispl, SEGMENT_SIZE, value);
455        }
456        /** Returns true if the two big arrays are elementwise equal.
457         *
458         * <P>This method uses a backward loop. It is significantly faster than the corresponding
459         * method in {@link java.util.Arrays}.
460         *
461         * @param a1 a big array.
462         * @param a2 another big array.
463         * @return true if the two big arrays are of the same length, and their elements are equal.
464         */
465        public static boolean equals(final int[][] a1, final int a2[][]) {
466         if (length(a1) != length(a2)) return false;
467         int i = a1.length, j;
468         int[] t, u;
469         while(i-- != 0) {
470          t = a1[i];
471          u = a2[i];
472          j = t.length;
473          while(j-- != 0) if (! ( (t[j]) == (u[j]) )) return false;
474         }
475         return true;
476        }
477        /* Returns a string representation of the contents of the specified big array.
478         *
479         * The string representation consists of a list of the big array's elements, enclosed in square brackets ("[]"). Adjacent elements are separated by the characters ", " (a comma followed by a space). Returns "null" if <code>a</code> is null.
480         * @param a the big array whose string representation to return.
481         * @return the string representation of <code>a</code>.
482         */
483        public static String toString(final int[][] a) {
484         if (a == null) return "null";
485         final long last = length(a) - 1;
486         if (last == - 1) return "[]";
487         final StringBuilder b = new StringBuilder();
488         b.append('[');
489         for (long i = 0; ; i++) {
490          b.append(String.valueOf(get(a, i)));
491          if (i == last) return b.append(']').toString();
492          b.append(", ");
493         }
494        }

559        /** Sorts the specified range of elements according to the order induced by the specified
560         * comparator using quicksort.
561         *
562         * <p>The sorting algorithm is a tuned quicksort adapted from Jon L. Bentley and M. Douglas
563         * McIlroy, &ldquo;Engineering a Sort Function&rdquo;, <i>Software: Practice and Experience</i>, 23(11), pages
564         * 1249&minus;1265, 1993.
565         *
566         * @param x the big array to be sorted.
567         * @param from the index of the first element (inclusive) to be sorted.
568         * @param to the index of the last element (exclusive) to be sorted.
569         * @param comp the comparator to determine the sorting order.
570         */
571        public static void quickSort(final int[][] x, final long from, final long to, final IntComparator comp) {
572         final long len = to - from;
573         // Selection sort on smallest arrays
574         if (len < SMALL) {
575          selectionSort(x, from, to, comp);
576          return;
577         }
578         // Choose a partition element, v
579         long m = from + len / 2; // Small arrays, middle element
580         if (len > SMALL) {
581          long l = from;
582          long n = to - 1;
583          if (len > MEDIUM) { // Big arrays, pseudomedian of 9
584           long s = len / 8;
585           l = med3(x, l, l + s, l + 2 * s, comp);
586           m = med3(x, m - s, m, m + s, comp);
587           n = med3(x, n - 2 * s, n - s, n, comp);
588          }
589          m = med3(x, l, m, n, comp); // Mid-size, med of 3
590         }
591         final int v = get(x, m);
592         // Establish Invariant: v* (<v)* (>v)* v*
593         long a = from, b = a, c = to - 1, d = c;
594         while(true) {
595          int comparison;
596          while (b <= c && (comparison = comp.compare(get(x, b), v)) <= 0) {
597           if (comparison == 0) swap(x, a++, b);
598           b++;
599          }
600          while (c >= b && (comparison = comp.compare(get(x, c), v)) >=0) {
601           if (comparison == 0) swap(x, c, d--);
602           c--;
603          }
604          if (b > c) break;
605          swap(x, b++, c--);
606         }
607         // Swap partition elements back to middle
608         long s, n = to;
609         s = Math.min(a - from, b - a);
610         vecSwap(x, from, b - s, s);
611         s = Math.min(d - c, n - d- 1);
612         vecSwap(x, b, n - s, s);
613         // Recursively sort non-partition-elements
614         if ((s = b - a) > 1) quickSort(x, from, from + s, comp);
615         if ((s = d - c) > 1) quickSort(x, n - s, n, comp);
616        }

853         *
854         * <p>The sorting algorithm is a tuned radix sort adapted from Peter M. McIlroy, Keith Bostic and M. Douglas
855         * McIlroy, &ldquo;Engineering radix sort&rdquo;, <i>Computing Systems</i>, 6(1), pages 5&minus;27 (1993),
856         * and further improved using the digit-oracle idea described by
857         * Juha K&auml;rkk&auml;inen and Tommi Rantala in &ldquo;Engineering radix sort for strings&rdquo;,
858         * <i>String Processing and Information Retrieval, 15th International Symposium</i>, volume 5280 of
859         * Lecture Notes in Computer Science, pages 3&minus;14, Springer (2008).
860         *
861         * <p>This implementation is significantly faster than quicksort
862         * already at small sizes (say, more than 10000 elements), but it can only
863         * sort in ascending order.
864         * It will allocate a support array of bytes with the same number of elements as the array to be sorted.
865         *
866         * @param a the big array to be sorted.
867         * @param from the index of the first element (inclusive) to be sorted.
868         * @param to the index of the last element (exclusive) to be sorted.
869         */
870        public static void radixSort(final int[][] a, final long from, final long to) {
871         final int maxLevel = DIGITS_PER_ELEMENT - 1;
872         final int stackSize = ((1 << DIGIT_BITS) - 1) * (DIGITS_PER_ELEMENT - 1) + 1;
873         final long[] offsetStack = new long[stackSize];
874         int offsetPos = 0;
875         final long[] lengthStack = new long[stackSize];
876         int lengthPos = 0;
877         final int[] levelStack = new int[stackSize];
878         int levelPos = 0;
879         offsetStack[offsetPos++] = from;
880         lengthStack[lengthPos++] = to - from;
881         levelStack[levelPos++] = 0;
882         final long[] count = new long[1 << DIGIT_BITS];
883         final long[] pos = new long[1 << DIGIT_BITS];
884         final byte[][] digit = ByteBigArrays.newBigArray(to - from);
885         while(offsetPos > 0) {
886          final long first = offsetStack[--offsetPos];
887          final long length = lengthStack[--lengthPos];
888          final int level = levelStack[--levelPos];
889          final int signMask = level % DIGITS_PER_ELEMENT == 0 ? 1 << DIGIT_BITS - 1 : 0;
890          if (length < MEDIUM) {
891           selectionSort(a, first, first + length);
892           continue;
893          }
894          final int shift = (DIGITS_PER_ELEMENT - 1 - level % DIGITS_PER_ELEMENT) * DIGIT_BITS; // This is the shift that extract the right byte from a key
895          // Count keys.
896          for(long i = length; i-- != 0;) ByteBigArrays.set(digit, i, (byte)((((IntBigArrays.get(a, first + i)) >>> shift) & DIGIT_MASK) ^ signMask));
897          for(long i = length; i-- != 0;) count[ByteBigArrays.get(digit, i) & 0xFF]++;
898          // Compute cumulative distribution and push non-singleton keys on stack.
899          int lastUsed = -1;
900          long p = 0;
901          for(int i = 0; i < 1 << DIGIT_BITS; i++) {
902           if (count[i] != 0) {
903            lastUsed = i;
904            if (level < maxLevel && count[i] > 1){
905             //System.err.println(" Pushing " + new StackEntry(first + pos[i - 1], first + pos[i], level + 1));
906             offsetStack[offsetPos++] = p + first;
907             lengthStack[lengthPos++] = count[i];
908             levelStack[levelPos++] = level + 1;
909            }
910           }
911           pos[i] = (p += count[i]);
912          }
913          // When all slots are OK, the last slot is necessarily OK.
914          final long end = length - count[lastUsed];
915          count[lastUsed] = 0;
916          // i moves through the start of each block
917          int c = -1;
918          for(long i = 0, d; i < end; i += count[c], count[c] = 0) {
919           int t = IntBigArrays.get(a, i +first);
920           c = ByteBigArrays.get(digit, i) & 0xFF;
921           while((d = --pos[c]) > i) {
922            final int z = t;
923            final int zz = c;
924            t = IntBigArrays.get(a, d + first);
925            c = ByteBigArrays.get(digit, d) & 0xFF;
926            IntBigArrays.set(a, d + first, z);
927            ByteBigArrays.set(digit, d, (byte)zz);
928           }
929           IntBigArrays.set(a, i + first, t);
930          }
931         }
932        }

1001         final int[] levelStack = new int[stackSize];
1002         int levelPos = 0;
1003         offsetStack[offsetPos++] = from;
1004         lengthStack[lengthPos++] = to - from;
1005         levelStack[levelPos++] = 0;
1006         final long[] count = new long[1 << DIGIT_BITS];
1007         final long[] pos = new long[1 << DIGIT_BITS];
1008         final byte[][] digit = ByteBigArrays.newBigArray(to - from);
1009         while(offsetPos > 0) {
1010          final long first = offsetStack[--offsetPos];
1011          final long length = lengthStack[--lengthPos];
1012          final int level = levelStack[--levelPos];
1013          final int signMask = level % DIGITS_PER_ELEMENT == 0 ? 1 << DIGIT_BITS - 1 : 0;
1014          if (length < MEDIUM) {
1015           selectionSort(a, b, first, first + length);
1016           continue;
1017          }
1018          final int[][] k = level < DIGITS_PER_ELEMENT ? a : b; // This is the key array
1019          final int shift = (DIGITS_PER_ELEMENT - 1 - level % DIGITS_PER_ELEMENT) * DIGIT_BITS; // This is the shift that extract the right byte from a key
1020          // Count keys.
1021          for(long i = length; i-- != 0;) ByteBigArrays.set(digit, i, (byte)((((IntBigArrays.get(k, first + i)) >>> shift) & DIGIT_MASK) ^ signMask));
1022          for(long i = length; i-- != 0;) count[ByteBigArrays.get(digit, i) & 0xFF]++;
1023          // Compute cumulative distribution and push non-singleton keys on stack.
1024          int lastUsed = -1;
1025          long p = 0;
1026          for(int i = 0; i < 1 << DIGIT_BITS; i++) {
1027           if (count[i] != 0) {
1028            lastUsed = i;
1029            if (level < maxLevel && count[i] > 1){
1030             offsetStack[offsetPos++] = p + first;
1031             lengthStack[lengthPos++] = count[i];
1032             levelStack[levelPos++] = level + 1;
1033            }
1034           }
1035           pos[i] = (p += count[i]);
1036          }
1037          // When all slots are OK, the last slot is necessarily OK.
1038          final long end = length - count[lastUsed];
1039          count[lastUsed] = 0;
1040          // i moves through the start of each block
1041          int c = -1;
1042          for(long i = 0, d; i < end; i += count[c], count[c] = 0) {
1043           int t = IntBigArrays.get(a, i + first);
1044           int u = IntBigArrays.get(b, i + first);
1045           c = ByteBigArrays.get(digit, i) & 0xFF;
1046           while((d = --pos[c]) > i) {
1047            int z = t;
1048            final int zz = c;
1049            t = IntBigArrays.get(a, d + first);
1050            IntBigArrays.set(a, d + first, z);
1051            z = u;
1052            u = IntBigArrays.get(b, d + first);
1053            IntBigArrays.set(b, d + first, z);
1054            c = ByteBigArrays.get(digit, d) & 0xFF;
1055            ByteBigArrays.set(digit, d, (byte)zz);
1056           }
1057           IntBigArrays.set(a, i + first, t);
1058           IntBigArrays.set(b, i + first, u);
1059          }
1060         }
1061        }
1062        /** Shuffles the specified big array fragment using the specified pseudorandom number generator.
1063         *
1064         * @param a the big array to be shuffled.
1065         * @param from the index of the first element (inclusive) to be shuffled.
1066         * @param to the index of the last element (exclusive) to be shuffled.
1067         * @param random a pseudorandom number generator (please use a <a href="http://dsiutils.dsi.unimi.it/docs/it/unimi/dsi/util/XorShiftStarRandom.html">XorShift*</a> generator).
1068         * @return <code>a</code>.
1069         */
1070        public static int[][] shuffle(final int[][] a, final long from, final long to, final Random random) {
1071         for(long i = to - from; i-- != 0;) {
1072          final long p = (random.nextLong() & 0x7FFFFFFFFFFFFFFFL) % (i + 1);
1073          final int t = get(a, from + i);
1074          set(a, from + i, get(a, from + p));
1075          set(a, from + p, t);
1076         }
1077         return a;
1078        }

1080         *
1081         * @param a the big array to be shuffled.

1083         * @return <code>a</code>.
