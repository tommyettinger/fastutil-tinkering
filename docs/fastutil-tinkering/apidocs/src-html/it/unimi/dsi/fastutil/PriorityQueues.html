022import java.util.Comparator;
023import java.util.NoSuchElementException;
024
025/** A class providing static methods and objects that do useful things with priority queues.
026 *
027 * @see it.unimi.dsi.fastutil.PriorityQueue
028 */
029
030public class PriorityQueues {
031
032        private PriorityQueues() {}
033
034        /** An immutable class representing the empty priority queue.
035         *
036         * <P>This class may be useful to implement your own in case you subclass
037         * {@link PriorityQueue}.
038         */
039
040        @SuppressWarnings("rawtypes")
041        public static class EmptyPriorityQueue extends AbstractPriorityQueue implements Serializable {
042                private static final long serialVersionUID = 0L;
043
044                protected EmptyPriorityQueue() {}
045
046                @Override
047                public void enqueue(Object o) { throw new UnsupportedOperationException(); }
048
049                @Override
050                public Object dequeue() { throw new NoSuchElementException(); }
051
052                @Override
053                public boolean isEmpty() { return true; }
054
055                @Override
056                public int size() { return 0; }
057
058                @Override
059                public void clear() {}
060
061                @Override
062                public Object first() { throw new NoSuchElementException(); }
063
064                @Override
065                public Object last() { throw new NoSuchElementException(); }
066
067                @Override
068                public void changed() { throw new NoSuchElementException(); }
069
070                @Override
071                public Comparator<?> comparator() { return null; }
072
073                @Override
074                public Object clone() { return EMPTY_QUEUE; }
075
076                @Override
077                public boolean equals(final Object o) { return o instanceof PriorityQueue && ((PriorityQueue)o).isEmpty(); }
078
079                private Object readResolve() { return EMPTY_QUEUE; }
080        }
081
082        /** An empty indirect priority queue (immutable). */
083
084        public final static EmptyPriorityQueue EMPTY_QUEUE = new EmptyPriorityQueue();
085
086        /** Returns an empty queue (immutable). It is serializable and cloneable.
087         *
088         * <P>This method provides a typesafe access to {@link #EMPTY_QUEUE}.
089         * @param <K> the class of the objects in the queue.
090         * @return an empty queue (immutable).
091         */
092        @SuppressWarnings("unchecked")
093        public static <K> PriorityQueue<K> emptyQueue() {
094                return EMPTY_QUEUE;
095        }
096
097        /** A synchronized wrapper class for priority queues. */
098
099        public static class SynchronizedPriorityQueue<K> implements PriorityQueue<K>, Serializable {
100                public static final long serialVersionUID = -7046029254386353129L;
101
102                final protected PriorityQueue <K> q;
103                final protected Object sync;
104
105                protected SynchronizedPriorityQueue(final PriorityQueue <K> q, final Object sync) {
106                        this.q = q;
107                        this.sync = sync;
108                }
109
110                protected SynchronizedPriorityQueue(final PriorityQueue <K> q) {
111                        this.q = q;
112                        this.sync = this;
113                }
114
115                @Override
116                public void enqueue(K x) { synchronized(sync) { q.enqueue(x); } }
117
118                @Override
119                public K dequeue() { synchronized(sync) { return q.dequeue(); } }
120
121                @Override
122                public K first() { synchronized(sync) { return q.first(); } }
123
124                @Override
125                public K last() { synchronized(sync) { return q.last(); } }
126
127                @Override
128                public boolean isEmpty() { synchronized(sync) { return q.isEmpty(); } }
129
130                @Override
131                public int size() { synchronized(sync) { return q.size(); } }
132
133                @Override
134                public void clear() { synchronized(sync) { q.clear(); } }
135
136                @Override
137                public void changed() { synchronized(sync) { q.changed(); } }
138
139                @Override
140                public Comparator <? super K> comparator() { synchronized(sync) { return q.comparator(); } }
141
142                @Override
143                public String toString() { synchronized(sync) { return q.toString(); } }
144
145                @Override
146                public int hashCode() { synchronized(sync) { return q.hashCode(); } }
147
148                @Override
149                public boolean equals(final Object o) { if (o == this) return true; synchronized(sync) { return q.equals(o); } }
150
151                private void writeObject(java.io.ObjectOutputStream s) throws java.io.IOException {
152                        synchronized(sync) { s.defaultWriteObject(); }
153                }
154        }
155
156
157        /** Returns a synchronized priority queue backed by the specified priority queue.
158         *
159         * @param <K> the class of the objects in the queue.
160         * @param q the priority queue to be wrapped in a synchronized priority queue.
161         * @return a synchronized view of the specified priority queue.
162         */
163        public static <K> PriorityQueue <K> synchronize(final PriorityQueue <K> q) {    return new SynchronizedPriorityQueue<K>(q); }
164
165        /** Returns a synchronized priority queue backed by the specified priority queue, using an assigned object to synchronize.
166         *
167         * @param <K> the class of the objects in the queue.
168         * @param q the priority queue to be wrapped in a synchronized priority queue.
169         * @param sync an object that will be used to synchronize the access to the priority queue.
170         * @return a synchronized view of the specified priority queue.
171         */
172
173        public static <K> PriorityQueue <K> synchronize(final PriorityQueue <K> q, final Object sync) { return new SynchronizedPriorityQueue<K>(q, sync); }
174}
