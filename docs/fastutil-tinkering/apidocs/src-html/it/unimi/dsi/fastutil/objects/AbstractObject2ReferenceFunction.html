016package it.unimi.dsi.fastutil.objects;
017/** An abstract class providing basic methods for functions implementing a type-specific interface.
018        *
019        * <P>Optional operations just throw an {@link
020        * UnsupportedOperationException}. Generic versions of accessors delegate to
021        * the corresponding type-specific counterparts following the interface rules
022        * (they take care of returning {@code null} on a missing key).
023        *
024        * <P>This class handles directly a default return
025        * value (including {@linkplain #defaultReturnValue() methods to access
026        * it}). Instances of classes inheriting from this class have just to return
027        * <code>defRetValue</code> to denote lack of a key in type-specific methods. The value
028        * is serialized.
029        *
030        * <P>Implementing subclasses have just to provide type-specific <code>get()</code>,
031        * type-specific <code>containsKey()</code>, and <code>size()</code> methods.
032        *
033        */
034public abstract class AbstractObject2ReferenceFunction <K,V> implements Object2ReferenceFunction <K,V>, java.io.Serializable {
035        private static final long serialVersionUID = -4940583368468432370L;
036        protected AbstractObject2ReferenceFunction() {}
037        /**
038         * The default return value for <code>get()</code>, <code>put()</code> and
039         * <code>remove()</code>.
040         */
041        protected V defRetValue;
042        @Override
043        public void defaultReturnValue(final V rv) {
044         defRetValue = rv;
045        }
046        @Override
047        public V defaultReturnValue() {
048         return defRetValue;
049        }
050        /** {@inheritDoc}
051         *
052         * <p>This implementation always throws an {@link UnsupportedOperationException}.
053         */
054        public V put(K key, V value) {
055         throw new UnsupportedOperationException();
056        }
057        /** {@inheritDoc}
058         *
059         * <p>This implementation always throws an {@link UnsupportedOperationException}.
060         */
061        public V remove(Object key) {
062         throw new UnsupportedOperationException();
063        }
064        /** {@inheritDoc}
065         *
066         * <p>This implementation always throws an {@link UnsupportedOperationException}.
067         */
068        public void clear() {
069         throw new UnsupportedOperationException();
070        }
071}
