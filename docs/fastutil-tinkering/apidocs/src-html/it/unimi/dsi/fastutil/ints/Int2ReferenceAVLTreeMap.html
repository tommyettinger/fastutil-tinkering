029/** A type-specific AVL tree map with a fast, small-footprint implementation.
030        *
031        * <P>The iterators provided by the views of this class are type-specific {@linkplain
032        * it.unimi.dsi.fastutil.BidirectionalIterator bidirectional iterators}.
033        * Moreover, the iterator returned by <code>iterator()</code> can be safely cast
034        * to a type-specific {@linkplain java.util.ListIterator list iterator}.
035        */
036public class Int2ReferenceAVLTreeMap <V> extends AbstractInt2ReferenceSortedMap <V> implements java.io.Serializable, Cloneable {

071        /** Generates the comparator that will be actually used.
072         *
073         * <P>When a specific {@link Comparator} is specified and stored in {@link
074         * #storedComparator}, we must check whether it is type-specific.  If it is
075         * so, we can used directly, and we store it in {@link #actualComparator}. Otherwise,
076         * we generate on-the-fly an anonymous class that wraps the non-specific {@link Comparator}
077         * and makes it into a type-specific one.
078         */
079        private void setActualComparator() {
080         /* If the provided comparator is already type-specific, we use it. Otherwise,
081                   we use a wrapper anonymous class to fake that it is type-specific. */
082         if (storedComparator == null || storedComparator instanceof IntComparator) actualComparator = (IntComparator)storedComparator;
083         else actualComparator = new IntComparator () {
084           public int compare(int k1, int k2) {
085            return storedComparator.compare((Integer.valueOf(k1)), (Integer.valueOf(k2)));
086           }
087           public int compare(Integer ok1, Integer ok2) {
088            return storedComparator.compare(ok1, ok2);
089           }
090          };
091        }

157         *

163        /** Compares two keys in the right way.
164         *
165         * <P>This method uses the {@link #actualComparator} if it is non-{@code null}.
166         * Otherwise, it resorts to primitive type comparisons or to {@link Comparable#compareTo(Object) compareTo()}.
167         *
168         * @param k1 the first key.
169         * @param k2 the second key.
170         * @return a number smaller than, equal to or greater than 0, as usual
171         * (i.e., when k1 &lt; k2, k1 = k2 or k1 &gt; k2, respectively).
172         */
173
174        final int compare(final int k1, final int k2) {
175         return actualComparator == null ? ( Integer.compare((k1),(k2)) ) : actualComparator.compare(k1, k2);
176        }
177        /** Returns the entry corresponding to the given key, if it is in the tree; {@code null}, otherwise.
178         *
179         * @param k the key to search for.
180         * @return the corresponding entry, or {@code null} if no entry with the given key exists.
181         */
182        final Entry <V> findKey(final int k) {
183         Entry <V> e = tree;
184         int cmp;
185         while (e != null && (cmp = compare(k, e.key)) != 0) e = cmp < 0 ? e.left() : e.right();
186         return e;
187        }
188        /** Locates a key.
189         *
190         * @param k a key.
191         * @return the last entry on a search for the given key; this will be
192         * the given key, if it present; otherwise, it will be either the smallest greater key or the greatest smaller key.
193         */
194        final Entry <V> locateKey(final int k) {
195         Entry <V> e = tree, last = tree;
196         int cmp = 0;
197         while (e != null && (cmp = compare(k, e.key)) != 0) {
198          last = e;
199          e = cmp < 0 ? e.left() : e.right();
200         }
201         return cmp == 0 ? e : last;
202        }
203        /** This vector remembers the directions followed during
204         * the current insertion. It suffices for about 2<sup>32</sup> entries. */
205        private transient boolean dirPath[];
206        private void allocatePaths() {
207         dirPath = new boolean[48];
208        }
209        @Override
210        public V put(final int k, final V v) {
211         Entry <V> e = add(k);
212         final V oldValue = e.value;
213         e.value = v;
214         return oldValue;
215        }
216        /** Returns a node with key k in the balanced tree, creating one with defRetValue if necessary.
217        *
218        * @param k the key
219        * @return a node with key k. If a node with key k already exists, then that node is returned,
220        *                               otherwise a new node with defRetValue is created ensuring that the tree is balanced
221                                                after creation of the node.
222        */
223        private Entry <V> add(final int k) {
224          /* After execution of this method, modified is true iff a new entry has
225                been inserted. */
226         modified = false;
227         Entry <V> e = null;
228         if (tree == null) { // The case of the empty tree is treated separately.
229          count++;
230          e = tree = lastEntry = firstEntry = new Entry <V>(k, defRetValue);
231          modified = true;
232         }
233         else {
234          Entry <V> p = tree, q = null, y = tree, z = null, w = null;
235          int cmp, i = 0;
236          while(true) {
237           if ((cmp = compare(k, p.key)) == 0) {
238            return p;
239           }
240           if (p.balance() != 0) {
241            i = 0;
242            z = q;
243            y = p;
244           }
245           if (dirPath[i++] = cmp > 0) {
246            if (p.succ()) {
247             count++;
248             e = new Entry <V>(k, defRetValue);
249             modified = true;
250             if (p.right == null) lastEntry = e;
251             e.left = p;
252             e.right = p.right;
253             p.right(e);
254             break;
255            }
256            q = p;
257            p = p.right;
258           }
259           else {
260            if (p.pred()) {
261             count++;
262             e = new Entry <V>(k, defRetValue);
263             modified = true;
264             if (p.left == null) firstEntry = e;
265             e.right = p;
266             e.left = p.left;
267             p.left(e);
268             break;
269            }
270            q = p;
271            p = p.left;
272           }
273          }
274          p = y;
275          i = 0;
276          while(p != e) {
277           if (dirPath[i]) p.incBalance();
278           else p.decBalance();
279           p = dirPath[i++] ? p.right : p.left;
280          }
281          if (y.balance() == -2) {
282           Entry <V> x = y.left;
283           if (x.balance() == -1) {
284            w = x;
285            if (x.succ()) {
286             x.succ(false);
287             y.pred(x);
288            }
289            else y.left = x.right;
290            x.right = y;
291            x.balance(0);
292            y.balance(0);
293           }
294           else {
295            if (ASSERTS) assert x.balance() == 1;
296            w = x.right;
297            x.right = w.left;
298            w.left = x;
299            y.left = w.right;
300            w.right = y;
301            if (w.balance() == -1) {
302             x.balance(0);
303             y.balance(1);
304            }
305            else if (w.balance() == 0) {
306             x.balance(0);
307             y.balance(0);
308            }
309            else {
310             x.balance(-1);
311             y.balance(0);
312            }
313            w.balance(0);
314            if (w.pred()) {
315             x.succ(w);
316             w.pred(false);
317            }
318            if (w.succ()) {
319             y.pred(w);
320             w.succ(false);
321            }
322           }
323          }
324          else if (y.balance() == +2) {
325           Entry <V> x = y.right;
326           if (x.balance() == 1) {
327            w = x;
328            if (x.pred()) {
329             x.pred(false);
330             y.succ(x);
331            }
332            else y.right = x.left;
333            x.left = y;
334            x.balance(0);
335            y.balance(0);
336           }
337           else {
338            if (ASSERTS) assert x.balance() == -1;
339            w = x.left;
340            x.left = w.right;
341            w.right = x;
342            y.right = w.left;
343            w.left = y;
344            if (w.balance() == 1) {
345             x.balance(0);
346             y.balance(-1);
347            }
348            else if (w.balance() == 0) {
349             x.balance(0);
350             y.balance(0);
351            }
352            else {
353             x.balance(1);
354             y.balance(0);
355            }
356            w.balance(0);
357            if (w.pred()) {
358             y.succ(w);
359             w.pred(false);
360            }
361            if (w.succ()) {
362             x.pred(w);
363             w.succ(false);
364            }
365           }
366          }
367          else return e;
368          if (z == null) tree = w;
369          else {
370           if (z.left == y) z.left = w;
371           else z.right = w;
372          }
373         }
374         if (ASSERTS) checkTree(tree);
375         return e;
376        }
377        /** Finds the parent of an entry.
378         *
379         * @param e a node of the tree.
380         * @return the parent of the given node, or {@code null} for the root.
381         */
382        private Entry <V> parent(final Entry <V> e) {
383         if (e == tree) return null;
384         Entry <V> x, y, p;
385         x = y = e;
386         while(true) {
387          if (y.succ()) {
388           p = y.right;
389           if (p == null || p.left != e) {
390            while(! x.pred()) x = x.left;
391            p = x.left;
392           }
393           return p;
394          }
395          else if (x.pred()) {
396           p = x.left;
397           if (p == null || p.right != e) {
398            while(! y.succ()) y = y.right;
399            p = y.right;
400           }
401           return p;
402          }
403          x = x.left;
404          y = y.right;
405         }
406        }
407        /* After execution of this method, {@link #modified} is true iff an entry
408        has been deleted. */

410        public V remove(final int k) {
411         modified = false;
412         if (tree == null) return defRetValue;
413         int cmp;
414         Entry <V> p = tree, q = null;
415         boolean dir = false;
416         final int kk = k;
417         while(true) {
418          if ((cmp = compare(kk, p.key)) == 0) break;
419          else if (dir = cmp > 0) {
420           q = p;
421           if ((p = p.right()) == null) return defRetValue;
422          }
423          else {
424           q = p;
425           if ((p = p.left()) == null) return defRetValue;
426          }
427         }
428         if (p.left == null) firstEntry = p.next();
429         if (p.right == null) lastEntry = p.prev();
430         if (p.succ()) {
431          if (p.pred()) {
432           if (q != null) {
433            if (dir) q.succ(p.right);
434            else q.pred(p.left);
435           }
436           else tree = dir ? p.right : p.left;
437          }
438          else {
439           p.prev().right = p.right;
440           if (q != null) {
441            if (dir) q.right = p.left;
442            else q.left = p.left;
443           }
444           else tree = p.left;
445          }
446         }
447         else {
448          Entry <V> r = p.right;
449          if (r.pred()) {
450           r.left = p.left;
451           r.pred(p.pred());
452           if (! r.pred()) r.prev().right = r;
453           if (q != null) {
454            if (dir) q.right = r;
455            else q.left = r;
456           }
457           else tree = r;
458           r.balance(p.balance());
459           q = r;
460           dir = true;
461          }
462          else {
463           Entry <V> s;
464           while(true) {
465            s = r.left;
466            if (s.pred()) break;
467            r = s;
468           }
469           if (s.succ()) r.pred(s);
470           else r.left = s.right;
471           s.left = p.left;
472           if (! p.pred()) {
473            p.prev().right = s;
474            s.pred(false);
475           }
476           s.right = p.right;
477           s.succ(false);
478           if (q != null) {
479            if (dir) q.right = s;
480            else q.left = s;
481           }
482           else tree = s;
483           s.balance(p.balance());
484           q = r;
485           dir = false;
486          }
487         }
488         Entry <V> y;
489         while(q != null) {
490          y = q;
491          q = parent(y);
492          if (! dir) {
493           dir = q != null && q.left != y;
494           y.incBalance();
495           if (y.balance() == 1) break;
496           else if (y.balance() == 2) {
497            Entry <V> x = y.right;
498            if (ASSERTS) assert x != null;
499            if (x.balance() == -1) {
500             Entry <V> w;
501             if (ASSERTS) assert x.balance() == -1;
502             w = x.left;
503             x.left = w.right;
504             w.right = x;
505             y.right = w.left;
506             w.left = y;
507             if (w.balance() == 1) {
508              x.balance(0);
509              y.balance(-1);
510             }
511             else if (w.balance() == 0) {
512              x.balance(0);
513              y.balance(0);
514             }
515             else {
516              if (ASSERTS) assert w.balance() == -1;
517              x.balance(1);
518              y.balance(0);
519             }
520             w.balance(0);
521             if (w.pred()) {
522              y.succ(w);
523              w.pred(false);
524             }
525             if (w.succ()) {
526              x.pred(w);
527              w.succ(false);
528             }
529             if (q != null) {
530              if (dir) q.right = w;
531              else q.left = w;
532             }
533             else tree = w;
534            }
535            else {
536             if (q != null) {
537              if (dir) q.right = x;
538              else q.left = x;
539             }
540             else tree = x;
541             if (x.balance() == 0) {
542              y.right = x.left;
543              x.left = y;
544              x.balance(-1);
545              y.balance(+1);
546              break;
547             }
548             if (ASSERTS) assert x.balance() == 1;
549             if (x.pred()) {
550              y.succ(true);
551              x.pred(false);
552             }
553             else y.right = x.left;
554             x.left = y;
555             y.balance(0);
556             x.balance(0);
557            }
558           }
559          }
560          else {
561           dir = q != null && q.left != y;
562           y.decBalance();
563           if (y.balance() == -1) break;
564           else if (y.balance() == -2) {
565            Entry <V> x = y.left;
566            if (ASSERTS) assert x != null;
567            if (x.balance() == 1) {
568             Entry <V> w;
569             if (ASSERTS) assert x.balance() == 1;
570             w = x.right;
571             x.right = w.left;
572             w.left = x;
573             y.left = w.right;
574             w.right = y;
575             if (w.balance() == -1) {
576              x.balance(0);
577              y.balance(1);
578             }
579             else if (w.balance() == 0) {
580              x.balance(0);
581              y.balance(0);
582             }
583             else {
584              if (ASSERTS) assert w.balance() == 1;
585              x.balance(-1);
586              y.balance(0);
587             }
588             w.balance(0);
589             if (w.pred()) {
590              x.succ(w);
591              w.pred(false);
592             }
593             if (w.succ()) {
594              y.pred(w);
595              w.succ(false);
596             }
597             if (q != null) {
598              if (dir) q.right = w;
599              else q.left = w;
600             }
601             else tree = w;
602            }
603            else {
604             if (q != null) {
605              if (dir) q.right = x;
606              else q.left = x;
607             }
608             else tree = x;
609             if (x.balance() == 0) {
610              y.left = x.right;
611              x.right = y;
612              x.balance(+1);
613              y.balance(-1);
614              break;
615             }
616             if (ASSERTS) assert x.balance() == -1;
617             if (x.succ()) {
618              y.pred(true);
619              x.succ(false);
620             }
621             else y.left = x.right;
622             x.right = y;
623             y.balance(0);
624             x.balance(0);
625            }
626           }
627          }
628         }
629         modified = true;
630         count--;
631         if (ASSERTS) checkTree(tree);
632         return p.value;
633        }
634        @Override
635        public boolean containsValue(final Object v) {
636         final ValueIterator i = new ValueIterator();
637         V ev;
638         int j = count;
639         while(j-- != 0) {
640          ev = i.next();
641          if (( (ev) == (v) )) return true;
642         }
643         return false;
644        }

654        /** This class represent an entry in a tree map.
655         *
656         * <P>We use the only "metadata", i.e., {@link Entry#info}, to store
657         * information about balance, predecessor status and successor status.
658         *
659         * <P>Note that since the class is recursive, it can be
660         * considered equivalently a tree.
661         */
662        private static final class Entry <V> extends AbstractInt2ReferenceMap.BasicEntry <V> implements Cloneable {
663         /** If the bit in this mask is true, {@link #right} points to a successor. */
664         private final static int SUCC_MASK = 1 << 31;
665         /** If the bit in this mask is true, {@link #left} points to a predecessor. */
666         private final static int PRED_MASK = 1 << 30;
667         /** The bits in this mask hold the node balance info. You can get it just by casting to byte. */
668         private final static int BALANCE_MASK = 0xFF;
669         /** The pointers to the left and right subtrees. */
670         Entry <V> left, right;
671         /** This integers holds different information in different bits (see {@link #SUCC_MASK}, {@link #PRED_MASK} and {@link #BALANCE_MASK}). */
672         int info;
673         Entry() {
674          super((0), (null));
675         }
676         /** Creates a new entry with the given key and value.
677                 *
678                 * @param k a key.
679                 * @param v a value.
680                 */
681         Entry(final int k, final V v) {
682          super(k, v);
683          info = SUCC_MASK | PRED_MASK;
684         }
685         /** Returns the left subtree.
686                 *
687                 * @return the left subtree ({@code null} if the left
688                 * subtree is empty).
689                 */
690         Entry <V> left() {
691          return (info & PRED_MASK) != 0 ? null : left;
692         }
693         /** Returns the right subtree.
694                 *
695                 * @return the right subtree ({@code null} if the right
696                 * subtree is empty).
697                 */

896         /** The entry that will be returned by the next call to {@link java.util.ListIterator#previous()} (or {@code null} if no previous entry exists). */
897         Entry <V> prev;
898         /** The entry that will be returned by the next call to {@link java.util.ListIterator#next()} (or {@code null} if no next entry exists). */
899         Entry <V> next;
900         /** The last entry that was returned (or {@code null} if we did not iterate or used {@link #remove()}). */
901         Entry <V> curr;
902         /** The current index (in the sense of a {@link java.util.ListIterator}). Note that this value is not meaningful when this {@link TreeIterator} has been created using the nonempty constructor.*/

906         }

908          if ((next = locateKey(k)) != null) {
909           if (compare(next.key, k) <= 0) {
910            prev = next;
911            next = next.next();
912           }
913           else prev = next.prev();
914          }
915         }
916         public boolean hasNext() { return next != null; }
917         public boolean hasPrevious() { return prev != null; }
918         void updateNext() {
919          next = next.next();
920         }
921         Entry <V> nextEntry() {
922          if (! hasNext()) throw new NoSuchElementException();
923          curr = prev = next;
924          index++;
925          updateNext();
926          return curr;
927         }
928         void updatePrevious() {
929          prev = prev.prev();
930         }
931         Entry <V> previousEntry() {
932          if (! hasPrevious()) throw new NoSuchElementException();
933          curr = next = prev;
934          index--;
935          updatePrevious();
936          return curr;
937         }

939          return index;
940         }
941         public int previousIndex() {
942          return index - 1;
943         }

945          if (curr == null) throw new IllegalStateException();
946          /* If the last operation was a next(), we are removing an entry that preceeds
947                           the current index, and thus we must decrement it. */
948          if (curr == prev) index--;
949          next = prev = curr;
950          updatePrevious();
951          updateNext();
952          Int2ReferenceAVLTreeMap.this.remove(curr.key);
953          curr = null;
954         }
955         public int skip(final int n) {
956          int i = n;
957          while(i-- != 0 && hasNext()) nextEntry();
958          return n - i - 1;
959         }
960         public int back(final int n) {
961          int i = n;
962          while(i-- != 0 && hasPrevious()) previousEntry();
963          return n - i - 1;
964         }

1366         /** Locates the first entry.
1367                 *
1368                 * @return the first entry of this submap, or {@code null} if the submap is empty.
1369                 */
1370         public Int2ReferenceAVLTreeMap.Entry <V> firstEntry() {
1371          if (tree == null) return null;
1372          // If this submap goes to -infinity, we return the main map first entry; otherwise, we locate the start of the map.
1373          Int2ReferenceAVLTreeMap.Entry <V> e;
1374          if (bottom) e = firstEntry;
1375          else {
1376           e = locateKey(from);
1377           // If we find either the start or something greater we're OK.
1378           if (compare(e.key, from) < 0) e = e.next();
1379          }
1380          // Finally, if this subset doesn't go to infinity, we check that the resulting key isn't greater than the end.
1381          if (e == null || ! top && compare(e.key, to) >= 0) return null;
1382          return e;
1383         }

1399          if (e == null || ! bottom && compare(e.key, from) < 0) return null;

1414         /** An iterator for subranges.
1415                 *
1416                 * <P>This class inherits from {@link TreeIterator}, but overrides the methods that
1417                 * update the pointer after a {@link java.util.ListIterator#next()} or {@link java.util.ListIterator#previous()}. If we would
1418                 * move out of the range of the submap we just overwrite the next or previous
1419                 * entry with {@code null}.
1420                 */
1421         private class SubmapIterator extends TreeIterator {
1422          SubmapIterator() {
1423           next = firstEntry();
1424          }
1425          SubmapIterator(final int k) {
1426           this();
1427           if (next != null) {
1428            if (! bottom && compare(k, next.key) < 0) prev = null;
1429            else if (! top && compare(k, (prev = lastEntry()).key) >= 0) next = null;
1430            else {
1431             next = locateKey(k);
1432             if (compare(next.key, k) <= 0) {
1433              prev = next;
1434              next = next.next();
1435             }
1436             else prev = next.prev();
1437            }
1438           }
1439          }
1440          void updatePrevious() {
1441           prev = prev.prev();
1442           if (! bottom && prev != null && Int2ReferenceAVLTreeMap.this.compare(prev.key, from) < 0) prev = null;
1443          }
1444          void updateNext() {
1445           next = next.next();
1446           if (! top && next != null && Int2ReferenceAVLTreeMap.this.compare(next.key, to) >= 0) next = null;
1447          }
1448         }

1589        /** Reads the given number of entries from the input stream, returning the corresponding tree.
1590         *
1591         * @param s the input stream.
1592         * @param n the (positive) number of entries to read.
1593         * @param pred the entry containing the key that preceeds the first key in the tree.
1594         * @param succ the entry containing the key that follows the last key in the tree.
1595         */
1596        @SuppressWarnings("unchecked")
1597        private Entry <V> readTree(final java.io.ObjectInputStream s, final int n, final Entry <V> pred, final Entry <V> succ) throws java.io.IOException, ClassNotFoundException {
1598         if (n == 1) {
1599          final Entry <V> top = new Entry <V>( s.readInt(), (V) s.readObject());
1600          top.pred(pred);
1601          top.succ(succ);
1602          return top;
1603         }
1604         if (n == 2) {
1605          /* We handle separately this case so that recursion will
1606                         *always* be on nonempty subtrees. */
1607          final Entry <V> top = new Entry <V>( s.readInt(), (V) s.readObject());
1608          top.right(new Entry <V>( s.readInt(), (V) s.readObject()));
1609          top.right.pred(top);
1610          top.balance(1);
1611          top.pred(pred);
1612          top.right.succ(succ);
1613          return top;
1614         }
1615         // The right subtree is the largest one.
1616         final int rightN = n / 2, leftN = n - rightN - 1;
1617         final Entry <V> top = new Entry <V>();
1618         top.left(readTree(s, leftN, pred, top));
1619         top.key = s.readInt();
1620         top.value = (V) s.readObject();
1621         top.right(readTree(s, rightN, top, succ));
1622         if (n == (n & -n)) top.balance(1); // Quick test for determining whether n is a power of 2.
1623         return top;
1624        }
1625        private void readObject(java.io.ObjectInputStream s) throws java.io.IOException, ClassNotFoundException {
1626         s.defaultReadObject();
1627         /* The storedComparator is now correctly set, but we must restore
1628                   on-the-fly the actualComparator. */
1629         setActualComparator();
1630         allocatePaths();
1631         if (count != 0) {
1632          tree = readTree(s, count, null, null);
1633          Entry <V> e;
1634          e = tree;
1635          while(e.left() != null) e = e.left();
1636          firstEntry = e;
1637          e = tree;
1638          while(e.right() != null) e = e.right();
1639          lastEntry = e;
1640         }
1641         if (ASSERTS) checkTree(tree);
1642        }
