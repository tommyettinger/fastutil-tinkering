036/**
037 * A class providing static methods and objects that do useful things with big
038 * arrays.
039 *
040 * <h2>Introducing big arrays</h2>
041 *
042 * <p>
043 * A <em>big array</em> is an array-of-arrays representation of an array. The
044 * length of a big array is bounded by {@link #SEGMENT_SIZE} *
045 * {@link Integer#MAX_VALUE} = {@value #SEGMENT_SIZE} * (2<sup>31</sup> &minus;
046 * 1) rather than {@link Integer#MAX_VALUE}. The type of a big array is that of
047 * an array-of-arrays, so a big array of integers is of type
048 * <code>int[][]</code>. Note that {@link #SEGMENT_SIZE} has been chosen so that
049 * a single segment is smaller than 2<sup>31</sup> bytes independently of the
050 * data type. It might be enlarged in the future.
051 *
052 * <p>
053 * If <code>a</code> is a big array, <code>a[0]</code>, <code>a[1]</code>,
054 * &hellip; are called the <em>segments</em> of the big array. All segments,
055 * except possibly for the last one, are of length {@link #SEGMENT_SIZE}. Given
056 * an index <code>i</code> into a big array, there is an associated
057 * <em>{@linkplain #segment(long) segment}</em> and an associated
058 * <em>{@linkplain #displacement(long)
059 * displacement}</em> into that segment. Access to single members happens by
060 * means of accessors defined in the type-specific versions (see, e.g.,
061 * {@link IntBigArrays#get(int[][], long)} and
062 * {@link IntBigArrays#set(int[][], long, int)}), but you can also use the
063 * methods {@link #segment(long)}/{@link #displacement(long)} to access entries
064 * manually.
065 *
066 * <h2>Scanning big arrays</h2>
067 *
068 * <p>
069 * You can scan a big array using the following idiomatic form:
070 *
071 * <pre>
072 * for(int s = 0; s &lt; a.length; s++) {
073 *     final int[] t = a[s];
074 *     final int l = t.length;
075 *     for(int d = 0; d &lt; l; d++) {
076 *          do something with t[d]
077 *     }
078 * }
079 * </pre>
080 *
081 * or using the (simpler and usually faster) reversed version:
082 *
083 * <pre>
084 * for(int s = a.length; s-- != 0;) {
085 *     final int[] t = a[s];
086 *     for(int d = t.length; d-- != 0;) {
087 *         do something with t[d]
088 *     }
089 * }
090 * </pre>
091 * <p>
092 * Inside the inner loop, the original index in <code>a</code> can be retrieved
093 * using {@link #index(int, int) index(segment, displacement)}. You can also
094 * use an additional long to keep track of the index.
095 *
096 * <p>
097 * Note that caching is essential in making these loops essentially as fast as
098 * those scanning standard arrays (as iterations of the outer loop happen very
099 * rarely). Using loops of this kind is extremely faster than using a standard
100 * loop and accessors.
101 *
102 * <p>
103 * In some situations, you might want to iterate over a part of a big array
104 * having an offset and a length. In this case, the idiomatic loops are as
105 * follows:
106 *
107 * <pre>
108 * for(int s = segment(offset); s &lt; segment(offset + length + SEGMENT_MASK); s++) {
109 *     final int[] t = a[s];
110 *     final int l = (int)Math.min(t.length, offset + length - start(s));
111 *     for(int d = (int)Math.max(0, offset - start(s)); d &lt; l; d++) {
112 *         do something with t[d]
113 *     }
114 * }
115 * </pre>
116 *
117 * or, in a reversed form,
118 *
119 * <pre>
120 * for(int s = segment(offset + length + SEGMENT_MASK); s-- != segment(offset);) {
121 *     final int[] t = a[s];
122 *     final int b = (int)Math.max(0, offset - start(s));
123 *     for(int d = (int)Math.min(t.length, offset + length - start(s)); d-- != b ;) {
124 *         do something with t[d]
125 *     }
126 * }
127 * </pre>
128 *
129 * <h2>Literal big arrays</h2>
130 *
131 * <p>
132 * A literal big array can be easily created by using the suitable type-specific
133 * <code>wrap()</code> method (e.g., {@link IntBigArrays#wrap(int[])}) around a
134 * literal standard array. Alternatively, for very small arrays you can just
135 * declare a literal array-of-array (e.g., <code>new int[][] { { 1, 2 } }</code>
136 *). Be warned, however, that this can lead to creating illegal big arrays if
137 * for some reason (e.g., stress testing) {@link #SEGMENT_SIZE} is set to a
138 * value smaller than the inner array length.
139 *
140 * <h2>Big alternatives</h2>
141 *
142 * <p>
143 * If you find the kind of &ldquo;bare hands&rdquo; approach to big arrays not
144 * enough object-oriented, please use big lists based on big arrays (.e.g,
145 * {@link IntBigArrayBigList}). Big arrays follow the Java tradition of
146 * considering arrays as a &ldquo;legal alien&rdquo;&mdash;something in-between
147 * an object and a primitive type. This approach lacks the consistency of a full
148 * object-oriented approach, but provides some significant performance gains.
149 *
150 * <h2>Additional methods</h2>
151 *
152 * <p>
153 * In addition to commodity methods, this class contains {@link BigSwapper}
154 * -based implementations of
155 * {@linkplain #quickSort(long, long, LongComparator, BigSwapper) quicksort} and
156 * of a stable, in-place
157 * {@linkplain #mergeSort(long, long, LongComparator, BigSwapper) mergesort}.
158 * These generic sorting methods can be used to sort any kind of list, but they
159 * find their natural usage, for instance, in sorting big arrays in parallel.
160 *
161 * @see it.unimi.dsi.fastutil.Arrays
162 */
163
164public class BigArrays {
165        /**
166         * The shift used to compute the segment associated with an index
167         * (equivalently, the logarithm of the segment size).
168         */
169        public final static int SEGMENT_SHIFT = 27;
170        /**
171         * The current size of a segment (2<sup>27</sup>) is the largest size that
172         * makes the physical memory allocation for a single segment strictly
173         * smaller than 2<sup>31</sup> bytes.
174         */
175        public final static int SEGMENT_SIZE = 1 << SEGMENT_SHIFT;
176        /** The mask used to compute the displacement associated to an index. */
177        public final static int SEGMENT_MASK = SEGMENT_SIZE - 1;

179        protected BigArrays() {
180        }

182        /**
183         * Computes the segment associated with a given index.
184         *
185         * @param index
186         *            an index into a big array.
187         * @return the associated segment.
188         */
189        public static int segment(final long index) {
190                return (int) (index >>> SEGMENT_SHIFT);
191        }
192
193        /**
194         * Computes the displacement associated with a given index.
195         *
196         * @param index
197         *            an index into a big array.
198         * @return the associated displacement (in the associated
199         *         {@linkplain #segment(long) segment}).
200         */
201        public static int displacement(final long index) {
202                return (int) (index & SEGMENT_MASK);
203        }

205        /**
206         * Computes the starting index of a given segment.
207         *
208         * @param segment
209         *            the segment of a big array.
210         * @return the starting index of the segment.
211         */
212        public static long start(final int segment) {
213                return (long) segment << SEGMENT_SHIFT;
214        }

216        /**
217         * Computes the index associated with given segment and displacement.
218         *
219         * @param segment
220         *            the segment of a big array.
221         * @param displacement
222         *            the displacement into the segment.
223         * @return the associated index: that is, {@link #segment(long)
224         *         segment(index(segment, displacement)) == segment} and
225         *         {@link #displacement(long) displacement(index(segment,
226         *         displacement)) == displacement}.
227         */
228        public static long index(final int segment, final int displacement) {
229                return start(segment) + displacement;
230        }

232        /**
233         * Ensures that a range given by its first (inclusive) and last (exclusive)
234         * elements fits a big array of given length.
235         *
236         * <P>
237         * This method may be used whenever a big array range check is needed.
238         *
239         * @param bigArrayLength
240         *            a big-array length.
241         * @param from
242         *            a start index (inclusive).
243         * @param to
244         *            an end index (inclusive).
245         * @throws IllegalArgumentException
246         *             if <code>from</code> is greater than <code>to</code>.
247         * @throws ArrayIndexOutOfBoundsException
248         *             if <code>from</code> or <code>to</code> are greater than
249         *             <code>bigArrayLength</code> or negative.
250         */
251        public static void ensureFromTo(final long bigArrayLength, final long from, final long to) {
252                if (from < 0) throw new ArrayIndexOutOfBoundsException("Start index (" + from + ") is negative");
253                if (from > to) throw new IllegalArgumentException("Start index (" + from + ") is greater than end index (" + to + ")");
254                if (to > bigArrayLength) throw new ArrayIndexOutOfBoundsException("End index (" + to + ") is greater than big-array length (" + bigArrayLength + ")");
255        }

257        /**
258         * Ensures that a range given by an offset and a length fits a big array of
259         * given length.
260         *
261         * <P>
262         * This method may be used whenever a big array range check is needed.
263         *
264         * @param bigArrayLength
265         *            a big-array length.
266         * @param offset
267         *            a start index for the fragment
268         * @param length
269         *            a length (the number of elements in the fragment).
270         * @throws IllegalArgumentException
271         *             if <code>length</code> is negative.
272         * @throws ArrayIndexOutOfBoundsException
273         *             if <code>offset</code> is negative or <code>offset</code> +
274         *             <code>length</code> is greater than
275         *             <code>bigArrayLength</code>.
276         */
277        public static void ensureOffsetLength(final long bigArrayLength, final long offset, final long length) {
278                if (offset < 0) throw new ArrayIndexOutOfBoundsException("Offset (" + offset + ") is negative");
279                if (length < 0) throw new IllegalArgumentException("Length (" + length + ") is negative");
280                if (offset + length > bigArrayLength) throw new ArrayIndexOutOfBoundsException("Last index (" + (offset + length) + ") is greater than big-array length (" + bigArrayLength + ")");
281        }

284         * Ensures that a big-array length is legal.
285         *
286         * @param bigArrayLength
287         *            a big-array length.
288         * @throws IllegalArgumentException
289         *             if <code>length</code> is negative, or larger than or equal
290         *             to {@link #SEGMENT_SIZE} * {@link Integer#MAX_VALUE}.
291         */
292        public static void ensureLength(final long bigArrayLength) {
293                if (bigArrayLength < 0) throw new IllegalArgumentException("Negative big-array size: " + bigArrayLength);
294                if (bigArrayLength >= (long) Integer.MAX_VALUE << SEGMENT_SHIFT) throw new IllegalArgumentException("Big-array size too big: " + bigArrayLength);
295        }

297        private static final int SMALL = 7;
298        private static final int MEDIUM = 40;

300        /**
301         * Transforms two consecutive sorted ranges into a single sorted range. The
302         * initial ranges are <code>[first, middle)</code> and
303         * <code>[middle, last)</code>, and the resulting range is
304         * <code>[first, last)</code>. Elements in the first input range will
305         * precede equal elements in the second.
306         */
307        private static void inPlaceMerge(final long from, long mid, final long to, final LongComparator comp, final BigSwapper swapper) {
308                if (from >= mid || mid >= to) return;
309                if (to - from == 2) {
310                        if (comp.compare(mid, from) < 0) {
311                                swapper.swap(from, mid);
312                        }
313                        return;
314                }
315                long firstCut;
316                long secondCut;
317                if (mid - from > to - mid) {
318                        firstCut = from + (mid - from) / 2;
319                        secondCut = lowerBound(mid, to, firstCut, comp);
320                } else {
321                        secondCut = mid + (to - mid) / 2;
322                        firstCut = upperBound(from, mid, secondCut, comp);
323                }
324
325                long first2 = firstCut;
326                long middle2 = mid;
327                long last2 = secondCut;
328                if (middle2 != first2 && middle2 != last2) {
329                        long first1 = first2;
330                        long last1 = middle2;
331                        while (first1 < --last1)
332                                swapper.swap(first1++, last1);
333                        first1 = middle2;
334                        last1 = last2;
335                        while (first1 < --last1)
336                                swapper.swap(first1++, last1);
337                        first1 = first2;
338                        last1 = last2;
339                        while (first1 < --last1)
340                                swapper.swap(first1++, last1);
341                }
342
343                mid = firstCut + (secondCut - mid);
344                inPlaceMerge(from, firstCut, mid, comp, swapper);
345                inPlaceMerge(mid, secondCut, to, comp, swapper);
346        }

348        /**
349         * Performs a binary search on an already sorted range: finds the first
350         * position where an element can be inserted without violating the ordering.
351         * Sorting is by a user-supplied comparison function.
352         *
353         * @param mid
354         *            Beginning of the range.
355         * @param to
356         *            One past the end of the range.
357         * @param firstCut
358         *            Element to be searched for.
359         * @param comp
360         *            Comparison function.
361         * @return The largest index i such that, for every j in the range
362         *         <code>[first, i)</code>, <code>comp.apply(array[j], x)</code> is
363         *         <code>true</code>.
364         */
365        private static long lowerBound(long mid, final long to, final long firstCut, final LongComparator comp) {
366                long len = to - mid;
367                while (len > 0) {
368                        long half = len / 2;
369                        long middle = mid + half;
370                        if (comp.compare(middle, firstCut) < 0) {
371                                mid = middle + 1;
372                                len -= half + 1;
373                        } else {
374                                len = half;
375                        }
376                }
377                return mid;
378        }

380        /** Returns the index of the median of three elements. */
381        private static long med3(final long a, final long b, final long c, final LongComparator comp) {
382                final int ab = comp.compare(a, b);
383                final int ac = comp.compare(a, c);
384                final int bc = comp.compare(b, c);
385                return (ab < 0 ? (bc < 0 ? b : ac < 0 ? c : a) : (bc > 0 ? b : ac > 0 ? c : a));
386        }

388        /**
389         * Sorts the specified range of elements using the specified big swapper and
390         * according to the order induced by the specified comparator using
391         * mergesort.
392         *
393         * <p>
394         * This sort is guaranteed to be <i>stable</i>: equal elements will not be
395         * reordered as a result of the sort. The sorting algorithm is an in-place
396         * mergesort that is significantly slower than a standard mergesort, as its
397         * running time is
398         * <i>O</i>(<var>n</var>&nbsp;(log&nbsp;<var>n</var>)<sup>2</sup>), but it
399         * does not allocate additional memory; as a result, it can be used as a
400         * generic sorting algorithm.
401         *
402         * @param from
403         *            the index of the first element (inclusive) to be sorted.
404         * @param to
405         *            the index of the last element (exclusive) to be sorted.
406         * @param comp
407         *            the comparator to determine the order of the generic data
408         *            (arguments are positions).
409         * @param swapper
410         *            an object that knows how to swap the elements at any two
411         *            positions.
412         */
413        public static void mergeSort(final long from, final long to, final LongComparator comp, final BigSwapper swapper) {
414                final long length = to - from;
415
416                // Insertion sort on smallest arrays
417                if (length < SMALL) {
418                        for (long i = from; i < to; i++) {
419                                for (long j = i; j > from && (comp.compare(j - 1, j) > 0); j--) {
420                                        swapper.swap(j, j - 1);
421                                }
422                        }
423                        return;
424                }
425
426                // Recursively sort halves
427                long mid = (from + to) >>> 1;
428                mergeSort(from, mid, comp, swapper);
429                mergeSort(mid, to, comp, swapper);
430
431                // If list is already sorted, nothing left to do. This is an
432                // optimization that results in faster sorts for nearly ordered lists.
433                if (comp.compare(mid - 1, mid) <= 0) return;
434
435                // Merge sorted halves
436                inPlaceMerge(from, mid, to, comp, swapper);
437        }

439        /**
440         * Sorts the specified range of elements using the specified big swapper and
441         * according to the order induced by the specified comparator using
442         * quicksort.
443         *
444         * <p>
445         * The sorting algorithm is a tuned quicksort adapted from Jon L. Bentley
446         * and M. Douglas McIlroy, &ldquo;Engineering a Sort Function&rdquo;,
447         * <i>Software: Practice and Experience</i>, 23(11), pages 1249&minus;1265,
448         * 1993.
449         *
450         * @param from
451         *            the index of the first element (inclusive) to be sorted.
452         * @param to
453         *            the index of the last element (exclusive) to be sorted.
454         * @param comp
455         *            the comparator to determine the order of the generic data.
456         * @param swapper
457         *            an object that knows how to swap the elements at any two
458         *            positions.
459         */
460        public static void quickSort(final long from, final long to, final LongComparator comp, final BigSwapper swapper) {
461                final long len = to - from;
462                // Insertion sort on smallest arrays
463                if (len < SMALL) {
464                        for (long i = from; i < to; i++)
465                                for (long j = i; j > from && (comp.compare(j - 1, j) > 0); j--) {
466                                        swapper.swap(j, j - 1);
467                                }
468                        return;
469                }
470
471                // Choose a partition element, v
472                long m = from + len / 2; // Small arrays, middle element
473                if (len > SMALL) {
474                        long l = from, n = to - 1;
475                        if (len > MEDIUM) { // Big arrays, pseudomedian of 9
476                                long s = len / 8;
477                                l = med3(l, l + s, l + 2 * s, comp);
478                                m = med3(m - s, m, m + s, comp);
479                                n = med3(n - 2 * s, n - s, n, comp);
480                        }
481                        m = med3(l, m, n, comp); // Mid-size, med of 3
482                }
483                // long v = x[m];
484
485                long a = from, b = a, c = to - 1, d = c;
486                // Establish Invariant: v* (<v)* (>v)* v*
487                while (true) {
488                        int comparison;
489                        while (b <= c && ((comparison = comp.compare(b, m)) <= 0)) {
490                                if (comparison == 0) {
491                                        if (a == m) m = b; // moving target; DELTA to JDK !!!
492                                        else if (b == m) m = a; // moving target; DELTA to JDK !!!
493                                        swapper.swap(a++, b);
494                                }
495                                b++;
496                        }
497                        while (c >= b && ((comparison = comp.compare(c, m)) >= 0)) {
498                                if (comparison == 0) {
499                                        if (c == m) m = d; // moving target; DELTA to JDK !!!
500                                        else if (d == m) m = c; // moving target; DELTA to JDK !!!
501                                        swapper.swap(c, d--);
502                                }
503                                c--;
504                        }
505                        if (b > c) break;
506                        if (b == m) m = d; // moving target; DELTA to JDK !!!
507                        else if (c == m) m = c; // moving target; DELTA to JDK !!!
508                        swapper.swap(b++, c--);
509                }
510
511                // Swap partition elements back to middle
512                long s;
513                long n = from + len;
514                s = Math.min(a - from, b - a);
515                vecSwap(swapper, from, b - s, s);
516                s = Math.min(d - c, n - d - 1);
517                vecSwap(swapper, b, n - s, s);
518
519                // Recursively sort non-partition-elements
520                if ((s = b - a) > 1) quickSort(from, from + s, comp, swapper);
521                if ((s = d - c) > 1) quickSort(n - s, n, comp, swapper);
522        }

524        /**
525         * Performs a binary search on an already-sorted range: finds the last
526         * position where an element can be inserted without violating the ordering.
527         * Sorting is by a user-supplied comparison function.
528         *
529         * @param from
530         *            Beginning of the range.
531         * @param mid
532         *            One past the end of the range.
533         * @param secondCut
534         *            Element to be searched for.
535         * @param comp
536         *            Comparison function.
537         * @return The largest index i such that, for every j in the range
538         *         <code>[first, i)</code>, <code>comp.apply(x, array[j])</code> is
539         *         <code>false</code>.
540         */
541        private static long upperBound(long from, final long mid, final long secondCut, final LongComparator comp) {
542                long len = mid - from;
543                while (len > 0) {
544                        long half = len / 2;
545                        long middle = from + half;
546                        if (comp.compare(secondCut, middle) < 0) {
547                                len = half;
548                        } else {
549                                from = middle + 1;
550                                len -= half + 1;
551                        }
552                }
553                return from;
554        }
555
556        /** Swaps x[a .. (a+n-1)] with x[b .. (b+n-1)]. */
557        private static void vecSwap(final BigSwapper swapper, long from, long l, final long s) {
558                for (int i = 0; i < s; i++, from++, l++)
559                        swapper.swap(from, l);
560        }

562        public static void main(final String arg[]) {
563                int[][] a = IntBigArrays.newBigArray(1L << Integer.parseInt(arg[0]));
564                long x, y, z, start;
565
566                for (int k = 10; k-- != 0;) {
567
568                        start = -System.currentTimeMillis();
569
570                        x = 0;
571                        for (long i = IntBigArrays.length(a); i-- != 0;)
572                                x ^= i ^ IntBigArrays.get(a, i);
573                        if (x == 0) System.err.println();
574
575                        System.out.println("Single loop: " + (start + System.currentTimeMillis()) + "ms");
576
577                        start = -System.currentTimeMillis();
578
579                        y = 0;
580                        for (int i = a.length; i-- != 0;) {
581                                final int[] t = a[i];
582                                for (int d = t.length; d-- != 0;)
583                                        y ^= t[d] ^ index(i, d);
584                        }
585                        if (y == 0) System.err.println();
586                        if (x != y) throw new AssertionError();
587
588                        System.out.println("Double loop: " + (start + System.currentTimeMillis()) + "ms");
589
590                        z = 0;
591                        long j = IntBigArrays.length(a);
592                        for (int i = a.length; i-- != 0;) {
593                                final int[] t = a[i];
594                                for (int d = t.length; d-- != 0;)
595                                        y ^= t[d] ^ --j;
596                        }
597                        if (z == 0) System.err.println();
598                        if (x != z) throw new AssertionError();
599
600                        System.out.println("Double loop (with additional index): " + (start + System.currentTimeMillis()) + "ms");
601                }
602        }
