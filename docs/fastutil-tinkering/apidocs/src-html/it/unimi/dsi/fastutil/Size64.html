019import java.util.Collection;
020
021/** An interface for data structures whose size can exceed {@link Integer#MAX_VALUE}.
022 *
023 * <P>The only methods specified by this interfaces are {@link #size64()}, and
024 * a deprecated {@link #size()} identical to {@link Collection#size()}. Implementations
025 * can work around the type problem of {@link java.util.Collection#size()}
026 * (e.g., not being able to return more than {@link Integer#MAX_VALUE}) by implementing this
027 * interface. Callers interested in large structures
028 * can use a reflective call to <code>instanceof</code> to check for the presence of {@link #size64()}.
029 *
030 * <p>We remark that it is always a good idea to implement both {@link #size()} <em>and</em> {@link #size64()},
031 * as the former might be implemented by a superclass in an incompatible way. If you implement this interface,
032 * just implement {@link #size()} as a <em>deprecated</em> method returning <code>Math.min(Integer.MAX_VALUE, size64())</code>.
033 */
034
035public interface Size64 {
036        /** Returns the size of this data structure as a long.
037         *
038         * @return  the size of this data structure.
039         */
040        long size64();
041
042        /** Returns the size of this data structure, minimized with {@link Integer#MAX_VALUE}.
043         *
044         * @return the size of this data structure, minimized with {@link Integer#MAX_VALUE}.
045         * @see java.util.Collection#size()
046         * @deprecated Use {@link #size64()} instead.
047         */
048        @Deprecated
049        int size();
050}
