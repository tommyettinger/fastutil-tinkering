020import java.util.Comparator;
021import java.util.NoSuchElementException;
022
023/** A class providing static methods and objects that do useful things with indirect priority queues.
024 *
025 * @see IndirectPriorityQueue
026 */
027
028public class IndirectPriorityQueues {
029
030        private IndirectPriorityQueues() {}
031
032        /** An immutable class representing the empty indirect priority queue.
033         *
034         * <P>This class may be useful to implement your own in case you subclass
035         * {@link IndirectPriorityQueue}.
036         */
037
038        @SuppressWarnings("rawtypes")
039        public static class EmptyIndirectPriorityQueue extends AbstractIndirectPriorityQueue {
040
041                protected EmptyIndirectPriorityQueue() {}
042
043                public void enqueue(final int i) { throw new UnsupportedOperationException(); }
044                public int dequeue() { throw new NoSuchElementException(); }
045                public boolean isEmpty() { return true; }
046                public int size() { return 0; }
047                public boolean contains(int index) { return false; }
048                public void clear() {}
049                public int first() { throw new NoSuchElementException(); }
050                public int last() { throw new NoSuchElementException(); }
051                public void changed() { throw new NoSuchElementException(); }
052                public void allChanged() {}
053                public Comparator<?> comparator() { return null; }
054                public void changed(final int i) { throw new IllegalArgumentException("Index " + i + " is not in the queue"); }
055                public boolean remove(final int i) { return false; }
056                public int front(int[] a) { return 0; }
057
058        }
059
060        /** An empty indirect priority queue (immutable).
061         */
062
063        public final static EmptyIndirectPriorityQueue EMPTY_QUEUE = new EmptyIndirectPriorityQueue();
064
065
066        /** A synchronized wrapper class for indirect priority queues. */
067
068        public static class SynchronizedIndirectPriorityQueue<K> implements IndirectPriorityQueue<K> {
069
070                public static final long serialVersionUID = -7046029254386353129L;
071
072                final protected IndirectPriorityQueue<K> q;
073                final protected Object sync;
074
075                protected SynchronizedIndirectPriorityQueue(final IndirectPriorityQueue<K> q, final Object sync) {
076                        this.q = q;
077                        this.sync = sync;
078                }
079
080                protected SynchronizedIndirectPriorityQueue(final IndirectPriorityQueue<K> q) {
081                        this.q = q;
082                        this.sync = this;
083                }
084
085                public void enqueue(int x) { synchronized(sync) { q.enqueue(x); } }
086                public int dequeue() { synchronized(sync) { return q.dequeue(); } }
087                public boolean contains(final int index) { synchronized(sync) { return q.contains(index); } }
088                public int first() { synchronized(sync) { return q.first(); } }
089                public int last() { synchronized(sync) { return q.last(); } }
090                public boolean isEmpty() { synchronized(sync) { return q.isEmpty(); } }
091                public int size() { synchronized(sync) { return q.size(); } }
092                public void clear() { synchronized(sync) { q.clear(); } }
093                public void changed() { synchronized(sync) { q.changed(); } }
094                public void allChanged() { synchronized(sync) { q.allChanged(); } }
095                public void changed(int i) { synchronized(sync) { q.changed(i); } }
096                public boolean remove(int i) { synchronized(sync) { return q.remove(i); } }
097                public Comparator<? super K> comparator() { synchronized(sync) { return q.comparator(); } }
098                public int front(int[] a) { return q.front(a); }
099        }
100
101
102        /** Returns a synchronized type-specific indirect priority queue backed by the specified type-specific indirect priority queue.
103         *
104         * @param q the indirect priority queue to be wrapped in a synchronized indirect priority queue.
105         * @return a synchronized view of the specified indirect priority queue.
106         */
107        public static <K> IndirectPriorityQueue<K> synchronize(final IndirectPriorityQueue<K> q) {      return new SynchronizedIndirectPriorityQueue<K>(q); }
108
109        /** Returns a synchronized type-specific indirect priority queue backed by the specified type-specific indirect priority queue, using an assigned object to synchronize.
110         *
111         * @param q the indirect priority queue to be wrapped in a synchronized indirect priority queue.
112         * @param sync an object that will be used to synchronize the access to the indirect priority queue.
113         * @return a synchronized view of the specified indirect priority queue.
114         */
115
116        public static <K> IndirectPriorityQueue<K> synchronize(final IndirectPriorityQueue<K> q, final Object sync) { return new SynchronizedIndirectPriorityQueue<K>(q, sync); }
117
118}
