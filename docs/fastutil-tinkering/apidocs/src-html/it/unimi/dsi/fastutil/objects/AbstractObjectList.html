016package it.unimi.dsi.fastutil.objects;
017import it.unimi.dsi.fastutil.Stack;
018import java.util.List;
019import java.util.Iterator;
020import java.util.ListIterator;
021import java.util.Collection;
022import java.util.NoSuchElementException;
023/**  An abstract class providing basic methods for lists implementing a type-specific list interface.
024        *
025        * <P>As an additional bonus, this class implements on top of the list operations a type-specific stack.
026        */
027public abstract class AbstractObjectList <K> extends AbstractObjectCollection <K> implements ObjectList <K>, Stack <K> {
028        protected AbstractObjectList() {}
029        /** Ensures that the given index is nonnegative and not greater than the list size.
030         *
031         * @param index an index.
032         * @throws IndexOutOfBoundsException if the given index is negative or greater than the list size.
033         */
034        protected void ensureIndex(final int index) {
035         if (index < 0) throw new IndexOutOfBoundsException("Index (" + index + ") is negative");
036         if (index > size()) throw new IndexOutOfBoundsException("Index (" + index + ") is greater than list size (" + (size()) + ")");
037        }
038        /** Ensures that the given index is nonnegative and smaller than the list size.
039         *
040         * @param index an index.
041         * @throws IndexOutOfBoundsException if the given index is negative or not smaller than the list size.
042         */
043        protected void ensureRestrictedIndex(final int index) {
044         if (index < 0) throw new IndexOutOfBoundsException("Index (" + index + ") is negative");
045         if (index >= size()) throw new IndexOutOfBoundsException("Index (" + index + ") is greater than or equal to list size (" + (size()) + ")");
046        }
047        /** {@inheritDoc}
048         *
049         * <p>This implementation always throws an {@link UnsupportedOperationException}.
050         */
051        @Override
052        public void add(final int index, final K k) {
053         throw new UnsupportedOperationException();
054        }
055        /** {@inheritDoc}
056         *
057         * <p>This implementation delegates to the type-specific version of {@link List#add(int, Object)}.
058         */
059        @Override
060        public boolean add(final K k) {
061         add(size(), k);
062         return true;
063        }
064        /** {@inheritDoc}
065         *
066         * <p>This implementation always throws an {@link UnsupportedOperationException}.
067         */
068        @Override
069        public K remove(int i) {
070         throw new UnsupportedOperationException();
071        }
072        /** {@inheritDoc}
073         *
074         * <p>This implementation always throws an {@link UnsupportedOperationException}.
075         */
076        @Override
077        public K set(final int index, final K k) {
078         throw new UnsupportedOperationException();
079        }
080        /** Adds all of the elements in the specified collection to this list (optional operation). */
081        @Override
082        public boolean addAll(int index, final Collection<? extends K> c) {
083         ensureIndex(index);
084         final Iterator<? extends K> i = c.iterator();
085         final boolean retVal = i.hasNext();
086         while(i.hasNext()) add(index++, i.next());
087         return retVal;
088        }
089        /** {@inheritDoc}
090         *
091         * <p>This implementation delegates to the type-specific version of {@link List#addAll(int, Collection)}.
092         */
093        @Override
094        public boolean addAll(final Collection<? extends K> c) {
095         return addAll(size(), c);
096        }
097        /** {@inheritDoc}
098         *
099         * <p>This implementation delegates to the new covariantly stronger generic method  {@link #listIterator()}.
100         * @deprecated As of <code>fastutil</code> 5, replaced by {@link #listIterator()}.
101         */
102        @Deprecated
103        @Override
104        public ObjectListIterator <K> objectListIterator() {
105         return listIterator();
106        }
107        /** {@inheritDoc}
108         *
109         * <p>This implementation delegates to the new covariantly stronger generic method {@link #listIterator(int)}.
110         * @deprecated As of <code>fastutil</code> 5, replaced by {@link #listIterator()}.
111         */
112        @Deprecated
113        @Override
114        public ObjectListIterator <K> objectListIterator(final int index) {
115         return listIterator(index);
116        }
117        /** {@inheritDoc}
118         *
119         * <p>This implementation delegates to {@link #listIterator()}.
120         */
121        @Override
122        public ObjectListIterator <K> iterator() {
123         return listIterator();
124        }
125        /** {@inheritDoc}
126         *
127         * <p>This implementation delegates to {@link #listIterator(int) listIterator(0)}.
128         */
129        @Override
130        public ObjectListIterator <K> listIterator() {
131         return listIterator(0);
132        }
133        /** {@inheritDoc}
134         * <P>This implementation is based on the random-access methods. */
135        @Override
136        public ObjectListIterator <K> listIterator(final int index) {
137         ensureIndex(index);
138         return new AbstractObjectListIterator <K>() {
139           int pos = index, last = -1;
140           @Override
141           public boolean hasNext() { return pos < AbstractObjectList.this.size(); }
142           @Override
143           public boolean hasPrevious() { return pos > 0; }
144           @Override
145           public K next() { if (! hasNext()) throw new NoSuchElementException(); return AbstractObjectList.this.get(last = pos++); }
146           @Override
147           public K previous() { if (! hasPrevious()) throw new NoSuchElementException(); return AbstractObjectList.this.get(last = --pos); }
148           @Override
149           public int nextIndex() { return pos; }
150           @Override
151           public int previousIndex() { return pos - 1; }
152           @Override
153           public void add(K k) {
154            AbstractObjectList.this.add(pos++, k);
155            last = -1;
156           }
157           @Override
158           public void set(K k) {
159            if (last == -1) throw new IllegalStateException();
160            AbstractObjectList.this.set(last, k);
161           }
162           @Override
163           public void remove() {
164            if (last == -1) throw new IllegalStateException();
165            AbstractObjectList.this.remove(last);
166            /* If the last operation was a next(), we are removing an element *before* us, and we must decrease pos correspondingly. */
167            if (last < pos) pos--;
168            last = -1;
169           }
170          };
171        }
172        /** Returns true if this list contains the specified element.
173         * <p>This implementation delegates to {@code indexOf()}.
174         * @see List#contains(Object)
175         */
176        @Override
177        public boolean contains(final Object k) {
178         return indexOf(k) >= 0;
179        }
180        @Override
181        public int indexOf(final Object k) {
182         final ObjectListIterator <K> i = listIterator();
183         K e;
184         while(i.hasNext()) {
185          e = i.next();
186          if (( (k) == null ? (e) == null : (k).equals(e) )) return i.previousIndex();
187         }
188         return -1;
189        }
190        @Override
191        public int lastIndexOf(final Object k) {
192         ObjectListIterator <K> i = listIterator(size());
193         K e;
194         while(i.hasPrevious()) {
195          e = i.previous();
196          if (( (k) == null ? (e) == null : (k).equals(e) )) return i.nextIndex();
197         }
198         return -1;
199        }
200        @Override
201        public void size(final int size) {
202         int i = size();
203         if (size > i) while(i++ < size) add((null));
204         else while(i-- != size) remove(i);
205        }
206        @Override
207        public ObjectList <K> subList(final int from, final int to) {
208         ensureIndex(from);
209         ensureIndex(to);
210         if (from > to) throw new IndexOutOfBoundsException("Start index (" + from + ") is greater than end index (" + to + ")");
211         return new ObjectSubList <K>(this, from, to);
212        }
213        /** {@inheritDoc}
214         *
215         * <p>This implementation delegates to the new covariantly stronger generic method  {@link #subList(int, int)}.
216         * @deprecated As of <code>fastutil</code> 5, replaced by {@link #subList(int, int)}.
217         */

223        /** {@inheritDoc}
224         *
225         * <P>This is a trivial iterator-based implementation. It is expected that
226         * implementations will override this method with a more optimized version.
227         */
228        @Override
229        public void removeElements(final int from, final int to) {
230         ensureIndex(to);
231         ObjectListIterator <K> i = listIterator(from);
232         int n = to - from;
233         if (n < 0) throw new IllegalArgumentException("Start index (" + from + ") is greater than end index (" + to + ")");
234         while(n-- != 0) {
235          i.next();
236          i.remove();
237         }
238        }
239        /** {@inheritDoc}
240         *
241         * <P>This is a trivial iterator-based implementation. It is expected that
242         * implementations will override this method with a more optimized version.
243         */
244        @Override
245        public void addElements(int index, final K a[], int offset, int length) {
246         ensureIndex(index);
247         if (offset < 0) throw new ArrayIndexOutOfBoundsException("Offset (" + offset + ") is negative");
248         if (offset + length > a.length) throw new ArrayIndexOutOfBoundsException("End index (" + (offset + length) + ") is greater than array length (" + a.length + ")");
249         while(length-- != 0) add(index++, a[offset++]);
250        }
251        /** {@inheritDoc}
252         *
253         * <P>This implementation delegates to the analogous method for array fragments.
254         */
255        @Override
256        public void addElements(final int index, final K a[]) {
257         addElements(index, a, 0, a.length);
258        }
259        /** {@inheritDoc}
260         *
261         * <P>This is a trivial iterator-based implementation. It is expected that
262         * implementations will override this method with a more optimized version.
263         */
264        @Override
265        public void getElements(final int from, final Object a[], int offset, int length) {
266         ObjectListIterator <K> i = listIterator(from);
267         if (offset < 0) throw new ArrayIndexOutOfBoundsException("Offset (" + offset + ") is negative");
268         if (offset + length > a.length) throw new ArrayIndexOutOfBoundsException("End index (" + (offset + length) + ") is greater than array length (" + a.length + ")");
269         if (from + length > size()) throw new IndexOutOfBoundsException("End index (" + (from + length) + ") is greater than list size (" + size() + ")");
270         while(length-- != 0) a[offset++] = i.next();
271        }
272        /** {@inheritDoc}
273         * <P>This implementation delegates to {@link #removeElements(int, int)}.*/
274        @Override
275        public void clear() {
276         removeElements(0, size());
277        }
278        private boolean valEquals(final Object a, final Object b) {
279         return a == null ? b == null : a.equals(b);
280        }
281        /** Returns the hash code for this list, which is identical to {@link java.util.List#hashCode()}.
282         *
283         * @return the hash code for this list.
284         */
285        @Override
286        public int hashCode() {
287         ObjectIterator <K> i = iterator();
288         int h = 1, s = size();
289         while (s-- != 0) {
290          K k = i.next();
291          h = 31 * h + ( (k) == null ? 0 : (k).hashCode() );
292         }
293         return h;
294        }
295        @Override
296        public boolean equals(final Object o) {
297         if (o == this) return true;
298         if (! (o instanceof List)) return false;
299         final List<?> l = (List<?>)o;
300         int s = size();
301         if (s != l.size()) return false;
302         final ListIterator<?> i1 = listIterator(), i2 = l.listIterator();
303         while(s-- != 0) if (! valEquals(i1.next(), i2.next())) return false;
304         return true;
305        }
306        /** Compares this list to another object. If the
307         * argument is a {@link java.util.List}, this method performs a lexicographical comparison; otherwise,
308         * it throws a <code>ClassCastException</code>.
309         *
310         * @param l a list.
311         * @return if the argument is a {@link java.util.List}, a negative integer,
312         * zero, or a positive integer as this list is lexicographically less than, equal
313         * to, or greater than the argument.
314         * @throws ClassCastException if the argument is not a list.
315         */
316        @SuppressWarnings("unchecked")
317        @Override
318        public int compareTo(final List<? extends K> l) {
319         if (l == this) return 0;
320         if (l instanceof ObjectList) {
321          final ObjectListIterator <K> i1 = listIterator(), i2 = ((ObjectList <K>)l).listIterator();
322          int r;
323          K e1, e2;
324          while(i1.hasNext() && i2.hasNext()) {
325           e1 = i1.next();
326           e2 = i2.next();
327           if ((r = ( ((Comparable<K>)(e1)).compareTo(e2) )) != 0) return r;
328          }
329          return i2.hasNext() ? -1 : (i1.hasNext() ? 1 : 0);
330         }
331         ListIterator<? extends K> i1 = listIterator(), i2 = l.listIterator();
332         int r;
333         while(i1.hasNext() && i2.hasNext()) {
334          if ((r = ((Comparable<? super K>)i1.next()).compareTo(i2.next())) != 0) return r;
335         }
336         return i2.hasNext() ? -1 : (i1.hasNext() ? 1 : 0);
337        }
338        @Override
339        public void push(K o) {
340         add(o);
341        }
342        @Override
343        public K pop() {
344         if (isEmpty()) throw new NoSuchElementException();
345         return remove(size() - 1);
346        }
347        @Override
348        public K top() {
349         if (isEmpty()) throw new NoSuchElementException();
350         return get(size() - 1);
351        }
352        @Override
353        public K peek(int i) {
354         return get(size() - 1 - i);
355        }
356        @Override
357        public String toString() {
358         final StringBuilder s = new StringBuilder();
359         final ObjectIterator <K> i = iterator();
360         int n = size();
361         K k;
362         boolean first = true;
363         s.append("[");
364         while(n-- != 0) {
365          if (first) first = false;
366          else s.append(", ");
367          k = i.next();
368          if (this == k) s.append("(this list)"); else
369           s.append(String.valueOf(k));
370         }
371         s.append("]");
372         return s.toString();
373        }
374        /** A class implementing a sublist view. */
375        public static class ObjectSubList <K> extends AbstractObjectList <K> implements java.io.Serializable {
376         private static final long serialVersionUID = -7046029254386353129L;
377         /** The list this sublist restricts. */
378         protected final ObjectList <K> l;
379         /** Initial (inclusive) index of this sublist. */
380         protected final int from;
381         /** Final (exclusive) index of this sublist. */
382         protected int to;
383         public ObjectSubList(final ObjectList <K> l, final int from, final int to) {
384          this.l = l;
385          this.from = from;
386          this.to = to;
387         }
388         private boolean assertRange() {
389          assert from <= l.size();
390          assert to <= l.size();
391          assert to >= from;
392          return true;
393         }
394         @Override
395         public boolean add(final K k) {
396          l.add(to, k);
397          to++;
398          assert assertRange();
399          return true;
400         }
401         @Override
402         public void add(final int index, final K k) {
403          ensureIndex(index);
404          l.add(from + index, k);
405          to++;
406          assert assertRange();
407         }
408         @Override
409         public boolean addAll(final int index, final Collection<? extends K> c) {
410          ensureIndex(index);
411          to += c.size();
412          return l.addAll(from + index, c);
413         }
414         @Override
415         public K get(int index) {
416          ensureRestrictedIndex(index);
417          return l.get(from + index);
418         }
419         @Override
420         public K remove(int index) {
421          ensureRestrictedIndex(index);
422          to--;
423          return l.remove(from + index);
424         }
425         @Override
426         public K set(int index, K k) {
427          ensureRestrictedIndex(index);
428          return l.set(from + index, k);
429         }
430         @Override
431         public int size() {
432          return to - from;
433         }
434         @Override
435         public void getElements(final int from, final Object[] a, final int offset, final int length) {
436          ensureIndex(from);
437          if (from + length > size()) throw new IndexOutOfBoundsException("End index (" + from + length + ") is greater than list size (" + size() + ")");
438          l.getElements(this.from + from, a, offset, length);
439         }
440         @Override
441         public void removeElements(final int from, final int to) {
442          ensureIndex(from);
443          ensureIndex(to);
444          l.removeElements(this.from + from, this.from + to);
445          this.to -= (to - from);
446          assert assertRange();
447         }
448         @Override
449         public void addElements(int index, final K a[], int offset, int length) {
450          ensureIndex(index);
451          l.addElements(this.from + index, a, offset, length);
452          this.to += length;
453          assert assertRange();
454         }
455         @Override
456         public ObjectListIterator <K> listIterator(final int index) {
457          ensureIndex(index);
458          return new AbstractObjectListIterator <K>() {
459            int pos = index, last = -1;
460            public boolean hasNext() { return pos < size(); }
461            public boolean hasPrevious() { return pos > 0; }
462            public K next() { if (! hasNext()) throw new NoSuchElementException(); return l.get(from + (last = pos++)); }
463            public K previous() { if (! hasPrevious()) throw new NoSuchElementException(); return l.get(from + (last = --pos)); }
464            public int nextIndex() { return pos; }
465            public int previousIndex() { return pos - 1; }
466            public void add(K k) {
467             if (last == -1) throw new IllegalStateException();
468             ObjectSubList.this.add(pos++, k);
469             last = -1;
470             assert assertRange();
471            }
472            public void set(K k) {
473             if (last == -1) throw new IllegalStateException();
474             ObjectSubList.this.set(last, k);
475            }
476            public void remove() {
477             if (last == -1) throw new IllegalStateException();
478             ObjectSubList.this.remove(last);
479             /* If the last operation was a next(), we are removing an element *before* us, and we must decrease pos correspondingly. */
480             if (last < pos) pos--;
481             last = -1;
482             assert assertRange();
483            }
484           };
485         }
486         @Override
487         public ObjectList <K> subList(final int from, final int to) {
488          ensureIndex(from);
489          ensureIndex(to);
490          if (from > to) throw new IllegalArgumentException("Start index (" + from + ") is greater than end index (" + to + ")");
491          return new ObjectSubList <K>(this, from, to);
492         }
493        }
