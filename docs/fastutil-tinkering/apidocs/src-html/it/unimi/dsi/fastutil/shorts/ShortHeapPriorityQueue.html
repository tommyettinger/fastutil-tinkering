016package it.unimi.dsi.fastutil.shorts;
017import java.util.Iterator;
018import java.util.Collection;
019import java.util.NoSuchElementException;
020/** A type-specific heap-based priority queue.
021        *
022        * <P>Instances of this class represent a priority queue using a heap. The heap is enlarged as needed, but
023        * it is never shrunk. Use the {@link #trim()} method to reduce its size, if necessary.
024        */
025public class ShortHeapPriorityQueue extends AbstractShortPriorityQueue implements java.io.Serializable {
026        private static final long serialVersionUID = 1L;
027        /** The heap array. */
028
029        protected transient short[] heap = ShortArrays.EMPTY_ARRAY;
030        /** The number of elements in this queue. */
031        protected int size;
032        /** The type-specific comparator used in this queue. */
033        protected ShortComparator c;
034        /** Creates a new empty queue with a given capacity and comparator.
035         *
036         * @param capacity the initial capacity of this queue.
037         * @param c the comparator used in this queue, or {@code null} for the natural order.
038         */
039
040        public ShortHeapPriorityQueue(int capacity, ShortComparator c) {
041         if (capacity > 0) this.heap = new short[capacity];
042         this.c = c;
043        }
044        /** Creates a new empty queue with a given capacity and using the natural order.
045         *
046         * @param capacity the initial capacity of this queue.
047         */
048        public ShortHeapPriorityQueue(int capacity) {
049         this(capacity, null);
050        }
051        /** Creates a new empty queue with a given comparator.
052         *
053         * @param c the comparator used in this queue, or {@code null} for the natural order.
054         */
055        public ShortHeapPriorityQueue(ShortComparator c) {
056         this(0, c);
057        }
058        /** Creates a new empty queue using the natural order.
059         */
060        public ShortHeapPriorityQueue() {
061         this(0, null);
062        }
063        /** Wraps a given array in a queue using a given comparator.
064         *
065         * <P>The queue returned by this method will be backed by the given array.
066         * The first <code>size</code> element of the array will be rearranged so to form a heap (this is
067         * more efficient than enqueing the elements of <code>a</code> one by one).
068         *
069         * @param a an array.
070         * @param size the number of elements to be included in the queue.
071         * @param c the comparator used in this queue, or {@code null} for the natural order.
072         */
073        public ShortHeapPriorityQueue(final short[] a, int size, final ShortComparator c) {
074         this(c);
075         this.heap = a;
076         this.size = size;
077         ShortHeaps.makeHeap(a, size, c);
078        }
079        /** Wraps a given array in a queue using a given comparator.
080         *
081         * <P>The queue returned by this method will be backed by the given array.
082         * The elements of the array will be rearranged so to form a heap (this is
083         * more efficient than enqueing the elements of <code>a</code> one by one).
084         *
085         * @param a an array.
086         * @param c the comparator used in this queue, or {@code null} for the natural order.
087         */
088        public ShortHeapPriorityQueue(final short[] a, final ShortComparator c) {
089         this(a, a.length, c);
090        }
091        /** Wraps a given array in a queue using the natural order.
092         *
093         * <P>The queue returned by this method will be backed by the given array.
094         * The first <code>size</code> element of the array will be rearranged so to form a heap (this is
095         * more efficient than enqueing the elements of <code>a</code> one by one).
096         *
097         * @param a an array.
098         * @param size the number of elements to be included in the queue.
099         */
100        public ShortHeapPriorityQueue(final short[] a, int size) {
101         this(a, size, null);
102        }
103        /** Wraps a given array in a queue using the natural order.
104         *
105         * <P>The queue returned by this method will be backed by the given array.
106         * The elements of the array will be rearranged so to form a heap (this is
107         * more efficient than enqueing the elements of <code>a</code> one by one).
108         *
109         * @param a an array.
110         */
111        public ShortHeapPriorityQueue(final short[] a) {
112         this(a, a.length);
113        }
114        /** Creates a queue using the elements in a type-specific collection using a given comparator.
115         *
116         * <P>This constructor is more efficient than enqueing the elements of <code>collection</code> one by one.
117         *
118         * @param collection a collection; its elements will be used to initialize the queue.
119         * @param c the comparator used in this queue, or {@code null} for the natural order.
120         */
121        public ShortHeapPriorityQueue(final ShortCollection collection, final ShortComparator c) {
122         this(collection.toShortArray(), c);
123        }
124        /** Creates a queue using the elements in a type-specific collection using the natural order.
125         *
126         * <P>This constructor is
127         * more efficient than enqueing the elements of <code>collection</code> one by one.
128         *
129         * @param collection a collection; its elements will be used to initialize the queue.
130         */
131        public ShortHeapPriorityQueue(final ShortCollection collection) {
132         this(collection, null);
133        }
134        /** Creates a queue using the elements in a collection using a given comparator.
135         *
136         * <P>This constructor is more efficient than enqueing the elements of <code>collection</code> one by one.
137         *
138         * @param collection a collection; its elements will be used to initialize the queue.
139         * @param c the comparator used in this queue, or {@code null} for the natural order.
140         */
141        public ShortHeapPriorityQueue(final Collection<? extends Short> collection, final ShortComparator c) {
142         this(collection.size(), c);
143         final Iterator<? extends Short> iterator = collection.iterator();
144         final int size = collection.size();
145         for(int i = 0 ; i < size; i++) heap[i] = ((((Short)(iterator.next())).shortValue()));
146        }
147        /** Creates a queue using the elements in a collection using the natural order.
148         *
149         * <P>This constructor is
150         * more efficient than enqueing the elements of <code>collection</code> one by one.
151         *
152         * @param collection a collection; its elements will be used to initialize the queue.
153         */
154        public ShortHeapPriorityQueue(final Collection<? extends Short> collection) {
155         this(collection, null);
156        }
157        @Override
158         public void enqueue(short x) {
159         if (size == heap.length) heap = ShortArrays.grow(heap, size + 1);
160         heap[size++] = x;
161         ShortHeaps.upHeap(heap, size, size - 1, c);
162        }
163        @Override
164        public short dequeueShort() {
165         if (size == 0) throw new NoSuchElementException();
166         final short result = heap[0];
167         heap[0] = heap[--size];
168         if (size != 0) ShortHeaps.downHeap(heap, size, 0, c);
169         return result;
170        }
171        @Override
172        public short firstShort() {
173         if (size == 0) throw new NoSuchElementException();
174         return heap[0];
175        }
176        @Override
177        public void changed() {
178         ShortHeaps.downHeap(heap, size, 0, c);
179        }
180        @Override
181        public int size() { return size; }
182        @Override
183        public void clear() {
184         size = 0;
185        }
186        /** Trims the underlying heap array so that it has exactly {@link #size()} elements. */
187        public void trim() {
188         heap = ShortArrays.trim(heap, size);
189        }
190        @Override
191        public ShortComparator comparator() { return c; }
192        private void writeObject(java.io.ObjectOutputStream s) throws java.io.IOException {
193         s.defaultWriteObject();
194         s.writeInt(heap.length);
195         for(int i = 0; i < size; i++) s.writeShort(heap[i]);
196        }
197
198        private void readObject(java.io.ObjectInputStream s) throws java.io.IOException, ClassNotFoundException {
199         s.defaultReadObject();
200         heap = new short[s.readInt()];
201         for(int i = 0; i < size; i++) heap[i] = s.readShort();
202        }
203}
