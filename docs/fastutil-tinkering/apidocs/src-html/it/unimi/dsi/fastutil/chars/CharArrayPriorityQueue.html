016package it.unimi.dsi.fastutil.chars;
017import java.util.NoSuchElementException;
018/** A type-specific array-based priority queue.
019        *
020        * <P>Instances of this class represent a priority queue using a backing
021        * array&mdash;all operations are performed directly on the array. The array is
022        * enlarged as needed, but it is never shrunk. Use the {@link #trim()} method
023        * to reduce its size, if necessary.
024        *
025        * <P>This implementation is extremely inefficient, but it is difficult to beat
026        * when the size of the queue is very small.
027        */
028public class CharArrayPriorityQueue extends AbstractCharPriorityQueue implements java.io.Serializable {
029        private static final long serialVersionUID = 1L;
030        /** The backing array. */
031
032        protected transient char array[] = CharArrays.EMPTY_ARRAY;
033        /** The number of elements in this queue. */
034        protected int size;
035        /** The type-specific comparator used in this queue. */
036        protected CharComparator c;
037        /** The first index, cached, if {@link #firstIndexValid} is true. */
038        transient protected int firstIndex;
039        /** Whether {@link #firstIndex} contains a valid value. */
040        transient protected boolean firstIndexValid;
041        /** Creates a new empty queue with a given capacity and comparator.
042         *
043         * @param capacity the initial capacity of this queue.
044         * @param c the comparator used in this queue, or {@code null} for the natural order.
045         */
046
047        public CharArrayPriorityQueue(int capacity, CharComparator c) {
048         if (capacity > 0) this.array = new char[capacity];
049         this.c = c;
050        }
051        /** Creates a new empty queue with a given capacity and using the natural order.
052         *
053         * @param capacity the initial capacity of this queue.
054         */
055        public CharArrayPriorityQueue(int capacity) {
056         this(capacity, null);
057        }
058        /** Creates a new empty queue with a given comparator.
059         *
060         * @param c the comparator used in this queue, or {@code null} for the natural order.
061         */
062        public CharArrayPriorityQueue(CharComparator c) {
063         this(0, c);
064        }
065        /** Creates a new empty queue using the natural order.
066         */
067        public CharArrayPriorityQueue() {
068         this(0, null);
069        }
070        /** Wraps a given array in a queue using a given comparator.
071         *
072         * <P>The queue returned by this method will be backed by the given array.
073         *
074         * @param a an array.
075         * @param size the number of elements to be included in the queue.
076         * @param c the comparator used in this queue, or {@code null} for the natural order.
077         */
078        public CharArrayPriorityQueue(final char[] a, int size, final CharComparator c) {
079         this(c);
080         this.array = a;
081         this.size = size;
082        }
083        /** Wraps a given array in a queue using a given comparator.
084         *
085         * <P>The queue returned by this method will be backed by the given array.
086         *
087         * @param a an array.
088         * @param c the comparator used in this queue, or {@code null} for the natural order.
089         */
090        public CharArrayPriorityQueue(final char[] a, final CharComparator c) {
091         this(a, a.length, c);
092        }
093        /** Wraps a given array in a queue using the natural order.
094         *
095         * <P>The queue returned by this method will be backed by the given array.
096         *
097         * @param a an array.
098         * @param size the number of elements to be included in the queue.
099         */
100        public CharArrayPriorityQueue(final char[] a, int size) {
101         this(a, size, null);
102        }
103        /** Wraps a given array in a queue using the natural order.
104         *
105         * <P>The queue returned by this method will be backed by the given array.
106         *
107         * @param a an array.
108         */
109        public CharArrayPriorityQueue(final char[] a) {
110         this(a, a.length);
111        }
112        /** Returns the index of the smallest element. */
113
114        private int findFirst() {
115         if (firstIndexValid) return this.firstIndex;
116         firstIndexValid = true;
117         int i = size;
118         int firstIndex = --i;
119         char first = array[firstIndex];
120         if (c == null) { while(i-- != 0) if (( (array[i]) < (first) )) first = array[firstIndex = i]; }
121         else while(i-- != 0) { if (c.compare(array[i], first) < 0) first = array[firstIndex = i]; }
122         return this.firstIndex = firstIndex;
123        }
124        private void ensureNonEmpty() {
125         if (size == 0) throw new NoSuchElementException();
126        }
127        @Override
128
129        public void enqueue(char x) {
130         if (size == array.length) array = CharArrays.grow(array, size + 1);
131         if (firstIndexValid) {
132          if (c == null) { if (( (x) < (array[firstIndex]) )) firstIndex = size; }
133          else if (c.compare(x, array[firstIndex]) < 0) firstIndex = size;
134         }
135         else firstIndexValid = false;
136         array[size++] = x;
137        }
138        @Override
139        public char dequeueChar() {
140         ensureNonEmpty();
141         final int first = findFirst();
142         final char result = array[first];
143         System.arraycopy(array, first + 1, array, first, --size - first);
144         firstIndexValid = false;
145         return result;
146        }
147        @Override
148        public char firstChar() {
149         ensureNonEmpty();
150         return array[findFirst()];
151        }
152        @Override
153        public void changed() {
154         ensureNonEmpty();
155         firstIndexValid = false;
156        }
157        @Override
158        public int size() { return size; }
159        @Override
160        public void clear() {
161         size = 0;
162         firstIndexValid = false;
163        }
164        /** Trims the underlying array so that it has exactly {@link #size()} elements. */
165        public void trim() {
166         array = CharArrays.trim(array, size);
167        }
168        @Override
169        public CharComparator comparator() { return c; }
170        private void writeObject(java.io.ObjectOutputStream s) throws java.io.IOException {
171         s.defaultWriteObject();
172         s.writeInt(array.length);
173         for(int i = 0; i < size; i++) s.writeChar(array[i]);
174        }
175
176        private void readObject(java.io.ObjectInputStream s) throws java.io.IOException, ClassNotFoundException {
177         s.defaultReadObject();
178         array = new char[s.readInt()];
179         for(int i = 0; i < size; i++) array[i] = s.readChar();
180        }
181}
