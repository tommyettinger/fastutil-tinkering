027        /** Shuffles the specified list using the specified pseudorandom number generator.
028         *
029         * @param l the list to be shuffled.
030         * @param random a pseudorandom number generator (please use a <a href="http://dsiutils.dsi.unimi.it/docs/it/unimi/dsi/util/XorShiftStarRandom.html">XorShift*</a> generator).
031         * @return <code>l</code>.
032         */
033        public static <K> ObjectList <K> shuffle(final ObjectList <K> l, final Random random) {
034         for(int i = l.size(); i-- != 0;) {
035          final int p = random.nextInt(i + 1);
036          final K t = l.get(i);
037          l.set(i, l.get(p));
038          l.set(p, t);
039         }
040         return l;
041        }

133        public static class Singleton <K> extends AbstractObjectList <K> implements RandomAccess, java.io.Serializable, Cloneable {
134         private static final long serialVersionUID = -7046029254386353129L;
135         private final K element;
136         private Singleton(final K element) {
137          this.element = element;
138         }
139         @Override
140         public K get(final int i) { if (i == 0) return element; throw new IndexOutOfBoundsException(); }
141         @Override
142         public boolean remove(Object k) { throw new UnsupportedOperationException(); }
143         @Override
144         public K remove(final int i) { throw new UnsupportedOperationException(); }
145         @Override
146         public boolean contains(final Object k) { return ( (k) == null ? (element) == null : (k).equals(element) ); }
147         /* Slightly optimized w.r.t. the one in ABSTRACT_SET. */
148         @Override
149         public Object[] toArray() {
150          Object a[] = new Object[1];
151          a[0] = element;
152          return a;
153         }
154         @Override
155         public ObjectListIterator <K> listIterator() { return ObjectIterators.singleton(element); }
156         @Override
157         public ObjectListIterator <K> iterator() { return listIterator(); }
158         @Override
159         public ObjectListIterator <K> listIterator(int i) {
160          if (i > 1 || i < 0) throw new IndexOutOfBoundsException();
161          ObjectListIterator <K> l = listIterator();
162          if (i == 1) l.next();
163          return l;
164         }
165         @Override
166         @SuppressWarnings("unchecked")
167         public ObjectList <K> subList(final int from, final int to) {
168          ensureIndex(from);
169          ensureIndex(to);
170          if (from > to) throw new IndexOutOfBoundsException("Start index (" + from + ") is greater than end index (" + to + ")");
171          if (from != 0 || to != 1) return EMPTY_LIST;
172          return this;
173         }
174         @Override
175         public boolean addAll(int i, Collection<? extends K> c) { throw new UnsupportedOperationException(); }
176         @Override
177         public boolean addAll(final Collection<? extends K> c) { throw new UnsupportedOperationException(); }
178         @Override
179         public boolean removeAll(final Collection<?> c) { throw new UnsupportedOperationException(); }
180         @Override
181         public boolean retainAll(final Collection<?> c) { throw new UnsupportedOperationException(); }
182         @Override
183         public int size() { return 1; }
184         @Override
185         public void size(final int size) { throw new UnsupportedOperationException(); }
186         @Override
187         public void clear() { throw new UnsupportedOperationException(); }
188         @Override
189         public Object clone() { return this; }
190        }

197        /** A synchronized wrapper class for lists. */

204         }
205         protected SynchronizedList(final ObjectList <K> l) {
206          super(l);
207          this.list = l;
208         }
209         @Override

211         @Override
212         public K set(final int i, final K k) { synchronized(sync) { return list.set(i, k); } }
213         @Override
214         public void add(final int i, final K k) { synchronized(sync) { list.add(i, k); } }
215         @Override
216         public K remove(final int i) { synchronized(sync) { return list.remove(i); } }

220         public int lastIndexOf(final Object k) { synchronized(sync) { return list.lastIndexOf(k); } }
221         @Override

223         @Override
224         public void getElements(final int from, final Object a[], final int offset, final int length) { synchronized(sync) { list.getElements(from, a, offset, length); } }
225         @Override
226         public void removeElements(final int from, final int to) { synchronized(sync) { list.removeElements(from, to); } }
227         @Override
228         public void addElements(int index, final K a[], int offset, int length) { synchronized(sync) { list.addElements(index, a, offset, length); } }
229         @Override
230         public void addElements(int index, final K a[]) { synchronized(sync) { list.addElements(index, a); } }
231         @Override
232         public void size(final int size) { synchronized(sync) { list.size(size); } }
233         @Override
234         public ObjectListIterator <K> listIterator() { return list.listIterator(); }
235         @Override
236         public ObjectListIterator <K> iterator() { return listIterator(); }
237         @Override

239         /** Returns a type-specific list iterator on the list.
240                 *
241                 * @see #listIterator()
242                 * @deprecated As of <code>fastutil</code> 5, replaced by {@link #listIterator()}.
243                 */
244         @Deprecated
245         @Override
246         public ObjectListIterator <K> objectListIterator() { return listIterator(); }
247         /** Returns a type-specific list iterator on the list.
248                 *
249                 * @see #listIterator()

251                 */
252         @Deprecated
253         @Override
254         public ObjectListIterator <K> objectListIterator(final int i) { return listIterator(i); }
255         @Override
256         public ObjectList <K> subList(final int from, final int to) { synchronized(sync) { return new SynchronizedList <K>(list.subList(from, to), sync); } }
257         /* {@inheritDoc}
258                 * @deprecated As of <code>fastutil</code> 5, replaced by {@link #subList(int,int)}.
259                 */
260         @Deprecated
261         @Override
262         public ObjectList <K> objectSubList(final int from, final int to) { return subList(from, to); }
263         @Override
264         public boolean equals(final Object o) { if (o == this) return true; synchronized(sync) { return collection.equals(o); } }
265         @Override
266         public int hashCode() { synchronized(sync) { return collection.hashCode(); } }
267         @Override
268         public int compareTo(final List<? extends K> o) { synchronized(sync) { return list.compareTo(o); } }
269         private void writeObject(java.io.ObjectOutputStream s) throws java.io.IOException {
270          synchronized(sync) { s.defaultWriteObject(); }
271         }
272        }

285        /** Returns a synchronized type-specific list backed by the given type-specific list.
286         *
287         * @param l the list to be wrapped in a synchronized list.
288         * @return a synchronized view of the specified list.
289         * @see java.util.Collections#synchronizedList(List)
