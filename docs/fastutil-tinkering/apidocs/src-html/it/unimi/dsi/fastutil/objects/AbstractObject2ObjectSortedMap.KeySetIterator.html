016package it.unimi.dsi.fastutil.objects;
017import it.unimi.dsi.fastutil.objects.ObjectCollection;
018import it.unimi.dsi.fastutil.objects.AbstractObjectCollection;
019import it.unimi.dsi.fastutil.objects.AbstractObjectIterator;
020import it.unimi.dsi.fastutil.objects.ObjectIterator;
021import it.unimi.dsi.fastutil.objects.ObjectBidirectionalIterator;
022import it.unimi.dsi.fastutil.objects.ObjectSortedSet;
023import java.util.Map;
024import java.util.Comparator;
025/** An abstract class providing basic methods for sorted maps implementing a type-specific interface. */
026public abstract class AbstractObject2ObjectSortedMap <K,V> extends AbstractObject2ObjectMap <K,V> implements Object2ObjectSortedMap <K,V> {
027        private static final long serialVersionUID = -1773560792952436569L;
028        protected AbstractObject2ObjectSortedMap() {}
029        /** {@inheritDoc}
030         *
031         * <P>The view is backed by the sorted set returned by {@link #entrySet()}. Note that
032         * <em>no attempt is made at caching the result of this method</em>, as this would
033         * require adding some attributes that lightweight implementations would
034         * not need. Subclasses may easily override this policy by calling
035         * this method and caching the result, but implementors are encouraged to
036         * write more efficient ad-hoc implementations.
037         *
038         * @return a sorted set view of the keys of this map; it may be safely cast to a type-specific interface.
039         */
040        @Override
041        public ObjectSortedSet <K> keySet() {
042         return new KeySet();
043        }
044        /** A wrapper exhibiting the keys of a map. */
045        protected class KeySet extends AbstractObjectSortedSet <K> {
046         @Override
047         public boolean contains(final Object k) { return containsKey(k); }
048         @Override
049         public int size() { return AbstractObject2ObjectSortedMap.this.size(); }
050         @Override
051         public void clear() { AbstractObject2ObjectSortedMap.this.clear(); }
052         @Override
053         public Comparator <? super K> comparator() { return AbstractObject2ObjectSortedMap.this.comparator(); }
054         @Override
055         public K first() { return firstKey(); }
056         @Override
057         public K last() { return lastKey(); }
058         @Override
059         public ObjectSortedSet <K> headSet(final K to) { return headMap(to).keySet(); }
060         @Override
061         public ObjectSortedSet <K> tailSet(final K from) { return tailMap(from).keySet(); }
062         @Override
063         public ObjectSortedSet <K> subSet(final K from, final K to) { return subMap(from, to).keySet(); }
064         @Override
065         public ObjectBidirectionalIterator <K> iterator(final K from) { return new KeySetIterator <K,V>(object2ObjectEntrySet().iterator(new BasicEntry <K,V>(from, (null)))); }
066         @Override
067         public ObjectBidirectionalIterator <K> iterator() { return new KeySetIterator <K,V>(object2ObjectEntrySet().iterator()); }
068        }
069        /** A wrapper exhibiting a map iterator as an iterator on keys.
070         *
071         * <P>To provide an iterator on keys, just create an instance of this
072         * class using the corresponding iterator on entries.
073         */
074        protected static class KeySetIterator <K,V> extends AbstractObjectBidirectionalIterator <K> {
075         protected final ObjectBidirectionalIterator<Object2ObjectMap.Entry <K,V> > i;
076         public KeySetIterator(ObjectBidirectionalIterator<Object2ObjectMap.Entry <K,V> > i) {
077          this.i = i;
078         }
079         @Override
080         public K next() { return i.next().getKey(); };
081         @Override
082         public K previous() { return i.previous().getKey(); };
083         @Override
084         public boolean hasNext() { return i.hasNext(); }
085         @Override
086         public boolean hasPrevious() { return i.hasPrevious(); }
087        }
088        /** {@inheritDoc}
089         *
090         * <P>The view is backed by the sorted set returned by {@link #entrySet()}. Note that
091         * <em>no attempt is made at caching the result of this method</em>, as this would
092         * require adding some attributes that lightweight implementations would
093         * not need. Subclasses may easily override this policy by calling
094         * this method and caching the result, but implementors are encouraged to
095         * write more efficient ad-hoc implementations.
096         *
097         * @return a type-specific collection view of the values contained in this map.
098         */
099        @Override
100        public ObjectCollection <V> values() {
101         return new ValuesCollection();
102        }
103        /** A wrapper exhibiting the values of a map. */
104        protected class ValuesCollection extends AbstractObjectCollection <V> {
105         @Override
106         public ObjectIterator <V> iterator() { return new ValuesIterator <K,V>(object2ObjectEntrySet().iterator()); }
107         @Override
108         public boolean contains(final Object k) { return containsValue(k); }
109         @Override
110         public int size() { return AbstractObject2ObjectSortedMap.this.size(); }
111         @Override
112         public void clear() { AbstractObject2ObjectSortedMap.this.clear(); }
113        }
114        /** A wrapper exhibiting a map iterator as an iterator on values.
115         *
116         * <P>To provide an iterator on values, just create an instance of this
117         * class using the corresponding iterator on entries.
118         */
119        protected static class ValuesIterator <K,V> extends AbstractObjectIterator <V> {
120         protected final ObjectBidirectionalIterator<Object2ObjectMap.Entry <K,V> > i;
121         public ValuesIterator(ObjectBidirectionalIterator<Object2ObjectMap.Entry <K,V> > i) {
122          this.i = i;
123         }
124         @Override
125         public V next() { return i.next().getValue(); };
126         @Override
127         public boolean hasNext() { return i.hasNext(); }
128        }
129        /** {@inheritDoc}
130         * <p>This implementation delegates to the corresponding type-specific method. */
131        @SuppressWarnings({ "unchecked", "rawtypes" })
132        @Override
133        public ObjectSortedSet<Map.Entry<K, V>> entrySet() {
134         return (ObjectSortedSet)object2ObjectEntrySet();
135        }
136}
