016package it.unimi.dsi.fastutil.longs;
017import it.unimi.dsi.fastutil.objects.AbstractObjectListIterator;
018import it.unimi.dsi.fastutil.objects.AbstractObjectList;
019import it.unimi.dsi.fastutil.objects.ObjectListIterator;
020import it.unimi.dsi.fastutil.longs.LongArrays;
021import java.io.Serializable;
022import java.util.Iterator;
023import java.util.Collection;
024import java.util.NoSuchElementException;
025import java.util.RandomAccess;
026/** Compact storage of lists of arrays using front coding.
027        *
028        * <P>This class stores immutably a list of arrays in a single large array
029        * using front coding (of course, the compression will be reasonable only if
030        * the list is sorted lexicographically&mdash;see below). It implements an
031        * immutable type-specific list that returns the <var>i</var>-th array when
032        * calling {@link #get(int) get(<var>i</var>)}. The returned array may be
033        * freely modified.
034        *
035        * <P>Front coding is based on the idea that if the <var>i</var>-th and the
036        * (<var>i</var>+1)-th array have a common prefix, we might store the length
037        * of the common prefix, and then the rest of the second array.
038        *
039        * <P>This approach, of course, requires that once in a while an array is
040        * stored entirely.  The <em>ratio</em> of a front-coded list defines how
041        * often this happens (once every {@link #ratio()} arrays). A higher ratio
042        * means more compression, but means also a longer access time, as more arrays
043        * have to be probed to build the result. Note that we must build an array
044        * every time {@link #get(int)} is called, but this class provides also methods
045        * that extract one of the stored arrays in a given array, reducing garbage
046        * collection. See the documentation of the family of <code>get()</code>
047        * methods.
048        *
049        * <P>By setting the ratio to 1 we actually disable front coding: however, we
050        * still have a data structure storing large list of arrays with a reduced
051        * overhead (just one integer per array, plus the space required for lengths).
052        *
053        * <P>Note that the typical usage of front-coded lists is under the form of
054        * serialized objects; usually, the data that has to be compacted is processed
055        * offline, and the resulting structure is stored permanently. Since the
056        * pointer array is not stored, the serialized format is very small.
057        *
058        * <H2>Implementation Details</H2>
059        *
060        * <P>All arrays are stored in a {@linkplain it.unimi.dsi.fastutil.BigArrays big array}. A separate array of pointers
061        * indexes arrays whose position is a multiple of the ratio: thus, a higher ratio
062        * means also less pointers.
063        *
064        * <P>More in detail, an array whose position is a multiple of the ratio is
065        * stored as the array length, followed by the elements of the array. The array
066        * length is coded by a simple variable-length list of <var>k</var>-1 bit
067        * blocks, where <var>k</var> is the number of bits of the underlying primitive
068        * type.  All other arrays are stored as follows: let <code>common</code> the
069        * length of the maximum common prefix between the array and its predecessor.
070        * Then we store the array length decremented by <code>common</code>, followed
071        * by <code>common</code>, followed by the array elements whose index is
072        * greater than or equal to <code>common</code>. For instance, if we store
073        * <code>foo</code>, <code>foobar</code>, <code>football</code> and
074        * <code>fool</code> in a front-coded character-array list with ratio 3, the
075        * character array will contain
076        *
077        * <pre>
078        * <b>3</b> f o o <b>3</b> <b>3</b> b a r <b>5</b> <b>3</b> t b a l l <b>4</b> f o o l
079        * </pre>
080        */
081public class LongArrayFrontCodedList extends AbstractObjectList<long[]> implements Serializable, Cloneable, RandomAccess {
082        private static final long serialVersionUID = 1L;
083        /** The number of arrays in the list. */
084        protected final int n;
085        /** The ratio of this front-coded list. */
086        protected final int ratio;
087        /** The big array containing the compressed arrays. */
088        protected final long[][] array;
089        /** The pointers to entire arrays in the list. */
090        protected transient long[] p;
091        /** Creates a new front-coded list containing the arrays returned by the given iterator.
092         *
093         * @param arrays an iterator returning arrays.
094         * @param ratio the desired ratio.
095         */
096        public LongArrayFrontCodedList(final Iterator<long[]> arrays, final int ratio) {
097         if (ratio < 1) throw new IllegalArgumentException("Illegal ratio (" + ratio + ")");
098         long[][] array = LongBigArrays.EMPTY_BIG_ARRAY;
099         long[] p = LongArrays.EMPTY_ARRAY;
100         long[][] a = new long[2][];
101         long curSize = 0;
102         int n = 0, b = 0, common, length, minLength;
103         while(arrays.hasNext()) {
104          a[b] = arrays.next();
105          length = a[b].length;
106          if (n % ratio == 0) {
107           p = LongArrays.grow(p, n / ratio + 1);
108           p[n / ratio] = curSize;
109           array = LongBigArrays.grow(array, curSize + count(length) + length, curSize);
110           curSize += writeInt(array, length, curSize);
111           LongBigArrays.copyToBig(a[b], 0, array, curSize, length);
112           curSize += length;
113          }
114          else {
115           minLength = a[1 - b].length;
116           if (length < minLength) minLength = length;
117           for(common = 0; common < minLength; common++) if (a[0][common] != a[1][common]) break;
118           length -= common;
119           array = LongBigArrays.grow(array, curSize + count(length) + count(common) + length, curSize);
120           curSize += writeInt(array, length, curSize);
121           curSize += writeInt(array, common, curSize);
122           LongBigArrays.copyToBig(a[b], common, array, curSize, length);
123           curSize += length;
124          }
125          b = 1 - b;
126          n++;
127         }
128         this.n = n;
129         this.ratio = ratio;
130         this.array = LongBigArrays.trim(array, curSize);
131         this.p = LongArrays.trim(p, (n + ratio - 1) / ratio);
132        }
133        /** Creates a new front-coded list containing the arrays in the given collection.
134         *
135         * @param c a collection containing arrays.
136         * @param ratio the desired ratio.
137         */
138        public LongArrayFrontCodedList(final Collection<long[]> c, final int ratio) {
139         this(c.iterator(), ratio);
140        }
141        /* The following (rather messy) methods implements the encoding of arbitrary integers inside a big array.
142         * Unfortunately, we have to specify different codes for almost every type. */
143        /** Reads a coded length.
144         * @param a the data big array.
145         * @param pos the starting position.
146         * @return the length coded at <code>pos</code>.
147         */
148        private static int readInt(final long a[][], long pos) {
149         return (int)LongBigArrays.get(a, pos);
150        }
151        /** Computes the number of elements coding a given length.
152         * @param length the length to be coded.
153         * @return the number of elements coding <code>length</code>.
154         */
155        private static int count(final int length) {
156         return 1;
157        }
158        /** Writes a length.
159         * @param a the data array.
160         * @param length the length to be written.
161         * @param pos the starting position.
162         * @return the number of elements coding <code>length</code>.
163         */
164        private static int writeInt(final long a[][], int length, long pos) {
165         LongBigArrays.set(a, pos, length);
166         return 1;
167        }
168        /** Returns the ratio of this list.
169         *
170         * @return the ratio of this list.
171         */
172        public int ratio() {
173         return ratio;
174        }
175        /** Computes the length of the array at the given index.
176         *
177         * <P>This private version of {@link #arrayLength(int)} does not check its argument.
178         *
179         * @param index an index.
180         * @return the length of the <code>index</code>-th array.
181         */
182        private int length(final int index) {
183         final long[][] array = this.array;
184         final int delta = index % ratio; // The index into the p array, and the delta inside the block.
185         long pos = p[index / ratio]; // The position into the array of the first entire word before the index-th.
186         int length = readInt(array, pos);
187         if (delta == 0) return length;
188         // First of all, we recover the array length and the maximum amount of copied elements.
189         int common;
190         pos += count(length) + length;
191         length = readInt(array, pos);
192         common = readInt(array, pos + count(length));
193         for(int i = 0; i < delta - 1; i++) {
194          pos += count(length) + count(common) + length;
195          length = readInt(array, pos);
196          common = readInt(array, pos + count(length));
197         }
198         return length + common;
199        }
200        /** Computes the length of the array at the given index.
201         *
202         * @param index an index.
203         * @return the length of the <code>index</code>-th array.
204         */
205        public int arrayLength(final int index) {
206         ensureRestrictedIndex(index);
207         return length(index);
208        }
209        /** Extracts the array at the given index.
210         *
211         * @param index an index.
212         * @param a the array that will store the result (we assume that it can hold the result).
213         * @param offset an offset into <code>a</code> where elements will be store.
214         * @param length a maximum number of elements to store in <code>a</code>.
215         * @return the length of the extracted array.
216         */
217        private int extract(final int index, final long a[], final int offset, final int length) {
218         final int delta = index % ratio; // The delta inside the block.
219         final long startPos = p[index / ratio]; // The position into the array of the first entire word before the index-th.
220         long pos, prevArrayPos;
221         int arrayLength = readInt(array, pos = startPos), currLen = 0, actualCommon;
222         if (delta == 0) {
223          pos = p[index / ratio] + count(arrayLength);
224          LongBigArrays.copyFromBig(array, pos, a, offset, Math.min(length, arrayLength));
225          return arrayLength;
226         }
227         int common = 0;
228         for(int i = 0; i < delta; i++) {
229          prevArrayPos = pos + count(arrayLength) + (i != 0 ? count(common) : 0);
230          pos = prevArrayPos + arrayLength;
231          arrayLength = readInt(array, pos);
232          common = readInt(array, pos + count(arrayLength));
233          actualCommon = Math.min(common, length);
234          if (actualCommon <= currLen) currLen = actualCommon;
235          else {
236           LongBigArrays.copyFromBig(array, prevArrayPos, a, currLen + offset, actualCommon - currLen);
237           currLen = actualCommon;
238          }
239         }
240         if (currLen < length) LongBigArrays.copyFromBig(array, pos + count(arrayLength) + count(common), a, currLen + offset, Math.min(arrayLength, length - currLen));
241         return arrayLength + common;
242        }
243        /** {@inheritDoc}
244         * <P>This implementation delegates to {@link #getArray(int)}. */
245        @Override
246        public long[] get(final int index) {
247         return getArray(index);
248        }
249        /** Returns an array stored in this front-coded list.
250         *
251         * @param index an index.
252         * @return the corresponding array stored in this front-coded list.
253         */
254        public long[] getArray(final int index) {
255         ensureRestrictedIndex(index);
256         final int length = length(index);
257         final long a[] = new long[length];
258         extract(index, a, 0, length);
259         return a;
260        }
261        /** Stores in the given array elements from an array stored in this front-coded list.
262         *
263         * @param index an index.
264         * @param a the array that will store the result.
265         * @param offset an offset into <code>a</code> where elements will be store.
266         * @param length a maximum number of elements to store in <code>a</code>.
267         * @return if <code>a</code> can hold the extracted elements, the number of extracted elements;
268         * otherwise, the number of remaining elements with the sign changed.
269         */
270        public int get(final int index, final long[] a, final int offset, final int length) {
271         ensureRestrictedIndex(index);
272         LongArrays.ensureOffsetLength(a, offset, length);
273         final int arrayLength = extract(index, a, offset, length);
274         if (length >= arrayLength) return arrayLength;
275         return length - arrayLength;
276        }
277        /** Stores in the given array an array stored in this front-coded list.
278         *
279         * @param index an index.
280         * @param a the array that will store the content of the result (we assume that it can hold the result).
281         * @return if <code>a</code> can hold the extracted elements, the number of extracted elements;
282         * otherwise, the number of remaining elements with the sign changed.
283         */
284        public int get(final int index, final long[] a) {
285         return get(index, a, 0, a.length);
286        }
287        @Override
288        public int size() {
289         return n;
290        }
291        @Override
292        public ObjectListIterator<long[]> listIterator(final int start) {
293         ensureIndex(start);
294         return new AbstractObjectListIterator<long[]>() {
295           long s[] = LongArrays.EMPTY_ARRAY;
296           int i = 0;
297           long pos = 0;
298           boolean inSync; // Whether the current value in a is the string just before the next to be produced.
299           {
300            if (start != 0) {
301             if (start == n) i = start; // If we start at the end, we do nothing.
302             else {
303              pos = p[start / ratio];
304              int j = start % ratio;
305              i = start - j;
306              while(j-- != 0) next();
307             }
308            }
309           }
310           @Override
311           public boolean hasNext() {
312            return i < n;
313           }
314           @Override
315           public boolean hasPrevious() {
316            return i > 0;
317           }
318           @Override
319           public int previousIndex() {
320            return i - 1;
321           }
322           @Override
323           public int nextIndex() {
324            return i;
325           }
326           @Override
327           public long[] next() {
328            int length, common;
329            if (! hasNext()) throw new NoSuchElementException();
330            if (i % ratio == 0) {
331             pos = p[i / ratio];
332             length = readInt(array, pos);
333             s = LongArrays.ensureCapacity(s, length, 0);
334             LongBigArrays.copyFromBig(array, pos + count(length), s, 0, length);
335             pos += length + count(length);
336             inSync = true;
337            }
338            else {
339             if (inSync) {
340              length = readInt(array, pos);
341              common = readInt(array, pos + count(length));
342              s = LongArrays.ensureCapacity(s, length + common, common);
343              LongBigArrays.copyFromBig(array, pos + count(length) + count (common), s, common, length);
344              pos += count(length) + count(common) + length;
345              length += common;
346             }
347             else {
348              s = LongArrays.ensureCapacity(s, length = length(i), 0);
349              extract(i, s, 0, length);
350             }
351            }
352            i++;
353            return LongArrays.copy(s, 0, length);
354           }
355           @Override
356           public long[] previous() {
357            if (! hasPrevious()) throw new NoSuchElementException();
358            inSync = false;
359            return getArray(--i);
360           }
361          };
362        }
363        /** Returns a copy of this list.
364         *
365         *  @return a copy of this list.
366         */
367        @Override
368        public LongArrayFrontCodedList clone() {
369         return this;
370        }
371        @Override
372        public String toString() {
373         final StringBuffer s = new StringBuffer();
374         s.append("[");
375         for(int i = 0; i < n; i++) {
376          if (i != 0) s.append(", ");
377          s.append(LongArrayList.wrap(getArray(i)).toString());
378         }
379         s.append("]");
380         return s.toString();
381        }
382        /** Computes the pointer array using the currently set ratio, number of elements and underlying array.
383         *
384         * @return the computed pointer array.
385         */
386        protected long[] rebuildPointerArray() {
387         final long[] p = new long[(n + ratio - 1) / ratio];
388         final long a[][] = array;
389         int length, count;
390         long pos = 0;
391         for(int i = 0, j = 0, skip = ratio - 1; i < n; i++) {
392          length = readInt(a, pos);
393          count = count(length);
394          if (++skip == ratio) {
395           skip = 0;
396           p[j++] = pos;
397           pos += count + length;
398          }
399          else pos += count + count(readInt(a, pos + count)) + length;
400         }
401         return p;
402        }
403        private void readObject(java.io.ObjectInputStream s) throws java.io.IOException, ClassNotFoundException {
404         s.defaultReadObject();
405         // Rebuild pointer array
406         p = rebuildPointerArray();
407        }
