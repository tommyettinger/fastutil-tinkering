025        /** An immutable class representing an empty type-specific collection.

064        /** A synchronized wrapper class for collections. */
065        public static class SynchronizedCollection implements ShortCollection , java.io.Serializable {
066         private static final long serialVersionUID = -7046029254386353129L;
067         protected final ShortCollection collection;
068         protected final Object sync;
069         protected SynchronizedCollection(final ShortCollection c, final Object sync) {
070          if (c == null) throw new NullPointerException();
071          this.collection = c;
072          this.sync = sync;
073         }
074         protected SynchronizedCollection(final ShortCollection c) {
075          if (c == null) throw new NullPointerException();
076          this.collection = c;
077          this.sync = this;
078         }
079         @Override
080         public boolean add(final short k) { synchronized(sync) { return collection.add(k); } }
081         @Override
082         public boolean contains(final short k) { synchronized(sync) { return collection.contains(k); } }
083         @Override
084         public boolean rem(final short k) { synchronized(sync) { return collection.rem(k); } }
085         @Override
086         public int size() { synchronized(sync) { return collection.size(); } }
087         @Override
088         public boolean isEmpty() { synchronized(sync) { return collection.isEmpty(); } }
089         @Override
090         public short[] toShortArray() { synchronized(sync) { return collection.toShortArray(); } }
091         @Override
092         public Object[] toArray() { synchronized(sync) { return collection.toArray(); } }
093         /* {@inheritDoc}
094                 * @deprecated Please use {@code toArray()} instead&mdash;this method is redundant.
095                 */
096         @Deprecated
097         @Override
098         public short[] toShortArray(final short[] a) { return toArray(a); }
099         @Override
100         public short[] toArray(final short[] a) { synchronized(sync) { return collection.toArray(a); } }
101         @Override
102         public boolean addAll(final ShortCollection c) { synchronized(sync) { return collection.addAll(c); } }
103         @Override
104         public boolean containsAll(final ShortCollection c) { synchronized(sync) { return collection.containsAll(c); } }
105         @Override
106         public boolean removeAll(final ShortCollection c) { synchronized(sync) { return collection.removeAll(c); } }
107         @Override
108         public boolean retainAll(final ShortCollection c) { synchronized(sync) { return collection.retainAll(c); } }
109         @Override
110         @Deprecated
111         public boolean add(final Short k) { synchronized(sync) { return collection.add(k); } }
112         @Override
113         @Deprecated
114         public boolean contains(final Object k) { synchronized(sync) { return collection.contains(k); } }
115         @Override
116         @Deprecated
117         public boolean remove(final Object k) { synchronized(sync) { return collection.remove(k); } }
118         @Override
119         public <T> T[] toArray(final T[] a) { synchronized(sync) { return collection.toArray(a); } }
120         @Override
121         public ShortIterator iterator() { return collection.iterator(); }
122         /** Returns a type-specific iterator on this elements of this collection.
123                 *
124                 * @see #iterator()
125                 * @deprecated As of <code>fastutil</code> 5, replaced by {@link #iterator()}.
126                 */
127         @Override
128         @Deprecated
129         public ShortIterator shortIterator() { return iterator(); }
130         @Override
131         public boolean addAll(final Collection<? extends Short> c) { synchronized(sync) { return collection.addAll(c); } }
132         @Override
133         public boolean containsAll(final Collection<?> c) { synchronized(sync) { return collection.containsAll(c); } }
134         @Override
135         public boolean removeAll(final Collection<?> c) { synchronized(sync) { return collection.removeAll(c); } }
136         @Override
137         public boolean retainAll(final Collection<?> c) { synchronized(sync) { return collection.retainAll(c); } }
138         @Override
139         public void clear() { synchronized(sync) { collection.clear(); } }
140         @Override
141         public String toString() { synchronized(sync) { return collection.toString(); } }
142         @Override
143         public int hashCode() { synchronized(sync) { return collection.hashCode(); } }
144         @Override
145         public boolean equals(final Object o) { if (o == this) return true; synchronized(sync) { return collection.equals(o); } }
146         private void writeObject(java.io.ObjectOutputStream s) throws java.io.IOException {
147          synchronized(sync) { s.defaultWriteObject(); }
148         }
149        }

164        public static ShortCollection synchronize(final ShortCollection c, final Object sync) { return new SynchronizedCollection (c, sync); }

246        /** A collection wrapper class for iterables. */
247        public static class IterableCollection extends AbstractShortCollection implements java.io.Serializable {
248         private static final long serialVersionUID = -7046029254386353129L;
249         protected final ShortIterable iterable;
250         protected IterableCollection(final ShortIterable iterable) {
251          if (iterable == null) throw new NullPointerException();
252          this.iterable = iterable;
253         }
254         @Override
255         public int size() {
256          int c = 0;
257          final ShortIterator iterator = iterator();
258          while(iterator.hasNext()) {
259           iterator.nextShort();
260           c++;
261          }
262          return c;
263         }
264         @Override
265         public boolean isEmpty() { return ! iterable.iterator().hasNext(); }
266         @Override
267         public ShortIterator iterator() { return iterable.iterator(); }
268        }
269        /** Returns an unmodifiable collection backed by the specified iterable.
270         *
271         * @param iterable the iterable object to be wrapped in an unmodifiable collection.
272         * @return an unmodifiable collection view of the specified iterable.
273         */
274        public static ShortCollection asCollection(final ShortIterable iterable) {
275         if (iterable instanceof ShortCollection) return (ShortCollection )iterable;
276         return new IterableCollection (iterable);
277        }
