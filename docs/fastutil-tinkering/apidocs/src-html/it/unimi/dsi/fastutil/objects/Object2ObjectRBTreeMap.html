029/** A type-specific red-black tree map with a fast, small-footprint implementation.
030        *
031        * <P>The iterators provided by the views of this class are type-specific {@linkplain
032        * it.unimi.dsi.fastutil.BidirectionalIterator bidirectional iterators}.
033        * Moreover, the iterator returned by <code>iterator()</code> can be safely cast
034        * to a type-specific {@linkplain java.util.ListIterator list iterator}.
035        *
036        */
037public class Object2ObjectRBTreeMap <K,V> extends AbstractObject2ObjectSortedMap <K,V> implements java.io.Serializable, Cloneable {

145        /*
146         * The following methods implements some basic building blocks used by
147         * all accessors.  They are (and should be maintained) identical to those used in RBTreeSet.drv.
148         *
149         * The put()/remove() code is derived from Ben Pfaff's GNU libavl
150         * (http://www.msu.edu/~pfaffben/avl/). If you want to understand what's
151         * going on, you should have a look at the literate code contained therein
152         * first.
153         */
154        /** Compares two keys in the right way.
155         *
156         * <P>This method uses the {@link #actualComparator} if it is non-{@code null}.
157         * Otherwise, it resorts to primitive type comparisons or to {@link Comparable#compareTo(Object) compareTo()}.
158         *
159         * @param k1 the first key.
160         * @param k2 the second key.
161         * @return a number smaller than, equal to or greater than 0, as usual
162         * (i.e., when k1 &lt; k2, k1 = k2 or k1 &gt; k2, respectively).
163         */
164        @SuppressWarnings("unchecked")
165        final int compare(final K k1, final K k2) {
166         return actualComparator == null ? ( ((Comparable<K>)(k1)).compareTo(k2) ) : actualComparator.compare(k1, k2);
167        }
168        /** Returns the entry corresponding to the given key, if it is in the tree; {@code null}, otherwise.
169         *
170         * @param k the key to search for.
171         * @return the corresponding entry, or {@code null} if no entry with the given key exists.
172         */
173        final Entry <K,V> findKey(final K k) {
174         Entry <K,V> e = tree;
175         int cmp;
176         while (e != null && (cmp = compare(k, e.key)) != 0) e = cmp < 0 ? e.left() : e.right();
177         return e;
178        }
179        /** Locates a key.
180         *
181         * @param k a key.
182         * @return the last entry on a search for the given key; this will be
183         * the given key, if it present; otherwise, it will be either the smallest greater key or the greatest smaller key.
184         */
185        final Entry <K,V> locateKey(final K k) {
186         Entry <K,V> e = tree, last = tree;
187         int cmp = 0;
188         while (e != null && (cmp = compare(k, e.key)) != 0) {
189          last = e;
190          e = cmp < 0 ? e.left() : e.right();
191         }
192         return cmp == 0 ? e : last;
193        }
194        /** This vector remembers the path and the direction followed during the
195         *  current insertion. It suffices for about 2<sup>32</sup> entries. */
196        private transient boolean dirPath[];
197        private transient Entry <K,V> nodePath[];
198        @SuppressWarnings("unchecked")
199        private void allocatePaths() {
200         dirPath = new boolean[64];
201         nodePath = new Entry[64];
202        }
203        @Override
204        public V put(final K k, final V v) {
205         Entry <K,V> e = add(k);
206         final V oldValue = e.value;
207         e.value = v;
208         return oldValue;
209        }
210        /** Returns a node with key k in the balanced tree, creating one with defRetValue if necessary.
211         *
212         * @param k the key
213         * @return a node with key k. If a node with key k already exists, then that node is returned,
214         * otherwise a new node with defRetValue is created ensuring that the tree is balanced
215         * after creation of the node.
216         */
217        private Entry <K,V> add(final K k) {
218         /* After execution of this method, modified is true iff a new entry has been inserted. */
219         modified = false;
220         int maxDepth = 0;
221         Entry <K,V> e;
222         if (tree == null) { // The case of the empty tree is treated separately.
223          count++;
224          e = tree = lastEntry = firstEntry = new Entry <K,V>(k, defRetValue);
225         }
226         else {
227          Entry <K,V> p = tree;
228          int cmp, i = 0;
229          while(true) {
230           if ((cmp = compare(k, p.key)) == 0) {
231            // We clean up the node path, or we could have stale references later.
232            while(i-- != 0) nodePath[i] = null;
233            return p;
234           }
235           nodePath[i] = p;
236           if (dirPath[i++] = cmp > 0) {
237            if (p.succ()) {
238             count++;
239             e = new Entry <K,V>(k, defRetValue);
240             if (p.right == null) lastEntry = e;
241             e.left = p;
242             e.right = p.right;
243             p.right(e);
244             break;
245            }
246            p = p.right;
247           }
248           else {
249            if (p.pred()) {
250             count++;
251             e = new Entry <K,V>(k, defRetValue);
252             if (p.left == null) firstEntry = e;
253             e.right = p;
254             e.left = p.left;
255             p.left(e);
256             break;
257            }
258            p = p.left;
259           }
260          }
261          modified = true;
262          maxDepth = i--;
263          while(i > 0 && ! nodePath[i].black()) {
264           if (! dirPath[i - 1]) {
265            Entry <K,V> y = nodePath[i - 1].right;
266            if (! nodePath[i - 1].succ() && ! y.black()) {
267             nodePath[i].black(true);
268             y.black(true);
269             nodePath[i - 1].black(false);
270             i -= 2;
271            }
272            else {
273             Entry <K,V> x;
274             if (! dirPath[i]) y = nodePath[i];
275             else {
276              x = nodePath[i];
277              y = x.right;
278              x.right = y.left;
279              y.left = x;
280              nodePath[i - 1].left = y;
281              if (y.pred()) {
282               y.pred(false);
283               x.succ(y);
284              }
285             }
286             x = nodePath[i - 1];
287             x.black(false);
288             y.black(true);
289             x.left = y.right;
290             y.right = x;
291             if (i < 2) tree = y;
292             else {
293              if (dirPath[i - 2]) nodePath[i - 2].right = y;
294              else nodePath[i - 2].left = y;
295             }
296             if (y.succ()) {
297              y.succ(false);
298              x.pred(y);
299             }
300             break;
301            }
302           }
303           else {
304            Entry <K,V> y = nodePath[i - 1].left;
305            if (! nodePath[i - 1].pred() && ! y.black()) {
306             nodePath[i].black(true);
307             y.black(true);
308             nodePath[i - 1].black(false);
309             i -= 2;
310            }
311            else {
312             Entry <K,V> x;
313             if (dirPath[i]) y = nodePath[i];
314             else {
315              x = nodePath[i];
316              y = x.left;
317              x.left = y.right;
318              y.right = x;
319              nodePath[i - 1].right = y;
320              if (y.succ()) {
321               y.succ(false);
322               x.pred(y);
323              }
324             }
325             x = nodePath[i - 1];
326             x.black(false);
327             y.black(true);
328             x.right = y.left;
329             y.left = x;
330             if (i < 2) tree = y;
331             else {
332              if (dirPath[i - 2]) nodePath[i - 2].right = y;
333              else nodePath[i - 2].left = y;
334             }
335             if (y.pred()){
336              y.pred(false);
337              x.succ(y);
338             }
339             break;
340            }
341           }
342          }
343         }
344         tree.black(true);
345         // We clean up the node path, or we could have stale references later.
346         while(maxDepth-- != 0) nodePath[maxDepth] = null;
347         if (ASSERTS) {
348          checkNodePath();
349          checkTree(tree, 0, -1);
350         }
351         return e;
352        }
353        /* After execution of this method, {@link #modified} is true iff an entry
354        has been deleted. */
355        @SuppressWarnings("unchecked")
356        @Override
357        public V remove(final Object k) {
358         modified = false;
359         if (tree == null) return defRetValue;
360         Entry <K,V> p = tree;
361         int cmp;
362         int i = 0;
363         final K kk = (K) k;
364         while(true) {
365          if ((cmp = compare(kk, p.key)) == 0) break;
366          dirPath[i] = cmp > 0;
367          nodePath[i] = p;
368          if (dirPath[i++]) {
369           if ((p = p.right()) == null) {
370            // We clean up the node path, or we could have stale references later.
371            while(i-- != 0) nodePath[i] = null;
372            return defRetValue;
373           }
374          }
375          else {
376           if ((p = p.left()) == null) {
377            // We clean up the node path, or we could have stale references later.
378            while(i-- != 0) nodePath[i] = null;
379            return defRetValue;
380           }
381          }
382         }
383         if (p.left == null) firstEntry = p.next();
384         if (p.right == null) lastEntry = p.prev();
385         if (p.succ()) {
386          if (p.pred()) {
387           if (i == 0) tree = p.left;
388           else {
389            if (dirPath[i - 1]) nodePath[i - 1].succ(p.right);
390            else nodePath[i - 1].pred(p.left);
391           }
392          }
393          else {
394           p.prev().right = p.right;
395           if (i == 0) tree = p.left;
396           else {
397            if (dirPath[i - 1]) nodePath[i - 1].right = p.left;
398            else nodePath[i - 1].left = p.left;
399           }
400          }
401         }
402         else {
403          boolean color;
404          Entry <K,V> r = p.right;
405          if (r.pred()) {
406           r.left = p.left;
407           r.pred(p.pred());
408           if (! r.pred()) r.prev().right = r;
409           if (i == 0) tree = r;
410           else {
411            if (dirPath[i - 1]) nodePath[i - 1].right = r;
412            else nodePath[i - 1].left = r;
413           }
414           color = r.black();
415           r.black(p.black());
416           p.black(color);
417           dirPath[i] = true;
418           nodePath[i++] = r;
419          }
420          else {
421           Entry <K,V> s;
422           int j = i++;
423           while(true) {
424            dirPath[i] = false;
425            nodePath[i++] = r;
426            s = r.left;
427            if (s.pred()) break;
428            r = s;
429           }
430           dirPath[j] = true;
431           nodePath[j] = s;
432           if (s.succ()) r.pred(s);
433           else r.left = s.right;
434           s.left = p.left;
435           if (! p.pred()) {
436            p.prev().right = s;
437            s.pred(false);
438           }
439           s.right(p.right);
440           color = s.black();
441           s.black(p.black());
442           p.black(color);
443           if (j == 0) tree = s;
444           else {
445            if (dirPath[j - 1]) nodePath[j - 1].right = s;
446            else nodePath[j - 1].left = s;
447           }
448          }
449         }
450         int maxDepth = i;
451         if (p.black()) {
452          for(; i > 0; i--) {
453           if (dirPath[i - 1] && ! nodePath[i - 1].succ() ||
454             ! dirPath[i - 1] && ! nodePath[i - 1].pred()) {
455            Entry <K,V> x = dirPath[i - 1] ? nodePath[i - 1].right : nodePath[i - 1].left;
456            if (! x.black()) {
457             x.black(true);
458             break;
459            }
460           }
461           if (! dirPath[i - 1]) {
462            Entry <K,V> w = nodePath[i - 1].right;
463            if (! w.black()) {
464             w.black(true);
465             nodePath[i - 1].black(false);
466             nodePath[i - 1].right = w.left;
467             w.left = nodePath[i - 1];
468             if (i < 2) tree = w;
469             else {
470              if (dirPath[i - 2]) nodePath[i - 2].right = w;
471              else nodePath[i - 2].left = w;
472             }
473             nodePath[i] = nodePath[i - 1];
474             dirPath[i] = false;
475             nodePath[i - 1] = w;
476             if (maxDepth == i++) maxDepth++;
477             w = nodePath[i - 1].right;
478            }
479            if ((w.pred() || w.left.black()) &&
480              (w.succ() || w.right.black())) {
481             w.black(false);
482            }
483            else {
484             if (w.succ() || w.right.black()) {
485              Entry <K,V> y = w.left;
486              y.black (true);
487              w.black(false);
488              w.left = y.right;
489              y.right = w;
490              w = nodePath[i - 1].right = y;
491              if (w.succ()) {
492               w.succ(false);
493               w.right.pred(w);
494              }
495             }
496             w.black(nodePath[i - 1].black());
497             nodePath[i - 1].black(true);
498             w.right.black(true);
499             nodePath[i - 1].right = w.left;
500             w.left = nodePath[i - 1];
501             if (i < 2) tree = w;
502             else {
503              if (dirPath[i - 2]) nodePath[i - 2].right = w;
504              else nodePath[i - 2].left = w;
505             }
506             if (w.pred()) {
507              w.pred(false);
508              nodePath[i - 1].succ(w);
509             }
510             break;
511            }
512           }
513           else {
514            Entry <K,V> w = nodePath[i - 1].left;
515            if (! w.black()) {
516             w.black (true);
517             nodePath[i - 1].black(false);
518             nodePath[i - 1].left = w.right;
519             w.right = nodePath[i - 1];
520             if (i < 2) tree = w;
521             else {
522              if (dirPath[i - 2]) nodePath[i - 2].right = w;
523              else nodePath[i - 2].left = w;
524             }
525             nodePath[i] = nodePath[i - 1];
526             dirPath[i] = true;
527             nodePath[i - 1] = w;
528             if (maxDepth == i++) maxDepth++;
529             w = nodePath[i - 1].left;
530            }
531            if ((w.pred() || w.left.black()) &&
532              (w.succ() || w.right.black())) {
533             w.black(false);
534            }
535            else {
536             if (w.pred() || w.left.black()) {
537              Entry <K,V> y = w.right;
538              y.black(true);
539              w.black (false);
540              w.right = y.left;
541              y.left = w;
542              w = nodePath[i - 1].left = y;
543              if (w.pred()) {
544               w.pred(false);
545               w.left.succ(w);
546              }
547             }
548             w.black(nodePath[i - 1].black());
549             nodePath[i - 1].black(true);
550             w.left.black(true);
551             nodePath[i - 1].left = w.right;
552             w.right = nodePath[i - 1];
553             if (i < 2) tree = w;
554             else {
555              if (dirPath[i - 2]) nodePath[i - 2].right = w;
556              else nodePath[i - 2].left = w;
557             }
558             if (w.succ()) {
559              w.succ(false);
560              nodePath[i - 1].pred(w);
561             }
562             break;
563            }
564           }
565          }
566          if (tree != null) tree.black(true);
567         }
568         modified = true;
569         count--;
570         // We clean up the node path, or we could have stale references later.
571         while(maxDepth-- != 0) nodePath[maxDepth] = null;
572         if (ASSERTS) {
573          checkNodePath();
574          checkTree(tree, 0, -1);
575         }
576         return p.value;
577        }
578        @Override
579        public boolean containsValue(final Object v) {
580         final ValueIterator i = new ValueIterator();
581         Object ev;
582         int j = count;
583         while(j-- != 0) {
584          ev = i.next();
585          if (( (ev) == null ? (v) == null : (ev).equals(v) )) return true;
586         }
587         return false;
588        }
589        @Override
590        public void clear() {
591         count = 0;
592         tree = null;
593         entries = null;
594         values = null;
595         keys = null;
596         firstEntry = lastEntry = null;

598        /** This class represent an entry in a tree map.
599         *
600         * <P>We use the only "metadata", i.e., {@link Entry#info}, to store
601         * information about color, predecessor status and successor status.
602         *
603         * <P>Note that since the class is recursive, it can be
604         * considered equivalently a tree.
605         */
606        private static final class Entry <K,V> extends AbstractObject2ObjectMap.BasicEntry <K,V> implements Cloneable {
607         /** The the bit in this mask is true, the node is black. */
608         private final static int BLACK_MASK = 1;
609         /** If the bit in this mask is true, {@link #right} points to a successor. */
610         private final static int SUCC_MASK = 1 << 31;
611         /** If the bit in this mask is true, {@link #left} points to a predecessor. */
612         private final static int PRED_MASK = 1 << 30;
613         /** The pointers to the left and right subtrees. */
614         Entry <K,V> left, right;
615         /** This integers holds different information in different bits (see {@link #SUCC_MASK} and {@link #PRED_MASK}. */
616         int info;
617         Entry() {
618          super((null), (null));
619         }
620         /** Creates a new entry with the given key and value.
621                 *
622                 * @param k a key.
623                 * @param v a value.
624                 */
625         Entry(final K k, final V v) {
626          super(k, v);
627          info = SUCC_MASK | PRED_MASK;
628         }
629         /** Returns the left subtree.
630                 *
631                 * @return the left subtree ({@code null} if the left
632                 * subtree is empty).
633                 */
634         Entry <K,V> left() {
635          return (info & PRED_MASK) != 0 ? null : left;
636         }
637         /** Returns the right subtree.
638                 *
639                 * @return the right subtree ({@code null} if the right
640                 * subtree is empty).
641                 */
642         Entry <K,V> right() {
643          return (info & SUCC_MASK) != 0 ? null : right;
644         }
645         /** Checks whether the left pointer is really a predecessor.
646                 * @return true if the left pointer is a predecessor.
647                 */
648         boolean pred() {

826         *
827         * <P>This class can iterate in both directions on a threaded tree.
828         */
829        private class TreeIterator {
830         /** The entry that will be returned by the next call to {@link java.util.ListIterator#previous()} (or {@code null} if no previous entry exists). */
831         Entry <K,V> prev;
832         /** The entry that will be returned by the next call to {@link java.util.ListIterator#next()} (or {@code null} if no next entry exists). */
833         Entry <K,V> next;
834         /** The last entry that was returned (or {@code null} if we did not iterate or used {@link #remove()}). */
835         Entry <K,V> curr;
836         /** The current index (in the sense of a {@link java.util.ListIterator}). Note that this value is not meaningful when this {@link TreeIterator} has been created using the nonempty constructor.*/
837         int index = 0;
838         TreeIterator() {
839          next = firstEntry;
840         }
841         TreeIterator(final K k) {
842          if ((next = locateKey(k)) != null) {
843           if (compare(next.key, k) <= 0) {
844            prev = next;
845            next = next.next();
846           }
847           else prev = next.prev();
848          }
849         }
850         public boolean hasNext() { return next != null; }
851         public boolean hasPrevious() { return prev != null; }
852         void updateNext() {
853          next = next.next();
854         }
855         Entry <K,V> nextEntry() {
856          if (! hasNext()) throw new NoSuchElementException();
857          curr = prev = next;
858          index++;
859          updateNext();
860          return curr;
861         }
862         void updatePrevious() {
863          prev = prev.prev();
864         }
865         Entry <K,V> previousEntry() {
866          if (! hasPrevious()) throw new NoSuchElementException();
867          curr = next = prev;
868          index--;
869          updatePrevious();
870          return curr;
871         }
872         public int nextIndex() {
873          return index;
874         }
875         public int previousIndex() {
876          return index - 1;
877         }
878         public void remove() {
879          if (curr == null) throw new IllegalStateException();
880          /* If the last operation was a next(), we are removing an entry that preceeds
881                           the current index, and thus we must decrement it. */
882          if (curr == prev) index--;
883          next = prev = curr;
884          updatePrevious();
885          updateNext();
886          Object2ObjectRBTreeMap.this.remove(curr.key);
887          curr = null;
888         }
889         public int skip(final int n) {
890          int i = n;
891          while(i-- != 0 && hasNext()) nextEntry();
892          return n - i - 1;
893         }
894         public int back(final int n) {
895          int i = n;
896          while(i-- != 0 && hasPrevious()) previousEntry();
897          return n - i - 1;
898         }

1279         /** Locates the first entry.
1280                 *
1281                 * @return the first entry of this submap, or {@code null} if the submap is empty.
1282                 */
1283         public Object2ObjectRBTreeMap.Entry <K,V> firstEntry() {
1284          if (tree == null) return null;
1285          // If this submap goes to -infinity, we return the main map first entry; otherwise, we locate the start of the map.
1286          Object2ObjectRBTreeMap.Entry <K,V> e;
1287          if (bottom) e = firstEntry;
1288          else {
1289           e = locateKey(from);
1290           // If we find either the start or something greater we're OK.
1291           if (compare(e.key, from) < 0) e = e.next();
1292          }
1293          // Finally, if this submap doesn't go to infinity, we check that the resulting key isn't greater than the end.
1294          if (e == null || ! top && compare(e.key, to) >= 0) return null;
1295          return e;
1296         }

1304          Object2ObjectRBTreeMap.Entry <K,V> e;
1305          if (top) e = lastEntry;

1312          if (e == null || ! bottom && compare(e.key, from) < 0) return null;
1313          return e;
1314         }

1327         /** An iterator for subranges.
1328                 *
1329                 * <P>This class inherits from {@link TreeIterator}, but overrides the methods that
1330                 * update the pointer after a {@link java.util.ListIterator#next()} or {@link java.util.ListIterator#previous()}. If we would
1331                 * move out of the range of the submap we just overwrite the next or previous
1332                 * entry with {@code null}.
1333                 */
1334         private class SubmapIterator extends TreeIterator {
1335          SubmapIterator() {
1336           next = firstEntry();
1337          }
1338          SubmapIterator(final K k) {
1339           this();
1340           if (next != null) {
1341            if (! bottom && compare(k, next.key) < 0) prev = null;
1342            else if (! top && compare(k, (prev = lastEntry()).key) >= 0) next = null;
1343            else {
1344             next = locateKey(k);
1345             if (compare(next.key, k) <= 0) {
1346              prev = next;
1347              next = next.next();
1348             }
1349             else prev = next.prev();
1350            }
1351           }
1352          }
1353          void updatePrevious() {
1354           prev = prev.prev();
1355           if (! bottom && prev != null && Object2ObjectRBTreeMap.this.compare(prev.key, from) < 0) prev = null;
1356          }
1357          void updateNext() {
1358           next = next.next();
1359           if (! top && next != null && Object2ObjectRBTreeMap.this.compare(next.key, to) >= 0) next = null;
1360          }
1361         }

1416         *

1418         * set, however, is not cloned. Note that this makes a difference only for object keys.
1419         *
1420         * @return a deep copy of this tree map.
1421         */
1422        @Override

1424        public Object2ObjectRBTreeMap <K,V> clone() {
1425         Object2ObjectRBTreeMap <K,V> c;
1426         try {
1427          c = (Object2ObjectRBTreeMap <K,V>)super.clone();
1428         }
1429         catch(CloneNotSupportedException cantHappen) {
1430          throw new InternalError();
1431         }
1432         c.keys = null;
1433         c.values = null;
1434         c.entries = null;
1435         c.allocatePaths();
1436         if (count != 0) {
1437          // Also this apparently unfathomable code is derived from GNU libavl.
1438          Entry <K,V> e, p, q, rp = new Entry <K,V>(), rq = new Entry <K,V>();
1439          p = rp;
1440          rp.left(tree);
1441          q = rq;
1442          rq.pred(null);
1443          while(true) {
1444           if (! p.pred()) {
1445            e = p.left.clone();
1446            e.pred(q.left);
1447            e.succ(q);
1448            q.left(e);
1449            p = p.left;
1450            q = q.left;
1451           }
1452           else {
1453            while(p.succ()) {
1454             p = p.right;
1455             if (p == null) {
1456              q.right = null;
1457              c.tree = rq.left;
1458              c.firstEntry = c.tree;
1459              while(c.firstEntry.left != null) c.firstEntry = c.firstEntry.left;
1460              c.lastEntry = c.tree;
1461              while(c.lastEntry.right != null) c.lastEntry = c.lastEntry.right;
1462              return c;
1463             }
1464             q = q.right;
1465            }
1466            p = p.right;
1467            q = q.right;
1468           }
1469           if (! p.succ()) {
1470            e = p.right.clone();

1490        /** Reads the given number of entries from the input stream, returning the corresponding tree.
1491         *
1492         * @param s the input stream.
1493         * @param n the (positive) number of entries to read.
1494         * @param pred the entry containing the key that preceeds the first key in the tree.
1495         * @param succ the entry containing the key that follows the last key in the tree.
1496         */
1497        @SuppressWarnings("unchecked")
1498        private Entry <K,V> readTree(final java.io.ObjectInputStream s, final int n, final Entry <K,V> pred, final Entry <K,V> succ) throws java.io.IOException, ClassNotFoundException {
1499         if (n == 1) {
1500          final Entry <K,V> top = new Entry <K,V>((K) s.readObject(), (V) s.readObject());
1501          top.pred(pred);
1502          top.succ(succ);
1503          top.black(true);
1504          return top;
1505         }
1506         if (n == 2) {
1507          /* We handle separately this case so that recursion will
1508                         *always* be on nonempty subtrees. */
1509          final Entry <K,V> top = new Entry <K,V>((K) s.readObject(), (V) s.readObject());
1510          top.black(true);
1511          top.right(new Entry <K,V>((K) s.readObject(), (V) s.readObject()));
1512          top.right.pred(top);
1513          top.pred(pred);
1514          top.right.succ(succ);
1515          return top;
1516         }
1517         // The right subtree is the largest one.
1518         final int rightN = n / 2, leftN = n - rightN - 1;
1519         final Entry <K,V> top = new Entry <K,V>();
1520         top.left(readTree(s, leftN, pred, top));
1521         top.key = (K) s.readObject();
1522         top.value = (V) s.readObject();
1523         top.black(true);
1524         top.right(readTree(s, rightN, top, succ));
1525         if (n + 2 == ((n + 2) & -(n + 2))) top.right.black(false); // Quick test for determining whether n + 2 is a power of 2.
1526         return top;
1527        }
1528        private void readObject(java.io.ObjectInputStream s) throws java.io.IOException, ClassNotFoundException {
1529         s.defaultReadObject();
1530         /* The storedComparator is now correctly set, but we must restore
1531                   on-the-fly the actualComparator. */
1532         setActualComparator();
1533         allocatePaths();
1534         if (count != 0) {
1535          tree = readTree(s, count, null, null);
1536          Entry <K,V> e;
1537          e = tree;
1538          while(e.left() != null) e = e.left();
1539          firstEntry = e;
1540          e = tree;
1541          while(e.right() != null) e = e.right();
1542          lastEntry = e;
1543         }
1544         if (ASSERTS) checkTree(tree, 0, -1);
1545        }
