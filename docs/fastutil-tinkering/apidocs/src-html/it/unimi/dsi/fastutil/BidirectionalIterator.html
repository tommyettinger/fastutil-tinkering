020import java.util.Iterator;
021import java.util.ListIterator;
022
023/** A bidirectional {@link Iterator}.
024 *
025 * <P>This kind of iterator is essentially a {@link ListIterator} that
026 * does not support {@link ListIterator#previousIndex()} and {@link
027 * ListIterator#nextIndex()}. It is useful for those maps that can easily
028 * provide bidirectional iteration, but provide no index.
029 *
030 * <P>Note that iterators returned by <code>fastutil</code> classes are more
031 * specific, and support skipping. This class serves the purpose of organising
032 * in a cleaner way the relationships between various iterators.
033 *
034 * @see Iterator
035 * @see ListIterator
036 */
037
038public interface BidirectionalIterator<K> extends Iterator<K> {
039
040        /** Returns the previous element from the collection.
041         *
042         * @return the previous element from the collection.
043         * @see java.util.ListIterator#previous()
044         */
045
046        K previous();
047
048        /** Returns whether there is a previous element.
049         *
050         * @return whether there is a previous element.
051         * @see java.util.ListIterator#hasPrevious()
052         */
053
054        boolean hasPrevious();
055}
