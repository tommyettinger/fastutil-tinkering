031/** Provides static methods to perform easily binary I/O.
032        *
033        * <P>This class fills some gaps in the Java API. First of all, you have two
034        * buffered, easy-to-use methods to {@linkplain #storeObject(Object,CharSequence) store an object to a file}
035        * or {@linkplain #loadObject(CharSequence) load an object from a file},
036        * and two
037        * buffered, easy-to-use methods to {@linkplain #storeObject(Object,OutputStream) store an object to an output stream}
038        * or to {@linkplain #loadObject(InputStream) load an object from an input stream}.
039        *
040        * <p>Second, a natural operation on sequences of primitive elements is to load or
041        * store them in binary form using the {@link DataInput} conventions.  This
042        * method is much more flexible than storing arrays as objects, as it allows
043        * for partial load, partial store, and makes it easy to read the
044        * resulting files from other languages.
045        *
046        * <P>For each primitive type, this class provides methods that read elements
047        * from a {@link DataInput} or from a filename into an array. Analogously, there are
048        * methods that store the content of an array (fragment) or the elements
049        * returned by an iterator to a {@link DataOutput} or to a given filename. Files
050        * are buffered using {@link FastBufferedInputStream} and {@link FastBufferedOutputStream}.
051        *
052        * <P>Since bytes can be read from or written to any stream, additional methods
053        * makes it possible to {@linkplain #loadBytes(InputStream,byte[]) load bytes from} and
054        * {@linkplain #storeBytes(byte[],OutputStream) store bytes to} a stream. Such methods
055        * use the bulk-read methods of {@link InputStream} and {@link OutputStream}, but they
056        * also include a workaround for <a href="http://bugs.sun.com/bugdatabase/view_bug.do?bug_id=6478546">bug #6478546</a>.
057        *
058        * <P>Finally, there are useful wrapper methods that {@linkplain #asIntIterator(CharSequence)
059        * exhibit a file as a type-specific iterator}.
060        *
061        * @since 4.4
062        */
063public class BinIO {
064        private BinIO() {}

137         return result;
138        }

177/** Loads elements from a given data input, storing them in a given array fragment.
178        *
179        * @param dataInput a data input.
180        * @param array an array which will be filled with data from <code>dataInput</code>.

182        * @param length the number of elements of <code>array</code> to be filled.

188        try {
189         for(i = 0; i < length; i++) array[i + offset] = dataInput.readBoolean();
190        }
191        catch(EOFException itsOk) {}
192        return i;

285        for(int i = 0; i < length; i++) array[i] = dis.readBoolean();
286        dis.close();
287        return array;

363/** Loads elements from a given data input, storing them in a given big-array fragment.
364        *
365        * @param dataInput a data input.
366        * @param array a big array which will be filled with data from <code>dataInput</code>.
367        * @param offset the index of the first element of <code>bigArray</code> to be filled.
368        * @param length the number of elements of <code>bigArray</code> to be filled.
369        * @return the number of elements actually read from <code>dataInput</code> (it might be less than <code>length</code> if <code>dataInput</code> ends).
370        */
371public static long loadBooleans(final DataInput dataInput, final boolean[][] array, final long offset, final long length) throws IOException {
372        it.unimi.dsi.fastutil.booleans.BooleanBigArrays.ensureOffsetLength(array, offset, length);
373        long c = 0;
374        try {
375         for(int i = segment(offset); i < segment(offset + length + SEGMENT_MASK); i++) {
376          final boolean[] t = array[i];
377          final int l = (int)Math.min(t.length, offset + length - start(i));
378          for(int d = (int)Math.max(0, offset - start(i)); d < l; d++) {
379           t[d] = dataInput.readBoolean();
380           c++;
381          }
382         }
383        }
384        catch(EOFException itsOk) {}
385        return c;
386}

1849final private static class ShortDataInputWrapper extends AbstractShortIterator {
1850        final private DataInput dataInput;
1851        private boolean toAdvance = true;
1852        private boolean endOfProcess = false;
1853        private short next;
1854        public ShortDataInputWrapper(final DataInput dataInput) {
1855         this.dataInput = dataInput;
1856        }
1857        public boolean hasNext() {
1858         if (! toAdvance) return ! endOfProcess;
1859         toAdvance = false;
1860         try {
1861          next = dataInput.readShort();
1862         }
1863         catch(EOFException eof) {
1864          endOfProcess = true;
1865         }
1866         catch(IOException rethrow) { throw new RuntimeException(rethrow); }
1867         return ! endOfProcess;
1868        }
1869        public short nextShort() {
1870         if (! hasNext()) throw new NoSuchElementException();
1871         toAdvance = true;
1872         return next;
1873        }
1874}
