017import it.unimi.dsi.fastutil.Hash;
018import it.unimi.dsi.fastutil.HashCommon;
019import static it.unimi.dsi.fastutil.HashCommon.arraySize;

021import java.util.Arrays;
022import java.util.Collection;

025/**  A type-specific hash set with with a fast, small-footprint implementation.
026        *
027        * <P>Instances of this class use a hash table to represent a set. The table is
028        * enlarged as needed by doubling its size when new entries are created, but it is <em>never</em> made
029        * smaller (even on a {@link #clear()}). A family of {@linkplain #trim() trimming
030        * methods} lets you control the size of the table; this is particularly useful
031        * if you reuse instances of this class.
032        *
033        * @see Hash
034        * @see HashCommon
035        */
036public class ReferenceOpenHashSet <K> extends AbstractReferenceSet <K> implements java.io.Serializable, Cloneable, Hash {
037        private static final long serialVersionUID = 0L;
038        private static final boolean ASSERTS = false;
039        /** The array of keys. */
040        protected transient K[] key;
041        /** The mask for wrapping a position counter. */
042        protected transient int mask;
043        /** Whether this set contains the null key. */
044        protected transient boolean containsNull;
045        /** The current table size. Note that an additional element is allocated for storing the null key. */
046        protected transient int n;
047        /** Threshold after which we rehash. It must be the table size times {@link #f}. */
048        protected transient int maxFill;
049        /** Number of entries in the set (including the null key, if present). */
050        protected int size;
051        /** The acceptable load factor. */
052        protected final float f;
053        /** Creates a new hash set.
054         *
055         * <p>The actual table size will be the least power of two greater than <code>expected</code>/<code>f</code>.
056         *
057         * @param expected the expected number of elements in the hash set.
058         * @param f the load factor.
059         */
060        @SuppressWarnings("unchecked")
061        public ReferenceOpenHashSet(final int expected, final float f) {
062         if (f <= 0 || f > 1) throw new IllegalArgumentException("Load factor must be greater than 0 and smaller than or equal to 1");
063         if (expected < 0) throw new IllegalArgumentException("The expected number of elements must be nonnegative");
064         this.f = f;
065         n = arraySize(expected, f);
066         mask = n - 1;
067         maxFill = maxFill(n, f);
068         key = (K[]) new Object[n + 1];
069        }

075         this(expected, DEFAULT_LOAD_FACTOR);

133        /** Creates a new hash set and fills it with the elements of a given array.

170        private int realSize() {
171         return containsNull ? size - 1 : size;
172        }
173        private void ensureCapacity(final int capacity) {
174         final int needed = arraySize(capacity, f);
175         if (needed > n) rehash(needed);
176        }
177        private void tryCapacity(final long capacity) {
178         final int needed = (int)Math.min(1 << 30, Math.max(2, HashCommon.nextPowerOfTwo((long)Math.ceil(capacity / f))));
179         if (needed > n) rehash(needed);
180        }
181        @Override
182        public boolean addAll(Collection<? extends K> c) {
183         // The resulting collection will be at least c.size() big
184         if (f <= .5) ensureCapacity(c.size()); // The resulting collection will be sized for c.size() elements
185         else tryCapacity(size() + c.size()); // The resulting collection will be tentatively sized for size() + c.size() elements
186         return super.addAll(c);
187        }
188        @Override
189        public boolean add(final K k) {
190         int pos;
191         if (( (k) == (null) )) {
192          if (containsNull) return false;
193          containsNull = true;
194         }
195         else {
196          K curr;
197          final K[] key = this.key;
198          // The starting point.
199          if (! ( (curr = key[pos = ( it.unimi.dsi.fastutil.HashCommon.mix( System.identityHashCode(k) ) ) & mask]) == (null) )) {
200           if (( (curr) == (k) )) return false;
201           while(! ( (curr = key[pos = (pos + 1) & mask]) == (null) ))
202            if (( (curr) == (k) )) return false;
203          }
204          key[pos] = k;
205         }
206         if (size++ >= maxFill) rehash(arraySize(size + 1, f));
207         if (ASSERTS) checkTable();
208         return true;
209        }
210        /** Shifts left entries with the specified hash code, starting at the specified position,
211         * and empties the resulting free entry.
212         *
213         * @param pos a starting position.
214         */
215        protected final void shiftKeys(int pos) {
216         // Shift entries with the same hash.
217         int last, slot;
218         K curr;
219         final K[] key = this.key;
220         for(;;) {
221          pos = ((last = pos) + 1) & mask;
222          for(;;) {
223           if (( (curr = key[pos]) == (null) )) {
224            key[last] = (null);
225            return;
226           }
227           slot = ( it.unimi.dsi.fastutil.HashCommon.mix( System.identityHashCode(curr) ) ) & mask;
228           if (last <= pos ? last >= slot || slot > pos : last >= slot && slot > pos) break;
229           pos = (pos + 1) & mask;
230          }
231          key[last] = curr;
232         }
233        }
234        private boolean removeEntry(final int pos) {
235         size--;
236         shiftKeys(pos);
237         if (size < maxFill / 4 && n > DEFAULT_INITIAL_SIZE) rehash(n / 2);
238         return true;
239        }
240        private boolean removeNullEntry() {
241         containsNull = false;
242         key[n] = (null);
243         size--;
244         if (size < maxFill / 4 && n > DEFAULT_INITIAL_SIZE) rehash(n / 2);
245         return true;
246        }
247        @SuppressWarnings("unchecked")
248        @Override
249        public boolean remove(final Object k) {
250         if (( ((K) k) == (null) )) {
251          if (containsNull) return removeNullEntry();
252          return false;
253         }
254         K curr;
255         final K[] key = this.key;
256         int pos;
257         // The starting point.
258         if (( (curr = key[pos = ( it.unimi.dsi.fastutil.HashCommon.mix( System.identityHashCode(k) ) ) & mask]) == (null) )) return false;
259         if (( (k) == (curr) )) return removeEntry(pos);
260         while(true) {
261          if (( (curr = key[pos = (pos + 1) & mask]) == (null) )) return false;
262          if (( (k) == (curr) )) return removeEntry(pos);
263         }
264        }
265        @SuppressWarnings("unchecked")
266        @Override
267        public boolean contains(final Object k) {
268         if (( ((K) k) == (null) )) return containsNull;
269         K curr;
270         final K[] key = this.key;
271         int pos;
272         // The starting point.
273         if (( (curr = key[pos = ( it.unimi.dsi.fastutil.HashCommon.mix( System.identityHashCode(k) ) ) & mask]) == (null) )) return false;
274         if (( (k) == (curr) )) return true;
275         while(true) {
276          if (( (curr = key[pos = (pos + 1) & mask]) == (null) )) return false;
277          if (( (k) == (curr) )) return true;
278         }
279        }
280        /* Removes all elements from this set.
281         *
282         * <P>To increase object reuse, this method does not change the table size.
283         * If you want to reduce the table size, you must use {@link #trim()}.
284         *
285         */
286        @Override
287        public void clear() {
288         if (size == 0) return;
289         size = 0;
290         containsNull = false;
291         Arrays.fill(key, (null));
292        }

318        /** An iterator over a hash set. */
319        private class SetIterator extends AbstractObjectIterator <K> {
320         /** The index of the last entry returned, if positive or zero; initially, {@link #n}. If negative, the last
321                        element returned was that of index {@code - pos - 1} from the {@link #wrapped} list. */
322         int pos = n;
323         /** The index of the last entry that has been returned (more precisely, the value of {@link #pos} if {@link #pos} is positive,
324                        or {@link Integer#MIN_VALUE} if {@link #pos} is negative). It is -1 if either
325                        we did not return an entry yet, or the last returned entry has been removed. */
326         int last = -1;
327         /** A downward counter measuring how many entries must still be returned. */
328         int c = size;
329         /** A boolean telling us whether we should return the null key. */
330         boolean mustReturnNull = ReferenceOpenHashSet.this.containsNull;
331         /** A lazily allocated list containing elements that have wrapped around the table because of removals. */
332         ReferenceArrayList <K> wrapped;
333         public boolean hasNext() {
334          return c != 0;
335         }
336         public K next() {
337          if (! hasNext()) throw new NoSuchElementException();
338          c--;
339          if (mustReturnNull) {
340           mustReturnNull = false;
341           last = n;
342           return key[n];
343          }
344          final K key[] = ReferenceOpenHashSet.this.key;
345          for(;;) {
346           if (--pos < 0) {
347            // We are just enumerating elements from the wrapped list.
348            last = Integer.MIN_VALUE;
349            return wrapped.get(- pos - 1);
350           }
351           if (! ( (key[pos]) == (null) )) return key[last = pos];
352          }
353         }
354         /** Shifts left entries with the specified hash code, starting at the specified position,
355                 * and empties the resulting free entry.
356                 *
357                 * @param pos a starting position.
358                 */
359         private final void shiftKeys(int pos) {
360          // Shift entries with the same hash.
361          int last, slot;
362          K curr;
363          final K[] key = ReferenceOpenHashSet.this.key;
364          for(;;) {
365           pos = ((last = pos) + 1) & mask;
366           for(;;) {
367            if (( (curr = key[pos]) == (null) )) {
368             key[last] = (null);
369             return;
370            }
371            slot = ( it.unimi.dsi.fastutil.HashCommon.mix( System.identityHashCode(curr) ) ) & mask;
372            if (last <= pos ? last >= slot || slot > pos : last >= slot && slot > pos) break;
373            pos = (pos + 1) & mask;
374           }
375           if (pos < last) { // Wrapped entry.
376            if (wrapped == null) wrapped = new ReferenceArrayList <K>(2);
377            wrapped.add(key[pos]);
378           }
379           key[last] = curr;
380          }
381         }
382         public void remove() {
383          if (last == -1) throw new IllegalStateException();
384          if (last == n) {
385           ReferenceOpenHashSet.this.containsNull = false;
386           ReferenceOpenHashSet.this.key[n] = (null);
387          }
388          else if (pos >= 0) shiftKeys(last);
389          else {
390           // We're removing wrapped entries.
391           ReferenceOpenHashSet.this.remove(wrapped.set(- pos - 1, null));
392           last = -1; // Note that we must not decrement size
393           return;
394          }
395          size--;
396          last = -1; // You can no longer remove this entry.
397          if (ASSERTS) checkTable();
398         }
399        }
400        @Override
401        public ObjectIterator <K> iterator() {
402         return new SetIterator();

404        /** A no-op for backward compatibility. The kind of tables implemented by
405         * this class never need rehashing.
406         *
407         * <P>If you need to reduce the table size to fit exactly
408         * this set, use {@link #trim()}.
409         *
410         * @return true.
411         * @see #trim()
412         * @deprecated A no-op.
413         */
414        @Deprecated

416         return true;
417        }
418        /** Rehashes this set, making the table as small as possible.
419         *
420         * <P>This method rehashes the table to the smallest size satisfying the
421         * load factor. It can be used when the set will not be changed anymore, so
422         * to optimize access speed and size.
423         *
424         * <P>If the table size is already the minimum possible, this method
425         * does nothing.
426         *
427         * @return true if there was enough memory to trim the set.
428         * @see #trim(int)
429         */
430        public boolean trim() {
431         final int l = arraySize(size, f);
432         if (l >= n || size > maxFill(l, f)) return true;
433         try {
434          rehash(l);
435         }
436         catch(OutOfMemoryError cantDoIt) { return false; }
437         return true;
438        }
439        /** Rehashes this set if the table is too large.
440         *
441         * <P>Let <var>N</var> be the smallest table size that can hold
442         * <code>max(n,{@link #size()})</code> entries, still satisfying the load factor. If the current
443         * table size is smaller than or equal to <var>N</var>, this method does
444         * nothing. Otherwise, it rehashes this set in a table of size
445         * <var>N</var>.
446         *
447         * <P>This method is useful when reusing sets.  {@linkplain #clear() Clearing a
448         * set} leaves the table size untouched. If you are reusing a set
449         * many times, you can call this method with a typical
450         * size to avoid keeping around a very large table just
451         * because of a few large transient sets.
452         *
453         * @param n the threshold for the trimming.
454         * @return true if there was enough memory to trim the set.
455         * @see #trim()
456         */
457        public boolean trim(final int n) {
458         final int l = HashCommon.nextPowerOfTwo((int)Math.ceil(n / f));
459         if (l >= n || size > maxFill(l, f)) return true;
460         try {
461          rehash(l);
462         }
463         catch(OutOfMemoryError cantDoIt) { return false; }
464         return true;
465        }
466        /** Rehashes the set.
467         *
468         * <P>This method implements the basic rehashing strategy, and may be
469         * overriden by subclasses implementing different rehashing strategies (e.g.,
470         * disk-based rehashing). However, you should not override this method
471         * unless you understand the internal workings of this class.
472         *
473         * @param newN the new size
474         */
475        @SuppressWarnings("unchecked")
476        protected void rehash(final int newN) {
477         final K key[] = this.key;
478         final int mask = newN - 1; // Note that this is used by the hashing macro
479         final K newKey[] = (K[]) new Object[newN + 1];
480         int i = n, pos;
481         for(int j = realSize(); j-- != 0;) {
482          while(( (key[--i]) == (null) ));
483          if (! ( (newKey[pos = ( it.unimi.dsi.fastutil.HashCommon.mix( System.identityHashCode(key[i]) ) ) & mask]) == (null) ))
484           while (! ( (newKey[pos = (pos + 1) & mask]) == (null) ));
485          newKey[pos] = key[i];
486         }
487         n = newN;
488         this.mask = mask;
489         maxFill = maxFill(n, f);
490         this.key = newKey;
491        }
492        /** Returns a deep copy of this set.
493         *
494         * <P>This method performs a deep copy of this hash set; the data stored in the
495         * set, however, is not cloned. Note that this makes a difference only for object keys.
496         *
497         *  @return a deep copy of this set.
498         */
499        @Override
500        @SuppressWarnings("unchecked")
501        public ReferenceOpenHashSet <K> clone() {
502         ReferenceOpenHashSet <K> c;
503         try {
504          c = (ReferenceOpenHashSet <K>)super.clone();
505         }
506         catch(CloneNotSupportedException cantHappen) {
507          throw new InternalError();
508         }
509         c.key = key.clone();
510         c.containsNull = containsNull;
511         return c;
512        }
513        /** Returns a hash code for this set.
514         *
515         * This method overrides the generic method provided by the superclass.
516         * Since <code>equals()</code> is not overriden, it is important
517         * that the value returned by this method is the same value as
518         * the one returned by the overriden method.
519         *
520         * @return a hash code for this set.
521         */
522        @Override
523        public int hashCode() {
524         int h = 0;
525         for(int j = realSize(), i = 0; j-- != 0;) {
526          while(( (key[i]) == (null) )) i++;
527          if (this != key[i])
528           h += ( System.identityHashCode(key[i]) );
529          i++;
530         }
531         // Zero / null have hash zero.
532         return h;
533        }
534        private void writeObject(java.io.ObjectOutputStream s) throws java.io.IOException {
535         final ObjectIterator <K> i = iterator();
536         s.defaultWriteObject();
537         for(int j = size; j-- != 0;) s.writeObject(i.next());
538        }
539        @SuppressWarnings("unchecked")
540        private void readObject(java.io.ObjectInputStream s) throws java.io.IOException, ClassNotFoundException {
541         s.defaultReadObject();
542         n = arraySize(size, f);
543         maxFill = maxFill(n, f);
544         mask = n - 1;
545         final K key[] = this.key = (K[]) new Object[n + 1];
546         K k;
547         for(int i = size, pos; i-- != 0;) {
548          k = (K) s.readObject();
549          if (( (k) == (null) )) {
550           pos = n;
551           containsNull = true;
552          }
553          else {
554           if (! ( (key[pos = ( it.unimi.dsi.fastutil.HashCommon.mix( System.identityHashCode(k) ) ) & mask]) == (null) ))
555            while (! ( (key[pos = (pos + 1) & mask]) == (null) ));
556          }
557          key[pos] = k;
558         }
559         if (ASSERTS) checkTable();
560        }
561        private void checkTable() {}
