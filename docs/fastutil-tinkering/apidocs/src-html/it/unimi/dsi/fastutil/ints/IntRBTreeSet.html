022/** A type-specific red-black tree set with a fast, small-footprint implementation.
023        *
024        * <P>The iterators provided by this class are type-specific {@link
025        * it.unimi.dsi.fastutil.BidirectionalIterator bidirectional iterators}.
026        * Moreover, the iterator returned by <code>iterator()</code> can be safely cast
027        * to a type-specific {@linkplain java.util.ListIterator list iterator}.
028        */
029public class IntRBTreeSet extends AbstractIntSortedSet implements java.io.Serializable, Cloneable, IntSortedSet {
030        /** A reference to the root entry. */
031        protected transient Entry tree;
032        /** Number of elements in this set. */
033        protected int count;
034        /** The entry of the first element of this set. */
035        protected transient Entry firstEntry;
036        /** The entry of the last element of this set. */
037        protected transient Entry lastEntry;
038        /** This set's comparator, as provided in the constructor. */
039        protected Comparator<? super Integer> storedComparator;
040        /** This set's actual comparator; it may differ from {@link #storedComparator} because it is
041                always a type-specific comparator, so it could be derived from the former by wrapping. */
042        protected transient IntComparator actualComparator;
043        private static final long serialVersionUID = -7046029254386353130L;
044        private static final boolean ASSERTS = false;

048        /** Creates a new empty tree set.
049         */
050        public IntRBTreeSet() {
051         tree = null;
052         count = 0;
053        }
054        /** Generates the comparator that will be actually used.
055         *
056         * <P>When a specific {@link Comparator} is specified and stored in {@link
057         * #storedComparator}, we must check whether it is type-specific.  If it is
058         * so, we can used directly, and we store it in {@link #actualComparator}. Otherwise,
059         * we generate on-the-fly an anonymous class that wraps the non-specific {@link Comparator}
060         * and makes it into a type-specific one.
061         */
062        private void setActualComparator() {
063         /* If the provided comparator is already type-specific, we use it. Otherwise,
064                   we use a wrapper anonymous class to fake that it is type-specific. */
065         if (storedComparator == null || storedComparator instanceof IntComparator) actualComparator = (IntComparator)storedComparator;
066         else actualComparator = new IntComparator () {
067           public int compare(int k1, int k2) {
068            return storedComparator.compare((Integer.valueOf(k1)), (Integer.valueOf(k2)));
069           }
070           public int compare(Integer ok1, Integer ok2) {
071            return storedComparator.compare(ok1, ok2);
072           }
073          };
074        }

171        /*
172         * The following methods implements some basic building blocks used by
173         * all accessors.  They are (and should be maintained) identical to those used in RBTreeMap.drv.
174         *
175         * The add()/remove() code is derived from Ben Pfaff's GNU libavl
176         * (http://www.msu.edu/~pfaffben/avl/). If you want to understand what's
177         * going on, you should have a look at the literate code contained therein
178         * first.
179         */
180        /** Compares two keys in the right way.
181         *
182         * <P>This method uses the {@link #actualComparator} if it is non-{@code null}.
183         * Otherwise, it resorts to primitive type comparisons or to {@link Comparable#compareTo(Object) compareTo()}.
184         *
185         * @param k1 the first key.
186         * @param k2 the second key.
187         * @return a number smaller than, equal to or greater than 0, as usual
188         * (i.e., when k1 &lt; k2, k1 = k2 or k1 &gt; k2, respectively).
189         */
190
191        final int compare(final int k1, final int k2) {
192         return actualComparator == null ? ( Integer.compare((k1),(k2)) ) : actualComparator.compare(k1, k2);
193        }
194        /** Returns the entry corresponding to the given key, if it is in the tree; {@code null}, otherwise.
195         *
196         * @param k the key to search for.
197         * @return the corresponding entry, or {@code null} if no entry with the given key exists.
198         */
199        private Entry findKey(final int k) {
200         Entry e = tree;
201         int cmp;
202         while (e != null && (cmp = compare(k, e.key)) != 0)
203          e = cmp < 0 ? e.left() : e.right();
204         return e;
205        }
206        /** Locates a key.
207         *
208         * @param k a key.
209         * @return the last entry on a search for the given key; this will be
210         * the given key, if it present; otherwise, it will be either the smallest greater key or the greatest smaller key.
211         */
212        final Entry locateKey(final int k) {
213         Entry e = tree, last = tree;
214         int cmp = 0;
215         while (e != null && (cmp = compare(k, e.key)) != 0) {
216          last = e;
217          e = cmp < 0 ? e.left() : e.right();
218         }
219         return cmp == 0 ? e : last;
220        }
221        /** This vector remembers the path and the direction followed during the
222         *  current insertion. It suffices for about 2<sup>32</sup> entries. */
223        private transient boolean dirPath[];
224        private transient Entry nodePath[];
225
226        private void allocatePaths() {
227         dirPath = new boolean[64];
228         nodePath = new Entry[64];
229        }
230        @Override
231        public boolean add(final int k) {
232         int maxDepth = 0;
233         if (tree == null) { // The case of the empty tree is treated separately.
234          count++;
235          tree = lastEntry = firstEntry = new Entry (k);
236         }
237         else {
238          Entry p = tree, e;
239          int cmp, i = 0;
240          while(true) {
241           if ((cmp = compare(k, p.key)) == 0) {
242            // We clean up the node path, or we could have stale references later.
243            while(i-- != 0) nodePath[i] = null;
244            return false;
245           }
246           nodePath[i] = p;
247           if (dirPath[i++] = cmp > 0) {
248            if (p.succ()) {
249             count++;
250             e = new Entry (k);
251             if (p.right == null) lastEntry = e;
252             e.left = p;
253             e.right = p.right;
254             p.right(e);
255             break;
256            }
257            p = p.right;
258           }
259           else {
260            if (p.pred()) {
261             count++;
262             e = new Entry (k);
263             if (p.left == null) firstEntry = e;
264             e.right = p;
265             e.left = p.left;
266             p.left(e);
267             break;
268            }
269            p = p.left;
270           }
271          }
272          maxDepth = i--;
273          while(i > 0 && ! nodePath[i].black()) {
274           if (! dirPath[i - 1]) {
275            Entry y = nodePath[i - 1].right;
276            if (! nodePath[i - 1].succ() && ! y.black()) {
277             nodePath[i].black(true);
278             y.black(true);
279             nodePath[i - 1].black(false);
280             i -= 2;
281            }
282            else {
283             Entry x;
284             if (! dirPath[i]) y = nodePath[i];
285             else {
286              x = nodePath[i];
287              y = x.right;
288              x.right = y.left;
289              y.left = x;
290              nodePath[i - 1].left = y;
291              if (y.pred()) {
292               y.pred(false);
293               x.succ(y);
294              }
295             }
296             x = nodePath[i - 1];
297             x.black(false);
298             y.black(true);
299             x.left = y.right;
300             y.right = x;
301             if (i < 2) tree = y;
302             else {
303              if (dirPath[i - 2]) nodePath[i - 2].right = y;
304              else nodePath[i - 2].left = y;
305             }
306             if (y.succ()) {
307              y.succ(false);
308              x.pred(y);
309             }
310             break;
311            }
312           }
313           else {
314            Entry y = nodePath[i - 1].left;
315            if (! nodePath[i - 1].pred() && ! y.black()) {
316             nodePath[i].black(true);
317             y.black(true);
318             nodePath[i - 1].black(false);
319             i -= 2;
320            }
321            else {
322             Entry x;
323             if (dirPath[i]) y = nodePath[i];
324             else {
325              x = nodePath[i];
326              y = x.left;
327              x.left = y.right;
328              y.right = x;
329              nodePath[i - 1].right = y;
330              if (y.succ()) {
331               y.succ(false);
332               x.pred(y);
333              }
334             }
335             x = nodePath[i - 1];
336             x.black(false);
337             y.black(true);
338             x.right = y.left;
339             y.left = x;
340             if (i < 2) tree = y;
341             else {
342              if (dirPath[i - 2]) nodePath[i - 2].right = y;
343              else nodePath[i - 2].left = y;
344             }
345             if (y.pred()){
346              y.pred(false);
347              x.succ(y);
348             }
349             break;
350            }
351           }
352          }
353         }
354         tree.black(true);
355         // We clean up the node path, or we could have stale references later.
356         while(maxDepth-- != 0) nodePath[maxDepth] = null;
357         if (ASSERTS) {
358          checkNodePath();
359          checkTree(tree, 0, -1);
360         }
361         return true;
362        }

364        @Override
365        public boolean remove(final int k) {
366         if (tree == null) return false;
367         Entry p = tree;
368         int cmp;
369         int i = 0;
370         final int kk = k;
371         while(true) {
372          if ((cmp = compare(kk, p.key)) == 0) break;
373          dirPath[i] = cmp > 0;
374          nodePath[i] = p;
375          if (dirPath[i++]) {
376           if ((p = p.right()) == null) {
377            // We clean up the node path, or we could have stale references later.
378            while(i-- != 0) nodePath[i] = null;
379            return false;
380           }
381          }
382          else {
383           if ((p = p.left()) == null) {
384            // We clean up the node path, or we could have stale references later.
385            while(i-- != 0) nodePath[i] = null;
386            return false;
387           }
388          }
389         }
390         if (p.left == null) firstEntry = p.next();
391         if (p.right == null) lastEntry = p.prev();
392         if (p.succ()) {
393          if (p.pred()) {
394           if (i == 0) tree = p.left;
395           else {
396            if (dirPath[i - 1]) nodePath[i - 1].succ(p.right);
397            else nodePath[i - 1].pred(p.left);
398           }
399          }
400          else {
401           p.prev().right = p.right;
402           if (i == 0) tree = p.left;
403           else {
404            if (dirPath[i - 1]) nodePath[i - 1].right = p.left;
405            else nodePath[i - 1].left = p.left;
406           }
407          }
408         }
409         else {
410          boolean color;
411          Entry r = p.right;
412          if (r.pred()) {
413           r.left = p.left;
414           r.pred(p.pred());
415           if (! r.pred()) r.prev().right = r;
416           if (i == 0) tree = r;
417           else {
418            if (dirPath[i - 1]) nodePath[i - 1].right = r;
419            else nodePath[i - 1].left = r;
420           }
421           color = r.black();
422           r.black(p.black());
423           p.black(color);
424           dirPath[i] = true;
425           nodePath[i++] = r;
426          }
427          else {
428           Entry s;
429           int j = i++;
430           while(true) {
431            dirPath[i] = false;
432            nodePath[i++] = r;
433            s = r.left;
434            if (s.pred()) break;
435            r = s;
436           }
437           dirPath[j] = true;
438           nodePath[j] = s;
439           if (s.succ()) r.pred(s);
440           else r.left = s.right;
441           s.left = p.left;
442           if (! p.pred()) {
443            p.prev().right = s;
444            s.pred(false);
445           }
446           s.right(p.right);
447           color = s.black();
448           s.black(p.black());
449           p.black(color);
450           if (j == 0) tree = s;
451           else {
452            if (dirPath[j - 1]) nodePath[j - 1].right = s;
453            else nodePath[j - 1].left = s;
454           }
455          }
456         }
457         int maxDepth = i;
458         if (p.black()) {
459          for(; i > 0; i--) {
460           if (dirPath[i - 1] && ! nodePath[i - 1].succ() ||
461             ! dirPath[i - 1] && ! nodePath[i - 1].pred()) {
462            Entry x = dirPath[i - 1] ? nodePath[i - 1].right : nodePath[i - 1].left;
463            if (! x.black()) {
464             x.black(true);
465             break;
466            }
467           }
468           if (! dirPath[i - 1]) {
469            Entry w = nodePath[i - 1].right;
470            if (! w.black()) {
471             w.black(true);
472             nodePath[i - 1].black(false);
473             nodePath[i - 1].right = w.left;
474             w.left = nodePath[i - 1];
475             if (i < 2) tree = w;
476             else {
477              if (dirPath[i - 2]) nodePath[i - 2].right = w;
478              else nodePath[i - 2].left = w;
479             }
480             nodePath[i] = nodePath[i - 1];
481             dirPath[i] = false;
482             nodePath[i - 1] = w;
483             if (maxDepth == i++) maxDepth++;
484             w = nodePath[i - 1].right;
485            }
486            if ((w.pred() || w.left.black()) &&
487              (w.succ() || w.right.black())) {
488             w.black(false);
489            }
490            else {
491             if (w.succ() || w.right.black()) {
492              Entry y = w.left;
493              y.black (true);
494              w.black(false);
495              w.left = y.right;
496              y.right = w;
497              w = nodePath[i - 1].right = y;
498              if (w.succ()) {
499               w.succ(false);
500               w.right.pred(w);
501              }
502             }
503             w.black(nodePath[i - 1].black());
504             nodePath[i - 1].black(true);
505             w.right.black(true);
506             nodePath[i - 1].right = w.left;
507             w.left = nodePath[i - 1];
508             if (i < 2) tree = w;
509             else {
510              if (dirPath[i - 2]) nodePath[i - 2].right = w;
511              else nodePath[i - 2].left = w;
512             }
513             if (w.pred()) {
514              w.pred(false);
515              nodePath[i - 1].succ(w);
516             }
517             break;
518            }
519           }
520           else {
521            Entry w = nodePath[i - 1].left;
522            if (! w.black()) {
523             w.black (true);
524             nodePath[i - 1].black(false);
525             nodePath[i - 1].left = w.right;
526             w.right = nodePath[i - 1];
527             if (i < 2) tree = w;
528             else {
529              if (dirPath[i - 2]) nodePath[i - 2].right = w;
530              else nodePath[i - 2].left = w;
531             }
532             nodePath[i] = nodePath[i - 1];
533             dirPath[i] = true;
534             nodePath[i - 1] = w;
535             if (maxDepth == i++) maxDepth++;
536             w = nodePath[i - 1].left;
537            }
538            if ((w.pred() || w.left.black()) &&
539              (w.succ() || w.right.black())) {
540             w.black(false);
541            }
542            else {
543             if (w.pred() || w.left.black()) {
544              Entry y = w.right;
545              y.black(true);
546              w.black (false);
547              w.right = y.left;
548              y.left = w;
549              w = nodePath[i - 1].left = y;
550              if (w.pred()) {
551               w.pred(false);
552               w.left.succ(w);
553              }
554             }
555             w.black(nodePath[i - 1].black());
556             nodePath[i - 1].black(true);
557             w.left.black(true);
558             nodePath[i - 1].left = w.right;
559             w.right = nodePath[i - 1];
560             if (i < 2) tree = w;
561             else {
562              if (dirPath[i - 2]) nodePath[i - 2].right = w;
563              else nodePath[i - 2].left = w;
564             }
565             if (w.succ()) {
566              w.succ(false);
567              nodePath[i - 1].pred(w);
568             }
569             break;
570            }
571           }
572          }
573          if (tree != null) tree.black(true);
574         }
575         count--;
576         // We clean up the node path, or we could have stale references later.
577         while(maxDepth-- != 0) nodePath[maxDepth] = null;
578         if (ASSERTS) {
579          checkNodePath();
580          checkTree(tree, 0, -1);
581         }
582         return true;
583        }

595        /** This class represent an entry in a tree set.
596         *
597         * <P>We use the only "metadata", i.e., {@link Entry#info}, to store
598         * information about color, predecessor status and successor status.
599         *
600         * <P>Note that since the class is recursive, it can be
601         * considered equivalently a tree.
602         */
603        private static final class Entry implements Cloneable {
604         /** The the bit in this mask is true, the node is black. */
605         private final static int BLACK_MASK = 1;
606         /** If the bit in this mask is true, {@link #right} points to a successor. */
607         private final static int SUCC_MASK = 1 << 31;
608         /** If the bit in this mask is true, {@link #left} points to a predecessor. */
609         private final static int PRED_MASK = 1 << 30;
610         /** The key of this entry. */
611         int key;
612         /** The pointers to the left and right subtrees. */
613         Entry left, right;
614         /** This integers holds different information in different bits (see {@link #SUCC_MASK}, {@link #PRED_MASK} and {@link #BLACK_MASK}). */
615         int info;
616         Entry() {}
617         /** Creates a new red entry with the given key.
618                 *
619                 * @param k a key.
620                 */
621         Entry(final int k) {
622          this.key = k;
623          info = SUCC_MASK | PRED_MASK;
624         }
625         /** Returns the left subtree.
626                 *
627                 * @return the left subtree ({@code null} if the left
628                 * subtree is empty).
629                 */
630         Entry left() {
631          return (info & PRED_MASK) != 0 ? null : left;
632         }
633         /** Returns the right subtree.
634                 *
635                 * @return the right subtree ({@code null} if the right
636                 * subtree is empty).
637                 */
638         Entry right() {
639          return (info & SUCC_MASK) != 0 ? null : right;
640         }
641         /** Checks whether the left pointer is really a predecessor.
642                 * @return true if the left pointer is a predecessor.
643                 */
644         boolean pred() {
645          return (info & PRED_MASK) != 0;
646         }
647         /** Checks whether the right pointer is really a successor.
648                 * @return true if the right pointer is a successor.
649                 */
650         boolean succ() {
651          return (info & SUCC_MASK) != 0;
652         }
653         /** Sets whether the left pointer is really a predecessor.
654                 * @param pred if true then the left pointer will be considered a predecessor.
655                 */
656         void pred(final boolean pred) {
657          if (pred) info |= PRED_MASK;
658          else info &= ~PRED_MASK;
659         }
660         /** Sets whether the right pointer is really a successor.
661                 * @param succ if true then the right pointer will be considered a successor.
662                 */
663         void succ(final boolean succ) {
664          if (succ) info |= SUCC_MASK;
665          else info &= ~SUCC_MASK;
666         }
667         /** Sets the left pointer to a predecessor.
668                 * @param pred the predecessr.
669                 */
670         void pred(final Entry pred) {
671          info |= PRED_MASK;
672          left = pred;
673         }
674         /** Sets the right pointer to a successor.
675                 * @param succ the successor.
676                 */
677         void succ(final Entry succ) {
678          info |= SUCC_MASK;
679          right = succ;
680         }

682                 * @param left the new left subtree.

685          info &= ~PRED_MASK;

703                 */

709                 *
710                 * @return the next entry ({@code null}) if this is the last entry).
711                 */
712         Entry next() {
713          Entry next = this.right;
714          if ((info & SUCC_MASK) == 0) while ((next.info & PRED_MASK) == 0) next = next.left;
715          return next;
716         }

718                 *
719                 * @return the previous entry ({@code null}) if this is the first entry).
720                 */
721         Entry prev() {
722          Entry prev = this.left;
723          if ((info & PRED_MASK) == 0) while ((prev.info & SUCC_MASK) == 0) prev = prev.right;
724          return prev;
725         }

801        /** An iterator on the whole range.
802         *
803         * <P>This class can iterate in both directions on a threaded tree.
804         */
805        private class SetIterator extends AbstractIntListIterator {
806         /** The entry that will be returned by the next call to {@link java.util.ListIterator#previous()} (or {@code null} if no previous entry exists). */
807         Entry prev;
808         /** The entry that will be returned by the next call to {@link java.util.ListIterator#next()} (or {@code null} if no next entry exists). */
809         Entry next;
810         /** The last entry that was returned (or {@code null} if we did not iterate or used {@link #remove()}). */
811         Entry curr;
812         /** The current index (in the sense of a {@link java.util.ListIterator}). Note that this value is not meaningful when this iterator has been created using the nonempty constructor.*/
813         int index = 0;
814         SetIterator() {
815          next = firstEntry;
816         }
817         SetIterator(final int k) {
818          if ((next = locateKey(k)) != null) {
819           if (compare(next.key, k) <= 0) {
820            prev = next;
821            next = next.next();
822           }
823           else prev = next.prev();
824          }
825         }
826         public boolean hasNext() { return next != null; }
827         public boolean hasPrevious() { return prev != null; }
828         void updateNext() {
829          next = next.next();
830         }
831         Entry nextEntry() {
832          if (! hasNext()) throw new NoSuchElementException();
833          curr = prev = next;
834          index++;
835          updateNext();
836          return curr;
837         }
838         public int nextInt() { return nextEntry().key; }
839         public int previousInt() { return previousEntry().key; }
840         void updatePrevious() {
841          prev = prev.prev();
842         }
843         Entry previousEntry() {
844          if (! hasPrevious()) throw new NoSuchElementException();
845          curr = next = prev;
846          index--;
847          updatePrevious();
848          return curr;
849         }
850         public int nextIndex() {
851          return index;
852         }
853         public int previousIndex() {
854          return index - 1;
855         }
856         public void remove() {
857          if (curr == null) throw new IllegalStateException();
858          /* If the last operation was a next(), we are removing an entry that preceeds
859                           the current index, and thus we must decrement it. */
860          if (curr == prev) index--;
861          next = prev = curr;
862          updatePrevious();
863          updateNext();
864          IntRBTreeSet.this.remove(curr.key);
865          curr = null;
866         }
867        }

892        /** A subset with given range.
893         *
894         * <P>This class represents a subset. One has to specify the left/right
895         * limits (which can be set to -&infin; or &infin;). Since the subset is a
896         * view on the set, at a given moment it could happen that the limits of
897         * the range are not any longer in the main set. Thus, things such as
898         * {@link java.util.SortedSet#first()} or {@link java.util.Collection#size()} must be always computed
899         * on-the-fly.
900         */
901        private final class Subset extends AbstractIntSortedSet implements java.io.Serializable, IntSortedSet {
902         private static final long serialVersionUID = -7046029254386353129L;
903         /** The start of the subset range, unless {@link #bottom} is true. */
904         int from;
905         /** The end of the subset range, unless {@link #top} is true. */
906         int to;
907         /** If true, the subset range starts from -&infin;. */
908         boolean bottom;
909         /** If true, the subset range goes to &infin;. */
910         boolean top;
911         /** Creates a new subset with given key range.
912                 *
913                 * @param from the start of the subset range.
914                 * @param bottom if true, the first parameter is ignored and the range starts from -&infin;.
915                 * @param to the end of the subset range.
916                 * @param top if true, the third parameter is ignored and the range goes to &infin;.
917                 */
918         public Subset(final int from, final boolean bottom, final int to, final boolean top) {
919          if (! bottom && ! top && IntRBTreeSet.this.compare(from, to) > 0) throw new IllegalArgumentException("Start element (" + from + ") is larger than end element (" + to + ")");
920          this.from = from;
921          this.bottom = bottom;
922          this.to = to;
923          this.top = top;
924         }
925         @Override
926         public void clear() {
927          final SubsetIterator i = new SubsetIterator();
928          while(i.hasNext()) {
929           i.nextInt();
930           i.remove();
931          }
932         }
933         /** Checks whether a key is in the subset range.
934                 * @param k a key.
935                 * @return true if is the key is in the subset range.
936                 */
937         final boolean in(final int k) {
938          return (bottom || IntRBTreeSet.this.compare(k, from) >= 0) &&
939           (top || IntRBTreeSet.this.compare(k, to) < 0);
940         }

950         }

996          if (! top) to = compare(to, this.to) < 0 ? to : this.to;
997          if (! bottom) from = compare(from, this.from) > 0 ? from : this.from;
998             if (! top && ! bottom && from == this.from && to == this.to) return this;

1001         /** Locates the first entry.
1002                 *
1003                 * @return the first entry of this subset, or {@code null} if the subset is empty.
1004                 */
1005         public IntRBTreeSet.Entry firstEntry() {
1006          if (tree == null) return null;
1007          // If this subset goes to -infinity, we return the main set first entry; otherwise, we locate the start of the set.
1008          IntRBTreeSet.Entry e;
1009          if (bottom) e = firstEntry;
1010          else {
1011           e = locateKey(from);
1012           // If we find either the start or something greater we're OK.
1013           if (compare(e.key, from) < 0) e = e.next();
1014          }
1015          // Finally, if this subset doesn't go to infinity, we check that the resulting key isn't greater than the end.
1016          if (e == null || ! top && compare(e.key, to) >= 0) return null;
1017          return e;
1018         }
1019         /** Locates the last entry.
1020                 *
1021                 * @return the last entry of this subset, or {@code null} if the subset is empty.
1022                 */

1024          if (tree == null) return null;
1025          // If this subset goes to infinity, we return the main set last entry; otherwise, we locate the end of the set.
1026          IntRBTreeSet.Entry e;
1027          if (top) e = lastEntry;
1028          else {
1029           e = locateKey(to);
1030           // If we find something smaller than the end we're OK.
1031           if (compare(e.key, to) >= 0) e = e.prev();
1032          }
1033          // Finally, if this subset doesn't go to -infinity, we check that the resulting key isn't smaller than the start.
1034          if (e == null || ! bottom && compare(e.key, from) < 0) return null;
1035          return e;
1036         }

1049         /** An iterator for subranges.
1050                 *
1051                 * <P>This class inherits from {@link SetIterator}, but overrides the methods that
1052                 * update the pointer after a {@link java.util.ListIterator#next()} or {@link java.util.ListIterator#previous()}. If we would
1053                 * move out of the range of the subset we just overwrite the next or previous
1054                 * entry with {@code null}.
1055                 */
1056         private final class SubsetIterator extends SetIterator {
1057          SubsetIterator() {
1058           next = firstEntry();
1059          }
1060          SubsetIterator(final int k) {
1061           this();
1062           if (next != null) {
1063            if (! bottom && compare(k, next.key) < 0) prev = null;
1064            else if (! top && compare(k, (prev = lastEntry()).key) >= 0) next = null;
1065            else {
1066             next = locateKey(k);
1067             if (compare(next.key, k) <= 0) {
1068              prev = next;
1069              next = next.next();
1070             }
1071             else prev = next.prev();
1072            }
1073           }
1074          }
1075          void updatePrevious() {
1076           prev = prev.prev();
1077           if (! bottom && prev != null && IntRBTreeSet.this.compare(prev.key, from) < 0) prev = null;
1078          }
1079          void updateNext() {
1080           next = next.next();
1081           if (! top && next != null && IntRBTreeSet.this.compare(next.key, to) >= 0) next = null;
1082          }
1083         }
1084        }

1086         *

1088         * set, however, is not cloned. Note that this makes a difference only for object keys.
1089         *
1090         * @return a deep copy of this tree set.

1103         if (count != 0) {

1106          p = rp;
1107          rp.left(tree);
1108          q = rq;

1130             }

1152        /** Reads the given number of entries from the input stream, returning the corresponding tree.
1153         *
1154         * @param s the input stream.
1155         * @param n the (positive) number of entries to read.
1156         * @param pred the entry containing the key that preceeds the first key in the tree.
1157         * @param succ the entry containing the key that follows the last key in the tree.
1158         */
1159
1160        private Entry readTree(final java.io.ObjectInputStream s, final int n, final Entry pred, final Entry succ) throws java.io.IOException, ClassNotFoundException {
1161         if (n == 1) {
1162          final Entry top = new Entry ( s.readInt());
1163          top.pred(pred);
1164          top.succ(succ);
1165          top.black(true);
1166          return top;
1167         }
1168         if (n == 2) {
1169          /* We handle separately this case so that recursion will
1170                         *always* be on nonempty subtrees. */
1171          final Entry top = new Entry ( s.readInt());
1172          top.black(true);
1173          top.right(new Entry ( s.readInt()));
1174          top.right.pred(top);
1175          top.pred(pred);
1176          top.right.succ(succ);
1177          return top;
1178         }
1179         // The right subtree is the largest one.
1180         final int rightN = n / 2, leftN = n - rightN - 1;
1181         final Entry top = new Entry ();
1182         top.left(readTree(s, leftN, pred, top));
1183         top.key = s.readInt();
1184         top.black(true);
1185         top.right(readTree(s, rightN, top, succ));
1186         if (n + 2 == ((n + 2) & -(n + 2))) top.right.black(false); // Quick test for determining whether n + 2 is a power of 2.
1187         return top;
1188        }
