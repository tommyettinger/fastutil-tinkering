040/** A class providing static methods and objects that do useful things with {@linkplain BigArrays big arrays}.
041        *
042        * <p>In particular, the <code>ensureCapacity()</code>, <code>grow()</code>,

046        * <P>Note that {@link it.unimi.dsi.fastutil.io.BinIO} and {@link it.unimi.dsi.fastutil.io.TextIO}
047        * contain several methods that make it possible to load and save big arrays of primitive types as sequences
048        * of elements in {@link java.io.DataInput} format (i.e., not as objects) or as sequences of lines of text.
049        *
050        * @see BigArrays

094        /** Copies a big array from the specified source big array, beginning at the specified position, to the specified position of the destination big array.
095         * Handles correctly overlapping regions of the same big array.
096         *
097         * @param srcArray the source big array.
098         * @param srcPos the starting position in the source big array.
099         * @param destArray the destination big array.
100         * @param destPos the starting position in the destination data.
101         * @param length the number of elements to be copied.
102         */
103        public static void copy(final boolean[][] srcArray, final long srcPos, final boolean[][] destArray, final long destPos, long length) {
104         if (destPos <= srcPos) {
105          int srcSegment = segment(srcPos);
106          int destSegment = segment(destPos);
107          int srcDispl = displacement(srcPos);
108          int destDispl = displacement(destPos);
109          int l;
110          while(length > 0) {
111           l = (int)Math.min(length, Math.min(srcArray[srcSegment].length - srcDispl, destArray[destSegment].length - destDispl));
112           System.arraycopy(srcArray[srcSegment], srcDispl, destArray[destSegment], destDispl, l);
113           if ((srcDispl += l) == SEGMENT_SIZE) {
114            srcDispl = 0;
115            srcSegment++;
116           }
117           if ((destDispl += l) == SEGMENT_SIZE) {
118            destDispl = 0;
119            destSegment++;
120           }
121           length -= l;
122          }
123         }
124         else {
125          int srcSegment = segment(srcPos + length);
126          int destSegment = segment(destPos + length);
127          int srcDispl = displacement(srcPos + length);
128          int destDispl = displacement(destPos + length);
129          int l;
130          while(length > 0) {
131           if (srcDispl == 0) {
132            srcDispl = SEGMENT_SIZE;
133            srcSegment--;
134           }
135           if (destDispl == 0) {
136            destDispl = SEGMENT_SIZE;
137            destSegment--;
138           }
139           l = (int)Math.min(length, Math.min(srcDispl, destDispl));
140           System.arraycopy(srcArray[srcSegment], srcDispl - l, destArray[destSegment], destDispl - l, l);
141           srcDispl -= l;
142           destDispl -= l;
143           length -= l;
144          }
145         }
146        }

148         *

151         * @param destArray the destination array.

153         * @param length the number of elements to be copied.

155        public static void copyFromBig(final boolean[][] srcArray, final long srcPos, final boolean[] destArray, int destPos, int length) {
156         int srcSegment = segment(srcPos);
157         int srcDispl = displacement(srcPos);
158         int l;
159         while(length > 0) {
160          l = Math.min(srcArray[srcSegment].length - srcDispl, length);
161          System.arraycopy(srcArray[srcSegment], srcDispl, destArray, destPos, l);
162          if ((srcDispl += l) == SEGMENT_SIZE) {
163           srcDispl = 0;
164           srcSegment++;
165          }
166          destPos += l;
167          length -= l;
168         }
169        }

184          System.arraycopy(srcArray, srcPos, destArray[destSegment], destDispl, l);

193        /** Creates a new big array.
194         *
195         * @param length the length of the new big array.
196         * @return a new big array of given length.
197         */
198        public static boolean[][] newBigArray(final long length) {
199         if (length == 0) return EMPTY_BIG_ARRAY;
200         ensureLength(length);
201         final int baseLength = (int)((length + SEGMENT_MASK) >>> SEGMENT_SHIFT);
202         boolean[][] base = new boolean[baseLength][];
203         final int residual = (int)(length & SEGMENT_MASK);
204         if (residual != 0) {
205          for(int i = 0; i < baseLength - 1; i++) base[i] = new boolean[SEGMENT_SIZE];
206          base[baseLength - 1] = new boolean[residual];
207         }
208         else for(int i = 0; i < baseLength; i++) base[i] = new boolean[SEGMENT_SIZE];
209         return base;
210        }
211        /** Turns a standard array into a big array.
212         *
213         * <P>Note that the returned big array might contain as a segment the original array.
214         *
215         * @param array an array.
216         * @return a new big array with the same length and content of <code>array</code>.
217         */
218        public static boolean[][] wrap(final boolean[] array) {
219         if (array.length == 0) return EMPTY_BIG_ARRAY;
220         if (array.length <= SEGMENT_SIZE) return new boolean[][] { array };
221         final boolean[][] bigArray = newBigArray(array.length);
222         for(int i = 0; i < bigArray.length; i++) System.arraycopy(array, (int)start(i), bigArray[i], 0, bigArray[i].length);
223         return bigArray;
224        }
225        /** Ensures that a big array can contain the given number of entries.
226         *
227         * <P>If you cannot foresee whether this big array will need again to be
228         * enlarged, you should probably use <code>grow()</code> instead.
229         *
230         * <p><strong>Warning:</strong> the returned array might use part of the segments of the original
231         * array, which must be considered read-only after calling this method.
232         *
233         * @param array a big array.
234         * @param length the new minimum length for this big array.
235         * @return <code>array</code>, if it contains <code>length</code> entries or more; otherwise,
236         * a big array with <code>length</code> entries whose first <code>length(array)</code>
237         * entries are the same as those of <code>array</code>.
238         */
239        public static boolean[][] ensureCapacity(final boolean[][] array, final long length) {
240         return ensureCapacity(array, length, length(array));
241        }
242        /** Ensures that a big array can contain the given number of entries, preserving just a part of the big array.
243         *
244         * <p><strong>Warning:</strong> the returned array might use part of the segments of the original
245         * array, which must be considered read-only after calling this method.
246         *
247         * @param array a big array.
248         * @param length the new minimum length for this big array.
249         * @param preserve the number of elements of the big array that must be preserved in case a new allocation is necessary.
250         * @return <code>array</code>, if it can contain <code>length</code> entries or more; otherwise,
251         * a big array with <code>length</code> entries whose first <code>preserve</code>
252         * entries are the same as those of <code>array</code>.
253         */
254        public static boolean[][] ensureCapacity(final boolean[][] array, final long length, final long preserve) {
255         final long oldLength = length(array);
256         if (length > oldLength) {
257          ensureLength(length);
258          final int valid = array.length - (array.length == 0 || array.length > 0 && array[array.length - 1].length == SEGMENT_SIZE ? 0 : 1);
259          final int baseLength = (int)((length + SEGMENT_MASK) >>> SEGMENT_SHIFT);
260          final boolean[][] base = Arrays.copyOf(array, baseLength);
261          final int residual = (int)(length & SEGMENT_MASK);
262          if (residual != 0) {
263           for(int i = valid; i < baseLength - 1; i++) base[i] = new boolean[SEGMENT_SIZE];
264           base[baseLength - 1] = new boolean[residual];
265          }
266          else for(int i = valid; i < baseLength; i++) base[i] = new boolean[SEGMENT_SIZE];
267          if (preserve - (valid * (long)SEGMENT_SIZE) > 0) copy(array, valid * (long)SEGMENT_SIZE, base, valid * (long)SEGMENT_SIZE, preserve - (valid * (long)SEGMENT_SIZE));
268          return base;
269         }
270         return array;
271        }
272        /** Grows the given big array to the maximum between the given length and
273         * the current length multiplied by two, provided that the given
274         * length is larger than the current length.
275         *
276         * <P>If you want complete control on the big array growth, you
277         * should probably use <code>ensureCapacity()</code> instead.
278         *
279         * <p><strong>Warning:</strong> the returned array might use part of the segments of the original
280         * array, which must be considered read-only after calling this method.
281         *
282         * @param array a big array.
283         * @param length the new minimum length for this big array.
284         * @return <code>array</code>, if it can contain <code>length</code>
285         * entries; otherwise, a big array with
286         * max(<code>length</code>,<code>length(array)</code>/&phi;) entries whose first
287         * <code>length(array)</code> entries are the same as those of <code>array</code>.
288         * */
289        public static boolean[][] grow(final boolean[][] array, final long length) {
290         final long oldLength = length(array);
291         return length > oldLength ? grow(array, length, oldLength) : array;
292        }
293        /** Grows the given big array to the maximum between the given length and
294         * the current length multiplied by two, provided that the given
295         * length is larger than the current length, preserving just a part of the big array.
296         *
297         * <P>If you want complete control on the big array growth, you
298         * should probably use <code>ensureCapacity()</code> instead.
299         *
300         * <p><strong>Warning:</strong> the returned array might use part of the segments of the original
301         * array, which must be considered read-only after calling this method.
302         *
303         * @param array a big array.
304         * @param length the new minimum length for this big array.
305         * @param preserve the number of elements of the big array that must be preserved in case a new allocation is necessary.
306         * @return <code>array</code>, if it can contain <code>length</code>
307         * entries; otherwise, a big array with
308         * max(<code>length</code>,<code>length(array)</code>/&phi;) entries whose first
309         * <code>preserve</code> entries are the same as those of <code>array</code>.

314        }
315        /** Trims the given big array to the given length.
316         *
317         * <p><strong>Warning:</strong> the returned array might use part of the segments of the original
318         * array, which must be considered read-only after calling this method.
319         *
320         * @param array a big array.
321         * @param length the new maximum length for the big array.
322         * @return <code>array</code>, if it contains <code>length</code>
323         * entries or less; otherwise, a big array with
324         * <code>length</code> entries whose entries are the same as
325         * the first <code>length</code> entries of <code>array</code>.
326         *
327         */
328        public static boolean[][] trim(final boolean[][] array, final long length) {
329         ensureLength(length);
330         final long oldLength = length(array);
331         if (length >= oldLength) return array;
332         final int baseLength = (int)((length + SEGMENT_MASK) >>> SEGMENT_SHIFT);
333         final boolean[][] base = Arrays.copyOf(array, baseLength);
334         final int residual = (int)(length & SEGMENT_MASK);
335         if (residual != 0) base[baseLength - 1] = BooleanArrays.trim(base[baseLength - 1], residual);
336         return base;
337        }
338        /** Sets the length of the given big array.
339         *
340         * <p><strong>Warning:</strong> the returned array might use part of the segments of the original
341         * array, which must be considered read-only after calling this method.
342         *
343         * @param array a big array.
344         * @param length the new length for the big array.
345         * @return <code>array</code>, if it contains exactly <code>length</code>
346         * entries; otherwise, if it contains <em>more</em> than
347         * <code>length</code> entries, a big array with <code>length</code> entries
348         * whose entries are the same as the first <code>length</code> entries of
349         * <code>array</code>; otherwise, a big array with <code>length</code> entries
350         * whose first <code>length(array)</code> entries are the same as those of
351         * <code>array</code>.
352         *
353         */
354        public static boolean[][] setLength(final boolean[][] array, final long length) {
355         final long oldLength = length(array);
356         if (length == oldLength) return array;
357         if (length < oldLength) return trim(array, length);
358         return ensureCapacity(array, length);
359        }

395        /** Fills a portion of the given big array with the given value.
396         *
397         * <P>If possible (i.e., <code>from</code> is 0) this method uses a
398         * backward loop. In this case, it is significantly faster than the
399         * corresponding method in {@link java.util.Arrays}.
400         *
401         * @param array a big array.
402         * @param from the starting index of the portion to fill.
403         * @param to the end index of the portion to fill.
404         * @param value the new value for all elements of the specified portion of the big array.
405         */
406        public static void fill(final boolean[][] array, final long from, long to, final boolean value) {
407         final long length = length(array);
408         BigArrays.ensureFromTo(length, from, to);
409         int fromSegment = segment(from);
410         int toSegment = segment(to);
411         int fromDispl = displacement(from);
412         int toDispl = displacement(to);
413         if (fromSegment == toSegment) {
414          Arrays.fill(array[fromSegment], fromDispl, toDispl, value);
415          return;
416         }
417         if (toDispl != 0) Arrays.fill(array[toSegment], 0, toDispl, value);
418         while(--toSegment > fromSegment) Arrays.fill(array[toSegment], value);
419         Arrays.fill(array[fromSegment], fromDispl, SEGMENT_SIZE, value);
420        }
421        /** Returns true if the two big arrays are elementwise equal.
422         *
423         * <P>This method uses a backward loop. It is significantly faster than the corresponding
424         * method in {@link java.util.Arrays}.
425         *
426         * @param a1 a big array.
427         * @param a2 another big array.
428         * @return true if the two big arrays are of the same length, and their elements are equal.
429         */
430        public static boolean equals(final boolean[][] a1, final boolean a2[][]) {
431         if (length(a1) != length(a2)) return false;
432         int i = a1.length, j;
433         boolean[] t, u;
434         while(i-- != 0) {
435          t = a1[i];
436          u = a2[i];
437          j = t.length;
438          while(j-- != 0) if (! ( (t[j]) == (u[j]) )) return false;
439         }
440         return true;
441        }
442        /* Returns a string representation of the contents of the specified big array.
443         *
444         * The string representation consists of a list of the big array's elements, enclosed in square brackets ("[]"). Adjacent elements are separated by the characters ", " (a comma followed by a space). Returns "null" if <code>a</code> is null.
445         * @param a the big array whose string representation to return.
446         * @return the string representation of <code>a</code>.
447         */

449         if (a == null) return "null";
450         final long last = length(a) - 1;
451         if (last == - 1) return "[]";
452         final StringBuilder b = new StringBuilder();

456          if (i == last) return b.append(']').toString();
457          b.append(", ");
458         }
459        }

486        /** A type-specific content-based hash strategy for big arrays. */
487        private static final class BigArrayHashStrategy implements Hash.Strategy<boolean[][]>, java.io.Serializable {
488         private static final long serialVersionUID = -7046029254386353129L;
489         public int hashCode(final boolean[][] o) {
490          return java.util.Arrays.deepHashCode(o);

493          return BooleanBigArrays.equals(a, b);

496        /** A type-specific content-based hash strategy for big arrays.
497         *
498         * <P>This hash strategy may be used in custom hash collections whenever keys are
499         * big arrays, and they must be considered equal by content. This strategy
500         * will handle {@code null} correctly, and it is serializable.
501         */
502        @SuppressWarnings({"rawtypes"})
503        public final static Hash.Strategy HASH_STRATEGY = new BigArrayHashStrategy();
504        private static final int SMALL = 7;
505        private static final int MEDIUM = 40;
506        private static void vecSwap(final boolean[][] x, long a, long b, final long n) {
507         for(int i = 0; i < n; i++, a++, b++) swap(x, a, b);
508        }
509        private static long med3(final boolean x[][], final long a, final long b, final long c, BooleanComparator comp) {
510         int ab = comp.compare(get(x, a), get(x, b));
511         int ac = comp.compare(get(x, a), get(x, c));
512         int bc = comp.compare(get(x, b), get(x, c));
513         return (ab < 0 ?
514          (bc < 0 ? b : ac < 0 ? c : a) :
515          (bc > 0 ? b : ac > 0 ? c : a));
516        }
517        private static void selectionSort(final boolean[][] a, final long from, final long to, final BooleanComparator comp) {
518         for(long i = from; i < to - 1; i++) {
519          long m = i;
520          for(long j = i + 1; j < to; j++) if (comp.compare(BooleanBigArrays.get(a, j), BooleanBigArrays.get(a, m)) < 0) m = j;
521          if (m != i) swap(a, i, m);
522         }
523        }
524        /** Sorts the specified range of elements according to the order induced by the specified
525         * comparator using quicksort.
526         *
527         * <p>The sorting algorithm is a tuned quicksort adapted from Jon L. Bentley and M. Douglas
528         * McIlroy, &ldquo;Engineering a Sort Function&rdquo;, <i>Software: Practice and Experience</i>, 23(11), pages
529         * 1249&minus;1265, 1993.
530         *
531         * @param x the big array to be sorted.
532         * @param from the index of the first element (inclusive) to be sorted.
533         * @param to the index of the last element (exclusive) to be sorted.
534         * @param comp the comparator to determine the sorting order.
535         */
536        public static void quickSort(final boolean[][] x, final long from, final long to, final BooleanComparator comp) {
537         final long len = to - from;
538         // Selection sort on smallest arrays
539         if (len < SMALL) {
540          selectionSort(x, from, to, comp);
541          return;
542         }
543         // Choose a partition element, v
544         long m = from + len / 2; // Small arrays, middle element
545         if (len > SMALL) {
546          long l = from;
547          long n = to - 1;
548          if (len > MEDIUM) { // Big arrays, pseudomedian of 9
549           long s = len / 8;
550           l = med3(x, l, l + s, l + 2 * s, comp);
551           m = med3(x, m - s, m, m + s, comp);
552           n = med3(x, n - 2 * s, n - s, n, comp);
553          }
554          m = med3(x, l, m, n, comp); // Mid-size, med of 3
555         }
556         final boolean v = get(x, m);
557         // Establish Invariant: v* (<v)* (>v)* v*
558         long a = from, b = a, c = to - 1, d = c;
559         while(true) {
560          int comparison;
561          while (b <= c && (comparison = comp.compare(get(x, b), v)) <= 0) {
562           if (comparison == 0) swap(x, a++, b);
563           b++;
564          }
565          while (c >= b && (comparison = comp.compare(get(x, c), v)) >=0) {
566           if (comparison == 0) swap(x, c, d--);
567           c--;
568          }
569          if (b > c) break;
570          swap(x, b++, c--);
571         }
572         // Swap partition elements back to middle
573         long s, n = to;
574         s = Math.min(a - from, b - a);
575         vecSwap(x, from, b - s, s);
576         s = Math.min(d - c, n - d- 1);
577         vecSwap(x, b, n - s, s);
578         // Recursively sort non-partition-elements
579         if ((s = b - a) > 1) quickSort(x, from, from + s, comp);
580         if ((s = d - c) > 1) quickSort(x, n - s, n, comp);
581        }

583        private static long med3(final boolean x[][], final long a, final long b, final long c) {
584         int ab = ( Boolean.compare((get(x, a)),(get(x, b))) );
585         int ac = ( Boolean.compare((get(x, a)),(get(x, c))) );
586         int bc = ( Boolean.compare((get(x, b)),(get(x, c))) );
587         return (ab < 0 ?
588          (bc < 0 ? b : ac < 0 ? c : a) :
589          (bc > 0 ? b : ac > 0 ? c : a));
590        }

592        private static void selectionSort(final boolean[][] a, final long from, final long to) {
593         for(long i = from; i < to - 1; i++) {
594          long m = i;
595          for(long j = i + 1; j < to; j++) if (( !(BooleanBigArrays.get(a, j)) && (BooleanBigArrays.get(a, m)) )) m = j;
596          if (m != i) swap(a, i, m);
597         }
598        }

681        /** Shuffles the specified big array fragment using the specified pseudorandom number generator.
682         *
683         * @param a the big array to be shuffled.
684         * @param from the index of the first element (inclusive) to be shuffled.
685         * @param to the index of the last element (exclusive) to be shuffled.
686         * @param random a pseudorandom number generator (please use a <a href="http://dsiutils.dsi.unimi.it/docs/it/unimi/dsi/util/XorShiftStarRandom.html">XorShift*</a> generator).
687         * @return <code>a</code>.
688         */
689        public static boolean[][] shuffle(final boolean[][] a, final long from, final long to, final Random random) {
690         for(long i = to - from; i-- != 0;) {
691          final long p = (random.nextLong() & 0x7FFFFFFFFFFFFFFFL) % (i + 1);
692          final boolean t = get(a, from + i);
693          set(a, from + i, get(a, from + p));
694          set(a, from + p, t);
695         }
696         return a;
697        }
