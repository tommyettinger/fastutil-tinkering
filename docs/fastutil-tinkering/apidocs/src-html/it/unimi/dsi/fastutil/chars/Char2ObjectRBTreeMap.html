029/** A type-specific red-black tree map with a fast, small-footprint implementation.
030        *
031        * <P>The iterators provided by the views of this class are type-specific {@linkplain
032        * it.unimi.dsi.fastutil.BidirectionalIterator bidirectional iterators}.
033        * Moreover, the iterator returned by <code>iterator()</code> can be safely cast
034        * to a type-specific {@linkplain java.util.ListIterator list iterator}.
035        *
036        */

072        /** Generates the comparator that will be actually used.
073         *
074         * <P>When a specific {@link Comparator} is specified and stored in {@link
075         * #storedComparator}, we must check whether it is type-specific.  If it is
076         * so, we can used directly, and we store it in {@link #actualComparator}. Otherwise,
077         * we generate on-the-fly an anonymous class that wraps the non-specific {@link Comparator}
078         * and makes it into a type-specific one.
079         */
080        private void setActualComparator() {
081         /* If the provided comparator is already type-specific, we use it. Otherwise,
082                   we use a wrapper anonymous class to fake that it is type-specific. */
083         if (storedComparator == null || storedComparator instanceof CharComparator) actualComparator = (CharComparator)storedComparator;
084         else actualComparator = new CharComparator () {
085           public int compare(char k1, char k2) {
086            return storedComparator.compare((Character.valueOf(k1)), (Character.valueOf(k2)));
087           }
088           public int compare(Character ok1, Character ok2) {
089            return storedComparator.compare(ok1, ok2);
090           }
091          };
092        }

164        /** Compares two keys in the right way.
165         *
166         * <P>This method uses the {@link #actualComparator} if it is non-{@code null}.
167         * Otherwise, it resorts to primitive type comparisons or to {@link Comparable#compareTo(Object) compareTo()}.
168         *
169         * @param k1 the first key.
170         * @param k2 the second key.
171         * @return a number smaller than, equal to or greater than 0, as usual
172         * (i.e., when k1 &lt; k2, k1 = k2 or k1 &gt; k2, respectively).
173         */
174
175        final int compare(final char k1, final char k2) {
176         return actualComparator == null ? ( Character.compare((k1),(k2)) ) : actualComparator.compare(k1, k2);
177        }
178        /** Returns the entry corresponding to the given key, if it is in the tree; {@code null}, otherwise.
179         *
180         * @param k the key to search for.
181         * @return the corresponding entry, or {@code null} if no entry with the given key exists.
182         */
183        final Entry <V> findKey(final char k) {
184         Entry <V> e = tree;
185         int cmp;
186         while (e != null && (cmp = compare(k, e.key)) != 0) e = cmp < 0 ? e.left() : e.right();
187         return e;
188        }
189        /** Locates a key.
190         *
191         * @param k a key.
192         * @return the last entry on a search for the given key; this will be
193         * the given key, if it present; otherwise, it will be either the smallest greater key or the greatest smaller key.
194         */
195        final Entry <V> locateKey(final char k) {
196         Entry <V> e = tree, last = tree;
197         int cmp = 0;
198         while (e != null && (cmp = compare(k, e.key)) != 0) {
199          last = e;
200          e = cmp < 0 ? e.left() : e.right();
201         }
202         return cmp == 0 ? e : last;
203        }
204        /** This vector remembers the path and the direction followed during the

206        private transient boolean dirPath[];
207        private transient Entry <V> nodePath[];
208        @SuppressWarnings("unchecked")
209        private void allocatePaths() {
210         dirPath = new boolean[64];
211         nodePath = new Entry[64];

213        @Override
214        public V put(final char k, final V v) {
215         Entry <V> e = add(k);
216         final V oldValue = e.value;
217         e.value = v;
218         return oldValue;
219        }
220        /** Returns a node with key k in the balanced tree, creating one with defRetValue if necessary.
221         *
222         * @param k the key
223         * @return a node with key k. If a node with key k already exists, then that node is returned,
224         * otherwise a new node with defRetValue is created ensuring that the tree is balanced
225         * after creation of the node.
226         */
227        private Entry <V> add(final char k) {
228         /* After execution of this method, modified is true iff a new entry has been inserted. */
229         modified = false;
230         int maxDepth = 0;
231         Entry <V> e;
232         if (tree == null) { // The case of the empty tree is treated separately.
233          count++;
234          e = tree = lastEntry = firstEntry = new Entry <V>(k, defRetValue);
235         }
236         else {
237          Entry <V> p = tree;
238          int cmp, i = 0;
239          while(true) {
240           if ((cmp = compare(k, p.key)) == 0) {
241            // We clean up the node path, or we could have stale references later.
242            while(i-- != 0) nodePath[i] = null;
243            return p;
244           }
245           nodePath[i] = p;
246           if (dirPath[i++] = cmp > 0) {
247            if (p.succ()) {
248             count++;
249             e = new Entry <V>(k, defRetValue);
250             if (p.right == null) lastEntry = e;
251             e.left = p;
252             e.right = p.right;
253             p.right(e);
254             break;
255            }
256            p = p.right;
257           }
258           else {
259            if (p.pred()) {
260             count++;
261             e = new Entry <V>(k, defRetValue);
262             if (p.left == null) firstEntry = e;
263             e.right = p;
264             e.left = p.left;
265             p.left(e);
266             break;
267            }
268            p = p.left;
269           }
270          }
271          modified = true;
272          maxDepth = i--;
273          while(i > 0 && ! nodePath[i].black()) {
274           if (! dirPath[i - 1]) {
275            Entry <V> y = nodePath[i - 1].right;
276            if (! nodePath[i - 1].succ() && ! y.black()) {
277             nodePath[i].black(true);
278             y.black(true);
279             nodePath[i - 1].black(false);
280             i -= 2;
281            }
282            else {
283             Entry <V> x;
284             if (! dirPath[i]) y = nodePath[i];
285             else {
286              x = nodePath[i];
287              y = x.right;
288              x.right = y.left;
289              y.left = x;
290              nodePath[i - 1].left = y;
291              if (y.pred()) {
292               y.pred(false);
293               x.succ(y);
294              }
295             }
296             x = nodePath[i - 1];
297             x.black(false);
298             y.black(true);
299             x.left = y.right;
300             y.right = x;
301             if (i < 2) tree = y;
302             else {
303              if (dirPath[i - 2]) nodePath[i - 2].right = y;
304              else nodePath[i - 2].left = y;
305             }
306             if (y.succ()) {
307              y.succ(false);
308              x.pred(y);
309             }
310             break;
311            }
312           }
313           else {
314            Entry <V> y = nodePath[i - 1].left;
315            if (! nodePath[i - 1].pred() && ! y.black()) {
316             nodePath[i].black(true);
317             y.black(true);
318             nodePath[i - 1].black(false);
319             i -= 2;
320            }
321            else {
322             Entry <V> x;
323             if (dirPath[i]) y = nodePath[i];
324             else {
325              x = nodePath[i];
326              y = x.left;
327              x.left = y.right;
328              y.right = x;
329              nodePath[i - 1].right = y;
330              if (y.succ()) {
331               y.succ(false);
332               x.pred(y);
333              }
334             }
335             x = nodePath[i - 1];
336             x.black(false);
337             y.black(true);
338             x.right = y.left;
339             y.left = x;
340             if (i < 2) tree = y;
341             else {
342              if (dirPath[i - 2]) nodePath[i - 2].right = y;
343              else nodePath[i - 2].left = y;
344             }
345             if (y.pred()){
346              y.pred(false);
347              x.succ(y);
348             }
349             break;
350            }
351           }
352          }
353         }
354         tree.black(true);
355         // We clean up the node path, or we could have stale references later.
356         while(maxDepth-- != 0) nodePath[maxDepth] = null;
357         if (ASSERTS) {
358          checkNodePath();
359          checkTree(tree, 0, -1);
360         }
361         return e;
362        }

366        @Override
367        public V remove(final char k) {
368         modified = false;
369         if (tree == null) return defRetValue;
370         Entry <V> p = tree;
371         int cmp;
372         int i = 0;
373         final char kk = k;
374         while(true) {
375          if ((cmp = compare(kk, p.key)) == 0) break;
376          dirPath[i] = cmp > 0;
377          nodePath[i] = p;
378          if (dirPath[i++]) {
379           if ((p = p.right()) == null) {
380            // We clean up the node path, or we could have stale references later.
381            while(i-- != 0) nodePath[i] = null;
382            return defRetValue;
383           }
384          }
385          else {
386           if ((p = p.left()) == null) {
387            // We clean up the node path, or we could have stale references later.
388            while(i-- != 0) nodePath[i] = null;
389            return defRetValue;
390           }
391          }
392         }
393         if (p.left == null) firstEntry = p.next();
394         if (p.right == null) lastEntry = p.prev();
395         if (p.succ()) {
396          if (p.pred()) {
397           if (i == 0) tree = p.left;
398           else {
399            if (dirPath[i - 1]) nodePath[i - 1].succ(p.right);
400            else nodePath[i - 1].pred(p.left);
401           }
402          }
403          else {
404           p.prev().right = p.right;
405           if (i == 0) tree = p.left;
406           else {
407            if (dirPath[i - 1]) nodePath[i - 1].right = p.left;
408            else nodePath[i - 1].left = p.left;
409           }
410          }
411         }
412         else {
413          boolean color;
414          Entry <V> r = p.right;
415          if (r.pred()) {
416           r.left = p.left;
417           r.pred(p.pred());
418           if (! r.pred()) r.prev().right = r;
419           if (i == 0) tree = r;
420           else {
421            if (dirPath[i - 1]) nodePath[i - 1].right = r;
422            else nodePath[i - 1].left = r;
423           }
424           color = r.black();
425           r.black(p.black());
426           p.black(color);
427           dirPath[i] = true;
428           nodePath[i++] = r;
429          }
430          else {
431           Entry <V> s;
432           int j = i++;
433           while(true) {
434            dirPath[i] = false;
435            nodePath[i++] = r;
436            s = r.left;
437            if (s.pred()) break;
438            r = s;
439           }
440           dirPath[j] = true;
441           nodePath[j] = s;
442           if (s.succ()) r.pred(s);
443           else r.left = s.right;
444           s.left = p.left;
445           if (! p.pred()) {
446            p.prev().right = s;
447            s.pred(false);
448           }
449           s.right(p.right);
450           color = s.black();
451           s.black(p.black());
452           p.black(color);
453           if (j == 0) tree = s;
454           else {
455            if (dirPath[j - 1]) nodePath[j - 1].right = s;
456            else nodePath[j - 1].left = s;
457           }
458          }
459         }
460         int maxDepth = i;
461         if (p.black()) {
462          for(; i > 0; i--) {
463           if (dirPath[i - 1] && ! nodePath[i - 1].succ() ||
464             ! dirPath[i - 1] && ! nodePath[i - 1].pred()) {
465            Entry <V> x = dirPath[i - 1] ? nodePath[i - 1].right : nodePath[i - 1].left;
466            if (! x.black()) {
467             x.black(true);
468             break;
469            }
470           }
471           if (! dirPath[i - 1]) {
472            Entry <V> w = nodePath[i - 1].right;
473            if (! w.black()) {
474             w.black(true);
475             nodePath[i - 1].black(false);
476             nodePath[i - 1].right = w.left;
477             w.left = nodePath[i - 1];
478             if (i < 2) tree = w;
479             else {
480              if (dirPath[i - 2]) nodePath[i - 2].right = w;
481              else nodePath[i - 2].left = w;
482             }
483             nodePath[i] = nodePath[i - 1];
484             dirPath[i] = false;
485             nodePath[i - 1] = w;
486             if (maxDepth == i++) maxDepth++;
487             w = nodePath[i - 1].right;
488            }
489            if ((w.pred() || w.left.black()) &&
490              (w.succ() || w.right.black())) {
491             w.black(false);
492            }
493            else {
494             if (w.succ() || w.right.black()) {
495              Entry <V> y = w.left;
496              y.black (true);
497              w.black(false);
498              w.left = y.right;
499              y.right = w;
500              w = nodePath[i - 1].right = y;
501              if (w.succ()) {
502               w.succ(false);
503               w.right.pred(w);
504              }
505             }
506             w.black(nodePath[i - 1].black());
507             nodePath[i - 1].black(true);
508             w.right.black(true);
509             nodePath[i - 1].right = w.left;
510             w.left = nodePath[i - 1];
511             if (i < 2) tree = w;
512             else {
513              if (dirPath[i - 2]) nodePath[i - 2].right = w;
514              else nodePath[i - 2].left = w;
515             }
516             if (w.pred()) {
517              w.pred(false);
518              nodePath[i - 1].succ(w);
519             }
520             break;
521            }
522           }
523           else {
524            Entry <V> w = nodePath[i - 1].left;
525            if (! w.black()) {
526             w.black (true);
527             nodePath[i - 1].black(false);
528             nodePath[i - 1].left = w.right;
529             w.right = nodePath[i - 1];
530             if (i < 2) tree = w;
531             else {
532              if (dirPath[i - 2]) nodePath[i - 2].right = w;
533              else nodePath[i - 2].left = w;
534             }
535             nodePath[i] = nodePath[i - 1];
536             dirPath[i] = true;
537             nodePath[i - 1] = w;
538             if (maxDepth == i++) maxDepth++;
539             w = nodePath[i - 1].left;
540            }
541            if ((w.pred() || w.left.black()) &&
542              (w.succ() || w.right.black())) {
543             w.black(false);
544            }
545            else {
546             if (w.pred() || w.left.black()) {
547              Entry <V> y = w.right;
548              y.black(true);
549              w.black (false);
550              w.right = y.left;
551              y.left = w;
552              w = nodePath[i - 1].left = y;
553              if (w.pred()) {
554               w.pred(false);
555               w.left.succ(w);
556              }
557             }
558             w.black(nodePath[i - 1].black());
559             nodePath[i - 1].black(true);
560             w.left.black(true);
561             nodePath[i - 1].left = w.right;
562             w.right = nodePath[i - 1];
563             if (i < 2) tree = w;
564             else {
565              if (dirPath[i - 2]) nodePath[i - 2].right = w;
566              else nodePath[i - 2].left = w;
567             }
568             if (w.succ()) {
569              w.succ(false);
570              nodePath[i - 1].pred(w);
571             }
572             break;
573            }
574           }
575          }
576          if (tree != null) tree.black(true);
577         }
578         modified = true;
579         count--;
580         // We clean up the node path, or we could have stale references later.
581         while(maxDepth-- != 0) nodePath[maxDepth] = null;
582         if (ASSERTS) {
583          checkNodePath();
584          checkTree(tree, 0, -1);
585         }
586         return p.value;
587        }
588        @Override
589        public boolean containsValue(final Object v) {
590         final ValueIterator i = new ValueIterator();
591         Object ev;
592         int j = count;
593         while(j-- != 0) {
594          ev = i.next();
595          if (( (ev) == null ? (v) == null : (ev).equals(v) )) return true;
596         }
597         return false;
598        }

603         entries = null;

608        /** This class represent an entry in a tree map.
609         *
610         * <P>We use the only "metadata", i.e., {@link Entry#info}, to store
611         * information about color, predecessor status and successor status.
612         *
613         * <P>Note that since the class is recursive, it can be
614         * considered equivalently a tree.
615         */
616        private static final class Entry <V> extends AbstractChar2ObjectMap.BasicEntry <V> implements Cloneable {
617         /** The the bit in this mask is true, the node is black. */
618         private final static int BLACK_MASK = 1;
619         /** If the bit in this mask is true, {@link #right} points to a successor. */
620         private final static int SUCC_MASK = 1 << 31;
621         /** If the bit in this mask is true, {@link #left} points to a predecessor. */
622         private final static int PRED_MASK = 1 << 30;
623         /** The pointers to the left and right subtrees. */
624         Entry <V> left, right;
625         /** This integers holds different information in different bits (see {@link #SUCC_MASK} and {@link #PRED_MASK}. */
626         int info;
627         Entry() {
628          super(((char)0), (null));
629         }
630         /** Creates a new entry with the given key and value.
631                 *
632                 * @param k a key.
633                 * @param v a value.
634                 */
635         Entry(final char k, final V v) {
636          super(k, v);
637          info = SUCC_MASK | PRED_MASK;
638         }
639         /** Returns the left subtree.
640                 *
641                 * @return the left subtree ({@code null} if the left
642                 * subtree is empty).
643                 */
644         Entry <V> left() {
645          return (info & PRED_MASK) != 0 ? null : left;
646         }
647         /** Returns the right subtree.
648                 *
649                 * @return the right subtree ({@code null} if the right
650                 * subtree is empty).
651                 */
652         Entry <V> right() {

835        /** An abstract iterator on the whole range.
836         *
837         * <P>This class can iterate in both directions on a threaded tree.
838         */
839        private class TreeIterator {
840         /** The entry that will be returned by the next call to {@link java.util.ListIterator#previous()} (or {@code null} if no previous entry exists). */
841         Entry <V> prev;
842         /** The entry that will be returned by the next call to {@link java.util.ListIterator#next()} (or {@code null} if no next entry exists). */
843         Entry <V> next;
844         /** The last entry that was returned (or {@code null} if we did not iterate or used {@link #remove()}). */
845         Entry <V> curr;
846         /** The current index (in the sense of a {@link java.util.ListIterator}). Note that this value is not meaningful when this {@link TreeIterator} has been created using the nonempty constructor.*/
847         int index = 0;
848         TreeIterator() {
849          next = firstEntry;
850         }
851         TreeIterator(final char k) {
852          if ((next = locateKey(k)) != null) {
853           if (compare(next.key, k) <= 0) {
854            prev = next;
855            next = next.next();
856           }
857           else prev = next.prev();
858          }
859         }
860         public boolean hasNext() { return next != null; }
861         public boolean hasPrevious() { return prev != null; }
862         void updateNext() {
863          next = next.next();
864         }
865         Entry <V> nextEntry() {
866          if (! hasNext()) throw new NoSuchElementException();
867          curr = prev = next;
868          index++;
869          updateNext();
870          return curr;
871         }
872         void updatePrevious() {
873          prev = prev.prev();
874         }
875         Entry <V> previousEntry() {
876          if (! hasPrevious()) throw new NoSuchElementException();
877          curr = next = prev;
878          index--;
879          updatePrevious();
880          return curr;
881         }
882         public int nextIndex() {
883          return index;
884         }
885         public int previousIndex() {
886          return index - 1;
887         }
888         public void remove() {
889          if (curr == null) throw new IllegalStateException();
890          /* If the last operation was a next(), we are removing an entry that preceeds
891                           the current index, and thus we must decrement it. */
892          if (curr == prev) index--;
893          next = prev = curr;
894          updatePrevious();
895          updateNext();
896          Char2ObjectRBTreeMap.this.remove(curr.key);
897          curr = null;
898         }
899         public int skip(final int n) {
900          int i = n;
901          while(i-- != 0 && hasNext()) nextEntry();
902          return n - i - 1;
903         }
904         public int back(final int n) {
905          int i = n;
906          while(i-- != 0 && hasPrevious()) previousEntry();
907          return n - i - 1;
908         }

1310          if (tree == null) return null;
1311          // If this submap goes to -infinity, we return the main map first entry; otherwise, we locate the start of the map.
1312          Char2ObjectRBTreeMap.Entry <V> e;
1313          if (bottom) e = firstEntry;
1314          else {
1315           e = locateKey(from);
1316           // If we find either the start or something greater we're OK.
1317           if (compare(e.key, from) < 0) e = e.next();

1319          // Finally, if this submap doesn't go to infinity, we check that the resulting key isn't greater than the end.
1320          if (e == null || ! top && compare(e.key, to) >= 0) return null;

1353         /** An iterator for subranges.
1354                 *
1355                 * <P>This class inherits from {@link TreeIterator}, but overrides the methods that
1356                 * update the pointer after a {@link java.util.ListIterator#next()} or {@link java.util.ListIterator#previous()}. If we would
1357                 * move out of the range of the submap we just overwrite the next or previous
1358                 * entry with {@code null}.
1359                 */
1360         private class SubmapIterator extends TreeIterator {
1361          SubmapIterator() {
1362           next = firstEntry();
1363          }
1364          SubmapIterator(final char k) {
1365           this();
1366           if (next != null) {
1367            if (! bottom && compare(k, next.key) < 0) prev = null;
1368            else if (! top && compare(k, (prev = lastEntry()).key) >= 0) next = null;
1369            else {
1370             next = locateKey(k);
1371             if (compare(next.key, k) <= 0) {
1372              prev = next;
1373              next = next.next();
1374             }
1375             else prev = next.prev();
1376            }
1377           }
1378          }
1379          void updatePrevious() {
1380           prev = prev.prev();
1381           if (! bottom && prev != null && Char2ObjectRBTreeMap.this.compare(prev.key, from) < 0) prev = null;
1382          }
1383          void updateNext() {
1384           next = next.next();
1385           if (! top && next != null && Char2ObjectRBTreeMap.this.compare(next.key, to) >= 0) next = null;
1386          }
1387         }

1528        /** Reads the given number of entries from the input stream, returning the corresponding tree.
1529         *
1530         * @param s the input stream.
1531         * @param n the (positive) number of entries to read.
1532         * @param pred the entry containing the key that preceeds the first key in the tree.
1533         * @param succ the entry containing the key that follows the last key in the tree.
1534         */
1535        @SuppressWarnings("unchecked")
1536        private Entry <V> readTree(final java.io.ObjectInputStream s, final int n, final Entry <V> pred, final Entry <V> succ) throws java.io.IOException, ClassNotFoundException {
1537         if (n == 1) {
1538          final Entry <V> top = new Entry <V>( s.readChar(), (V) s.readObject());
1539          top.pred(pred);
1540          top.succ(succ);
1541          top.black(true);
1542          return top;
1543         }
1544         if (n == 2) {
1545          /* We handle separately this case so that recursion will
1546                         *always* be on nonempty subtrees. */
1547          final Entry <V> top = new Entry <V>( s.readChar(), (V) s.readObject());
1548          top.black(true);
1549          top.right(new Entry <V>( s.readChar(), (V) s.readObject()));
1550          top.right.pred(top);
1551          top.pred(pred);
1552          top.right.succ(succ);
1553          return top;
1554         }
1555         // The right subtree is the largest one.
1556         final int rightN = n / 2, leftN = n - rightN - 1;
1557         final Entry <V> top = new Entry <V>();
1558         top.left(readTree(s, leftN, pred, top));
1559         top.key = s.readChar();
1560         top.value = (V) s.readObject();
1561         top.black(true);
1562         top.right(readTree(s, rightN, top, succ));
1563         if (n + 2 == ((n + 2) & -(n + 2))) top.right.black(false); // Quick test for determining whether n + 2 is a power of 2.
1564         return top;
1565        }
1566        private void readObject(java.io.ObjectInputStream s) throws java.io.IOException, ClassNotFoundException {
1567         s.defaultReadObject();

1569                   on-the-fly the actualComparator. */
1570         setActualComparator();
1571         allocatePaths();
1572         if (count != 0) {
1573          tree = readTree(s, count, null, null);

1578          e = tree;
1579          while(e.right() != null) e = e.right();
1580          lastEntry = e;
1581         }
1582         if (ASSERTS) checkTree(tree, 0, -1);
1583        }
