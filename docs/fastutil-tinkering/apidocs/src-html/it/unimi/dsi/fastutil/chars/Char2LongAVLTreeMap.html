072        /** Generates the comparator that will be actually used.
073         *
074         * <P>When a specific {@link Comparator} is specified and stored in {@link
075         * #storedComparator}, we must check whether it is type-specific.  If it is
076         * so, we can used directly, and we store it in {@link #actualComparator}. Otherwise,
077         * we generate on-the-fly an anonymous class that wraps the non-specific {@link Comparator}
078         * and makes it into a type-specific one.
079         */
080        private void setActualComparator() {
081         /* If the provided comparator is already type-specific, we use it. Otherwise,
082                   we use a wrapper anonymous class to fake that it is type-specific. */
083         if (storedComparator == null || storedComparator instanceof CharComparator) actualComparator = (CharComparator)storedComparator;
084         else actualComparator = new CharComparator () {
085           public int compare(char k1, char k2) {
086            return storedComparator.compare((Character.valueOf(k1)), (Character.valueOf(k2)));
087           }
088           public int compare(Character ok1, Character ok2) {
089            return storedComparator.compare(ok1, ok2);
090           }
091          };
092        }

184         Entry e = tree;

189        /** Locates a key.
190         *
191         * @param k a key.
192         * @return the last entry on a search for the given key; this will be
193         * the given key, if it present; otherwise, it will be either the smallest greater key or the greatest smaller key.
194         */
195        final Entry locateKey(final char k) {
196         Entry e = tree, last = tree;
197         int cmp = 0;
198         while (e != null && (cmp = compare(k, e.key)) != 0) {
199          last = e;
200          e = cmp < 0 ? e.left() : e.right();
201         }
202         return cmp == 0 ? e : last;
203        }

234        /** Returns a node with key k in the balanced tree, creating one with defRetValue if necessary.
235        *
236        * @param k the key
237        * @return a node with key k. If a node with key k already exists, then that node is returned,
238        *                               otherwise a new node with defRetValue is created ensuring that the tree is balanced
239                                                after creation of the node.
240        */
241        private Entry add(final char k) {
242          /* After execution of this method, modified is true iff a new entry has
243                been inserted. */
244         modified = false;
245         Entry e = null;
246         if (tree == null) { // The case of the empty tree is treated separately.
247          count++;
248          e = tree = lastEntry = firstEntry = new Entry (k, defRetValue);
249          modified = true;
250         }
251         else {
252          Entry p = tree, q = null, y = tree, z = null, w = null;
253          int cmp, i = 0;
254          while(true) {
255           if ((cmp = compare(k, p.key)) == 0) {
256            return p;
257           }
258           if (p.balance() != 0) {
259            i = 0;
260            z = q;
261            y = p;
262           }
263           if (dirPath[i++] = cmp > 0) {
264            if (p.succ()) {
265             count++;
266             e = new Entry (k, defRetValue);
267             modified = true;
268             if (p.right == null) lastEntry = e;
269             e.left = p;
270             e.right = p.right;
271             p.right(e);
272             break;
273            }
274            q = p;
275            p = p.right;
276           }
277           else {
278            if (p.pred()) {
279             count++;
280             e = new Entry (k, defRetValue);
281             modified = true;
282             if (p.left == null) firstEntry = e;
283             e.right = p;
284             e.left = p.left;
285             p.left(e);
286             break;
287            }
288            q = p;
289            p = p.left;
290           }
291          }
292          p = y;
293          i = 0;
294          while(p != e) {
295           if (dirPath[i]) p.incBalance();
296           else p.decBalance();
297           p = dirPath[i++] ? p.right : p.left;
298          }
299          if (y.balance() == -2) {
300           Entry x = y.left;
301           if (x.balance() == -1) {
302            w = x;
303            if (x.succ()) {
304             x.succ(false);
305             y.pred(x);
306            }
307            else y.left = x.right;
308            x.right = y;
309            x.balance(0);
310            y.balance(0);
311           }
312           else {
313            if (ASSERTS) assert x.balance() == 1;
314            w = x.right;
315            x.right = w.left;
316            w.left = x;
317            y.left = w.right;
318            w.right = y;
319            if (w.balance() == -1) {
320             x.balance(0);
321             y.balance(1);
322            }
323            else if (w.balance() == 0) {
324             x.balance(0);
325             y.balance(0);
326            }
327            else {
328             x.balance(-1);
329             y.balance(0);
330            }
331            w.balance(0);
332            if (w.pred()) {
333             x.succ(w);
334             w.pred(false);
335            }
336            if (w.succ()) {
337             y.pred(w);
338             w.succ(false);
339            }
340           }
341          }
342          else if (y.balance() == +2) {
343           Entry x = y.right;
344           if (x.balance() == 1) {
345            w = x;
346            if (x.pred()) {
347             x.pred(false);
348             y.succ(x);
349            }
350            else y.right = x.left;
351            x.left = y;
352            x.balance(0);
353            y.balance(0);
354           }
355           else {
356            if (ASSERTS) assert x.balance() == -1;
357            w = x.left;
358            x.left = w.right;
359            w.right = x;
360            y.right = w.left;
361            w.left = y;
362            if (w.balance() == 1) {
363             x.balance(0);
364             y.balance(-1);
365            }
366            else if (w.balance() == 0) {
367             x.balance(0);
368             y.balance(0);
369            }
370            else {
371             x.balance(1);
372             y.balance(0);
373            }
374            w.balance(0);
375            if (w.pred()) {
376             y.succ(w);
377             w.pred(false);
378            }
379            if (w.succ()) {
380             x.pred(w);
381             w.succ(false);
382            }
383           }
384          }
385          else return e;
386          if (z == null) tree = w;
387          else {
388           if (z.left == y) z.left = w;
389           else z.right = w;
390          }
391         }
392         if (ASSERTS) checkTree(tree);
393         return e;
394        }
395        /** Finds the parent of an entry.
396         *
397         * @param e a node of the tree.
398         * @return the parent of the given node, or {@code null} for the root.
399         */
400        private Entry parent(final Entry e) {
401         if (e == tree) return null;
402         Entry x, y, p;
403         x = y = e;
404         while(true) {
405          if (y.succ()) {
406           p = y.right;
407           if (p == null || p.left != e) {
408            while(! x.pred()) x = x.left;
409            p = x.left;
410           }
411           return p;
412          }
413          else if (x.pred()) {
414           p = x.left;
415           if (p == null || p.right != e) {
416            while(! y.succ()) y = y.right;
417            p = y.right;
418           }
419           return p;
420          }
421          x = x.left;
422          y = y.right;
423         }
424        }

428        public long remove(final char k) {
429         modified = false;
430         if (tree == null) return defRetValue;
431         int cmp;
432         Entry p = tree, q = null;
433         boolean dir = false;
434         final char kk = k;
435         while(true) {
436          if ((cmp = compare(kk, p.key)) == 0) break;
437          else if (dir = cmp > 0) {
438           q = p;
439           if ((p = p.right()) == null) return defRetValue;
440          }
441          else {
442           q = p;
443           if ((p = p.left()) == null) return defRetValue;
444          }
445         }
446         if (p.left == null) firstEntry = p.next();
447         if (p.right == null) lastEntry = p.prev();
448         if (p.succ()) {
449          if (p.pred()) {
450           if (q != null) {
451            if (dir) q.succ(p.right);
452            else q.pred(p.left);
453           }
454           else tree = dir ? p.right : p.left;
455          }
456          else {
457           p.prev().right = p.right;
458           if (q != null) {
459            if (dir) q.right = p.left;
460            else q.left = p.left;
461           }
462           else tree = p.left;
463          }
464         }
465         else {
466          Entry r = p.right;
467          if (r.pred()) {
468           r.left = p.left;
469           r.pred(p.pred());
470           if (! r.pred()) r.prev().right = r;
471           if (q != null) {
472            if (dir) q.right = r;
473            else q.left = r;
474           }
475           else tree = r;
476           r.balance(p.balance());
477           q = r;
478           dir = true;
479          }
480          else {
481           Entry s;
482           while(true) {
483            s = r.left;
484            if (s.pred()) break;
485            r = s;
486           }
487           if (s.succ()) r.pred(s);
488           else r.left = s.right;
489           s.left = p.left;
490           if (! p.pred()) {
491            p.prev().right = s;
492            s.pred(false);
493           }
494           s.right = p.right;
495           s.succ(false);
496           if (q != null) {
497            if (dir) q.right = s;
498            else q.left = s;
499           }
500           else tree = s;
501           s.balance(p.balance());
502           q = r;
503           dir = false;
504          }
505         }
506         Entry y;
507         while(q != null) {
508          y = q;
509          q = parent(y);
510          if (! dir) {
511           dir = q != null && q.left != y;
512           y.incBalance();
513           if (y.balance() == 1) break;
514           else if (y.balance() == 2) {
515            Entry x = y.right;
516            if (ASSERTS) assert x != null;
517            if (x.balance() == -1) {
518             Entry w;
519             if (ASSERTS) assert x.balance() == -1;
520             w = x.left;
521             x.left = w.right;
522             w.right = x;
523             y.right = w.left;
524             w.left = y;
525             if (w.balance() == 1) {
526              x.balance(0);
527              y.balance(-1);
528             }
529             else if (w.balance() == 0) {
530              x.balance(0);
531              y.balance(0);
532             }
533             else {
534              if (ASSERTS) assert w.balance() == -1;
535              x.balance(1);
536              y.balance(0);
537             }
538             w.balance(0);
539             if (w.pred()) {
540              y.succ(w);
541              w.pred(false);
542             }
543             if (w.succ()) {
544              x.pred(w);
545              w.succ(false);
546             }
547             if (q != null) {
548              if (dir) q.right = w;
549              else q.left = w;
550             }
551             else tree = w;
552            }
553            else {
554             if (q != null) {
555              if (dir) q.right = x;
556              else q.left = x;
557             }
558             else tree = x;
559             if (x.balance() == 0) {
560              y.right = x.left;
561              x.left = y;
562              x.balance(-1);
563              y.balance(+1);
564              break;
565             }
566             if (ASSERTS) assert x.balance() == 1;
567             if (x.pred()) {
568              y.succ(true);
569              x.pred(false);
570             }
571             else y.right = x.left;
572             x.left = y;
573             y.balance(0);
574             x.balance(0);
575            }
576           }
577          }
578          else {
579           dir = q != null && q.left != y;
580           y.decBalance();
581           if (y.balance() == -1) break;
582           else if (y.balance() == -2) {
583            Entry x = y.left;
584            if (ASSERTS) assert x != null;
585            if (x.balance() == 1) {
586             Entry w;
587             if (ASSERTS) assert x.balance() == 1;
588             w = x.right;
589             x.right = w.left;
590             w.left = x;
591             y.left = w.right;
592             w.right = y;
593             if (w.balance() == -1) {
594              x.balance(0);
595              y.balance(1);
596             }
597             else if (w.balance() == 0) {
598              x.balance(0);
599              y.balance(0);
600             }
601             else {
602              if (ASSERTS) assert w.balance() == 1;
603              x.balance(-1);
604              y.balance(0);
605             }
606             w.balance(0);
607             if (w.pred()) {
608              x.succ(w);
609              w.pred(false);
610             }
611             if (w.succ()) {
612              y.pred(w);
613              w.succ(false);
614             }
615             if (q != null) {
616              if (dir) q.right = w;
617              else q.left = w;
618             }
619             else tree = w;
620            }
621            else {
622             if (q != null) {
623              if (dir) q.right = x;
624              else q.left = x;
625             }
626             else tree = x;
627             if (x.balance() == 0) {
628              y.left = x.right;
629              x.right = y;
630              x.balance(+1);
631              y.balance(-1);
632              break;
633             }
634             if (ASSERTS) assert x.balance() == -1;
635             if (x.succ()) {
636              y.pred(true);
637              x.succ(false);
638             }
639             else y.left = x.right;
640             x.right = y;
641             y.balance(0);
642             x.balance(0);
643            }
644           }
645          }
646         }
647         modified = true;
648         count--;
649         if (ASSERTS) checkTree(tree);
650         return p.value;
651        }

672        /** This class represent an entry in a tree map.
673         *
674         * <P>We use the only "metadata", i.e., {@link Entry#info}, to store
675         * information about balance, predecessor status and successor status.
676         *
677         * <P>Note that since the class is recursive, it can be
678         * considered equivalently a tree.
679         */
680        private static final class Entry extends AbstractChar2LongMap.BasicEntry implements Cloneable {
681         /** If the bit in this mask is true, {@link #right} points to a successor. */
682         private final static int SUCC_MASK = 1 << 31;
683         /** If the bit in this mask is true, {@link #left} points to a predecessor. */
684         private final static int PRED_MASK = 1 << 30;
685         /** The bits in this mask hold the node balance info. You can get it just by casting to byte. */
686         private final static int BALANCE_MASK = 0xFF;
687         /** The pointers to the left and right subtrees. */
688         Entry left, right;
689         /** This integers holds different information in different bits (see {@link #SUCC_MASK}, {@link #PRED_MASK} and {@link #BALANCE_MASK}). */
690         int info;
691         Entry() {
692          super(((char)0), (0));
693         }
694         /** Creates a new entry with the given key and value.
695                 *
696                 * @param k a key.
697                 * @param v a value.
698                 */

700          super(k, v);

965                           the current index, and thus we must decrement it. */
966          if (curr == prev) index--;

968          updatePrevious();

972         }

1635        /** Reads the given number of entries from the input stream, returning the corresponding tree.
1636         *
1637         * @param s the input stream.
1638         * @param n the (positive) number of entries to read.
1639         * @param pred the entry containing the key that preceeds the first key in the tree.
1640         * @param succ the entry containing the key that follows the last key in the tree.
1641         */
1642
1643        private Entry readTree(final java.io.ObjectInputStream s, final int n, final Entry pred, final Entry succ) throws java.io.IOException, ClassNotFoundException {
1644         if (n == 1) {
1645          final Entry top = new Entry ( s.readChar(), s.readLong());
1646          top.pred(pred);
1647          top.succ(succ);
1648          return top;
1649         }
1650         if (n == 2) {
1651          /* We handle separately this case so that recursion will
1652                         *always* be on nonempty subtrees. */
1653          final Entry top = new Entry ( s.readChar(), s.readLong());
1654          top.right(new Entry ( s.readChar(), s.readLong()));
1655          top.right.pred(top);
1656          top.balance(1);
1657          top.pred(pred);
1658          top.right.succ(succ);
1659          return top;
1660         }
1661         // The right subtree is the largest one.
1662         final int rightN = n / 2, leftN = n - rightN - 1;
1663         final Entry top = new Entry ();
1664         top.left(readTree(s, leftN, pred, top));
1665         top.key = s.readChar();
1666         top.value = s.readLong();
1667         top.right(readTree(s, rightN, top, succ));
1668         if (n == (n & -n)) top.balance(1); // Quick test for determining whether n is a power of 2.
1669         return top;
1670        }
