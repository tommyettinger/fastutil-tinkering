039/**  A type-specific linked hash set with with a fast, small-footprint implementation.
040
041        *
042
043        * <P>Instances of this class use a hash table to represent a set. The table is
044
045        * filled up to a specified <em>load factor</em>, and then doubled in size to
046
047        * accommodate new entries. If the table is emptied below <em>one fourth</em>
048
049        * of the load factor, it is halved in size. However, halving is
050
051        * not performed when deleting entries from an iterator, as it would interfere
052
053        * with the iteration process.
054
055        *
056
057        * <p>Note that {@link #clear()} does not modify the hash table size.
058
059        * Rather, a family of {@linkplain #trim() trimming
060
061        * methods} lets you control the size of the table; this is particularly useful
062
063        * if you reuse instances of this class.
064
065        *
066
067        * <P>Iterators generated by this set will enumerate elements in the same order in which they
068
069        * have been added to the set (addition of elements already present
070
071        * in the set does not change the iteration order). Note that this order has nothing in common with the natural
072
073        * order of the keys. The order is kept by means of a doubly linked list, represented
074
075        * <i>via</i> an array of longs parallel to the table.
076
077        *
078
079        * <P>This class implements the interface of a sorted set, so to allow easy
080
081        * access of the iteration order: for instance, you can get the first element
082
083        * in iteration order with {@code first()} without having to create an
084
085        * iterator; however, this class partially violates the {@link java.util.SortedSet}
086
087        * contract because all subset methods throw an exception and {@link
088
089        * #comparator()} returns always {@code null}.
090
091        *
092
093        * <p>Additional methods, such as <code>addAndMoveToFirst()</code>, make it easy
094
095        * to use instances of this class as a cache (e.g., with LRU policy).
096
097        *
098
099        * <P>The iterators provided by this class are type-specific {@linkplain
100
101        * java.util.ListIterator list iterators}, and can be started at any
102
103        * element <em>which is in the set</em> (if the provided element
104
105        * is not in the set, a {@link NoSuchElementException} exception will be thrown).
106
107        * If, however, the provided element is not the first or last element in the
108
109        * set, the first access to the list index will require linear time, as in the worst case
110
111        * the entire set must be scanned in iteration order to retrieve the positional
112
113        * index of the starting element. If you use just the methods of a type-specific {@link it.unimi.dsi.fastutil.BidirectionalIterator},
114
115        * however, all operations will be performed in constant time.
116
117        *
118
119        * @see Hash
120
121        * @see HashCommon
122
123        */

171        public ByteLinkedOpenCustomHashSet(final int expected, final float f, final it.unimi.dsi.fastutil.bytes.ByteHash.Strategy strategy) {
172         this.strategy = strategy;
173         if (f <= 0 || f > 1) throw new IllegalArgumentException("Load factor must be greater than 0 and smaller than or equal to 1");

178         maxFill = maxFill(n, f);
179         key = new byte[n + 1];
180         link = new long[n + 1];

389        private int realSize() {

392        private void ensureCapacity(final int capacity) {
393         final int needed = arraySize(capacity, f);
394         if (needed > n) rehash(needed);
395        }
396        private void tryCapacity(final long capacity) {
397         final int needed = (int)Math.min(1 << 30, Math.max(2, HashCommon.nextPowerOfTwo((long)Math.ceil(capacity / f))));
398         if (needed > n) rehash(needed);
399        }
400        @Override
401        public boolean addAll(ByteCollection c) {
402         if (f <= .5) ensureCapacity(c.size()); // The resulting collection will be sized for c.size() elements
403         else tryCapacity(size() + c.size()); // The resulting collection will be tentatively sized for size() + c.size() elements
404         return super.addAll(c);
405        }

409         if (f <= .5) ensureCapacity(c.size()); // The resulting collection will be sized for c.size() elements
410         else tryCapacity(size() + c.size()); // The resulting collection will be tentatively sized for size() + c.size() elements

413        @Override
414        public boolean add(final byte k) {
415         int pos;
416         if (( strategy.equals( (k), ((byte)0) ) )) {
417          if (containsNull) return false;
418          pos = n;
419          containsNull = true;
420          key[n] = k;
421         }
422         else {
423          byte curr;
424          final byte[] key = this.key;
425          // The starting point.
426          if (! ( (curr = key[pos = ( it.unimi.dsi.fastutil.HashCommon.mix( strategy.hashCode(k) ) ) & mask]) == ((byte)0) )) {
427           if (( strategy.equals( (curr), (k) ) )) return false;
428           while(! ( (curr = key[pos = (pos + 1) & mask]) == ((byte)0) ))
429            if (( strategy.equals( (curr), (k) ) )) return false;
430          }
431          key[pos] = k;
432         }
433         if (size == 0) {
434          first = last = pos;
435          // Special case of SET_UPPER_LOWER(link[pos], -1, -1);
436          link[pos] = -1L;
437         }
438         else {
439          link[last] ^= ( ( link[last] ^ ( pos & 0xFFFFFFFFL ) ) & 0xFFFFFFFFL );
440          link[pos] = ( ( last & 0xFFFFFFFFL ) << 32 ) | ( -1 & 0xFFFFFFFFL );
441          last = pos;
442         }
443         if (size++ >= maxFill) rehash(arraySize(size + 1, f));
444         if (ASSERTS) checkTable();
445         return true;
446        }
447        /** Shifts left entries with the specified hash code, starting at the specified position,
448
449         * and empties the resulting free entry.
450
451         *
452
453         * @param pos a starting position.
454
455         */
456        protected final void shiftKeys(int pos) {
457         // Shift entries with the same hash.
458         int last, slot;
459         byte curr;
460         final byte[] key = this.key;
461         for(;;) {
462          pos = ((last = pos) + 1) & mask;
463          for(;;) {
464           if (( (curr = key[pos]) == ((byte)0) )) {
465            key[last] = ((byte)0);
466            return;
467           }
468           slot = ( it.unimi.dsi.fastutil.HashCommon.mix( strategy.hashCode(curr) ) ) & mask;
469           if (last <= pos ? last >= slot || slot > pos : last >= slot && slot > pos) break;
470           pos = (pos + 1) & mask;
471          }
472          key[last] = curr;
473          fixPointers(pos, last);
474         }
475        }
476        private boolean removeEntry(final int pos) {
477         size--;
478         fixPointers(pos);
479         shiftKeys(pos);
480         if (size < maxFill / 4 && n > DEFAULT_INITIAL_SIZE) rehash(n / 2);
481         return true;
482        }
483        private boolean removeNullEntry() {
484         containsNull = false;
485         key[n] = ((byte)0);
486         size--;
487         fixPointers(n);
488         if (size < maxFill / 4 && n > DEFAULT_INITIAL_SIZE) rehash(n / 2);
489         return true;
490        }

492        @Override
493        public boolean remove(final byte k) {
494         if (( strategy.equals( ( k), ((byte)0) ) )) {
495          if (containsNull) return removeNullEntry();
496          return false;
497         }
498         byte curr;
499         final byte[] key = this.key;
500         int pos;
501         // The starting point.
502         if (( (curr = key[pos = ( it.unimi.dsi.fastutil.HashCommon.mix( strategy.hashCode(k) ) ) & mask]) == ((byte)0) )) return false;
503         if (( strategy.equals( (k), (curr) ) )) return removeEntry(pos);
504         while(true) {
505          if (( (curr = key[pos = (pos + 1) & mask]) == ((byte)0) )) return false;
506          if (( strategy.equals( (k), (curr) ) )) return removeEntry(pos);
507         }
508        }

510        @Override
511        public boolean contains(final byte k) {
512         if (( strategy.equals( ( k), ((byte)0) ) )) return containsNull;
513         byte curr;
514         final byte[] key = this.key;
515         int pos;
516         // The starting point.
517         if (( (curr = key[pos = ( it.unimi.dsi.fastutil.HashCommon.mix( strategy.hashCode(k) ) ) & mask]) == ((byte)0) )) return false;
518         if (( strategy.equals( (k), (curr) ) )) return true;
519         while(true) {
520          if (( (curr = key[pos = (pos + 1) & mask]) == ((byte)0) )) return false;
521          if (( strategy.equals( (k), (curr) ) )) return true;
522         }
523        }
524        /** Removes the first key in iteration order.
525
526         * @return the first key.
527
528         * @throws NoSuchElementException is this set is empty.
529
530         */
531        public byte removeFirstByte() {
532         if (size == 0) throw new NoSuchElementException();
533         final int pos = first;
534         // Abbreviated version of fixPointers(pos)
535         first = (int) link[pos];
536         if (0 <= first) {
537          // Special case of SET_PREV(link[first], -1)
538          link[first] |= (-1 & 0xFFFFFFFFL) << 32;
539         }
540         final byte k = key[pos];
541         size--;
542         if (( strategy.equals( (k), ((byte)0) ) )) {
543          containsNull = false;
544          key[n] = ((byte)0);
545         }
546         else shiftKeys(pos);
547         if (size < maxFill / 4 && n > DEFAULT_INITIAL_SIZE) rehash(n / 2);
548         return k;
549        }
550        /** Removes the the last key in iteration order.
551
552         * @return the last key.
553
554         * @throws NoSuchElementException is this set is empty.
555
556         */
557        public byte removeLastByte() {
558         if (size == 0) throw new NoSuchElementException();
559         final int pos = last;
560         // Abbreviated version of fixPointers(pos)
561         last = (int) ( link[pos] >>> 32 );
562         if (0 <= last) {
563          // Special case of SET_NEXT(link[last], -1)
564          link[last] |= -1 & 0xFFFFFFFFL;
565         }
566         final byte k = key[pos];
567         size--;
568         if (( strategy.equals( (k), ((byte)0) ) )) {
569          containsNull = false;
570          key[n] = ((byte)0);
571         }
572         else shiftKeys(pos);
573         if (size < maxFill / 4 && n > DEFAULT_INITIAL_SIZE) rehash(n / 2);
574         return k;
575        }
576        private void moveIndexToFirst(final int i) {
577         if (size == 1 || first == i) return;
578         if (last == i) {
579          last = (int) ( link[i] >>> 32 );
580          // Special case of SET_NEXT(link[last], -1);
581          link[last] |= -1 & 0xFFFFFFFFL;
582         }
583         else {
584          final long linki = link[i];
585          final int prev = (int) ( linki >>> 32 );
586          final int next = (int) linki;
587          link[prev] ^= ( ( link[prev] ^ ( linki & 0xFFFFFFFFL ) ) & 0xFFFFFFFFL );
588          link[next] ^= ( ( link[next] ^ ( linki & 0xFFFFFFFF00000000L ) ) & 0xFFFFFFFF00000000L );
589         }
590         link[first] ^= ( ( link[first] ^ ( ( i & 0xFFFFFFFFL ) << 32 ) ) & 0xFFFFFFFF00000000L );
591         link[i] = ( ( -1 & 0xFFFFFFFFL ) << 32 ) | ( first & 0xFFFFFFFFL );
592         first = i;
593        }
594        private void moveIndexToLast(final int i) {
595         if (size == 1 || last == i) return;
596         if (first == i) {
597          first = (int) link[i];
598          // Special case of SET_PREV(link[first], -1);
599          link[first] |= (-1 & 0xFFFFFFFFL) << 32;
600         }
601         else {
602          final long linki = link[i];
603          final int prev = (int) ( linki >>> 32 );
604          final int next = (int) linki;
605          link[prev] ^= ( ( link[prev] ^ ( linki & 0xFFFFFFFFL ) ) & 0xFFFFFFFFL );
606          link[next] ^= ( ( link[next] ^ ( linki & 0xFFFFFFFF00000000L ) ) & 0xFFFFFFFF00000000L );
607         }
608         link[last] ^= ( ( link[last] ^ ( i & 0xFFFFFFFFL ) ) & 0xFFFFFFFFL );
609         link[i] = ( ( last & 0xFFFFFFFFL ) << 32 ) | ( -1 & 0xFFFFFFFFL );
610         last = i;
611        }
612        /** Adds a key to the set; if the key is already present, it is moved to the first position of the iteration order.
613
614         *
615
616         * @param k the key.
617
618         * @return true if the key was not present.
619
620         */
621        public boolean addAndMoveToFirst(final byte k) {
622         int pos;
623         if (( strategy.equals( (k), ((byte)0) ) )) {
624          if (containsNull) {
625           moveIndexToFirst(n);
626           return false;
627          }
628          containsNull = true;
629          pos = n;
630         }
631         else {
632          // The starting point.
633          final byte key[] = this.key;
634          pos = ( it.unimi.dsi.fastutil.HashCommon.mix( strategy.hashCode(k) ) ) & mask;
635          // There's always an unused entry. TODO
636          while(! ( (key[pos]) == ((byte)0) )) {
637           if (( strategy.equals( (k), (key[pos]) ) )) {
638            moveIndexToFirst(pos);
639            return false;
640           }
641           pos = (pos + 1) & mask;
642          }
643         }
644         key[pos] = k;
645         if (size == 0) {
646          first = last = pos;
647          // Special case of SET_UPPER_LOWER(link[pos], -1, -1);
648          link[pos] = -1L;
649         }
650         else {
651          link[first] ^= ( ( link[first] ^ ( ( pos & 0xFFFFFFFFL ) << 32 ) ) & 0xFFFFFFFF00000000L );
652          link[pos] = ( ( -1 & 0xFFFFFFFFL ) << 32 ) | ( first & 0xFFFFFFFFL );
653          first = pos;
654         }
655         if (size++ >= maxFill) rehash(arraySize(size, f));
656         if (ASSERTS) checkTable();
657         return true;
658        }
659        /** Adds a key to the set; if the key is already present, it is moved to the last position of the iteration order.
660
661         *
662
663         * @param k the key.
664
665         * @return true if the key was not present.
666
667         */
668        public boolean addAndMoveToLast(final byte k) {
669         int pos;
670         if (( strategy.equals( (k), ((byte)0) ) )) {
671          if (containsNull) {
672           moveIndexToLast(n);
673           return false;
674          }
675          containsNull = true;
676          pos = n;
677         }
678         else {
679          // The starting point.
680          final byte key[] = this.key;
681          pos = ( it.unimi.dsi.fastutil.HashCommon.mix( strategy.hashCode(k) ) ) & mask;
682          // There's always an unused entry.
683          while(! ( (key[pos]) == ((byte)0) )) {
684           if (( strategy.equals( (k), (key[pos]) ) )) {
685            moveIndexToLast(pos);
686            return false;
687           }
688           pos = (pos + 1) & mask;
689          }
690         }
691         key[pos] = k;
692         if (size == 0) {
693          first = last = pos;
694          // Special case of SET_UPPER_LOWER(link[pos], -1, -1);
695          link[pos] = -1L;
696         }
697         else {
698          link[last] ^= ( ( link[last] ^ ( pos & 0xFFFFFFFFL ) ) & 0xFFFFFFFFL );
699          link[pos] = ( ( last & 0xFFFFFFFFL ) << 32 ) | ( -1 & 0xFFFFFFFFL );
700          last = pos;
701         }
702         if (size++ >= maxFill) rehash(arraySize(size, f));
703         if (ASSERTS) checkTable();
704         return true;
705        }

759        /** Modifies the {@link #link} vector so that the given entry is removed.
760
761         * This method will complete in constant time.
762
763         *
764
765         * @param i the index of an entry.
766
767         */
768        protected void fixPointers(final int i) {
769         if (size == 0) {
770          first = last = -1;
771          return;
772         }
773         if (first == i) {
774          first = (int) link[i];
775          if (0 <= first) {
776           // Special case of SET_PREV(link[first], -1)
777           link[first] |= (-1 & 0xFFFFFFFFL) << 32;
778          }
779          return;
780         }
781         if (last == i) {
782          last = (int) ( link[i] >>> 32 );
783          if (0 <= last) {
784           // Special case of SET_NEXT(link[last], -1)
785           link[last] |= -1 & 0xFFFFFFFFL;
786          }
787          return;
788         }
789         final long linki = link[i];
790         final int prev = (int) ( linki >>> 32 );
791         final int next = (int) linki;
792         link[prev] ^= ( ( link[prev] ^ ( linki & 0xFFFFFFFFL ) ) & 0xFFFFFFFFL );
793         link[next] ^= ( ( link[next] ^ ( linki & 0xFFFFFFFF00000000L ) ) & 0xFFFFFFFF00000000L );
794        }
795        /** Modifies the {@link #link} vector for a shift from s to d.
796
797         * This method will complete in constant time.
798
799         *
800
801         * @param s the source position.
802
803         * @param d the destination position.
804
805         */
806        protected void fixPointers(int s, int d) {
807         if (size == 1) {
808          first = last = d;
809          // Special case of SET(link[d], -1, -1)
810          link[d] = -1L;
811          return;
812         }
813         if (first == s) {
814          first = d;
815          link[(int) link[s]] ^= ( ( link[(int) link[s]] ^ ( ( d & 0xFFFFFFFFL ) << 32 ) ) & 0xFFFFFFFF00000000L );
816          link[d] = link[s];
817          return;
818         }
819         if (last == s) {
820          last = d;
821          link[(int) ( link[s] >>> 32 )] ^= ( ( link[(int) ( link[s] >>> 32 )] ^ ( d & 0xFFFFFFFFL ) ) & 0xFFFFFFFFL );
822          link[d] = link[s];
823          return;
824         }
825         final long links = link[s];
826         final int prev = (int) ( links >>> 32 );
827         final int next = (int) links;
828         link[prev] ^= ( ( link[prev] ^ ( d & 0xFFFFFFFFL ) ) & 0xFFFFFFFFL );
829         link[next] ^= ( ( link[next] ^ ( ( d & 0xFFFFFFFFL ) << 32 ) ) & 0xFFFFFFFF00000000L );
830         link[d] = links;
831        }

874        /** A list iterator over a linked set.
875
876         *
877
878         * <P>This class provides a list iterator over a linked hash set. The constructor runs in constant time.
879
880         */
881        private class SetIterator extends AbstractByteListIterator {
882         /** The entry that will be returned by the next call to {@link java.util.ListIterator#previous()} (or {@code null} if no previous entry exists). */
883         int prev = -1;
884         /** The entry that will be returned by the next call to {@link java.util.ListIterator#next()} (or {@code null} if no next entry exists). */
885         int next = -1;
886         /** The last entry that was returned (or -1 if we did not iterate or used {@link #remove()}). */
887         int curr = -1;
888         /** The current index (in the sense of a {@link java.util.ListIterator}). When -1, we do not know the current index.*/
889         int index = -1;
890         SetIterator() {
891          next = first;
892          index = 0;
893         }
894         SetIterator(byte from) {
895          if (( strategy.equals( (from), ((byte)0) ) )) {
896           if (ByteLinkedOpenCustomHashSet.this.containsNull) {
897            next = (int) link[n];
898            prev = n;
899            return;
900           }
901           else throw new NoSuchElementException("The key " + from + " does not belong to this set.");
902          }
903          if (( strategy.equals( (key[last]), (from) ) )) {
904           prev = last;
905           index = size;
906           return;
907          }
908          // The starting point.
909          final byte key[] = ByteLinkedOpenCustomHashSet.this.key;
910          int pos = ( it.unimi.dsi.fastutil.HashCommon.mix( strategy.hashCode(from) ) ) & mask;
911          // There's always an unused entry.
912          while(! ( (key[pos]) == ((byte)0) )) {
913           if (( strategy.equals( (key[pos]), (from) ) )) {
914            // Note: no valid index known.
915            next = (int) link[pos];
916            prev = pos;
917            return;
918           }
919           pos = (pos + 1) & mask;
920          }
921          throw new NoSuchElementException("The key " + from + " does not belong to this set.");
922         }
923         public boolean hasNext() { return next != -1; }
924         public boolean hasPrevious() { return prev != -1; }
925         public byte nextByte() {
926          if (! hasNext()) throw new NoSuchElementException();
927          curr = next;
928          next = (int) link[curr];
929          prev = curr;
930          if (index >= 0) index++;
931          if (ASSERTS) assert curr == n || ! ( (key[curr]) == ((byte)0) ) : "Position " + curr + " is not used";
932          return key[curr];
933         }
934         public byte previousByte() {
935          if (! hasPrevious()) throw new NoSuchElementException();
936          curr = prev;
937          prev = (int) ( link[curr] >>> 32 );
938          next = curr;
939          if (index >= 0) index--;
940          return key[curr];
941         }
942         private final void ensureIndexKnown() {
943          if (index >= 0) return;
944          if (prev == -1) {
945           index = 0;
946           return;
947          }
948          if (next == -1) {
949           index = size;
950           return;
951          }
952          int pos = first;
953          index = 1;
954          while(pos != prev) {
955           pos = (int) link[pos];
956           index++;
957          }
958         }
959         public int nextIndex() {
960          ensureIndexKnown();
961          return index;
962         }
963         public int previousIndex() {
964          ensureIndexKnown();
965          return index - 1;
966         }
967         public void remove() {
968          ensureIndexKnown();
969          if (curr == -1) throw new IllegalStateException();
970          if (curr == prev) {
971           /* If the last operation was a next(), we are removing an entry that preceeds
972
973                                   the current index, and thus we must decrement it. */
974           index--;
975           prev = (int) ( link[curr] >>> 32 );
976          }
977          else
978           next = (int) link[curr];
979          size--;
980          /* Now we manually fix the pointers. Because of our knowledge of next
981
982                           and prev, this is going to be faster than calling fixPointers(). */
983          if (prev == -1) first = next;
984          else
985           link[prev] ^= ( ( link[prev] ^ ( next & 0xFFFFFFFFL ) ) & 0xFFFFFFFFL );
986          if (next == -1) last = prev;
987          else
988           link[next] ^= ( ( link[next] ^ ( ( prev & 0xFFFFFFFFL ) << 32 ) ) & 0xFFFFFFFF00000000L );
989          int last, slot, pos = curr;
990          curr = -1;
991          if (pos == n) {
992           ByteLinkedOpenCustomHashSet.this.containsNull = false;
993           ByteLinkedOpenCustomHashSet.this.key[n] = ((byte)0);
994          }
995          else {
996           byte curr;
997           final byte[] key = ByteLinkedOpenCustomHashSet.this.key;
998           // We have to horribly duplicate the shiftKeys() code because we need to update next/prev.
999           for(;;) {
1000            pos = ((last = pos) + 1) & mask;
1001            for(;;) {
1002             if (( (curr = key[pos]) == ((byte)0) )) {
1003              key[last] = ((byte)0);
1004              return;
1005             }
1006             slot = ( it.unimi.dsi.fastutil.HashCommon.mix( strategy.hashCode(curr) ) ) & mask;
1007             if (last <= pos ? last >= slot || slot > pos : last >= slot && slot > pos) break;
1008             pos = (pos + 1) & mask;
1009            }
1010            key[last] = curr;
1011            if (next == pos) next = last;
1012            if (prev == pos) prev = last;
1013            fixPointers(pos, last);
1014           }
1015          }
1016         }
1017        }

1147        /** Rehashes the set.
1148
1149         *
1150
1151         * <P>This method implements the basic rehashing strategy, and may be
1152
1153         * overriden by subclasses implementing different rehashing strategies (e.g.,
1154
1155         * disk-based rehashing). However, you should not override this method
1156
1157         * unless you understand the internal workings of this class.
1158
1159         *
1160
1161         * @param newN the new size
1162
1163         */
1164
1165        protected void rehash(final int newN) {
1166         final byte key[] = this.key;
1167         final int mask = newN - 1; // Note that this is used by the hashing macro
1168         final byte newKey[] = new byte[newN + 1];
1169         int i = first, prev = -1, newPrev = -1, t, pos;
1170         final long link[] = this.link;
1171         final long newLink[] = new long[newN + 1];
1172         first = -1;
1173         for(int j = size; j-- != 0;) {
1174          if (( strategy.equals( (key[i]), ((byte)0) ) )) pos = newN;
1175          else {
1176           pos = ( it.unimi.dsi.fastutil.HashCommon.mix( strategy.hashCode(key[i]) ) ) & mask;
1177           while (! ( (newKey[pos]) == ((byte)0) )) pos = (pos + 1) & mask;
1178          }
1179          newKey[pos] = key[i];
1180          if (prev != -1) {
1181           newLink[newPrev] ^= ( ( newLink[newPrev] ^ ( pos & 0xFFFFFFFFL ) ) & 0xFFFFFFFFL );
1182           newLink[pos] ^= ( ( newLink[pos] ^ ( ( newPrev & 0xFFFFFFFFL ) << 32 ) ) & 0xFFFFFFFF00000000L );
1183           newPrev = pos;
1184          }
1185          else {
1186           newPrev = first = pos;
1187           // Special case of SET(newLink[pos], -1, -1);
1188           newLink[pos] = -1L;
1189          }
1190          t = i;
1191          i = (int) link[i];
1192          prev = t;
1193         }
1194         this.link = newLink;
1195         this.last = newPrev;
1196         if (newPrev != -1)
1197          // Special case of SET_NEXT(newLink[newPrev], -1);
1198          newLink[newPrev] |= -1 & 0xFFFFFFFFL;
1199         n = newN;
1200         this.mask = mask;
1201         maxFill = maxFill(n, f);
1202         this.key = newKey;
1203        }
