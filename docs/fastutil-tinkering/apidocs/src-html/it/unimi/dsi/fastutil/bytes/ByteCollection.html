016package it.unimi.dsi.fastutil.bytes;
017import java.util.Collection;
018/** A type-specific {@link Collection}; provides some additional methods
019        * that use polymorphism to avoid (un)boxing.
020        *
021        * <P>Additionally, this class defines strengthens (again) {@link #iterator()}.
022        *
023        * @see Collection
024        */
025public interface ByteCollection extends Collection<Byte>, ByteIterable {
026        /** Returns a type-specific iterator on the elements of this collection.
027         *
028         * <p>Note that this specification strengthens the one given in
029         * {@link java.lang.Iterable#iterator()}, which was already
030         * strengthened in the corresponding type-specific class,
031         * but was weakened by the fact that this interface extends {@link Collection}.
032         *
033         * @return a type-specific iterator on the elements of this collection.
034         */
035        @Override
036        ByteIterator iterator();
037        /** Returns a type-specific iterator on this elements of this collection.
038         *
039         * @see #iterator()
040         * @deprecated As of <code>fastutil</code> 5, replaced by {@link #iterator()}.
041         */
042        @Deprecated
043        ByteIterator byteIterator();
044        /** Ensures that this collection contains the specified element (optional operation).
045         * @see Collection#add(Object)
046         */
047        boolean add(byte key);
048        /** Returns {@code true} if this collection contains the specified element.
049         * @see Collection#contains(Object)
050         */
051        boolean contains(byte key);
052        /** Removes a single instance of the specified element from this
053         * collection, if it is present (optional operation).
054         *
055         * <p>Note that this method should be called {@link java.util.Collection#remove(Object) remove()}, but the clash
056         * with the similarly named index-based method in the {@link java.util.List} interface
057         * forces us to use a distinguished name. For simplicity, the set interfaces reinstates
058         * <code>remove()</code>.
059         *
060         * @see Collection#remove(Object)
061         */
062        boolean rem(byte key);
063        /** {@inheritDoc}
064         * @deprecated Please use the corresponding type-specific method instead.
065         */
066        @Deprecated
067        @Override
068        boolean add(Byte key);
069        /** {@inheritDoc}
070         * @deprecated Please use the corresponding type-specific method instead.
071         */
072        @Deprecated
073        @Override
074        boolean contains(Object key);
075        /** {@inheritDoc}
076         * @deprecated Please use (and implement) the {@code rem()} method instead.
077         */
078        @Deprecated
079        @Override
080        boolean remove(Object key);
081        /** Returns a primitive type array containing the items of this collection.
082         * @return a primitive type array containing the items of this collection.
083         * @see Collection#toArray()
084         */
085        byte[] toByteArray();
086        /** Returns a primitive type array containing the items of this collection.
087         *
088         * <p>Note that, contrarily to {@link Collection#toArray(Object[])}, this
089         * methods just writes all elements of this collection: no special
090         * value will be added after the last one.
091         *
092         * @param a if this array is big enough, it will be used to store this collection.
093         * @return a primitive type array containing the items of this collection.
094         * @see Collection#toArray(Object[])
095         * @deprecated Please use {@code toArray()} instead&mdash;this method is redundant.
096         */
097        @Deprecated
098        byte[] toByteArray(byte a[]);
099        /** Returns an array containing all of the elements in this collection; the runtime type of the returned array is that of the specified array.
100         *
101         * <p>Note that, contrarily to {@link Collection#toArray(Object[])}, this
102         * methods just writes all elements of this collection: no special
103         * value will be added after the last one.
104         *
105         * @param a if this array is big enough, it will be used to store this collection.
106         * @return a primitive type array containing the items of this collection.
107         * @see Collection#toArray(Object[])
108         */
109        byte[] toArray(byte a[]);
110        /** Adds all elements of the given type-specific collection to this collection.
111         *
112         * @param c a type-specific collection.
113         * @see Collection#addAll(Collection)
114         * @return <code>true</code> if this collection changed as a result of the call.
115         */
116        boolean addAll(ByteCollection c);
117        /** Checks whether this collection contains all elements from the given type-specific collection.
118         *
119         * @param c a type-specific collection.
120         * @see Collection#containsAll(Collection)
121         * @return <code>true</code> if this collection contains all elements of the argument.
122         */
123        boolean containsAll(ByteCollection c);
124        /** Remove from this collection all elements in the given type-specific collection.
125         *
126         * @param c a type-specific collection.
127         * @see Collection#removeAll(Collection)
128         * @return <code>true</code> if this collection changed as a result of the call.
129         */
130        boolean removeAll(ByteCollection c);
131        /** Retains in this collection only elements from the given type-specific collection.
132         *
133         * @param c a type-specific collection.
134         * @see Collection#retainAll(Collection)
135         * @return <code>true</code> if this collection changed as a result of the call.
136         */
137        boolean retainAll(ByteCollection c);
138}
