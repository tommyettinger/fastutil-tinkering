016package it.unimi.dsi.fastutil.longs;
017/** A class providing static methods and objects that do useful things with comparators.
018        */
019public class LongComparators {
020        private LongComparators() {}
021        /** A type-specific comparator mimicking the natural order. */
022        protected static class NaturalImplicitComparator extends AbstractLongComparator implements java.io.Serializable {
023         private static final long serialVersionUID = 1L;
024         @Override
025         public final int compare(final long a, final long b) {
026          return ( Long.compare((a),(b)) );
027         }
028         private Object readResolve() { return NATURAL_COMPARATOR; }
029        };
030
031        public static final LongComparator NATURAL_COMPARATOR = new NaturalImplicitComparator();
032        /** A type-specific comparator mimicking the opposite of the natural order. */
033        protected static class OppositeImplicitComparator extends AbstractLongComparator implements java.io.Serializable {
034         private static final long serialVersionUID = 1L;
035         @Override
036         public final int compare(final long a, final long b) {
037          return - ( Long.compare((a),(b)) );
038         }
039         private Object readResolve() { return OPPOSITE_COMPARATOR; }
040        };
041
042        public static final LongComparator OPPOSITE_COMPARATOR = new OppositeImplicitComparator();
043        protected static class OppositeComparator extends AbstractLongComparator implements java.io.Serializable {
044         private static final long serialVersionUID = 1L;
045         private final LongComparator comparator;
046         protected OppositeComparator(final LongComparator c) {
047          comparator = c;
048         }
049         @Override
050         public final int compare(final long a, final long b) {
051          return comparator.compare(b, a);
052         }
053        };
054        /** Returns a comparator representing the opposite order of the given comparator.
055         *
056         * @param c a comparator.
057         * @return a comparator representing the opposite order of <code>c</code>.
058         */
059        public static LongComparator oppositeComparator(final LongComparator c) {
060         return new OppositeComparator (c);
061        }
062}
