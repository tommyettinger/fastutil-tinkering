016package it.unimi.dsi.fastutil.ints;
017/** A class providing static methods and objects that do useful things with type-specific priority queues.
018        *
019        * @see it.unimi.dsi.fastutil.PriorityQueue
020        */
021public class IntPriorityQueues {
022        private IntPriorityQueues() {}
023        /** A synchronized wrapper class for priority queues. */
024        public static class SynchronizedPriorityQueue implements IntPriorityQueue {
025         final protected IntPriorityQueue q;
026         final protected Object sync;
027         protected SynchronizedPriorityQueue(final IntPriorityQueue q, final Object sync) {
028          this.q = q;
029          this.sync = sync;
030         }
031         protected SynchronizedPriorityQueue(final IntPriorityQueue q) {
032          this.q = q;
033          this.sync = this;
034         }
035         @Override
036         public void enqueue(int x) { synchronized(sync) { q.enqueue(x); } }
037         @Override
038         public int dequeueInt() { synchronized(sync) { return q.dequeueInt(); } }
039         @Override
040         public int firstInt() { synchronized(sync) { return q.firstInt(); } }
041         @Override
042         public int lastInt() { synchronized(sync) { return q.lastInt(); } }
043         @Override
044         public boolean isEmpty() { synchronized(sync) { return q.isEmpty(); } }
045         @Override
046         public int size() { synchronized(sync) { return q.size(); } }
047         @Override
048         public void clear() { synchronized(sync) { q.clear(); } }
049         @Override
050         public void changed() { synchronized(sync) { q.changed(); } }
051         @Override
052         public IntComparator comparator() { synchronized(sync) { return q.comparator(); } }
053         @Deprecated
054         @Override
055         public void enqueue(Integer x) { synchronized(sync) { q.enqueue(x); } }
056         @Deprecated
057         @Override
058         public Integer dequeue() { synchronized(sync) { return q.dequeue(); } }
059         @Deprecated
060         @Override
061         public Integer first() { synchronized(sync) { return q.first(); } }
062         @Deprecated
063         @Override
064         public Integer last() { synchronized(sync) { return q.last(); } }
065         @Override
066         public int hashCode() { synchronized(sync) { return q.hashCode(); } }
067         @Override
068         public boolean equals(final Object o) { if (o == this) return true; synchronized(sync) { return q.equals(o); } }
069         private void writeObject(java.io.ObjectOutputStream s) throws java.io.IOException {
070          synchronized(sync) { s.defaultWriteObject(); }
071         }
072        }
073        /** Returns a synchronized type-specific priority queue backed by the specified type-specific priority queue.
074         *
075         * @param q the priority queue to be wrapped in a synchronized priority queue.
076         * @return a synchronized view of the specified priority queue.
077         */
078        public static IntPriorityQueue synchronize(final IntPriorityQueue q) { return new SynchronizedPriorityQueue(q); }
079        /** Returns a synchronized type-specific priority queue backed by the specified type-specific priority queue, using an assigned object to synchronize.
080         *
081         * @param q the priority queue to be wrapped in a synchronized priority queue.
082         * @param sync an object that will be used to synchronize the access to the priority queue.
083         * @return a synchronized view of the specified priority queue.
084         */
085        public static IntPriorityQueue synchronize(final IntPriorityQueue q, final Object sync) { return new SynchronizedPriorityQueue(q, sync); }
086}
