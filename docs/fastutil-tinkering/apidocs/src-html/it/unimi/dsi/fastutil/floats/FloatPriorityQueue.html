016package it.unimi.dsi.fastutil.floats;
017import java.util.NoSuchElementException;
018import it.unimi.dsi.fastutil.PriorityQueue;
019/** A type-specific {@link PriorityQueue}; provides some additional methods that use polymorphism to avoid (un)boxing.
020        *
021        * <P>Additionally, this interface strengthens {@link #comparator()}.
022        */
023public interface FloatPriorityQueue extends PriorityQueue<Float> {
024        /** Enqueues a new element.
025         * @see #enqueue(Object)
026         * @param x the element to enqueue.
027         */
028        void enqueue(float x);
029        /** Dequeues the {@linkplain #first() first} element from the queue.
030         * @see #dequeue()
031         * @return the dequeued element.
032         * @throws NoSuchElementException if the queue is empty.
033         */
034        float dequeueFloat();
035        /** Returns the first element of the queue.
036         * @see #first()
037         * @return the first element.
038         * @throws NoSuchElementException if the queue is empty.
039         */
040        float firstFloat();
041        /** Returns the last element of the queue, that is, the element the would be dequeued last (optional operation).
042         * @see #last()
043         * @return the last element.
044         * @throws NoSuchElementException if the queue is empty.
045         */
046        float lastFloat();
047        /** Returns the comparator associated with this priority queue, or null if it uses its elements' natural ordering.
048         *
049         * <P>Note that this specification strengthens the one given in {@link PriorityQueue#comparator()}.
050         * @see PriorityQueue#comparator()
051         * @return the comparator associated with this priority queue.
052         */
053        @Override
054        FloatComparator comparator();
055        /** {@inheritDoc}
056         * @deprecated Please use the corresponding type-specific method instead. */
057        @Deprecated
058        @Override
059        void enqueue(Float x);
060        /** {@inheritDoc}
061         * @deprecated Please use the corresponding type-specific method instead. */
062        @Deprecated
063        @Override
064        Float dequeue();
065        /** {@inheritDoc}
066         * @deprecated Please use the corresponding type-specific method instead. */
067        @Deprecated
068        @Override
069        Float first();
070        /** {@inheritDoc}
071         * @deprecated Please use the corresponding type-specific method instead. */
072        @Deprecated
073        @Override
074        Float last();
075}
