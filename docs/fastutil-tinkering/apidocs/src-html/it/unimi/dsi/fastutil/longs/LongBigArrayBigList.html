016package it.unimi.dsi.fastutil.longs;
017import java.util.Collection;
018import java.util.Iterator;
019import java.util.RandomAccess;
020import java.util.NoSuchElementException;
021import it.unimi.dsi.fastutil.BigArrays;
022/** A type-specific big list based on a big array; provides some additional methods that use polymorphism to avoid (un)boxing.
023        *
024        * <P>This class implements a lightweight, fast, open, optimized,
025        * reuse-oriented version of big-array-based big lists. Instances of this class
026        * represent a big list with a big array that is enlarged as needed when new entries
027        * are created (by doubling the current length), but is
028        * <em>never</em> made smaller (even on a {@link #clear()}). A family of
029        * {@linkplain #trim() trimming methods} lets you control the size of the
030        * backing big array; this is particularly useful if you reuse instances of this class.
031        * Range checks are equivalent to those of {@link java.util}'s classes, but
032        * they are delayed as much as possible. The backing big array is exposed by the
033        * {@link #elements()} method.
034        *
035        * <p>This class implements the bulk methods <code>removeElements()</code>,
036        * <code>addElements()</code> and <code>getElements()</code> using
037        * high-performance system calls (e.g., {@link
038        * System#arraycopy(Object,int,Object,int,int) System.arraycopy()} instead of
039        * expensive loops.
040        *
041        * @see java.util.ArrayList
042        */
043public class LongBigArrayBigList extends AbstractLongBigList implements RandomAccess, Cloneable, java.io.Serializable {
044        private static final long serialVersionUID = -7046029254386353130L;
045        /** The initial default capacity of a big-array big list. */
046        public final static int DEFAULT_INITIAL_CAPACITY = 16;
047        /** The backing big array. */
048        protected transient long a[][];
049        /** The current actual size of the big list (never greater than the backing-array length). */
050        protected long size;
051        private static final boolean ASSERTS = false;
052        /** Creates a new big-array big list using a given array.
053         *
054         * <P>This constructor is only meant to be used by the wrapping methods.
055         *
056         * @param a the big array that will be used to back this big-array big list.
057         */
058        protected LongBigArrayBigList(final long a[][], boolean dummy) {
059         this.a = a;
060        }
061        /** Creates a new big-array big list with given capacity.
062         *
063         * @param capacity the initial capacity of the array list (may be 0).
064         */
065
066        public LongBigArrayBigList(final long capacity) {
067         if (capacity < 0) throw new IllegalArgumentException("Initial capacity (" + capacity + ") is negative");
068         a = LongBigArrays.newBigArray(capacity);
069        }
070        /** Creates a new big-array big list with {@link #DEFAULT_INITIAL_CAPACITY} capacity.
071         */
072        public LongBigArrayBigList() {
073         this(DEFAULT_INITIAL_CAPACITY);
074        }
075        /** Creates a new big-array big list and fills it with a given type-specific collection.
076         *
077         * @param c a type-specific collection that will be used to fill the array list.
078         */
079        public LongBigArrayBigList(final LongCollection c) {
080         this(c.size());
081         for(LongIterator i = c.iterator(); i.hasNext();) add(i.nextLong());
082        }
083        /** Creates a new big-array big list and fills it with a given type-specific list.
084         *
085         * @param l a type-specific list that will be used to fill the array list.
086         */
087        public LongBigArrayBigList(final LongBigList l) {
088         this(l.size64());
089         l.getElements(0, a, 0, size = l.size64());
090        }
091        /** Creates a new big-array big list and fills it with the elements of a given big array.
092         *
093         * <p>Note that this constructor makes it easy to build big lists from literal arrays
094         * declared as <code><var>type</var>[][] {{ <var>init_values</var> }}</code>.
095         * The only constraint is that the number of initialisation values is
096         * below {@link it.unimi.dsi.fastutil.BigArrays#SEGMENT_SIZE}.
097         *
098         * @param a a big array whose elements will be used to fill the array list.
099         */
100        public LongBigArrayBigList(final long a[][]) {
101         this(a, 0, LongBigArrays.length(a));
102        }
103        /** Creates a new big-array big list and fills it with the elements of a given big array.
104         *
105         * <p>Note that this constructor makes it easy to build big lists from literal arrays
106         * declared as <code><var>type</var>[][] {{ <var>init_values</var> }}</code>.
107         * The only constraint is that the number of initialisation values is
108         * below {@link it.unimi.dsi.fastutil.BigArrays#SEGMENT_SIZE}.
109         *
110         * @param a a big array whose elements will be used to fill the array list.
111         * @param offset the first element to use.
112         * @param length the number of elements to use.
113         */
114        public LongBigArrayBigList(final long a[][], final long offset, final long length) {
115         this(length);
116         LongBigArrays.copy(a, offset, this.a, 0, length);
117         size = length;
118        }
119        /** Creates a new big-array big list and fills it with the elements returned by an iterator..
120         *
121         * @param i an iterator whose returned elements will fill the array list.
122         */
123        public LongBigArrayBigList(final Iterator<? extends Long> i) {
124         this();
125         while(i.hasNext()) this.add(((i.next()).longValue()));
126        }
127        /** Creates a new big-array big list and fills it with the elements returned by a type-specific iterator..
128         *
129         * @param i a type-specific iterator whose returned elements will fill the array list.
130         */
131        public LongBigArrayBigList(final LongIterator i) {
132         this();
133         while(i.hasNext()) this.add(i.nextLong());
134        }
135        /** Returns the backing big array of this big list.
136         *
137         * @return the backing big array.
138         */
139        public long[][] elements() {
140         return a;
141        }
142        /** Wraps a given big array into a big-array list of given size.
143         *
144         * @param a a big array to wrap.
145         * @param length the length of the resulting big-array list.
146         * @return a new big-array list of the given size, wrapping the given big array.
147         */
148        public static LongBigArrayBigList wrap(final long a[][], final long length) {
149         if (length > LongBigArrays.length(a)) throw new IllegalArgumentException("The specified length (" + length + ") is greater than the array size (" + LongBigArrays.length(a) + ")");
150         final LongBigArrayBigList l = new LongBigArrayBigList (a, false);
151         l.size = length;
152         return l;
153        }
154        /** Wraps a given big array into a big-array big list.
155         *
156         * @param a a big array to wrap.
157         * @return a new big-array big list wrapping the given array.
158         */
159        public static LongBigArrayBigList wrap(final long a[][]) {
160         return wrap(a, LongBigArrays.length(a));
161        }
162        /** Ensures that this big-array big list can contain the given number of entries without resizing.
163         *
164         * @param capacity the new minimum capacity for this big-array big list.
165         */
166
167        public void ensureCapacity(final long capacity) {
168         a = LongBigArrays.ensureCapacity(a, capacity, size);
169         if (ASSERTS) assert size <= LongBigArrays.length(a);
170        }
171        /** Grows this big-array big list, ensuring that it can contain the given number of entries without resizing,
172         * and in case enlarging it at least by a factor of two.
173         *
174         * @param capacity the new minimum capacity for this big-array big list.
175         */
176
177        private void grow(final long capacity) {
178         a = LongBigArrays.grow(a, capacity, size);
179         if (ASSERTS) assert size <= LongBigArrays.length(a);
180        }
181        @Override
182        public void add(final long index, final long k) {
183         ensureIndex(index);
184         grow(size + 1);
185         if (index != size) LongBigArrays.copy(a, index, a, index + 1, size - index);
186         LongBigArrays.set(a, index, k);
187         size++;
188         if (ASSERTS) assert size <= LongBigArrays.length(a);
189        }
190        @Override
191        public boolean add(final long k) {
192         grow(size + 1);
193         LongBigArrays.set(a, size++, k);
194         if (ASSERTS) assert size <= LongBigArrays.length(a);
195         return true;
196        }
197        @Override
198        public long getLong(final long index) {
199         if (index >= size) throw new IndexOutOfBoundsException("Index (" + index + ") is greater than or equal to list size (" + size + ")");
200         return LongBigArrays.get(a, index);
201        }
202        @Override
203        public long indexOf(final long k) {
204         for(long i = 0; i < size; i++) if (( (k) == (LongBigArrays.get(a, i)) )) return i;
205         return -1;
206        }
207        @Override
208        public long lastIndexOf(final long k) {
209         for(long i = size; i-- != 0;) if (( (k) == (LongBigArrays.get(a, i)) )) return i;
210         return -1;
211        }
212        @Override
213        public long removeLong(final long index) {
214         if (index >= size) throw new IndexOutOfBoundsException("Index (" + index + ") is greater than or equal to list size (" + size + ")");
215         final long old = LongBigArrays.get(a, index);
216         size--;
217         if (index != size) LongBigArrays.copy(a, index + 1, a, index, size - index);
218         if (ASSERTS) assert size <= LongBigArrays.length(a);
219         return old;
220        }
221        @Override
222        public boolean rem(final long k) {
223         final long index = indexOf(k);
224         if (index == -1) return false;
225         removeLong(index);
226         if (ASSERTS) assert size <= LongBigArrays.length(a);
227         return true;
228        }
229        @Override
230        public long set(final long index, final long k) {
231         if (index >= size) throw new IndexOutOfBoundsException("Index (" + index + ") is greater than or equal to list size (" + size + ")");
232         long old = LongBigArrays.get(a, index);
233         LongBigArrays.set(a, index, k);
234         return old;
235        }
236        @Override
237        public boolean removeAll(final LongCollection c) {
238         long[] s = null, d = null;
239         int ss = -1, sd = BigArrays.SEGMENT_SIZE, ds = -1, dd = BigArrays.SEGMENT_SIZE;
240         for (long i = 0; i < size; i++) {
241          if (sd == BigArrays.SEGMENT_SIZE) {
242           sd = 0;
243           s = a[++ss];
244          }
245          if (!c.contains(s[sd])) {
246           if (dd == BigArrays.SEGMENT_SIZE) {
247            d = a[++ds];
248            dd = 0;
249           }
250           d[dd++] = s[sd];
251          }
252          sd++;
253         }
254         final long j = BigArrays.index(ds, dd);
255         final boolean modified = size != j;
256         size = j;
257         return modified;
258        }
259        @Override
260        public boolean removeAll(final Collection<?> c) {
261         long[] s = null, d = null;
262         int ss = -1, sd = BigArrays.SEGMENT_SIZE, ds = -1, dd = BigArrays.SEGMENT_SIZE;
263         for (long i = 0; i < size; i++) {
264          if (sd == BigArrays.SEGMENT_SIZE) {
265           sd = 0;
266           s = a[++ss];
267          }
268          if (!c.contains((Long.valueOf(s[sd])))) {
269           if (dd == BigArrays.SEGMENT_SIZE) {
270            d = a[++ds];
271            dd = 0;
272           }
273           d[dd++] = s[sd];
274          }
275          sd++;
276         }
277         final long j = BigArrays.index(ds, dd);
278         final boolean modified = size != j;
279         size = j;
280         return modified;
281        }
282        @Override
283        public void clear() {
284         size = 0;
285         if (ASSERTS) assert size <= LongBigArrays.length(a);
286        }
287        @Override
288        public long size64() {
289         return size;
290        }
291        @Override
292        public void size(final long size) {
293         if (size > LongBigArrays.length(a)) ensureCapacity(size);
294         if (size > this.size) LongBigArrays.fill(a, this.size, size, (0));
295         this.size = size;
296        }
297        @Override
298        public boolean isEmpty() {
299         return size == 0;
300        }
301        /** Trims this big-array big list so that the capacity is equal to the size.
302         *
303         * @see java.util.ArrayList#trimToSize()
304         */
305        public void trim() {
306         trim(0);
307        }
308        /** Trims the backing big array if it is too large.
309         *
310         * If the current big array length is smaller than or equal to
311         * <code>n</code>, this method does nothing. Otherwise, it trims the
312         * big-array length to the maximum between <code>n</code> and {@link #size64()}.
313         *
314         * <P>This method is useful when reusing big lists.  {@linkplain #clear() Clearing a
315         * big list} leaves the big-array length untouched. If you are reusing a big list
316         * many times, you can call this method with a typical
317         * size to avoid keeping around a very large big array just
318         * because of a few large transient big lists.
319         *
320         * @param n the threshold for the trimming.
321         */
322        public void trim(final long n) {
323         final long arrayLength = LongBigArrays.length(a);
324         if (n >= arrayLength || size == arrayLength) return;
325         a = LongBigArrays.trim(a, Math.max(n, size));
326         if (ASSERTS) assert size <= LongBigArrays.length(a);
327        }
328           /** Copies element of this type-specific list into the given big array using optimized system calls.
329         *
330         * @param from the start index (inclusive).
331         * @param a the destination big array.
332         * @param offset the offset into the destination array where to store the first element copied.
333         * @param length the number of elements to be copied.
334         */
335        @Override
336        public void getElements(final long from, final long[][] a, final long offset, final long length) {
337         LongBigArrays.copy(this.a, from, a, offset, length);
338        }
339        /** Removes elements of this type-specific list using optimized system calls.
340         *
341         * @param from the start index (inclusive).
342         * @param to the end index (exclusive).
343         */
344        @Override
345        public void removeElements(final long from, final long to) {
346         BigArrays.ensureFromTo(size, from, to);
347         LongBigArrays.copy(a, to, a, from, size - to);
348         size -= (to - from);
349        }
350        /** Adds elements to this type-specific list using optimized system calls.
351         *
352         * @param index the index at which to add elements.
353         * @param a the big array containing the elements.
354         * @param offset the offset of the first element to add.
355         * @param length the number of elements to add.
356         */
357        @Override
358        public void addElements(final long index, final long a[][], final long offset, final long length) {
359         ensureIndex(index);
360         LongBigArrays.ensureOffsetLength(a, offset, length);
361         grow(size + length);
362         LongBigArrays.copy(this.a, index, this.a, index + length, size - index);
363         LongBigArrays.copy(a, offset, this.a, index, length);
364         size += length;
365        }
366        @Override
367        public LongBigListIterator listIterator(final long index) {
368         ensureIndex(index);
369         return new AbstractLongBigListIterator () {
370           long pos = index, last = -1;
371           public boolean hasNext() { return pos < size; }
372           public boolean hasPrevious() { return pos > 0; }
373           public long nextLong() { if (! hasNext()) throw new NoSuchElementException(); return LongBigArrays.get(a, last = pos++); }
374           public long previousLong() { if (! hasPrevious()) throw new NoSuchElementException(); return LongBigArrays.get(a, last = --pos); }
375           public long nextIndex() { return pos; }
376           public long previousIndex() { return pos - 1; }
377           public void add(long k) {
378            LongBigArrayBigList.this.add(pos++, k);
379            last = -1;
380           }
381           public void set(long k) {
382            if (last == -1) throw new IllegalStateException();
383            LongBigArrayBigList.this.set(last, k);
384           }
385           public void remove() {
386            if (last == -1) throw new IllegalStateException();
387            LongBigArrayBigList.this.removeLong(last);
388            /* If the last operation was a next(), we are removing an element *before* us, and we must decrease pos correspondingly. */
389            if (last < pos) pos--;
390            last = -1;
391           }
392          };
393        }
394        @Override
395        public LongBigArrayBigList clone() {
396         LongBigArrayBigList c = new LongBigArrayBigList (size);
397         LongBigArrays.copy(a, 0, c.a, 0, size);
398         c.size = size;
399         return c;
400        }
401        /** Compares this type-specific big-array list to another one.
402         *
403         * <P>This method exists only for sake of efficiency. The implementation
404         * inherited from the abstract implementation would already work.
405         *
406         * @param l a type-specific big-array list.
407         * @return true if the argument contains the same elements of this type-specific big-array list.
408         */
409        public boolean equals(final LongBigArrayBigList l) {
410         if (l == this) return true;
411         long s = size64();
412         if (s != l.size64()) return false;
413         final long[][] a1 = a;
414         final long[][] a2 = l.a;
415         while(s-- != 0) if (LongBigArrays.get(a1, s) != LongBigArrays.get(a2, s)) return false;
416         return true;
417        }
418        /** Compares this big list to another big list.
419         *
420         * <P>This method exists only for sake of efficiency. The implementation
421         * inherited from the abstract implementation would already work.
422         *
423         * @param l a big list.
424         * @return a negative integer,
425         * zero, or a positive integer as this big list is lexicographically less than, equal
426         * to, or greater than the argument.
427         */
428
429        public int compareTo(final LongBigArrayBigList l) {
430         final long s1 = size64(), s2 = l.size64();
431         final long a1[][] = a, a2[][] = l.a;
432         long e1, e2;
433         int r, i;
434         for(i = 0; i < s1 && i < s2; i++) {
435          e1 = LongBigArrays.get(a1, i);
436          e2 = LongBigArrays.get(a2, i);
437          if ((r = ( Long.compare((e1),(e2)) )) != 0) return r;
438         }
439         return i < s2 ? -1 : (i < s1 ? 1 : 0);
440        }
441        private void writeObject(java.io.ObjectOutputStream s) throws java.io.IOException {
442         s.defaultWriteObject();
443         for(int i = 0; i < size; i++) s.writeLong(LongBigArrays.get(a, i));
444        }
445
446        private void readObject(java.io.ObjectInputStream s) throws java.io.IOException, ClassNotFoundException {
447         s.defaultReadObject();
448         a = LongBigArrays.newBigArray(size);
449         for(int i = 0; i < size; i++) LongBigArrays.set(a, i, s.readLong());
450        }
