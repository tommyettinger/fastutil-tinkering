016package it.unimi.dsi.fastutil.booleans;
017import it.unimi.dsi.fastutil.AbstractStack;
018/** An abstract class providing basic methods for implementing a type-specific stack interface.
019        *
020        * <P>To create a type-specific stack, you need both object methods and
021        * primitive-type methods. However, if you inherit from this class you need
022        * just one (anyone).
023        */
024public abstract class AbstractBooleanStack extends AbstractStack<Boolean> implements BooleanStack {
025        protected AbstractBooleanStack() {}
026        /** {@inheritDoc}
027         * <p>This implementation delegates to the corresponding generic method. */
028        @Override
029        public void push(boolean k) {
030         push(Boolean.valueOf(k));
031        }
032        /** {@inheritDoc}
033         * <p>This implementation delegates to the corresponding generic method. */
034        @Override
035        public boolean popBoolean() {
036         return pop().booleanValue();
037        }
038        /** {@inheritDoc}
039         * <p>This implementation delegates to the corresponding generic method. */
040        @Override
041        public boolean topBoolean() {
042         return top().booleanValue();
043        }
044        /** {@inheritDoc}
045         * <p>This implementation delegates to the corresponding generic method. */
046        @Override
047        public boolean peekBoolean(int i) {
048         return peek(i).booleanValue();
049        }
050        /** {@inheritDoc}
051         * <p>This implementation delegates to the corresponding type-specific method.
052         * @deprecated Please use the corresponding type-specific method instead. */
053        @Deprecated
054        @Override
055        public void push(Boolean o) {
056         push(o.booleanValue());
057        }
058        /** {@inheritDoc}
059         * <p>This implementation delegates to the corresponding type-specific method.
060         * @deprecated Please use the corresponding type-specific method instead. */
061        @Deprecated
062        @Override
063        public Boolean pop() {
064         return Boolean.valueOf(popBoolean());
065        }
066        /** {@inheritDoc}
067         * <p>This implementation delegates to the corresponding type-specific method.
068         * @deprecated Please use the corresponding type-specific method instead. */
069        @Deprecated
070        @Override
071        public Boolean top() {
072         return Boolean.valueOf(topBoolean());
073        }
074        /** {@inheritDoc}
075         * <p>This implementation delegates to the corresponding type-specific method.
076         * @deprecated Please use the corresponding type-specific method instead. */
077        @Deprecated
078        @Override
079        public Boolean peek(int i) {
080         return Boolean.valueOf(peekBoolean(i));
081        }
082}
