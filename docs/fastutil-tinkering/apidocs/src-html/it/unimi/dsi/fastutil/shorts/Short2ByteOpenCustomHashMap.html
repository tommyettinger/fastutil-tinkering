043/** A type-specific hash map with a fast, small-footprint implementation whose {@linkplain it.unimi.dsi.fastutil.Hash.Strategy hashing strategy}
044
045        * is specified at creation time.
046
047        *
048
049        * <P>Instances of this class use a hash table to represent a map. The table is
050
051        * filled up to a specified <em>load factor</em>, and then doubled in size to
052
053        * accommodate new entries. If the table is emptied below <em>one fourth</em>
054
055        * of the load factor, it is halved in size. However, halving is
056
057        * not performed when deleting entries from an iterator, as it would interfere
058
059        * with the iteration process.
060
061        *
062
063        * <p>Note that {@link #clear()} does not modify the hash table size.
064
065        * Rather, a family of {@linkplain #trim() trimming
066
067        * methods} lets you control the size of the table; this is particularly useful
068
069        * if you reuse instances of this class.
070
071        *
072
073        * @see Hash
074
075        * @see HashCommon
076
077        */

121        public Short2ByteOpenCustomHashMap(final int expected, final float f, final it.unimi.dsi.fastutil.shorts.ShortHash.Strategy strategy) {
122         this.strategy = strategy;
123         if (f <= 0 || f > 1) throw new IllegalArgumentException("Load factor must be greater than 0 and smaller than or equal to 1");
124         if (expected < 0) throw new IllegalArgumentException("The expected number of elements must be nonnegative");
125         this.f = f;
126         n = arraySize(expected, f);
127         mask = n - 1;
128         maxFill = maxFill(n, f);
129         key = new short[n + 1];
130         value = new byte[n + 1];
131        }

254        private int realSize() {

257        private void ensureCapacity(final int capacity) {
258         final int needed = arraySize(capacity, f);
259         if (needed > n) rehash(needed);
260        }
261        private void tryCapacity(final long capacity) {
262         final int needed = (int)Math.min(1 << 30, Math.max(2, HashCommon.nextPowerOfTwo((long)Math.ceil(capacity / f))));
263         if (needed > n) rehash(needed);
264        }
265        private byte removeEntry(final int pos) {
266         final byte oldValue = value[pos];
267         size--;
268         shiftKeys(pos);
269         if (size < maxFill / 4 && n > DEFAULT_INITIAL_SIZE) rehash(n / 2);
270         return oldValue;
271        }
272        private byte removeNullEntry() {
273         containsNullKey = false;
274         final byte oldValue = value[n];
275         size--;
276         if (size < maxFill / 4 && n > DEFAULT_INITIAL_SIZE) rehash(n / 2);
277         return oldValue;
278        }
279        @Override
280        public void putAll(Map<? extends Short,? extends Byte> m) {
281         if (f <= .5) ensureCapacity(m.size()); // The resulting map will be sized for m.size() elements
282         else tryCapacity(size() + m.size()); // The resulting map will be tentatively sized for size() + m.size() elements
283         super.putAll(m);
284        }
285        private int insert(final short k, final byte v) {
286         int pos;
287         if (( strategy.equals( (k), ((short)0) ) )) {
288          if (containsNullKey) return n;
289          containsNullKey = true;
290          pos = n;
291         }
292         else {
293          short curr;
294          final short[] key = this.key;
295          // The starting point.
296          if (! ( (curr = key[pos = ( it.unimi.dsi.fastutil.HashCommon.mix( strategy.hashCode(k) ) ) & mask]) == ((short)0) )) {
297           if (( strategy.equals( (curr), (k) ) )) return pos;
298           while(! ( (curr = key[pos = (pos + 1) & mask]) == ((short)0) ))
299            if (( strategy.equals( (curr), (k) ) )) return pos;
300          }
301         }
302         key[pos] = k;
303         value[pos] = v;
304         if (size++ >= maxFill) rehash(arraySize(size + 1, f));
305         if (ASSERTS) checkTable();
306         return -1;
307        }
308        @Override
309        public byte put(final short k, final byte v) {
310         final int pos = insert(k, v);
311         if (pos < 0) return defRetValue;
312         final byte oldValue = value[pos];
313         value[pos] = v;
314         return oldValue;
315        }
316        private byte addToValue(final int pos, final byte incr) {
317         final byte oldValue = value[pos];
318         value[pos] = (byte)(oldValue + incr);
319         return oldValue;

321        /** Adds an increment to value currently associated with a key.
322
323         *
324
325         * <P>Note that this method respects the {@linkplain #defaultReturnValue() default return value} semantics: when
326
327         * called with a key that does not currently appears in the map, the key
328
329         * will be associated with the default return value plus
330
331         * the given increment.
332
333         *
334
335         * @param k the key.
336
337         * @param incr the increment.
338
339         * @return the old value, or the {@linkplain #defaultReturnValue() default return value} if no value was present for the given key.
340
341         */
342        public byte addTo(final short k, final byte incr) {
343         int pos;
344         if (( strategy.equals( (k), ((short)0) ) )) {
345          if (containsNullKey) return addToValue(n, incr);
346          pos = n;
347          containsNullKey = true;
348         }
349         else {
350          short curr;
351          final short[] key = this.key;
352          // The starting point.
353          if (! ( (curr = key[pos = ( it.unimi.dsi.fastutil.HashCommon.mix( strategy.hashCode(k) ) ) & mask]) == ((short)0) )) {
354           if (( strategy.equals( (curr), (k) ) )) return addToValue(pos, incr);
355           while(! ( (curr = key[pos = (pos + 1) & mask]) == ((short)0) ))
356            if (( strategy.equals( (curr), (k) ) )) return addToValue(pos, incr);
357          }
358         }
359         key[pos] = k;
360         value[pos] = (byte)(defRetValue + incr);
361         if (size++ >= maxFill) rehash(arraySize(size + 1, f));
362         if (ASSERTS) checkTable();
363         return defRetValue;
364        }
365        /** Shifts left entries with the specified hash code, starting at the specified position,
366
367         * and empties the resulting free entry.
368
369         *
370
371         * @param pos a starting position.
372
373         */
374        protected final void shiftKeys(int pos) {
375         // Shift entries with the same hash.
376         int last, slot;
377         short curr;
378         final short[] key = this.key;
379         for(;;) {
380          pos = ((last = pos) + 1) & mask;
381          for(;;) {
382           if (( (curr = key[pos]) == ((short)0) )) {
383            key[last] = ((short)0);
384            return;
385           }
386           slot = ( it.unimi.dsi.fastutil.HashCommon.mix( strategy.hashCode(curr) ) ) & mask;
387           if (last <= pos ? last >= slot || slot > pos : last >= slot && slot > pos) break;
388           pos = (pos + 1) & mask;
389          }
390          key[last] = curr;
391          value[last] = value[pos];
392         }
393        }

395        public byte remove(final short k) {
396         if (( strategy.equals( ( k), ((short)0) ) )) {
397          if (containsNullKey) return removeNullEntry();
398          return defRetValue;
399         }
400         short curr;
401         final short[] key = this.key;
402         int pos;
403         // The starting point.
404         if (( (curr = key[pos = ( it.unimi.dsi.fastutil.HashCommon.mix( strategy.hashCode(k) ) ) & mask]) == ((short)0) )) return defRetValue;
405         if (( strategy.equals( (k), (curr) ) )) return removeEntry(pos);
406         while(true) {
407          if (( (curr = key[pos = (pos + 1) & mask]) == ((short)0) )) return defRetValue;
408          if (( strategy.equals( (k), (curr) ) )) return removeEntry(pos);
409         }
410        }

413        public byte get(final short k) {
414         if (( strategy.equals( ( k), ((short)0) ) )) return containsNullKey ? value[n] : defRetValue;
415         short curr;
416         final short[] key = this.key;
417         int pos;
418         // The starting point.
419         if (( (curr = key[pos = ( it.unimi.dsi.fastutil.HashCommon.mix( strategy.hashCode(k) ) ) & mask]) == ((short)0) )) return defRetValue;
420         if (( strategy.equals( (k), (curr) ) )) return value[pos];
421         // There's always an unused entry.
422         while(true) {
423          if (( (curr = key[pos = (pos + 1) & mask]) == ((short)0) )) return defRetValue;
424          if (( strategy.equals( (k), (curr) ) )) return value[pos];
425         }
426        }

571        /** An iterator over a hash map. */
572        private class MapIterator {
573         /** The index of the last entry returned, if positive or zero; initially, {@link #n}. If negative, the last
574
575                        entry returned was that of the key of index {@code - pos - 1} from the {@link #wrapped} list. */
576         int pos = n;
577         /** The index of the last entry that has been returned (more precisely, the value of {@link #pos} if {@link #pos} is positive,
578
579                        or {@link Integer#MIN_VALUE} if {@link #pos} is negative). It is -1 if either
580
581                        we did not return an entry yet, or the last returned entry has been removed. */
582         int last = -1;
583         /** A downward counter measuring how many entries must still be returned. */
584         int c = size;
585         /** A boolean telling us whether we should return the entry with the null key. */
586         boolean mustReturnNullKey = Short2ByteOpenCustomHashMap.this.containsNullKey;
587         /** A lazily allocated list containing keys of entries that have wrapped around the table because of removals. */
588         ShortArrayList wrapped;
589         public boolean hasNext() {
590          return c != 0;
591         }
592         public int nextEntry() {
593          if (! hasNext()) throw new NoSuchElementException();
594          c--;
595          if (mustReturnNullKey) {
596           mustReturnNullKey = false;
597           return last = n;
598          }
599          final short key[] = Short2ByteOpenCustomHashMap.this.key;
600          for(;;) {
601           if (--pos < 0) {
602            // We are just enumerating elements from the wrapped list.
603            last = Integer.MIN_VALUE;
604            final short k = wrapped.getShort(- pos - 1);
605            int p = ( it.unimi.dsi.fastutil.HashCommon.mix( strategy.hashCode(k) ) ) & mask;
606            while (! ( strategy.equals( (k), (key[p]) ) )) p = (p + 1) & mask;
607            return p;
608           }
609           if (! ( (key[pos]) == ((short)0) )) return last = pos;
610          }
611         }
612         /** Shifts left entries with the specified hash code, starting at the specified position,
613
614                 * and empties the resulting free entry.
615
616                 *
617
618                 * @param pos a starting position.
619
620                 */
621         private final void shiftKeys(int pos) {
622          // Shift entries with the same hash.
623          int last, slot;
624          short curr;
625          final short[] key = Short2ByteOpenCustomHashMap.this.key;
626          for(;;) {
627           pos = ((last = pos) + 1) & mask;
628           for(;;) {
629            if (( (curr = key[pos]) == ((short)0) )) {
630             key[last] = ((short)0);
631             return;
632            }
633            slot = ( it.unimi.dsi.fastutil.HashCommon.mix( strategy.hashCode(curr) ) ) & mask;
634            if (last <= pos ? last >= slot || slot > pos : last >= slot && slot > pos) break;
635            pos = (pos + 1) & mask;
636           }
637           if (pos < last) { // Wrapped entry.
638            if (wrapped == null) wrapped = new ShortArrayList (2);
639            wrapped.add(key[pos]);
640           }
641           key[last] = curr;
642           value[last] = value[pos];
643          }
644         }
645         public void remove() {
646          if (last == -1) throw new IllegalStateException();
647          if (last == n) {
648           containsNullKey = false;
649          }
650          else if (pos >= 0) shiftKeys(last);
651          else {
652           // We're removing wrapped entries.
653           Short2ByteOpenCustomHashMap.this.remove(wrapped.getShort(- pos - 1));
654           last = -1; // Note that we must not decrement size
655           return;
656          }
657          size--;
658          last = -1; // You can no longer remove this entry.
659          if (ASSERTS) checkTable();
660         }
661         public int skip(final int n) {
662          int i = n;
663          while(i-- != 0 && hasNext()) nextEntry();
664          return n - i - 1;
665         }
666        }
